// private Node type
typedef struct NodeObj* Node;

// private NodeObj type


typedef struct NodeObj{
   int data;
   Node prev;
   Node next;
} NodeObj;

// private ListObj type
typedef struct ListObj{
   Node front;
   Node back;
   Node cursor;
   int index;
   int length;
} ListObj;

// public List type
typedef struct ListObj* List;

// newNode()
// Returns reference to new Node object. Initializes next and data fields.
Node newNode(int data){
   Node N = malloc(sizeof(NodeObj));
   assert( N!=NULL );
   N->data = data;
   N->prev = NULL;
   N->next = NULL;
   return(N);
}

// freeNode()
// Frees heap memory pointed to by *pN, sets *pN to NULL.
void freeNode(Node* pN){
   if( pN!=NULL && *pN!=NULL ){


        free(*pN);
        *pN = NULL;
    }
}

// newList()
// Pre-condition: None
// Post-condition: Creates and returns a new empty List.
List newList() {
    List L = malloc(sizeof(ListObj));
    assert(L!= NULL);
    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->index = -1;
    L->length = 0;
    return(L);
}

// freeList()
// Pre-condition: None
// Post-condition: Frees all heap memory associated with *pL, and sets *pL to NULL
void freeList(List* pL) {
    if( pL!=NULL && *pL!=NULL ){
        clear(*pL);
        free(*pL);
        *pL = NULL;
    }
}

// length()
// Pre-condition: List argument is not NULL
// Post-condition: Returns the number of elements in L
int length(List L) {
    if( L==NULL ){
      fprintf(stderr, "List Error: calling length() on NULL reference\n");
      exit(EXIT_FAILURE);
    }
    return L->length;
} 

// index()
// Pre-condition: List argument is not NULL
// Post-condition: Returns index of cursor element if defined, -1 otherwise
int index(List L) {
    if( L==NULL ){


      fprintf(stderr, "List Error: calling index() on NULL reference\n");
      exit(EXIT_FAILURE);
    }
    return L->index;
}

// front()
// Pre-condition: List argument is not NULL; length()>0
// Post-condition: Returns front element of L.
int front(List L) {
    if( L==NULL ){
      fprintf(stderr, "List Error: calling front() on NULL reference\n");
      exit(EXIT_FAILURE);
    }
    if(length(L) <= 0){
        fprintf(stderr, "List Error: calling front() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    Node N = L->front;
    return N->data;
}

// back()
// Pre-condition: List argument is not NULL; length()>0
// Post-condition: Returns back element of L.
int back(List L) {
    if( L==NULL ){
      fprintf(stderr, "List Error: calling back() on NULL reference\n");
      exit(EXIT_FAILURE);
    }
    if(length(L) <= 0){
        fprintf(stderr, "List Error: calling back() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    Node N = L->back;
    return N->data;
}

// get()
// Pre-condition: List argument is not NULL; length()>0; index()>=0
// Post-condition: Returns cursor element of L
int get(List L) {
    if( L==NULL ){
      fprintf(stderr, "List Error: calling get() on NULL reference\n");
      exit(EXIT_FAILURE);
    }
    if(length(L) <= 0){
        fprintf(stderr, "List Error: calling get() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    if(index(L) < 0){
        fprintf(stderr, "List Error: calling get() on List with undefined cursor\n");
        exit(EXIT_FAILURE);
    }
    Node N = L->cursor;
    return N->data;
}

// equals()
// Pre-condition: List arguments are not NULL; length()>0; index()>=0
// Post-condition: Returns true iff Lists A and B contain the same sequence of elements, returns false otherwise
bool equals(List A, List B) {
    if( A==NULL ){
      fprintf(stderr, "List Error: calling equals() on NULL reference\n");
      exit(EXIT_FAILURE);
    }
    if( B==NULL ){
      fprintf(stderr, "List Error: calling equals() on NULL reference\n");
      exit(EXIT_FAILURE);
    }
    bool eq = true;
    if (length(A) != length(B)) {
        return false;
    }
    int i = 0;
    Node a_elem = A->front;
    Node b_elem = B->front;
    while (i < length(A)) {
        if (a_elem->data != b_elem->data) {
            eq = false;
        }
        a_elem = a_elem->next;
        b_elem = b_elem->next;
        i++;
    }
    return eq;
}

// clear()
// Pre-condition: List argument is not NULL
// Post-condition: Resets L to its original empty state
void clear(List L) {
    if( L==NULL ){
      fprintf(stderr, "List Error: calling clear() on NULL reference\n");
      exit(EXIT_FAILURE);
    }
    if (length(L) == 0) {
        L->front = NULL;
        L->back = NULL;
        L->cursor = NULL;
        L->index = -1;
    }
    else {
        int i = 0;
        Node N = L->front;
        while (i < length(L)) {
            if (N->next != NULL) {
                N = N->next;
                freeNode(&(N->prev));
            } else {
                freeNode(&N);
            }
            i++;
        }
        L->front = NULL;
        L->back = NULL;
        L->cursor = NULL;
        L->index = -1;
        L->length = L->length-i;
    }
}

// set()
// Pre-condition: List argument is not NULL, length()>0, index()>=0
// Post-condition: Overwrites the cursor elementâ€™s data with x
void set(List L, int x) {
    if( L==NULL ){
      fprintf(stderr, "List Error: calling set() on NULL reference\n");
      exit(EXIT_FAILURE);
    }
    if(length(L) <= 0){
        fprintf(stderr, "List Error: calling set() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    if(index(L) < 0){
        fprintf(stderr, "List Error: calling set() on List with undefined cursor\n");
        exit(EXIT_FAILURE);
    }
    Node N = L->cursor;
    N->data = x;
}

// moveFront()
// Pre-condition: List argument is not NULL
// Post-condition: If L is non-empty, sets cursor under the front element, otherwise does nothing.
void moveFront(List L) {
    if( L==NULL ){
      fprintf(stderr, "List Error: calling moveFront() on NULL reference\n");
      exit(EXIT_FAILURE);
    }
    if(length(L) > 0){
        L->cursor = L->front;
        L->index = 0;
    }   
}

// moveBack()
// Pre-condition: List argument is not NULL
// Post-condition: If L is non-empty, sets cursor under the back element, otherwise does nothing.
void moveBack(List L) {
    if( L==NULL ){
      fprintf(stderr, "List Error: calling moveBack() on NULL reference\n");
      exit(EXIT_FAILURE);
    }
    if(length(L) > 0){
        L->cursor = L->back;
        L->index = length(L)-1;
    }  
} 

// movePrev()
// Pre-condition: List argument is not NULL
// Post-condition:  If cursor is defined and not at front, move cursor one step toward the front of L; if cursor is defined and at front, 
// cursor becomes undefined; if cursor is undefined, does nothing
void movePrev(List L) {
    if( L==NULL ){
      fprintf(stderr, "List Error: calling movePrev() on NULL reference\n");
      exit(EXIT_FAILURE);
    }
    if (L->index > 0) {
        Node N = L->cursor;
        L->cursor = N->prev;
        L->index--;
    }
    else if (L->index == 0) {
        L->cursor = NULL;
        L->index = -1;
    }
}

// moveNext()
// Pre-condition: List argument is not NULL
// Post-condition:   If cursor is defined and not at back, move cursor one step toward the back of L; if cursor is defined and at back, 
// cursor becomes undefined; if cursor is undefined, does nothing
void moveNext(List L) {
    if( L==NULL ){
      fprintf(stderr, "List Error: calling moveNext() on NULL reference\n");
      exit(EXIT_FAILURE);
    }
    if ((L->index >= 0) && (L->index < length(L)-1)) {
        Node N = L->cursor;
        L->cursor = N->next;
        L->index++;
    }
    else if (L->index == length(L)-1) {
        L->cursor = NULL;
        L->index = -1;
    }
} 

// prepend()
// Pre-condition: List argument is not NULL
// Post-condition: Insert new element into L. If L is non-empty, insertion takes place before front element.
void prepend(List L, int x) {
    if( L==NULL ){
      fprintf(stderr, "List Error: calling prepend() on NULL reference\n");
      exit(EXIT_FAILURE);
    }
    if (length(L) == 0) {
        L->front = L->back = newNode(x);
        L->length++;
    } else {
        Node N = newNode(x);
        L->front->prev = N;
        N->next = L->front;
        L->front = N;
        L->length++;
        if (index(L) >= 0) {
            L->index++;
        }
    }
}

// append()
// Pre-condition: List argument is not NULL
// Post-condition: Insert new element into L. If L is non-empty, insertion takes place after back element.
void append(List L, int x) {
    if( L==NULL ){
      fprintf(stderr, "List Error: calling append() on NULL reference\n");
      exit(EXIT_FAILURE);
    }
    if (length(L) == 0) {
        L->front = L->back = newNode(x);
        L->length++;
    } else {
        Node N = newNode(x);
        L->back->next = N;
        N->prev = L->back;
        L->back = N;
        L->length++;
    }  
}

// insertBefore()
// Pre-condition: List argument is not NULL, length()>0, index()>=0
// Post-condition: Insert new element before cursor
void insertBefore(List L, int x) {
    if( L==NULL ){
      fprintf(stderr, "List Error: calling insertBefore() on NULL reference\n");
      exit(EXIT_FAILURE);
    }


    if(length(L) <= 0){
        fprintf(stderr, "List Error: calling insertBefore() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    if(index(L) < 0){
        fprintf(stderr, "List Error: calling insertBefore() on List with undefined cursor\n");
        exit(EXIT_FAILURE);
    }
    if (index(L) == 0) {
        prepend(L, x);
    } else {
        Node N = newNode(x);
        Node P = L->cursor->prev;
        N->prev = P;
        P->next = N;
        L->cursor->prev = N;
        N->next = L->cursor;
        L->length++;
        L->index++;
    }
}

// insertAfter()
// Pre-condition: List argument is not NULL, length()>0, index()>=0
// Post-condition: Insert new element after cursor
void insertAfter(List L, int x) {
    if( L==NULL ){
      fprintf(stderr, "List Error: calling insertAfter() on NULL reference\n");
      exit(EXIT_FAILURE);
    }
    if(length(L) <= 0){
        fprintf(stderr, "List Error: calling insertAfter() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    if(index(L) < 0){
        fprintf(stderr, "List Error: calling insertAfter() on List with undefined cursor\n");
        exit(EXIT_FAILURE);
    }
    if (index(L) == length(L)-1) {
        append(L, x);
    } else {
        Node N = newNode(x);
        Node NE = L->cursor->next;
        N->next = NE;
        NE->prev = N;
        L->cursor->next = N;
        N->prev = L->cursor;
        L->length++;
    }    
}

// deleteFront()
// Pre-condition: List argument is not NULL, length()>0
// Post-condition: Delete the front element
void deleteFront(List L) {
    if( L==NULL ){
      fprintf(stderr, "List Error: calling deleteFront() on NULL reference\n");
      exit(EXIT_FAILURE);
    }
    if(length(L) <= 0){
        fprintf(stderr, "List Error: calling deleteFront() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    if (index(L) > 0) {
        L->index--;
    }
    else if (index(L) == 0) {
        L->index = -1;
        L->cursor = NULL;
    } 
    if (length(L) == 1) {
        clear(L);
    } else {
        Node N = L->front;
        L->front = N->next;
        L->front->prev = NULL;
        freeNode(&N);
        L->length--;
    }
}

// deleteBack()
// Pre-condition: List argument is not NULL, length()>0
// Post-condition: Delete the back element
void deleteBack(List L) {
    if( L==NULL ){
      fprintf(stderr, "List Error: calling deleteBack() on NULL reference\n");
      exit(EXIT_FAILURE);
    }
    if(length(L) <= 0){
        fprintf(stderr, "List Error: calling deleteBack() on an empty List\n");
        exit(EXIT_FAILURE);
    }  
    if (L->cursor == L->back) {
        L->index = -1;
        L->cursor = NULL;
    } 
    if (length(L) == 1) {
        clear(L);
    } else {
        Node N = L->back;
        L->back = N->prev;
        L->back->next = NULL;
        freeNode(&N);
        L->length--;
    }
}

// delete()
// Pre-condition: List argument is not NULL, length()>0, index()>=0
// Post-condition: Delete cursor element, making cursor undefined
void delete(List L) {
    if( L==NULL ){
      fprintf(stderr, "List Error: calling delete() on NULL reference\n");
      exit(EXIT_FAILURE);
    }
    if(length(L) <= 0){
        fprintf(stderr, "List Error: calling delete() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    if(index(L) < 0){
        fprintf(stderr, "List Error: calling delete() on List with undefined cursor\n");
        exit(EXIT_FAILURE);
    }
    if (L->cursor == L->front) {
        deleteFront(L);
    }
    else if (L->cursor == L->back) {
        deleteBack(L);
    }
    else {
        Node P = L->cursor->prev;
        Node NE = L->cursor->next;
        P->next = NE;
        NE->prev = P;
        Node N = L->cursor;
        freeNode(&N);
        L->cursor = NULL;
        L->index = -1;
        L->length--;
    }
}

// printList()
// Pre-condition: List argument is not NULL
// Post-condition: Prints to the file pointed to by out, a string representation of L consisting of a space 
// separated sequence of integers, with front on left.
void printList(FILE* out, List L) {
    if( L==NULL ){
      fprintf(stderr, "List Error: calling printList() on NULL reference\n");
      exit(EXIT_FAILURE);
    }
    moveFront(L);
    while(index(L)>=0){
        fprintf(out, "%d ", get(L));
        moveNext(L);
    }
}

// copyList()
// Pre-condition: List argument is not NULL
// Post-condition:  Returns a new List representing the same integer sequence as L. The cursor in the new list is undefined, 
// regardless of the state of the cursor in L. The state of L is unchanged.
List copyList(List L) {
    if( L==NULL ){
      fprintf(stderr, "List Error: calling copyList() on NULL reference\n");
      exit(EXIT_FAILURE);
    }
    List NL = newList();
    int i = 0;
    Node orig_elem = L->front;
    while (i < length(L)) {
        append(NL, orig_elem->data);
        orig_elem = orig_elem->next;
        i++;
    }
    return NL;
}
