#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>
#include <string.h>
#include "List.h"

// Node structure


typedef struct NodeObj {
    ListElement data;
    struct NodeObj* next;
    struct NodeObj* prev;
} NodeObj;

// Node type
typedef NodeObj* Node;

// List structure
typedef struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
    int index;
} ListObj;

// List type
typedef ListObj* List;

// Constructors-Destructors ---------------------------------------------------
List newList(void) {
    List L = malloc(sizeof(ListObj));
    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->length = 0;
    L->index = -1;
    return L;
}

void freeList(List* pL) {
    if (pL != NULL && *pL != NULL) {
        while ((*pL)->length > 0) {
            deleteFront(*pL);
        }
        free(*pL);


        *pL = NULL;
    }
}

// Access functions -----------------------------------------------------------
int length(List L) {
    return L->length;
}

int index(List L) {
    return L->index;
}

int front(List L) {
    if (L->length == 0) {
        printf("List Error: calling front() on an empty List\n");
        exit(1);
    }
    return L->front->data;
}

int back(List L) {
    if (L->length == 0) {
        printf("List Error: calling back() on an empty List\n");
        exit(1);
    }
    return L->back->data;
}

int get(List L) {
    if (L->length == 0 || L->index == -1) {
        printf("List Error: calling get() on an empty List or undefined cursor\n");
        exit(1);
    }
    return L->cursor->data;
}

bool equals(List A, List B) {
    if (A->length != B->length) return false;
    Node nodeA = A->front;
    Node nodeB = B->front;
    while (nodeA != NULL) {
        if (nodeA->data != nodeB->data) return false;


        nodeA = nodeA->next;
        nodeB = nodeB->next;
    }
    return true;
}

// Manipulation procedures ----------------------------------------------------
void clear(List L) {
    while (L->length > 0) {
        deleteFront(L);
    }
}

void set(List L, int x) {
    if (L->length == 0 || L->index == -1) {
        printf("List Error: calling set() on an empty List or undefined cursor\n");
        exit(1);
    }
    L->cursor->data = x;
}

void moveFront(List L) {
    if (L->length > 0) {
        L->cursor = L->front;
        L->index = 0;
    }
}

void moveBack(List L) {
    if (L->length > 0) {
        L->cursor = L->back;


        L->index = L->length - 1;
    }
}

void movePrev(List L) {
    if (L->cursor != NULL && L->index != 0) {
        L->cursor = L->cursor->prev;
        L->index--;
    } else {
        L->cursor = NULL;
        L->index = -1;
    }
}

void moveNext(List L) {
    if (L->cursor != NULL && L->index != L->length - 1) {
        L->cursor = L->cursor->next;
        L->index++;
    } else {
        L->cursor = NULL;
        L->index = -1;
    }
}

void prepend(List L, int x) {
    Node N = malloc(sizeof(NodeObj));
    N->data = x;
    N->prev = NULL;
    N->next = L->front;
    if (L->front != NULL) L->front->prev = N;


    L->front = N;
    if (L->back == NULL) L->back = N;
    L->length++;
}

void append(List L, int x) {
    Node N = malloc(sizeof(NodeObj));
    N->data = x;
    N->next = NULL;
    N->prev = L->back;
    if (L->back != NULL) L->back->next = N;


    L->back = N;
    if (L->front == NULL) L->front = N;
    L->length++;
}

void insertBefore(List L, int x) {
    if (L->length == 0 || L->index == -1) {
        printf("List Error: calling insertBefore() on an empty List or undefined cursor\n");
        exit(1);
    }
    Node N = malloc(sizeof(NodeObj));
    N->data = x;
    N->prev = L->cursor->prev;
    N->next = L->cursor;


    if (L->cursor->prev != NULL) L->cursor->prev->next = N;
    L->cursor->prev = N;
    if (L->cursor == L->front) L->front = N;
    L->length++;
    L->index++;
}

void insertAfter(List L, int x) {
    if (L->length == 0 || L->index == -1) {
        printf("List Error: calling insertAfter() on an empty List or undefined cursor\n");
        exit(1);
    }
    Node N = malloc(sizeof(NodeObj));
    N->data = x;
    N->prev = L->cursor;
    N->next = L->cursor->next;


    if (L->cursor->next != NULL) L->cursor->next->prev = N;
    L->cursor->next = N;
    if (L->cursor == L->back) L->back = N;
    L->length++;
}

void deleteFront(List L) {
    if (L->length == 0) {
        printf("List Error: calling deleteFront() on an empty List\n");
        exit(1);
    }
    Node N = L->front;
    if (L->length == 1) {
        L->front = NULL;
        L->back = NULL;
    } else {
        L->front = L->front->next;
        L->front->prev = NULL;
    }
    free(N);
    L->length--;
}

void deleteBack(List L) {
    if (L->length == 0) {
        printf("List Error: calling deleteBack() on an empty List\n");
        exit(1);
    }
    Node N = L->back;
    if (L->length == 1) {
        L->front = NULL;
        L->back = NULL;
    } else {
        L->back = L->back->prev;
        L->back->next = NULL;
    }
    free(N);
    L->length--;
}

void delete(List L) {
    if (L->length == 0 || L->index == -1) {
        printf("List Error: calling delete() on an empty List or undefined cursor\n");
        exit(1);
    }
    Node N = L->cursor;
    if (N->prev != NULL) N->prev->next = N->next;
    if (N->next != NULL) N->next->prev = N->prev;
    if (N == L->front) L->front = N->next;
    if (N == L->back) L->back = N->prev;
    L->cursor = NULL;
    L->index = -1;
    free(N);
    L->length--;
}

// Other operations -----------------------------------------------------------
void printList(FILE* out, List L) {
    Node N = L->front;
    while (N != NULL) {
        fprintf(out, "%d ", N->data);
        N = N->next;
    }
}

List copyList(List L) {
    List newwList = newList();
    for (Node N = L->front; N != NULL; N = N->next) {
        append(newwList, N->data);
    }
    return newwList;
}
