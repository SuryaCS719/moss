//-----------------------------------------------------------------------------
// List.c
// Optimized implementation of List ADT with robust error handling and memory management
//-----------------------------------------------------------------------------

#include <stdio.h>
#include <stdlib.h>
#include "List.h"

// structs --------------------------------------------------------------------
// private Node type
typedef struct NodeObj* Node;

// private NodeObj type
typedef struct NodeObj {
    int data; // This is where we store the integer data for each node.
    Node previous;// A pointer to the previous node in the list.
    Node next;// A pointer to the next node in the list.

} NodeObj;

// private ListObj type


typedef struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
    int index;
} ListObj;

// Constructors-Destructors ---------------------------------------------------
// newNode()
// Returns reference to new Node object. Initializes next, previous, and data fields.
Node newNode(int data) {
    Node N = malloc(sizeof(NodeObj));
    if (N == NULL) {
        fprintf(stderr, "List Error: newNode() - memory allocation failed\n");
        exit(EXIT_FAILURE);
    }
    N->data = data;
    N->next = NULL;
    N->previous = NULL;
    return N;
}

// freeNode()
// Frees heap memory pointed to by *pN, sets *pN to NULL.


void freeNode(Node* pN) {
    if (pN != NULL && *pN != NULL) {
        free(*pN);
        *pN = NULL;
    }
}

// newList()
// Creates and returns a new empty List object.
List newList() {
    List L = malloc(sizeof(ListObj));
    if (L == NULL) {
        fprintf(stderr, "List Error: newList() - memory allocation failed\n");
        exit(EXIT_FAILURE);
    }
    L->front = L->back = L->cursor = NULL;
    L->length = 0;
    L->index = -1;
    return L;
}

// freeList()
// Frees all heap memory associated with *pL, and sets *pL to NULL.
void freeList(List* pL) {
    if (pL != NULL && *pL != NULL) {
        clear(*pL);
        free(*pL);
        *pL = NULL;
    }
}

// Access functions -----------------------------------------------------------
int length(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling length() on a NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    return L->length;
}

int index(List L) {


    if (L == NULL) {
        fprintf(stderr, "List Error: calling index() on a NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    return L->index;
}

int front(List L) {
    if (L == NULL || L->length <= 0) {
        fprintf(stderr, "List Error: calling front() on an empty or NULL List\n");
        exit(EXIT_FAILURE);
    }
    return L->front->data;
}

int back(List L) {
    if (L == NULL || L->length <= 0) {
        fprintf(stderr, "List Error: calling back() on an empty or NULL List\n");
        exit(EXIT_FAILURE);
    }
    return L->back->data;
}

int get(List L) {
    if (L == NULL || L->length <= 0 || L->index < 0) {
        fprintf(stderr, "List Error: calling get() on an undefined cursor or empty List\n");
        exit(EXIT_FAILURE);
    }
    return L->cursor->data;
}

bool equals(List A, List B) {
    if (A == NULL || B == NULL) {
        fprintf(stderr, "List Error: calling equals() on a NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (A->length != B->length) {
        return false;
    }
    Node N = A->front;
    Node M = B->front;
    while (N != NULL && M != NULL) {
        if (N->data != M->data) {
            return false;
        }
        N = N->next;
        M = M->next;
    }
    return true;
}

// Manipulation procedures ----------------------------------------------------
void clear(List L) {


    if (L == NULL) {
        fprintf(stderr, "List Error: calling clear() on a NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    while (L->length > 0) {
        deleteFront(L);
    }
    L->cursor = NULL;
    L->index = -1;
}

void set(List L, int x) {
    if (L == NULL || L->length <= 0 || L->index < 0) {
        fprintf(stderr, "List Error: calling set() on an undefined cursor or empty List\n");
        exit(EXIT_FAILURE);
    }
    L->cursor->data = x;
}

void moveFront(List L) {
    if (L != NULL && L->length > 0) {
        L->cursor = L->front;
        L->index = 0;
    }
}

void moveBack(List L) {
    if (L != NULL && L->length > 0) {


        L->cursor = L->back;
        L->index = L->length - 1;
    }
}

void movePrev(List L) {
    if (L != NULL && L->cursor != NULL) {
        if (L->cursor == L->front) {
            L->cursor = NULL;
            L->index = -1;
        } else {
            L->cursor = L->cursor->previous;
            L->index--;
        }
    }
}



void moveNext(List L) {
    if (L != NULL && L->cursor != NULL) {
        if (L->cursor == L->back) {
            L->cursor = NULL;
            L->index = -1;
        } else {
            L->cursor = L->cursor->next;
            L->index++;
        }
    }
}

void prepend(List L, int x) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling prepend() on a NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    Node N = newNode(x);
    if (L->length == 0) {
        L->front = L->back = N;
    } else {


        N->next = L->front;
        L->front->previous = N;
        L->front = N;
        if (L->index != -1) {
            L->index++;
        }
    }
    L->length++;
}

void append(List L, int x) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling append() on a NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    Node N = newNode(x);
    if (L->length == 0) {
        L->front = L->back = N;
    } else {
        N->previous = L->back;


        L->back->next = N;
        L->back = N;
    }
    L->length++;
}

void insertBefore(List L, int x) {
    if (L == NULL || L->length <= 0 || L->index < 0) {
        fprintf(stderr, "List Error: calling insertBefore() on an undefined cursor or empty List\n");
        exit(EXIT_FAILURE);
    }
    if (L->cursor == L->front) {
        prepend(L, x);
    } else {
        Node N = newNode(x);
        N->next = L->cursor;
        N->previous = L->cursor->previous;
        L->cursor->previous->next = N;


        L->cursor->previous = N;
        L->length++;
        L->index++;
    }
}

void insertAfter(List L, int x) {
    if (L == NULL || L->length <= 0 || L->index < 0) {
        fprintf(stderr, "List Error: calling insertAfter() on an undefined cursor or empty List\n");
        exit(EXIT_FAILURE);
    }
    if (L->cursor == L->back) {
        append(L, x);
    } else {
        Node N = newNode(x);
        N->previous = L->cursor;
        N->next = L->cursor->next;
        L->cursor->next->previous = N;


        L->cursor->next = N;
        L->length++;
    }
}

void deleteFront(List L) {
    if (L == NULL || L->length <= 0) {
        fprintf(stderr, "List Error: calling deleteFront() on an empty or NULL List\n");
        exit(EXIT_FAILURE);
    }
    Node N = L->front;
    if (L->length > 1) {
        L->front = L->front->next;
        L->front->previous = NULL;
    } else {


        L->front = L->back = NULL;
    }
    if (L->cursor == N) {
        L->cursor = NULL;
        L->index = -1;
    } else if (L->index >= 0) {
        L->index--;
    }
    freeNode(&N);
    L->length--;
}

void deleteBack(List L) {
    if (L == NULL || L->length <= 0) {
        fprintf(stderr, "List Error: calling deleteBack() on an empty or NULL List\n");
        exit(EXIT_FAILURE);
    }
    Node N = L->back;
    if (L->length > 1) {
        L->back = L->back->previous;
        L->back->next = NULL;
    } else {


        L->front = L->back = NULL;
    }
    if (L->cursor == N) {
        L->cursor = NULL;
        L->index = -1;
    }
    freeNode(&N);
    L->length--;
}

void delete(List L) {
    if (L == NULL || L->length <= 0 || L->index < 0) {
        fprintf(stderr, "List Error: calling delete() on an undefined cursor or empty List\n");
        exit(EXIT_FAILURE);
    }
    Node N = L->cursor;
    if (N == L->front) {
        deleteFront(L);
    } else if (N == L->back) {
        deleteBack(L);
    } else {


        N->previous->next = N->next;
        N->next->previous = N->previous;
        freeNode(&N);
        L->cursor = NULL;
        L->index = -1;
        L->length--;
    }
}

// Other operations -----------------------------------------------------------
void printList(FILE* out, List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling printList() on a NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    Node N = L->front;
    while (N != NULL) {
        fprintf(out, "%d ", N->data);
        N = N->next;
    }
}

List copyList(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling copyList() on a NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    List newL = newList();
    Node N = L->front;
    while (N != NULL) {
        append(newL, N->data);
        N = N->next;
    }
    return newL;
}
