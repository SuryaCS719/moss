#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>
#include "List.h"

typedef struct NodeObj {
    int data;


    struct NodeObj* next;
    struct NodeObj* prev;
} NodeObj;

typedef struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
    int index;
} ListObj;



//constructors-destructors
List newList(void) {
    List L = malloc(sizeof(ListObj));
    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->length = 0;
    L->index = -1;
    return L;
}

void freeList(List* pL) {
    if (pL != NULL && *pL != NULL) {


        clear(*pL);
        free(*pL);
        *pL = NULL;
    }
}



//access functions
int length(List L) {
    return L->length;
}

int index(List L) {
    return L->cursor == NULL ? -1 : L->index;
}

int front(List L) {
    return L->front->data;
}

int back(List L) {
    return L->back->data;
}

int get(List L) {
    return L->cursor->data;
}

bool equals(List A, List B) {
    if (A->length != B->length) return false;
    Node N = A->front;
    Node M = B->front;
    while (N != NULL) {
        if (N->data != M->data) return false;
        N = N->next;
        M = M->next;
    }
    return true;
}



//manipulation procedures
void clear(List L) {


    while (L->length > 0) {
        deleteFront(L);
    }
    L->cursor = NULL;
    L->index = -1;
}

void set(List L, int x) {
    L->cursor->data = x;
}

void moveFront(List L) {
    if (L->length > 0) {
        L->cursor = L->front;
        L->index = 0;
    }
}

void moveBack(List L) {
    if (L->length > 0) {


        L->cursor = L->back;
        L->index = L->length - 1;
    }
}

void movePrev(List L) {
    if (L->cursor != NULL) {
        L->cursor = L->cursor->prev;
        L->index = (L->cursor == NULL) ? -1 : L->index - 1;
    }
}

void moveNext(List L) {
    if (L->cursor != NULL) {
        L->cursor = L->cursor->next;
        L->index = (L->cursor == NULL) ? -1 : L->index + 1;
    }
}

void prepend(List L, int x) {
    Node N = malloc(sizeof(NodeObj));
    N->data = x;
    N->next = L->front;
    N->prev = NULL;


    if (L->front != NULL) L->front->prev = N;
    L->front = N;
    if (L->length == 0) L->back = N;
    L->length++;
    if (L->cursor != NULL) L->index++;
}

void append(List L, int x) {
    Node N = malloc(sizeof(NodeObj));
    N->data = x;
    N->next = NULL;
    N->prev = L->back;
    if (L->back != NULL) L->back->next = N;


    L->back = N;
    if (L->length == 0) L->front = N;
    L->length++;
}

void insertBefore(List L, int x) {
    Node N = malloc(sizeof(NodeObj));
    N->data = x;
    N->next = L->cursor;
    N->prev = L->cursor->prev;
    if (L->cursor->prev != NULL) L->cursor->prev->next = N;
    else L->front = N;


    L->cursor->prev = N;
    L->length++;
    L->index++;
}

void insertAfter(List L, int x) {
    Node N = malloc(sizeof(NodeObj));
    N->data = x;
    N->prev = L->cursor;
    N->next = L->cursor->next;


    if (L->cursor->next != NULL) L->cursor->next->prev = N;
    else L->back = N;
    L->cursor->next = N;
    L->length++;
}

void deleteFront(List L) {
    Node N = L->front;
    if (L->length > 1) {
        L->front = L->front->next;
        L->front->prev = NULL;
    } else {
        L->front = L->back = NULL;
    }
    if (L->cursor == N) {


        L->cursor = NULL;
        L->index = -1;
    } else if (L->cursor != NULL) {
        L->index--;
    }
    L->length--;
    free(N);
}

void deleteBack(List L) {
    Node N = L->back;
    if (L->length > 1) {
        L->back = L->back->prev;
        L->back->next = NULL;
    } else {
        L->front = L->back = NULL;
    }
    if (L->cursor == N) {
        L->cursor = NULL;
        L->index = -1;
    }
    L->length--;
    free(N);
}

void delete(List L) {
    Node N = L->cursor;
    if (N == L->front) {
        deleteFront(L);
    } else if (N == L->back) {
        deleteBack(L);
    } else {
        N->prev->next = N->next;
        N->next->prev = N->prev;
        L->cursor = NULL;
        L->index = -1;
        L->length--;
        free(N);
    }
}



//other operations
void printList(FILE* out, List L) {
    Node N = L->front;
    while (N != NULL) {
        fprintf(out, "%d ", N->data);
        N = N->next;
    }
    fprintf(out, "\n");
}

List copyList(List L) {
    List C = newList();
    Node N = L->front;
    while (N != NULL) {
        append(C, N->data);
        N = N->next;
    }
    return C;
}
