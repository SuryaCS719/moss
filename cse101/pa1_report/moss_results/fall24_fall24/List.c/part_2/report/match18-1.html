// Structs ----------


typedef struct NodeObj* Node;

typedef struct NodeObj{
	int data;
	Node prev;
	Node next;
} NodeObj;

typedef struct ListObj* List;

typedef struct ListObj{
        Node front;
        Node back;
	Node cursor;
	int index;
	int length;
} ListObj;

// Constructors-Destructors ----------
Node newNode(int data){
	Node N = calloc(1, sizeof(NodeObj));
	assert(N!=NULL);
	N->data = data;
	N->next = N->prev = NULL;
	return N;
}

void freeNode(Node *pN){
	if(pN != NULL && *pN != NULL){
		free(*pN);
		*pN = NULL;
	}
}

List newList(){
	List L = calloc(1, sizeof(ListObj));
	assert(L!=NULL);
	L->front = NULL;
	L-> back = NULL;
	L->cursor = NULL;
	L->index = -1;
	L->length = 0;
	return L;
}

void freeList(List *pL){
	if(pL!=NULL && *pL!=NULL) {
		while(!isEmpty(*pL)) {
			deleteFront(*pL);
		}
		free(*pL);
		*pL = NULL;
	}
}

		fprintf(stderr, "List Error: calling  get() on undefined cursor index\n");
                exit(EXIT_FAILURE);

}

bool equals(List A, List B){
	if(A==NULL || B==NULL){
		return false;
	}
	if(A->length != B->length){
		return false;
	}

	Node currentA = A->front;
	Node currentB = B->front;
	while(currentA!=NULL && currentB!=NULL){


		if(currentA->data != currentB->data){
			return false;
		}
		currentA = currentA->next;
		currentB = currentB->next;
	}
	return true;
}

// Manipulation procedures ----------------------------------------------------
void clear(List L){
	if(L!=NULL){
		while(!isEmpty(L)){
			deleteFront(L);
		}
	}
}

void set(List L, int x){
	if(L==NULL){
                fprintf(stderr, "List Error: calling  set() on NULL List reference\n");
                exit(EXIT_FAILURE);
        }
	if(isEmpty(L)){
                fprintf(stderr, "List Error: calling  set() on empty List\n");
                exit(EXIT_FAILURE);
        }
	if(L->index < 0){
                fprintf(stderr, "List Error: calling  set() on undefined cursor index\n");
                exit(EXIT_FAILURE);
        }
	L->cursor->data = x;
}

void moveFront(List L){
	if(L==NULL){
                fprintf(stderr, "List Error: calling moveFront() on NULL List reference\n");
                exit(EXIT_FAILURE);
        }
	if(!isEmpty(L)){
		L->cursor = L->front;
		L->index = 0;
	}
}

void moveBack(List L){
        if(L==NULL){
                fprintf(stderr, "List Error: calling moveBack() on NULL List reference\n");
                exit(EXIT_FAILURE);
        }
        if(!isEmpty(L)){
                L->cursor = L->back;
                L->index = L->length - 1;
        }
}

void movePrev(List L){
        if(L==NULL){
                fprintf(stderr, "List Error: calling movePrev() on NULL List reference\n");
                exit(EXIT_FAILURE);
        }
        if(L->cursor != NULL){
	       	if(L->cursor != L->front){ //if cursor is not at front, move cursor one step towards front
                	L->cursor = L->cursor->prev;
                	L->index -=  1;
		}
		else{ // otherwise, the cursor becomes undefined
			L->cursor = NULL;
			L->index = -1;
		}
        }
}

void moveNext(List L){
        if(L==NULL){
                fprintf(stderr, "List Error: calling moveNext() on NULL List reference\n");
                exit(EXIT_FAILURE);
        }
        if(L->cursor != NULL){
                if(L->cursor != L->back){ //if cursor is not at back, move cursor one step towards back
                        L->cursor = L->cursor->next;
                        L->index += 1;
                }
                else{ // otherwise, the cursor becomes undefined
                        L->cursor = NULL;
                        L->index = -1;
                }
        }
}

void prepend(List L, int x){
	if(L==NULL){
                fprintf(stderr, "List Error: calling prepend() on NULL List reference\n");
                exit(EXIT_FAILURE);
        }
	Node N = newNode(x);
	N->next = L->front;
	if(isEmpty(L)){ 
		L->back = N; //if the list is empty, front and back elements are both N
	}
	else{
		L->front->prev = N; //if the list is NOT empty, the previous front is now preceeded by N
	}
	L->front = N; //reassign what the new front element is (this will happen regardless of isEmpty)

	//handling the other List elements (length and index)
	L->length += 1;
	if(L->cursor != NULL){
		L->index += 1;
	}
}

void append(List L, int x){
        if(L==NULL){
                fprintf(stderr, "List Error: calling append() on NULL List reference\n");
                exit(EXIT_FAILURE);
        }
        Node N = newNode(x);
        N->prev = L->back;
        if(isEmpty(L)){
                L->front = N; //if the list is empty, front and back elements are both N
        }
        else{
                L->back->next = N; //if the list is NOT empty, the previous back is now proceeded by N
        }
        L->back = N; //reassign what the new back element is (this will happen regardless of isEmpty)

        //handling the other List elements (length changes, but index stays the same)
        L->length +=  1;
}

void insertBefore(List L, int x){
	if(L==NULL){
                fprintf(stderr, "List Error: calling insertBefore() on NULL List reference\n");
                exit(EXIT_FAILURE);
        }
        if(isEmpty(L)){
                fprintf(stderr, "List Error: calling insertBefore() on empty List\n");
                exit(EXIT_FAILURE);
        }
        if(L->index < 0){
                fprintf(stderr, "List Error: calling insertBefore() on undefined cursor index\n");
                exit(EXIT_FAILURE);
        }

	//prepping N for insertion
	Node N = newNode(x);
	N->prev = L->cursor->prev;
	N->next = L->cursor;

	
	if(L->cursor->prev != NULL){
		L->cursor->prev->next = N; //if cursor isnt the first element, change 'next' of element before it to N
	}
	else{
		L->front = N; //if cursor is the first element, reassign what the new first element is
	}
	L->cursor->prev = N; //change cursor's prev element

	
	//handling the other List elements (length and index)
        L->length += + 1;
        L->index += 1; 
}

void insertAfter(List L, int x){
        if(L==NULL){
                fprintf(stderr, "List Error: calling insertAfter() on NULL List reference\n");
                exit(EXIT_FAILURE);
        }
        if(isEmpty(L)){
                fprintf(stderr, "List Error: calling insertAfter() on empty List\n");
                exit(EXIT_FAILURE);
        }
        if(L->index < 0){
                fprintf(stderr, "List Error: calling insertAfter() on undefined cursor index\n");
                exit(EXIT_FAILURE);
        }

        Node N = newNode(x);
        N->prev = L->cursor;
        N->next = L->cursor->next;
        if(L->cursor->next != NULL){ //if cursor isnt the last element, change 'prev' of element after it to N


                L->cursor->next->prev = N;
        }
	else{
                L->back = N; //if cursor is the last element, reassign what the new last element is
        }
        L->cursor->next = N; //change cursor's next element

        //handling the other List elements (length changes, but index stays the same)
        L->length += 1;
}

void deleteFront(List L){
	if(L==NULL){
                fprintf(stderr, "List Error: calling deleteFront() on NULL List reference\n");
                exit(EXIT_FAILURE);
        }
        if(isEmpty(L)){
                fprintf(stderr, "List Error: calling deleteFront() on empty List\n");
                exit(EXIT_FAILURE);
        }

	Node oldFront = L->front; //store what the old 'front' node is

	if(L->cursor != NULL){
		if(L->cursor == oldFront){ //if cursor node is at the front, cursor element is now null
			L->cursor = NULL;
			L->index = -1;
		}
		else{
			L->index -= 1;
		}
	}

	if(L->length > 1){ //if there is more than 1 node,
                L->front = oldFront->next;
                L->front->prev = NULL;
        }
        else{ //if we started with only 1 node, and will have 0 after the operation
                L->front = L->back =  NULL;
        }

	freeNode(&oldFront); //free the front node and decrement length
	L->length -= 1;
}

void deleteBack(List L){
        if(L==NULL){
                fprintf(stderr, "List Error: calling deleteBack() on NULL List reference\n");
                exit(EXIT_FAILURE);
        }
        if(isEmpty(L)){
                fprintf(stderr, "List Error: calling deleteBack() on empty List\n");
                exit(EXIT_FAILURE);
        }

        Node newBack = L->back->prev; //store what the new 'front' node is

        if(L->cursor == L->back){
                L->cursor = NULL;
		L->index = -1;
        }
        freeNode(&(L->back)); //free the front node and decrement length
        L->length -= 1;

        if(L->length > 0){ //if there is 1 or more nodes left, new back node and update newBack's next
                L->back = newBack;
                L->back->next = NULL;
        }
        else{ //if there are 0 nodes left 
                L->front = L->back = NULL;
        }
}

void delete(List L){
	if(L==NULL){
                fprintf(stderr, "List Error: calling delete() on NULL List reference\n");
                exit(EXIT_FAILURE);
        }
        if(isEmpty(L)){
                fprintf(stderr, "List Error: calling delete() on empty List\n");
                exit(EXIT_FAILURE);
        }
        if(L->index < 0){
                fprintf(stderr, "List Error: calling delete() on undefined cursor index\n");
                exit(EXIT_FAILURE);
        }

	if(L->cursor == L->front){
		deleteFront(L);
	}
	else if(L->cursor == L->back){
		deleteBack(L);
	}
	else{


		L->cursor->prev->next = L->cursor->next;
		L->cursor->next->prev = L->cursor->prev;
		freeNode(&(L->cursor));
		
		L->cursor = NULL;
		L->index = -1;
		L->length -= 1;
	}
}

        if(L==NULL){
                fprintf(stderr, "List Error: calling copyList() on NULL List reference\n");
                exit(EXIT_FAILURE);
        }

        List out = newList();

        Node current = L->front;
