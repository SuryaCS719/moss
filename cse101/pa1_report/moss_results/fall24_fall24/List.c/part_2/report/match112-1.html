#include<stdio.h>
#include<assert.h>
#include<stdlib.h>
#include<string.h>
#include"List.h"

typedef struct NodeObj* Node;

// Node structure --------
typedef struct NodeObj {
    int data;
    struct NodeObj* next;
    struct NodeObj* prev;
} NodeObj;

// List struct -----------


typedef struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
    int index;
} ListObj;

// Constructors and Destructors ----------------------------

// Node Constructor
Node newNode(int data) {
    Node N = malloc(sizeof(NodeObj));

    // Precondition, (malloc) is not NULL
    if (N == NULL) {
        fprintf(stderr, "Memory allocation failed: NewNode\n");
        exit(EXIT_FAILURE); // Exit program if malloc fails
    }

    N->data = data;
    N->next = NULL;
    N->prev = NULL;
    return N;
}

// Node Destructor - Frees Node
void freeNode(Node* pN) {


    if (pN != NULL && *pN !=NULL) {
        free(*pN);
        *pN = NULL;
    }
}

// List Constructor
List newList(void) {
    List L = malloc(sizeof(ListObj));

    // Precondition, (malloc) is not NULL
     if (L == NULL) {
        fprintf(stderr, "Memory allocation failed: newList\n");
        exit(EXIT_FAILURE); // Exit program if malloc fails
    }


    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->length = 0;
    L->index = -1;
    return L;
}


// List Destructor - Frees list
void freeList(List* pL) {
    if (pL !=NULL && *pL != NULL) { // If the list is not empty i.e pL > 0
        while((*pL)->length > 0) {
             deleteFront(*pL);
        }
        free(*pL);
        *pL = NULL;

        }
}


// Access Functions -------------------------------------------------------

int length(List L) {
    return L->length;
}



int index(List L) {
    return L->index;
}

int front(List L) {
    if (L->length > 0) { // Precondition: Ensure list is not empty
        return L->front->data;
    }
    exit(EXIT_FAILURE);
}

int back(List L) {
    if (L->length > 0) { // Precondition: Ensure list is not empty
        return L->back->data;
    }
    exit(EXIT_FAILURE);
}

int get(List L) {
    if (L->length > 0 && L->index >=0) {
        return L->cursor->data;
    }
    exit(EXIT_FAILURE);
}

bool equals(List A, List B) {
    // Check if the lengths of the list are the same
    if (length(A) != length(B)) {
        return false;
    }

    moveFront(A);
    moveFront(B);

    // Go through both lists and check for equivalency
    while (index(A) >= 0 && index(B) >= 0){
        if (get(A) != get(B)) {
            return false;
        }
        moveNext(A);
        moveNext(B);
    }

    return true; // If no differences are found b/w lists, return true
}

// Manipulation Procedures -----------------------------------------

void clear(List L) {
    while (L->length > 0) {
        deleteFront(L);
    }
    L->cursor = NULL;
    L->index = -1;
}

void set(List L, int x) {
    if (L->length > 0 && L->index >=0) {
        L->cursor->data = x; 
    } else {
        printf("Error: Cannot set cursor data, list is empty / cursor undefined.\n");
        exit(EXIT_FAILURE);
    }
}

void moveFront(List L) {
    if (L->length > 0) {
        L->cursor = L->front;
        L->index = 0; // Set the cursor index to the front position (0)

    }
}

void moveBack(List L) {
    if (L->length > 0) {
        L->cursor = L->back;
        L->index = L->length - 1;

    }
}

void movePrev(List L) {


    if (L->cursor != NULL && L->cursor != L->front) {
        // Ensure the cursor is defined AND not at the front
        L->cursor = L->cursor->prev;
        L->index--;

    } else if (L->cursor != NULL && L->cursor == L->front) {
        L->cursor = NULL;
        L->index = -1;

    }

 }

void moveNext(List L) {
    if (L->cursor != NULL && L->cursor != L->back) {
        // Ensure the cursor is defined and NOT at the back
        L->cursor = L->cursor->next;
        L->index++;

    } else if (L->cursor != NULL && L->cursor == L->back) {
        L->cursor = NULL;
        L->index = -1;

    }
}

void prepend(List L, int x) {
    Node N = newNode(x); // Create a new node for data (x)
    if (L->length == 0) {
        L->front = L->back = N;
    } else {
        N->next = L->front;
        L->front->prev = N;
        L ->front = N;
    }
    L->length++;
    if (L->index != -1) {
        L->index++;
    }

    }

void append(List L, int x) {
    Node N = newNode(x);
    if (L->length ==0) {
        L->front = L->back = N;

    } else {
        N->prev = L->back;
        L->back->next = N;


        L->back = N;
    }
    L->length++;
}

void insertBefore(List L, int x) {
    if(L->length > 0 && L->index >= 0) { // If list is non-empty and cursor is defined
        if (L->cursor == L->front) {
            prepend(L, x);
        } else {
            Node N = newNode(x);
            N->next = L->cursor; // Places the new node (data) BEFORE the cursor
            N->prev = L->cursor->prev; // Sets the node prev to cursor, prev to the newNode
            L->cursor->prev->next = N; // The prev cursor node's next node is newNode (N)
            L->cursor->prev = N; // Ensures the element before the cursor is newNode(N)
            L->length++;
            L->index++;
        }
    
  }

}

void insertAfter(List L, int x) {
    if(L->length > 0 && L->index >=0) {
        if (L->cursor == L->back) {
            append(L, x);
        } else {
            Node N = newNode(x);
            N->prev = L->cursor; //Places the newNode after the cursor (Cursor is prev)
            N->next = L->cursor->next; //Ensures the element after newNode is the element that was after the cursor
            L->cursor->next->prev = N; // Ensures the newNode is before what was after the cursor
            L->cursor->next = N; // Ensures the cursors next node is pointing to newNode


            L->length++; 
            // Index does not get incremented, as the relative position of the cursor in the list is not affected

        }
    }
}

void deleteFront(List L) {
    if (L->length > 0) {
        Node N = L->front; // Create a placeholder Node (N) at the front of the list
        if (L->cursor == L->front) {
            L->cursor = NULL; // Ensure the cursors does not become invalid if present at the front node
            L->index = -1; // Indicate that the cursor is now undefined and not pointing to any node
        }
        if (L->length == 1) {
            L->front = L->back = NULL;
        } else {
            L->front = L->front->next;
            L->front->prev = NULL; // Now the first node, nothing is before it, its prev is NULL
        }
        freeNode(&N); // Free the temp Node which was placed at the front
        L->length--;

    }
}

void deleteBack(List L) {
    if (L->length > 0) {
        Node N = L->back; // Create a temp Node (N) at the back of the list
        if (L->cursor == L->back) {
            L->cursor = NULL;
            L->index = -1;
    }
    if (L->length == 1) {
        L->front = L->back = NULL;
    } else {
        L->back = L->back->prev;
        L->back->next = NULL;
    }
    freeNode(&N);
    L->length--;
    }
}

void delete(List L) {
    if(L->length > 0 && L->index>=0) {
        Node N = L->cursor;
        if (L->cursor == L->front) {
            deleteFront(L);
        } else if (L->cursor == L->back) {


            deleteBack(L);
        } else {
            N->prev->next = N->next;
            N->next->prev = N->prev;
            L->length--;
            freeNode(&N); // Free the deleted node
        }

        L->cursor = NULL;
        L->index = -1;
    }
}

void printList(FILE* out, List L) {
    Node N = L->front;

    while (N != NULL) {
        fprintf(out, "%d ", N->data);
        N = N->next;
    }
}

List copyList(List L) {
    List newL = newList(); // Create a new list newL
    Node N = L->front; 

    while(N != NULL) {
        append(newL, N->data);
        N = N->next;
        // Note that the cursor in the newL is undef as it is its default state
    }
      return newL;
}
