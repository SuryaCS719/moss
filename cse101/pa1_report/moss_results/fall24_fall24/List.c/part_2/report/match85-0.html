// Constructors-Destructors ---------------------------------------------------


typedef struct ListObj* List;

// Creates and returns a new empty List.
List newList(void)
{
        List L = malloc(sizeof(struct ListObj));  // Allocate memory for the List
        if (L == NULL)
        {       // Handle memory allocation error
                fprintf(stderr, "List Error: unable to allocate memory for List\n");
                exit(EXIT_FAILURE);
        }

        // Initialize all fields
        L->front = NULL;
        L->back = NULL;
        L->cursor = NULL;
        L->length = 0;


        L->index = -1;
        return L;
}

// Frees all heap memory associated with *pL, and sets *pL to NULL.
void freeList(List* pL)
{
        if (pL != NULL && *pL != NULL)
        {
                // Check if pL is not NULL and points to a valid List
                List L = *pL;                  // Dereference pL to get the List
                Node current = L->front;       // Start from the front

                // Traverse and free all nodes
                while (current != NULL)
                {
                        Node temp = current;        // Keep a reference to the current node
                        current = current->next;    // Move to the next node
                        free(temp);                 // Free the current node
                }

                // Free the List itself
                free(L);
                *pL = NULL;                     // Set the pointer to NULL
        }
}

        }

        return L->length; // Return the length of the list
}

// Returns front element of L. Pre: length()>0
int front(List L)
{
        if (L == NULL)
        {  // Check if the list is NULL
                fprintf(stderr, "List Error: front() called on NULL List\n");
                exit(EXIT_FAILURE);
        }

        if (L->length == 0)
        {  // Check if the list is empty
                fprintf(stderr, "List Error: front() called on empty List\n");
                exit(EXIT_FAILURE);
        }

        return L->front->data; // Return the data of the front node
}

// Returns back element of L. Pre: length()>0
int back(List L)

// Returns cursor element of L. Pre: length()>0, index()>=0
int get(List L)
{
        if (L == NULL)
        {  // Check if the list is NULL
                fprintf(stderr, "List Error: get() called on NULL List\n");
                exit(EXIT_FAILURE);
        }


        if (L->length == 0)
        {  // Check if the list is empty
                fprintf(stderr, "List Error: get() called on empty List\n");
                exit(EXIT_FAILURE);
        }
        if (L->index < 0)
        {  // Check if the cursor is undefined
                fprintf(stderr, "List Error: get() called with an undefined cursor\n");
                exit(EXIT_FAILURE);
        }

        return L->cursor->data; // Return the data of the cursor node
}

 // Returns true iff Lists A and B contain the same sequence of elements, returns false otherwise.
bool equals(List A, List B)
{
        if (A == NULL || B == NULL)
        {  // Check if either list is NULL
                fprintf(stderr, "List Error: equals() called on NULL List\n");
                exit(EXIT_FAILURE);
        }

        if (A->length != B->length)
        {  // If lengths are different, lists are not equal
                return false;
        }

        Node N1 = A->front;  // Start at the front of list A
        Node N2 = B->front;  // Start at the front of list B

        while (N1 != NULL && N2 != NULL)
        {
                if (N1->data != N2->data)
                {  // If any corresponding elements differ, return false
                        return false;
                }

                N1 = N1->next;  // Move to the next element in list A
                N2 = N2->next;  // Move to the next element in list B
        }

        return true;  // If all elements match, return true
}

// Resets L to its original empty state.
void clear(List L)
{


        if (L == NULL)
        {  // Check if the list is NULL
                fprintf(stderr, "List Error: clear() called on NULL List\n");
                exit(EXIT_FAILURE);
        }

        while (L->length > 0)
        {  // While the list is not empty
                deleteFront(L);  // Repeatedly delete the front element
        }

        // Reset cursor and other fields
        L->front = NULL;
        L->back = NULL;
        L->cursor = NULL;
        L->length = 0;
        L->index = -1;  // Reset index to -1 (undefined cursor)
}

void set(List L, int x)

        L->cursor->data = x;  // Overwrite the cursor element's data with x
}

// If L is non-empty, sets cursor under the front element, otherwise does nothing.
void moveFront(List L)
{
        if (L == NULL)
        {  // Check if the list is NULL
                fprintf(stderr, "List Error: moveFront() called on NULL List\n");
                exit(EXIT_FAILURE);
        }

        if (L->length > 0)
        {  // If the list is non-empty
                L->cursor = L->front;  // Set the cursor to the front element
                L->index = 0;  // Set the index to 0 (front element index)
        }
}

// If L is non-empty, sets cursor under the back element, otherwise does nothing.
void moveBack(List L)
{
        if (L == NULL)
        {  // Check if the list is NULL
                fprintf(stderr, "List Error: moveBack() called on NULL List\n");
                exit(EXIT_FAILURE);
        }

        if (L->length > 0)
        {  // If the list is non-empty
                L->cursor = L->back;  // Set the cursor to the back element


                L->index = L->length - 1;  // Set the index to the last element index
        }
}

 // If cursor is defined and not at front, move cursor one step toward the front of L; if cursor is defined and at front, cursor becomes undefined; if cursor is undefined do nothing
void movePrev(List L)
{
        if (L == NULL)
        {  // Check if the list is NULL
                fprintf(stderr, "List Error: movePrev() called on NULL List\n");
                exit(EXIT_FAILURE);
        }

        if (L->index >= 0)
        {  // If cursor is defined
                if (L->index == 0)
                {  // If cursor is at front
                        L->index = -1;  // Cursor becomes undefined
                        L->cursor = NULL;  // Optionally set cursor to NULL
                } else {  // Cursor is not at front
                        L->cursor = L->cursor->prev;  // Move cursor one step toward the front
                        L->index--;  // Decrement index
                }
        }
}

// If cursor is defined and not at back, move cursor one step toward the back of L; if cursor is defined and at back, cursor becomes undefined; if cursor is undefined do nothing
void moveNext(List L)
{
        if (L == NULL)
        {  // Check if the list is NULL
                fprintf(stderr, "List Error: moveNext() called on NULL List\n");
                exit(EXIT_FAILURE);
        }

        if (L->index >= 0)
        {  // If cursor is defined
                if (L->index == L->length - 1)
                {  // If cursor is at back
                        L->index = -1;  // Cursor becomes undefined
                        L->cursor = NULL;  // Optionally set cursor to NULL
                } else {  // Cursor is not at back
                        L->cursor = L->cursor->next;  // Move cursor one step toward the back
                        L->index++;  // Increment index
                }
        }
}

 // Insert new element into L. If L is non-empty, insertion takes place before front element.
void prepend(List L, int x)
{
        if (L == NULL)
        {  // Check if the list is NULL
                fprintf(stderr, "List Error: prepend() called on NULL List\n");
                exit(EXIT_FAILURE);
        }

        // Create a new node
        Node N = newNode(x);

        if (L->length == 0)
        {  // If the list is empty
                L->front = L->back = N;  // New node is both front and back
        } else {  // If the list is not empty
                N->next = L->front;  // New node's next is current front
                L->front->prev = N;  // Current front's previous is new node
                L->front = N;  // Update front to new node
        }

        L->length++;  // Increase the length of the list

        // If cursor was previously at the front, we must adjust the cursor index
        if (L->index != -1)
        {
                L->index++;  // Shift cursor index to account for the new front element
        }
}

// Insert new element into L. If L is non-empty, insertion takes place after back element.
void append(List L, int x)
{
        if (L == NULL)
        {  // Check if the list is NULL
                fprintf(stderr, "List Error: append() called on NULL List\n");
                exit(EXIT_FAILURE);
        }

        // Create a new node
        Node N = newNode(x);

        if (L->length == 0)
        {  // If the list is empty
                L->front = L->back = N;  // New node is both front and back
        } else {  // If the list is not empty
                N->prev = L->back;  // New node's prev is current back
                L->back->next = N;  // Current back's next is new node
                L->back = N;  // Update back to new node
        }

        L->length++;  // Increase the length of the list
}

// Insert new element before cursor.Pre: length()>0, index()>=0
void insertBefore(List L, int x)
{
        if (L == NULL)
        {  // Null list check
                fprintf(stderr, "List Error: insertBefore() called on NULL List\n");
                exit(EXIT_FAILURE);
        }
        if (L->length <= 0)
        {  // Precondition: non-empty list
                fprintf(stderr, "List Error: insertBefore() called on an empty List\n");
                exit(EXIT_FAILURE);
        }
        if (L->index < 0)
        {  // Precondition: cursor must be defined
                fprintf(stderr, "List Error: insertBefore() called with undefined cursor\n");
                exit(EXIT_FAILURE);
        }

        // Create a new node
        Node N = newNode(x);

        // Case 1: Insert before the front element (i.e., cursor is at the front)
        if (L->cursor == L->front)
        {
                N->next = L->front;
                L->front->prev = N;
                L->front = N;
        } else {  // Case 2: Insert before the cursor in the middle or back
                N->next = L->cursor;
                N->prev = L->cursor->prev;
                L->cursor->prev->next = N;
                L->cursor->prev = N;
        }

        L->length++;  // Increase the length of the list
        L->index++;   // Adjust the cursor index to reflect the new insertion
}

// Insert new element after cursor. Pre: length()>0, index()>=0
void insertAfter(List L, int x)
{
        if (L == NULL)
        {  // Null list check
                fprintf(stderr, "List Error: insertAfter() called on NULL List\n");
                exit(EXIT_FAILURE);
        }
        if (L->length <= 0)
        {  // Precondition: non-empty list
                fprintf(stderr, "List Error: insertAfter() called on an empty List\n");
                exit(EXIT_FAILURE);
        }
        if (L->index < 0)
        {  // Precondition: cursor must be defined
                fprintf(stderr, "List Error: insertAfter() called with undefined cursor\n");
                exit(EXIT_FAILURE);
        }

        // Create a new node
        Node N = newNode(x);

        // Case 1: Insert after the back element (i.e., cursor is at the back)
        if (L->cursor == L->back)
        {
                N->prev = L->back;
                L->back->next = N;
                L->back = N; // Update back pointer to the new node
        } else {  // Case 2: Insert after the cursor in the middle or front
                N->prev = L->cursor;
                N->next = L->cursor->next;
                L->cursor->next->prev = N; // Link the next node's prev to new node
                L->cursor->next = N; // Link cursor's next to the new node
        }



        L->length++;  // Increase the length of the list
}

// Delete the front element. Pre: length()>0
void deleteFront(List L)
{
        if (L == NULL)
        {  // Null list check
                fprintf(stderr, "List Error: deleteFront() called on NULL List\n");
                exit(EXIT_FAILURE);
        }
        
	// Check if the List has at least one element
	if (L->length > 0) 
	{
		// Create a reference to the current front node for deletion
		Node n = L->front;
        	// If the cursor is currently at the front, set the cursor to undefined
		if (L->cursor == L->front) {
            		L->cursor = NULL;
            		L->index = -1;// Mark the cursor as undefined
        	}
        	 // If there are more than one elements in the list
		if (L->length > 1) 
		{
			L->front = L->front->next;
			L->front->prev = NULL;
            		if(L->index!=-1)
			{
                		L->index--;
            		}
            
        		} else {
            			L->front = L->back = NULL;
            			if(L->index!=-1){
                		L->index--;
            		}
        	}
        	L->length--;
        	freeNode(&n);
    	}
}

// Delete the back element. Pre: length()>0
void deleteBack(List L)
{
    	// Check if the List is NULL
    	if (L == NULL)
    	{
        	// If the List is NULL, print an error message and exit the program
        	fprintf(stderr, "List Error: deleteBack() called on NULL List\n");
        	exit(EXIT_FAILURE);
    	}

    	// Check if the List is empty (length <= 0), which is a precondition violation
    	if (L->length <= 0)
    	{
        	// If the List is empty, print an error message and exit the program
        	fprintf(stderr, "List Error: deleteBack() called on an empty List\n");
        	exit(EXIT_FAILURE);
    	}

    	// Create a reference to the current back node for deletion
    	Node N = L->back;

    	// If the cursor is currently at the back, set the cursor to undefined


    	if (L->cursor == L->back)
    	{
        	L->cursor = NULL;
        	L->index = -1;  // Mark the cursor as undefined
    	}

    	// If there is only one node in the list
    	if (L->length == 1)
    	{
        	// Set both front and back pointers to NULL, as the list will become empty
        	L->front = NULL;
        	L->back = NULL;
        	// Set the length of the list to 0 since the only node will be removed
        	L->length = 0;
        	// Free the memory associated with the removed node
        	freeNode(&N);
    	}
    	else
    	{
        	// Set the back pointer to the previous node in the list
        	L->back = N->prev;
        	// Set the new back node's next pointer to NULL, as it's now the last node
        	L->back->next = NULL;
        	// Decrease the length of the list after removing the back node
        	L->length--;
        	// Free the memory associated with the removed node
        	freeNode(&N);
    	}
}

// Delete cursor element, making cursor undefined.  Pre: length()>0, index()>=0
void delete(List L)
{
        if (L == NULL)
        {  // Null list check
                fprintf(stderr, "List Error: delete() called on NULL List\n");
                exit(EXIT_FAILURE);
        }
        if (L->length <= 0)
        {  // Precondition: non-empty list
                fprintf(stderr, "List Error: delete() called on an empty List\n");
                exit(EXIT_FAILURE);
        }
        if (L->index < 0)
        {  // Precondition: cursor is defined
                fprintf(stderr, "List Error: delete() called with undefined cursor\n");
                exit(EXIT_FAILURE);
        }

        Node N = L->cursor;  // Get the cursor node

        // Case 1: If cursor is the only element in the list
        if (L->length == 1)
        {
                L->front = NULL; // The front becomes NULL
                L->back = NULL;  // The back also becomes NULL
                L->cursor = NULL; // Cursor becomes undefined
                L->index = -1;   // Set cursor index to -1
        }
        // Case 2: If cursor is at the front
        else if (L->index == 0)
        {
                L->front = L->front->next; // Update front to the next node
                if (L->front != NULL)
                { // If the list is not empty after update
                        L->front->prev = NULL; // Set the new front's previous pointer to NULL
                }
                L->cursor = NULL; // Make cursor undefined
                L->index = -1;    // Set cursor index to -1

        // Case 3: If cursor is at the back
        }else if (L->index == L->length - 1) {
                L->back = L->back->prev; // Update back to the previous node
                if (L->back != NULL)
                { // If the list is not empty after update
                        L->back->next = NULL; // Set the new back's next pointer to NULL
                }
                L->cursor = NULL; // Make cursor undefined
                L->index = -1;    // Set cursor index to -1

        // Case 4: Cursor is in the middle
        }else {
                // Bypass the cursor node
                if (N->prev != NULL)
                {
                        N->prev->next = N->next; // Link previous node to next
                }
                if (N->next != NULL)
                {
                        N->next->prev = N->prev; // Link next node to previous
                }
                L->cursor = NULL; // Make cursor undefined
                L->index = -1;    // Set cursor index to -1
        }

        free(N);  // Free the memory of the old cursor node
        L->length--;  // Decrease the length of the list
}

//Prints to the file pointed to by out, a string representation of L consisting of a space separated sequence of integers, with front on left.
void printList(FILE* out, List L)
{
        if (out == NULL || L == NULL)
        {  // Null check for output file and list
                fprintf(stderr, "List Error: printList() called with NULL parameters\n");
                return;
        }

        Node current = L->front;  // Start from the front of the list
        while (current != NULL)
        {  // Traverse until the end of the list
                fprintf(out, "%d", current->data); // Print current node's data

                current = current->next; // Move to the next node
                if (current != NULL)
                { // Print space only if there is a next element
                        fprintf(out, " ");
                }
        }
}

// Returns a new List representing the same integer sequence as L. The cursor in the new list is undefined,regardless of the state of the cursor in L. The state of L is unchanged.
List copyList(List L)
{
        if (L == NULL)
        {  // Check if the original list is NULL
                return NULL;  // Handle the error: return NULL for an invalid input
        }

        List nList = newList();  // Create a new empty list

        Node current = L->front;  // Start from the front of the original list
        while (current != NULL)
        {  // Traverse until the end of the list
                append(nList, current->data);  // Append each element to the new list
                current = current->next;  // Move to the next node
        }

        return nList;  // Return the new list
}

// Private function: newNode
// Allocates memory for a new Node and initializes its fields
Node newNode(int data) 
{
    	Node N = malloc(sizeof(struct NodeObj));  // Allocate memory for a new node
    	if (N == NULL) 
	{
        	fprintf(stderr, "List Error: unable to allocate memory for Node\n");
        	exit(EXIT_FAILURE);
    	}
    	N->data = data;    // Initialize node data
    	N->next = NULL;    // Initialize next pointer to NULL
    	N->prev = NULL;    // Initialize previous pointer to NULL
    	return N;
}

// Private function: freeNode
// Frees the memory associated with *pN and sets *pN to NULL
void freeNode(Node* pN) 
{
    	if (pN != NULL && *pN != NULL) 
	{
        	free(*pN);      // Free the memory pointed to by *pN
        	*pN = NULL;     // Set the pointer to NULL to avoid dangling references
    	}
}
