#include <stdio.h>
#include <assert.h>
#include <stdlib.h>
#include <stdbool.h>
#include "List.h"



typedef struct Node{
    int data;
    struct Node* prev;
    struct Node* next;
} Node;

typedef struct ListObj{
    Node* front;
    Node* back;
    Node* cursor;
    int index;
    int length;
} ListObj;

//Constructors/Destructors ---------------------------------------------------

List newList(void){ // Creates and returns a new empty List.
    List L;
    L = malloc(sizeof(ListObj));
    assert(L != NULL);
    L->front = L->back = NULL;
    L->length = 0;
    L->index = -1;
    L->cursor = NULL;
    return L;
}
void freeList(List* pL){ // Frees all heap memory associated with *pL, and sets *pL to NULL.
    if (pL == NULL || *pL == NULL){
        return;
    }
    
    List L = *pL;
    
    Node* currentNode = L->front;
    Node* nextNode;

    while(currentNode != NULL){
        nextNode = currentNode->next;
        free(currentNode);
        currentNode = nextNode;
    } 
    free(L);
    *pL = NULL;
    return;
}

Node* newNode(int data){
    Node* N = malloc(sizeof(Node));
    assert(N != NULL);
    N->prev = NULL;
    N->data = data;
    N->next = NULL;
    return N;
}

// Access functions -----------------------------------------------------------
int length(List L){ // Returns the number of elements in L.
    return L->length;
}

int index(List L){ // Returns index of cursor element if defined, -1 otherwise.
    if (L->index > L->length - 1 || L->index < 0){
        return(-1);
    }
    return(L->index);
}

int front(List L){ // Returns front element of L. Pre: length()>0
    if (L == NULL){
        fprintf(stderr, "List Error: Calling front() on a NULL list reference");
        exit(EXIT_FAILURE);
    }
    
    if (L->length <= 0){
        fprintf(stderr, "List Error: Calling front() on an empty list");
        exit(EXIT_FAILURE);
    }
    return L->front->data;
}

int back(List L){ // Returns back element of L. Pre: length()>0
    if (L == NULL){
        fprintf(stderr, "List Error: Calling back() on a NULL list reference");
        exit(EXIT_FAILURE);
    }
    
    if (L->length <= 0){


        fprintf(stderr, "List Error: Calling back() on an empty list");
        exit(EXIT_FAILURE);
    }
    return L->back->data;
}

int get(List L){ // Returns cursor element of L. Pre: length()>0, index()>=0
    if (L->length <= 0){
        fprintf(stderr, "List Error: Calling get() on an empty list");
        exit(EXIT_FAILURE);
    }
    if (L->index < 0){
        fprintf(stderr, "List Error: Calling get() with an undefined cursor");
        exit(EXIT_FAILURE);
    }
    
    return L->cursor->data;
}

bool equals(List A, List B){ // Returns true iff Lists A and B contain the same sequence of elements, returns false otherwise.
    if (A == NULL && B == NULL){
        return true;
    }
    if (A == NULL || B == NULL){
        return false;
    }
    if (A->length != B->length){
        return false;
    }

    Node* currentA = A->front;
    Node* currentB = B->front;

    while(currentA != NULL && currentB != NULL){
        if (currentA->data != currentB->data){
            return false;
        }
        currentA = currentA->next;
        currentB = currentB->next;
    }

    return true;
}

// Manipulation procedures ----------------------------------------------------
void clear(List L){ // Resets L to its original empty state.
    if (L == NULL){
        return;
    }

    Node* current = L->front;
    Node* temp;

    while(current != NULL){
        temp = current;
        current = current->next;
        free(temp);
    }

    L->front = NULL;
    L->back = NULL;
    L->index = -1;
    L->length = 0;
    L->cursor = NULL;
    return;
}

void set(List L, int x){ // Overwrites the cursor elementâ€™s data with x. Pre: length()>0, index()>=0
    if (L->length <= 0){
        fprintf(stderr, "List Error: Calling set() on an empty list");
        exit(EXIT_FAILURE);
    }
    if (L->cursor < 0){
        fprintf(stderr, "List Error: Calling set() with an undefined cursor");
        exit(EXIT_FAILURE);
    }
    
    L->cursor->data = x;
    return;
}

void moveFront(List L){ // If L is non-empty, sets cursor under the front element, otherwise does nothing.
    if (L->length <= 0){
        return;
    }
    L->index = 0;
    L->cursor = L->front;
    return;
}

void moveBack(List L){ // If L is non-empty, sets cursor under the back element, otherwise does nothing.
    if (L->length <= 0){
        return;
    }
    L->index = L->length-1 ;
    L->cursor = L->back;
    return;
}

void movePrev(List L){ // If cursor is defined and not at front, move cursor one step toward the front of L; if cursor is defined and at front, cursor becomes undefined; if cursor is undefined do nothing
    if (L->index == 0){
        L->index = -1;
        L->cursor = NULL;
        return;
    }
    if(L->index == -1){
        return;
    }
    L->index -= 1;
    L->cursor = L->cursor->prev;
    return;
}

void moveNext(List L){ // If cursor is defined and not at back, move cursor one step toward the back of L; if cursor is defined and at back, cursor becomes undefined; if cursor is undefined do nothing
    if (L->index == L->length - 1){
        L->index = -1;
        L->cursor = NULL;
        return;
    }
    if (L->index == -1){
        return;
    }
    L->index += 1;
    L->cursor = L->cursor->next;
    return;
}

void prepend(List L, int x){ // Insert new element into L. If L is non-empty, insertion takes place before front element.
    if (L == NULL){
        fprintf(stderr, "List Error: Calling prepend() on a NULL list reference");
        exit(EXIT_FAILURE);
    }

    Node* N = newNode(x);

    if (N == NULL){
        fprintf(stderr, "List Error: Memory allocation error in prepend()");
        exit(EXIT_FAILURE);
    }

    if(L->length == 0){
        L->front = N;
        L->back = N;
        N->prev = NULL;
        N->next = NULL;
    } else{
        L->front->prev = N;
        N->next = L->front;
        N->prev = NULL;
    }

    L->front = N;
    L->length++;
    if (L->index >= 0){
        L->index++;
    }
    return;
}

void append(List L, int x){ // Insert new element into L. If L is non-empty, insertion takes place after back element.
    if (L == NULL){
        fprintf(stderr, "List Error: Calling append() on a NULL list reference");
        exit(EXIT_FAILURE);
    }

    Node* N = newNode(x);

    if (L->length == 0){
        L->front = N;
        L->back = N;
    } else {
        L->back->next = N;
        N->prev = L->back;
    }

    L->back = N;
    L->length++;
    return;
}

void insertBefore(List L, int x){ // Insert new element before cursor. Pre: length()>0, index()>=0
    if (L->length <= 0){
        fprintf(stderr, "List Error: calling insertBefore() on an empty list");
        exit(EXIT_FAILURE);
    }
    if (L->index < 0){
        fprintf(stderr, "List Error: calling insertBefore() with an undefined cursor");
        exit(EXIT_FAILURE);
    }
    if (L == NULL){
        fprintf(stderr, "List Error: calling insertBefore() on a NULL list reference");
        exit(EXIT_FAILURE);
    }

    if (L->index == 0){
        prepend(L, x);
        return;
    }

    Node* N = newNode(x);

    N->next = L->cursor;
    N->prev = L->cursor->prev;
    L->cursor->prev->next = N;
    L->cursor->prev = N;

    L->length++;
    L->index++;
    return;
}

void insertAfter(List L, int x){ // Insert new element after cursor. Pre: length()>0, index()>=0
    if (L->length <= 0){
        fprintf(stderr, "List Error: calling insertAfter() on an empty list");
        exit(EXIT_FAILURE);
    }
    if (L->index < 0){
        fprintf(stderr, "List Error: calling insertAfter() with an undefined cursor");
        exit(EXIT_FAILURE);
    }
    if (L == NULL){
        fprintf(stderr, "List Error: calling insertAfter() on a NULL list reference");
        exit(EXIT_FAILURE);
    }

    if (L->index == L->length - 1){
        append(L, x);
        return;
    }

    Node* N = newNode(x);

    N->prev = L->cursor;
    N->next = L->cursor->next;
    if (L->cursor->next != NULL) {
       L->cursor->next->prev = N;
    }
    L->cursor->next = N;

    L->length++;
    return;
}

void deleteFront(List L){ // Delete the front element. Pre: length()>0
    if (L->length <= 0){
        fprintf(stderr, "List Error: Calling deleteFront() on an empty list");
        exit(EXIT_FAILURE);
    }
    if (L == NULL){
        fprintf(stderr, "List Error: Calling deleteFront() on a NULL list reference");
        exit(EXIT_FAILURE);
    }

    Node* nodeToDelete = L->front;

    if (L->length == 1){
        L->front = NULL;
        L->back = NULL;
    } else{
        L->front = L->front->next;
        L->front->prev = NULL;
    }

    if (L->index == 0){
        L->index = -1;
        L->cursor = NULL;
    } else if(L->index != -1){
        L->index--;
    }

    free(nodeToDelete);
    
    L->length--;
    return;
}

void deleteBack(List L){ // Delete the back element. Pre: length()>0
    if (L->length <= 0){
            fprintf(stderr, "List Error: Calling deleteBack() on an empty list");
            exit(EXIT_FAILURE);
    }
    if (L == NULL){
        fprintf(stderr, "List Error: Calling deleteBack() on a NULL list reference");
        exit(EXIT_FAILURE);
    }

    Node* nodeToDelete = L->back;

    if (L->cursor == L->back){
        L->cursor = NULL;
        L->index = -1;
    }

    if (L->back->prev != NULL){
        L->back = L->back->prev;
        L->back->next = NULL;
    } else {
        L->front = NULL;
        L->back = NULL;
    }

    free(nodeToDelete);
    nodeToDelete = NULL;
    
    L->length--;
    return;
}

void delete(List L){ // Delete cursor element, making cursor undefined. Pre: length()>0, index()>=0
    if (L == NULL){
        fprintf(stderr, "List Error: Calling delete() on a NULL list reference");
        exit(EXIT_FAILURE);
    }
    if (L->length <= 0){
        fprintf(stderr, "List Error: Calling delete() on an empty list");
        exit(EXIT_FAILURE);
    }
    if (L->cursor < 0){
        fprintf(stderr, "List Error: calling delete() with an undefined cursor");
        exit(EXIT_FAILURE);
    }

    Node* nodeToDelete = L->cursor;
    
    if (nodeToDelete->prev != NULL) {
        nodeToDelete->prev->next = nodeToDelete->next;
    } else {
        L->front = nodeToDelete->next;
    }

    if (nodeToDelete->next != NULL) {
        nodeToDelete->next->prev = nodeToDelete->prev;
    } else {
        L->back = nodeToDelete->prev;
    }

    free(nodeToDelete);
    
    L->index = -1;
    L->cursor = NULL;
    L->length--;
    return;
}

// Other operations -----------------------------------------------------------

void printList(FILE* out, List L){ // Prints to the file pointed to by out, a string representation of L consisting of a space separated sequence of integers, with front on left.
    if (L == NULL){
        fprintf(stderr, "List Error: Calling printList() on a NULL reference");
        exit(EXIT_FAILURE);
    }
    if (L->length == 0){
        fprintf(out, "\n");
        return;
    }

    moveFront(L);
    while(L->cursor != NULL){
        fprintf(out, "%d", L->cursor->data);
        moveNext(L);
        if (L->cursor != NULL){
            fprintf(out, " ");
        }
    }
    return;
}

List copyList(List L){ // Returns a new List representing the same integer sequence as L. The cursor in the new list is undefined, regardless of the state of the cursor in L. The state of L is unchanged.
    if (L == NULL){
        fprintf(stderr, "List Error: Calling copyList() with a NULL reference list");
        exit(EXIT_FAILURE);
    }
    
    List newList = malloc(sizeof(ListObj));
    if (newList == NULL){
        fprintf(stderr, "List Error: Calling copyList() and resulted in a memory allocation error");
        exit(EXIT_FAILURE);
    }

    newList->front = NULL;
    newList->back = NULL;
    newList->index = -1;
    newList->length = 0;

    for (Node* N = L->front; N != NULL; N = N->next){
        append(newList, N->data);
    }
    
    return newList;
}
