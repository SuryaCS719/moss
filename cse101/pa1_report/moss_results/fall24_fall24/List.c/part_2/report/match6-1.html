#include "List.h"



 typedef struct Nodeobj* Node;
   typedef struct Nodeobj {
    int data;
    struct Nodeobj* next;
    struct Nodeobj* prev;
} Nodeobj;



typedef struct ListObj{
    int length;
    int index;
    Node head;
    Node tail; 
    Node cursor;
} ListObj;


// Constructors-Destructors ---------------------------------------------------

// Creates and returns a new empty List.
List newList(void){
    List L = malloc(sizeof(ListObj));
    // Fail to allocate memory
    assert( L != NULL );
    L->length=0;
    L->index=-1;
    L->head=NULL;
    L->tail=NULL;
    L->cursor=NULL;
    return L;
}

// Frees all heap memory associated with *pL, and sets
// *pL to NULL.
void freeList(List* pL){
    // No list or empty list
    if(pL==NULL||*pL==NULL)
    {
        fprintf(stderr,"List Error: calling freeList(List* pL) on NULL List reference.\n");
        return;
    }
    // free each node in the list
    Node current = (*pL)->head;
    while(current!=NULL){
        Node temp = current->next;
        free(current);
        current = temp;
    }
    (*pL)->cursor = NULL;
    free(*pL); // free list
    *pL = NULL;
    pL = NULL;
}

// Creates and returns a new emoty node.
Node newNode(int data) {
    Node N = malloc(sizeof(Nodeobj)); // Allocate memory for a new Node
    assert(N != NULL); // Ensure memory allocation is successful
    N->data = data; // Set the data
    N->next = NULL; // Initialize next pointer to NULL
    N->prev = NULL; // Initialize previous pointer to NULL
    return N; // Return the newly created node
}

// Free the heap memory of the node and set node to NULL.
void freeeNode(Node node){
    if (node != NULL) {
        free(node);
        node = NULL;
    }
}



// Access functions -----------------------------------------------------------

// Returns the number of elements in L.
int length(List L){
    if(L==NULL)
    {
        fprintf(stderr, "List Error: calling length(List L) on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    return L->length;
    }

// Returns index of cursor element if defined, -1 otherwise.
int index(List L){
    if(L==NULL)
    {
        fprintf(stderr, "List Error: calling index(List L) on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    return L->index;
    }

// Returns front element of L. Pre: length()>0
int front(List L){
    if(L==NULL || L->length <=0)
    {
        fprintf(stderr, "List Error: calling front(List L) on NULL List reference\n");
        exit(EXIT_FAILURE);
    }

    return L->head->data;
}

// Returns back element of L. Pre: length()>0
int back(List L){
    if(L==NULL || L->length <= 0)
    {
        fprintf(stderr, "List Error: calling back(List L) on NULL List reference\n");
        exit(EXIT_FAILURE);
    }

    return L->tail->data;
    }
    
// Returns cursor element of L. Pre: length()>0, index()>=0
// Pre: length(L)>0&&index(L)>=0
int get(List L){
    if(L==NULL)
    {
        fprintf(stderr, "List Error: calling get(List L) on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if(length(L)>0&&index(L)>=0)
    {
        return L->cursor->data;
    }
    return -1;
    }

// Returns true iff Lists A and B contain the same
 // sequence of elements, returns false otherwise.
bool equals(List A, List B){
    if(A == NULL || B == NULL)
    {
        fprintf(stderr, "List Error: calling equals(List A, List B) on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    // Compare length of two lists
    if(length(A)!=length(B)){
        return false;
    }
    if(length(A)==length(B)){
        Node node1 = A->head;
        Node node2 = B->head;
        while(node1 != NULL && node2 != NULL){
            if (node1->data!=node2->data){
                return false;
            }
            // Compare next pair of nodes


            node1 = node1->next;
            node2 = node2->next;
        }    
    }
    return true;
    }

// Manipulation procedures ----------------------------------------------------

// Resets L to its original empty state.
void clear(List L){
    if(L==NULL){
        fprintf(stderr,"List Error: calling clear(List L) on NULL List reference\n");
        return;
    }
    Node current = L->head;
    while(current!=NULL){
        Node temp = current->next;
        free(current);
        current = temp;
    }
    L->head=NULL;
    L->tail=NULL;
    L->cursor=NULL;
    L->length=0;


    L->index=-1;
}

// Overwrites the cursor elementâ€™s data with x.
 // Pre: length()>0, index()>=0
void set(List L, int x){
    if(L==NULL)
    {
        fprintf(stderr, "List Error: calling set(List L, int x) on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if(L->length>0&&L->index>=0){
        L->cursor->data = x;
    }
    }

 // If L is non-empty, sets cursor under the front element,
 // otherwise does nothing.
 // Pre: L->length > 0
void moveFront(List L){
    if(L==NULL)
    {
        fprintf(stderr,"List Error: calling moveFront(List L) on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    // Check empty list
    if(L->length>0){
    L->cursor = L->head;
    L->index = 0;
    }
 }


 // If L is non-empty, sets cursor under the back element,
 // otherwise does nothing.
 // Pre: L->length > 0
void moveBack(List L){
    if(L==NULL)
    {
        fprintf(stderr, "List Error: calling moveBack(List L) on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    // Check empty list
    if(L->length>0){
    L->cursor = L->tail;


    L->index = L->length-1;
    }

} 

 // If cursor is defined and not at front, move cursor one
 // step toward the front of L; if cursor is defined and at
 // front, cursor becomes undefined; if cursor is undefined
 // do nothing
 // Pre: L->length > 0 && L->index >= 0
void movePrev(List L){
    if(L==NULL)
    {
        fprintf(stderr, "List Error: calling movePrev(List L) on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if(L->length>0&&L->index>0){
        L->cursor = L->cursor->prev;
        L->index--;
        return;
    }
    else if(L->length>0&&L->index == 0){
        L->cursor = NULL;
        L->index = -1;
        return;
    }
    
} 

 // If cursor is defined and not at back, move cursor one
 // step toward the back of L; if cursor is defined and at
 // back, cursor becomes undefined; if cursor is undefined
 // do nothing
 // Pre: L->length > 0 && L->index >= 0
void moveNext(List L){
    if(L==NULL)
    {
        fprintf(stderr, "List Error: calling moveNext(List L) on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if(L->length > 0 && L->index >= 0 && L->index != L->length-1) {
        L->cursor = L->cursor->next;
        L->index++;
        return;
    }

    if(L->length>0 && L->index == L->length-1){
        L->cursor = NULL;
        L->index = -1;
        return;
    }
} 

 // Insert new element into L. If L is non-empty,
 // insertion takes place before front element.
 // Pre: L->length > 0
void prepend(List L, int x){
    if(L==NULL)
    {
        fprintf(stderr, "List Error: calling prepend(List L, int x) on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    Node NewNode = newNode(x);
    // empty list
    if(L->length == 0){
        L->tail = L->head = NewNode;
        moveFront(L);
        L->length = 1;
        return;
    }
    if(L->length>0){
        NewNode->prev = NULL;
        NewNode->next = L->head;
        L->head->prev = NewNode;
        L->head = NewNode;
        L->length++;
        L->index++;
        return;
    }
} 
 // Insert new element into L. If L is non-empty,
 // insertion takes place before front element.
 // Pre: L->length >= 0
void append(List L, int x){
     if(L==NULL)
    {
        fprintf(stderr,"List Error: calling append(List L, int x) on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    Node NewNode = newNode(x);

    // empty list
    if(L->length == 0){
        L->tail = L->head = NewNode;
        L->length = 1;
        moveFront(L);
        return;
    }
    if(L->length>0){
        NewNode->next = NULL;
        NewNode->prev = L->tail;
        L->tail->next = NewNode;
        L->tail = NewNode;
        L->length++;
        return;
    }
} 

// Insert new element before cursor.
 // Pre: length()>0, index()>=0
void insertBefore(List L, int x){
    if(L==NULL)
    {
        fprintf(stderr, "List Error: calling insertBefore(List L, int x) on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->cursor == NULL) {
        fprintf(stderr, "List Error: calling insertAfter() with NULL cursor\n");
        exit(EXIT_FAILURE);
    }

    if(L->length>0&&L->index>=0){
         // Cursor is the first element
        if(L->index==0){
            prepend(L,x);
        }
        else{
            Node NewNode = newNode(x);
            NewNode->next = L->cursor;
            NewNode->prev = L->cursor->prev;
            L->cursor->prev->next = NewNode;
            L->cursor->prev = NewNode;
            L->index++;
            L->length++;
            
        }
    }
} 

 // Insert new element after cursor.
 // Pre: length()>0, index()>=0


void insertAfter(List L, int x){
     if(L==NULL)
    {
        fprintf(stderr, "List Error: calling insertAfter(List L, int x) on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->cursor == NULL) {
        fprintf(stderr, "List Error: calling insertAfter() with NULL cursor\n");
        exit(EXIT_FAILURE);
    }

    // Cursor is the last element
    if(L->length>0&&L->index>=0){
        if(L->index== L->length-1){
            append(L,x);
        }

        // Cursor is in the middle
        else{
            Node NewNode = newNode(x);
            NewNode->next = L->cursor->next;
            NewNode->prev = L->cursor;
            L->cursor->next->prev = NewNode;
            L->cursor->next = NewNode;


            L->length++;
        }
    }
} 

 // Delete the front element. Pre: length()>0
 // Pre: length()>0
void deleteFront(List L){
    if(L==NULL)
    {
        fprintf(stderr, "List Error: calling deleteFront(List L) on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if(L->length>0){
        // Only one element in the list, clear the list
        if(L->length==1){
            clear(L);
            return;
        }
        else{
            Node temp = L->head; // element to be deleted
            if(L->cursor == L->head){
                L->cursor = NULL;
                L->index = -1;
            }

            L->head->next->prev = NULL;
            L->head = L->head->next;
            free(temp);
            // If the cursor is still valid, decrement the index
            if(L->cursor != NULL &&L->index >0){
            L->index--;
            }
            L->length--;
        }
    }

} 

// Delete the back element. Pre: length()>0
void deleteBack(List L){
      if(L==NULL)
    {
        fprintf(stderr, "List Error: calling deleteBack(List L) on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if(L->length>0){
        if(L->length==1){
            clear(L);
            return;
        }
        else{
            Node temp = L->tail; // element to be deleted
            if(L->cursor == L->tail){
                L->cursor = NULL;
                L->index = -1;
            }
            // move the last element back one step
            L->tail = L->tail->prev;
            L->tail->next = NULL;
            free(temp);
            L->length--;
            return;
        }
    }
}

 // Delete cursor element, making cursor undefined.
 // Pre: length()>0, index()>=0
void delete(List L){
     if(L==NULL)
    {
        fprintf(stderr, "List Error: calling delete(List L) on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if(L->length>0&&L->index>=0){
        // Only one element in list
        if(L->length==1){
            clear(L);
            return;
        }

        // Cursor is at the front of the list
       else if(L->index == 0){
            deleteFront(L);
            return;
        }


        // CUrsor is at the back of the list
       else if(L->index == L->length-1){
            deleteBack(L);
            return;
        }

        // Cursor is in the middle of the list
        else{
            Node before = L->cursor->prev;
            Node after = L->cursor->next;
            after->prev = before;
            before->next = after;
            // Clear cursor and set index to -1
            free(L->cursor);
            L->index = -1;
            L->cursor = NULL;
            L->length--;
        }


    }
}

 // Prints to the file pointed to by out, a
 // string representation of L consisting
 // of a space separated sequence of integers,
 // with front on left.
void printList(FILE* out, List L){
    if(L==NULL)
    {
        fprintf(stderr, "List Error: calling printList(FILE* out, List L) on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    
    Node current = L->head;
    while(current!=NULL){


       fprintf(out, "%d\n", current->data);
       current = current->next;
    }
    

} 

// Returns a new List representing the same integer
// sequence as L. The cursor in the new list is undefined,
// regardless of the state of the cursor in L. The state
// of L is unchanged
// Pre: L->length>0 && L->head != NULL
List copyList(List L){
    if(L==NULL)
    {
        fprintf(stderr, "List Error: calling copyList(List L) on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    List NewList = newList();
    if(L->length>0&&L->head != NULL)
    {
        Node current = L->head;
        while(current != NULL){
            append(NewList, current->data);
            current = current->next;
        }
    }
    return NewList;
} 

// Returns a new List which is the concatenation of list1 and list2.
// The cursor in the new List is undefined, regardless of the states of the cursors in list1 and list2.
// The states of list1 and list2 are unchanged.
List concatList(List list1, List list2) {
    if (list1 == NULL || list2 == NULL) {
        fprintf(stderr, "List Error: calling concatList() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }

    // Create a new list to store the concatenated result
    List NewList = newList();

    // Traverse list1 and append its elements to the new list
    Node trav = list1->head;
    while(trav != NULL) {
        append(NewList, trav->data);
        trav = trav->next; 
    }

    // Traverse list2 and append its elements to the new list
    trav = list2->head;
     while(trav != NULL) {
        append(NewList, trav->data);
        trav = trav->next; 
    } 

    // The cursor of the new list should be undefined
    NewList->cursor = NULL;
    NewList->index = -1;

    // Return the concatenated list
    return NewList;
}
