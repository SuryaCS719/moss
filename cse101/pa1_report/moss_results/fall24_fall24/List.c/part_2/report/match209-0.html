#include <stdio.h>
#include <stdlib.h>
#include "List.h"

// Node struct definition (private)
// Represents a single element in the doubly linked list.
// Contains an integer 'data' and pointers to the next and previous nodes.
typedef struct NodeObj {
    int data;
    struct NodeObj* next;
    struct NodeObj* prev;
} NodeObj;



typedef NodeObj* Node; // A Node is a pointer to a NodeObj.

// List struct definition (private)
// Represents the List ADT as a doubly linked list with a cursor.
// The List has pointers to the front, back, and cursor elements, along with
// an integer tracking the length of the list and the index of the cursor.
typedef struct ListObj {
    Node front;    // Points to the front (first) element of the list.
    Node back;     // Points to the back (last) element of the list.
    Node cursor;   // Points to the current element under the cursor.
    int length;    // Tracks the number of elements in the list.
    int index;     // Index of the cursor (-1 if the cursor is undefined).
} ListObj;

// Constructors-Destructors ---------------------------------------------------

// newList()
// Creates and returns a new empty List.
List newList(void) {
    List L = malloc(sizeof(ListObj));
    if (L == NULL) {
        fprintf(stderr, "List Error: newList() failed to allocate memory\n");
        exit(EXIT_FAILURE);
    }
    L->front = L->back = L->cursor = NULL;  // Initialize an empty list.
    L->length = 0;
    L->index = -1;  // The cursor is undefined in an empty list.
    return L;
}

// freeList()
// Frees all heap memory associated with *pL, and sets *pL to NULL.
// This function ensures that all nodes in the list are deallocated.
void freeList(List* pL) {
    if (pL != NULL && *pL != NULL) {
        while ((*pL)->length > 0) {
            deleteFront(*pL);  // Delete each element from the front.
        }
        free(*pL);  // Free the List structure itself.
        *pL = NULL; // Set the pointer to NULL to avoid dangling references.
    }
}

// Access functions -----------------------------------------------------------

// length()
// Returns the number of elements in the list.
int length(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling length() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    return L->length;
}

// index()
// Returns the index of the cursor element, or -1 if the cursor is undefined.
int index(List L) {


    if (L == NULL) {
        fprintf(stderr, "List Error: calling index() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    return L->index;
}

// front()
// Returns the front element of the list. Precondition: length() > 0.
int front(List L) {
    if (L == NULL || L->length == 0) {
        fprintf(stderr, "List Error: calling front() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    return L->front->data;
}

// back()
// Returns the back element of the list. Precondition: length() > 0.
int back(List L) {
    if (L == NULL || L->length == 0) {
        fprintf(stderr, "List Error: calling back() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    return L->back->data;
}

// get()
// Returns the cursor element of the list. Precondition: length() > 0 and index() >= 0.
int get(List L) {
    if (L == NULL || L->length == 0 || L->index < 0) {
        fprintf(stderr, "List Error: calling get() on an undefined cursor\n");
        exit(EXIT_FAILURE);
    }
    return L->cursor->data;
}

// equals()
// Returns true if A and B contain the same sequence of elements, false otherwise.
bool equals(List A, List B) {
    if (A == NULL || B == NULL) {
        fprintf(stderr, "List Error: calling equals() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (A->length != B->length) return false;  // Lists must have the same length.

    Node aNode = A->front;
    Node bNode = B->front;
    while (aNode != NULL && bNode != NULL) {
        if (aNode->data != bNode->data) return false;  // Compare element by element.
        aNode = aNode->next;
        bNode = bNode->next;
    }
    return true;
}

// Manipulation procedures ----------------------------------------------------

// clear()
// Resets the list to its original empty state.
void clear(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling clear() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    while (L->length > 0) {
        deleteFront(L);  // Remove all elements from the front.
    }


    L->front = L->back = L->cursor = NULL;
    L->length = 0;
    L->index = -1;  // The cursor is undefined after clearing the list.
}

// set()
// Overwrites the data at the cursor with a new value. Precondition: length() > 0 and index() >= 0.
void set(List L, int x) {
    if (L == NULL || L->length == 0 || L->index < 0) {
        fprintf(stderr, "List Error: calling set() on an undefined cursor\n");
        exit(EXIT_FAILURE);
    }
    L->cursor->data = x;
}

// moveFront()
// Moves the cursor to the front of the list, if the list is non-empty.
void moveFront(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling moveFront() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->length > 0) {
        L->cursor = L->front;
        L->index = 0;  // The cursor is now at the first element.
    }
}

// moveBack()
// Moves the cursor to the back of the list, if the list is non-empty.
void moveBack(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling moveBack() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->length > 0) {
        L->cursor = L->back;
        L->index = L->length - 1;  // The cursor is now at the last element.
    }
}

// movePrev()
// Moves the cursor one step toward the front of the list. If the cursor is at the front, it becomes undefined.
void movePrev(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling movePrev() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->cursor != NULL) {
        if (L->cursor == L->front) {
            L->cursor = NULL;
            L->index = -1;  // Cursor becomes undefined.
        } else {
            L->cursor = L->cursor->prev;
            L->index--;  // Move cursor toward the front.
        }
    }
}

// moveNext()
// Moves the cursor one step toward the back of the list. If the cursor is at the back, it becomes undefined.
void moveNext(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling moveNext() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->cursor != NULL) {
        if (L->cursor == L->back) {
            L->cursor = NULL;
            L->index = -1;  // Cursor becomes undefined.
        } else {
            L->cursor = L->cursor->next;
            L->index++;  // Move cursor toward the back.
        }
    }
}

// prepend()
// Inserts a new element at the front of the list.
void prepend(List L, int x) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling prepend() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    Node N = malloc(sizeof(NodeObj));  // Create a new node.


    if (N == NULL) {
        fprintf(stderr, "List Error: prepend() failed to allocate memory\n");
        exit(EXIT_FAILURE);
    }
    N->data = x;
    N->next = L->front;
    N->prev = NULL;

    if (L->front != NULL) {
        L->front->prev = N;
    } else {
        L->back = N;  // If the list was empty, the new node becomes both front and back.
    }
    L->front = N;
    L->length++;
    if (L->cursor != NULL) {
        L->index++;  // Adjust cursor index if the cursor is defined.
    }
}

// append()
// Inserts a new element at the back of the list.
void append(List L, int x) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling append() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    Node N = malloc(sizeof(NodeObj));  // Create a new node.
    if (N == NULL) {
        fprintf(stderr, "List Error: append() failed to allocate memory\n");
        exit(EXIT_FAILURE);
    }
    N->data = x;
    N->next = NULL;
    N->prev = L->back;

    if (L->back != NULL) {
        L->back->next = N;
    } else {
        L->front = N;  // If the list was empty, the new node becomes both front and back.
    }
    L->back = N;
    L->length++;
}

// insertBefore()
// Inserts a new element before the cursor. Precondition: length() > 0 and index() >= 0.
void insertBefore(List L, int x) {
    if (L == NULL || L->length == 0 || L->index < 0) {
        fprintf(stderr, "List Error: calling insertBefore() on an undefined cursor\n");
        exit(EXIT_FAILURE);
    }
    if (L->cursor == L->front) {
        prepend(L, x);  // Special case: insert at the front.
    } else {
        Node N = malloc(sizeof(NodeObj));  // Create a new node.
        if (N == NULL) {
            fprintf(stderr, "List Error: insertBefore() failed to allocate memory\n");
            exit(EXIT_FAILURE);
        }
        N->data = x;
        N->next = L->cursor;
        N->prev = L->cursor->prev;
        L->cursor->prev->next = N;
        L->cursor->prev = N;
        L->length++;
        L->index++;
    }
}

// insertAfter()
// Inserts a new element after the cursor. Precondition: length() > 0 and index() >= 0.
void insertAfter(List L, int x) {
    if (L == NULL || L->length == 0 || L->index < 0) {
        fprintf(stderr, "List Error: calling insertAfter() on an undefined cursor\n");
        exit(EXIT_FAILURE);
    }
    if (L->cursor == L->back) {
        append(L, x);  // Special case: insert at the back.
    } else {


        Node N = malloc(sizeof(NodeObj));  // Create a new node.
        if (N == NULL) {
            fprintf(stderr, "List Error: insertAfter() failed to allocate memory\n");
            exit(EXIT_FAILURE);
        }
        N->data = x;
        N->next = L->cursor->next;
        N->prev = L->cursor;
        L->cursor->next->prev = N;
        L->cursor->next = N;


        L->length++;
    }
}

// deleteFront()
// Deletes the front element. Precondition: length() > 0.
void deleteFront(List L) {
    if (L == NULL || L->length == 0) {
        fprintf(stderr, "List Error: calling deleteFront() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    Node N = L->front;
    if (L->front == L->back) {  // Special case: only one element.
        L->front = L->back = NULL;
    } else {
        L->front = L->front->next;
        L->front->prev = NULL;
    }
    if (L->cursor == N) {
        L->cursor = NULL;
        L->index = -1;  // Cursor becomes undefined if it was at the front.
    } else if (L->cursor != NULL) {
        L->index--;  // Adjust cursor index.
    }
    L->length--;
    free(N);  // Free the node's memory.
}

// deleteBack()
// Deletes the back element. Precondition: length() > 0.
void deleteBack(List L) {
    if (L == NULL || L->length == 0) {
        fprintf(stderr, "List Error: calling deleteBack() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    Node N = L->back;
    if (L->front == L->back) {  // Special case: only one element.
        L->front = L->back = NULL;
    } else {
        L->back = L->back->prev;
        L->back->next = NULL;
    }
    if (L->cursor == N) {
        L->cursor = NULL;
        L->index = -1;  // Cursor becomes undefined if it was at the back.
    }
    L->length--;
    free(N);  // Free the node's memory.
}

// delete()
// Deletes the cursor element, making the cursor undefined. Precondition: length() > 0 and index() >= 0.
void delete(List L) {
    if (L == NULL || L->length == 0 || L->index < 0) {
        fprintf(stderr, "List Error: calling delete() on an undefined cursor\n");
        exit(EXIT_FAILURE);
    }
    Node N = L->cursor;
    if (N == L->front) {
        deleteFront(L);  // Special case: deleting the front element.
    } else if (N == L->back) {
        deleteBack(L);  // Special case: deleting the back element.
    } else {
        N->prev->next = N->next;
        N->next->prev = N->prev;
        L->length--;
        free(N);  // Free the node's memory.
    }
    L->cursor = NULL;
    L->index = -1;  // Cursor becomes undefined after deletion.
}

// Other operations -----------------------------------------------------------

// printList()
// Prints the list to the file pointed to by 'out', with elements separated by spaces.
void printList(FILE* out, List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling printList() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    Node N = L->front;
    while (N != NULL) {
        fprintf(out, "%d ", N->data);  // Print each element in sequence.
        N = N->next;
    }
}

// copyList()
// Creates and returns a new list that is a copy of the original list.
List copyList(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling copyList() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    List copy = newList();  // Create a new empty list.
    Node N = L->front;
    while (N != NULL) {
        append(copy, N->data);  // Append each element from the original list.
        N = N->next;
    }
    return copy;
}
