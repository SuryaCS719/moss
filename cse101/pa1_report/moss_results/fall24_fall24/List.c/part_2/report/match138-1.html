#include <stdio.h>
#include <stdlib.h>
#include "List.h"

typedef struct NodeObj {
    int data;
    struct NodeObj* next;
    struct NodeObj* prev;
} NodeObj;

typedef NodeObj* Node;

typedef struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
    int index;
} ListObj;

List newList(void) {
    List L = malloc(sizeof(ListObj));
    L->front = L->back = L->cursor = NULL;
    L->length = 0;
    L->index = -1;
    return L;
}

void freeList(List* pL) {
    if (pL != NULL && *pL != NULL) {
        clear(*pL);
        free(*pL);
        *pL = NULL;
    }
}

int length(List L) { return L->length; }
int index(List L) { return L->index; }


int front(List L) { return L->front->data; }
int back(List L) { return L->back->data; }
int get(List L) { return L->cursor->data; }

void clear(List L) {
    while (L->length > 0) {
        deleteFront(L);
    }
    L->index = -1;
}

void moveFront(List L) {
    if (L->length > 0) {
        L->cursor = L->front;
        L->index = 0;
    }
}

void moveBack(List L) {
    if (L->length > 0) {
        L->cursor = L->back;
        L->index = L->length - 1;
    }
}

void movePrev(List L) {
    if (L->cursor != NULL) {
        if (L->cursor == L->front) {
            L->cursor = NULL;
            L->index = -1;
        } else {
            L->cursor = L->cursor->prev;
            L->index--;
        }
    }
}

void moveNext(List L) {
    if (L->cursor != NULL) {
        if (L->cursor == L->back) {
            L->cursor = NULL;
            L->index = -1;
        } else {
            L->cursor = L->cursor->next;
            L->index++;
        }
    }
}

void prepend(List L, int x) {
    Node N = malloc(sizeof(NodeObj));
    N->data = x;
    N->prev = NULL;
    N->next = L->front;
    if (L->length == 0) {
        L->back = N;
    } else {


        L->front->prev = N;
    }
    L->front = N;
    L->length++;
}

void append(List L, int x) {
    Node N = malloc(sizeof(NodeObj));
    N->data = x;
    N->next = NULL;
    N->prev = L->back;
    if (L->length == 0) {
        L->front = N;
    } else {
        L->back->next = N;
    }
    L->back = N;
    L->length++;
}

void insertBefore(List L, int x) {
    if (L->length > 0 && L->index >= 0) {
        Node N = malloc(sizeof(NodeObj));


        N->data = x;
        N->prev = L->cursor->prev;
        N->next = L->cursor;
        if (L->cursor == L->front) {
            L->front = N;
        } else {
            L->cursor->prev->next = N;
        }
        L->cursor->prev = N;
        L->length++;
        L->index++;
    }
}

void insertAfter(List L, int x) {
    if (L->length > 0 && L->index >= 0) {
        Node N = malloc(sizeof(NodeObj));


        N->data = x;
        N->next = L->cursor->next;
        N->prev = L->cursor;
        if (L->cursor == L->back) {
            L->back = N;
        } else {
            L->cursor->next->prev = N;
        }
        L->cursor->next = N;
        L->length++;
    }
}

void deleteFront(List L) {
    if (L->length > 0) {
        Node N = L->front;
        if (L->cursor == L->front) {
            L->cursor = NULL;
            L->index = -1;
        }
        if (L->length > 1) {


            L->front = L->front->next;
            L->front->prev = NULL;
        } else {
            L->front = L->back = NULL;
        }
        free(N);
        L->length--;
    }
}

void deleteBack(List L) {
    if (L->length > 0) {
        Node N = L->back;
        if (L->cursor == L->back) {
            L->cursor = NULL;
            L->index = -1;
        }
        if (L->length > 1) {


            L->back = L->back->prev;
            L->back->next = NULL;
        } else {
            L->front = L->back = NULL;
        }
        free(N);
        L->length--;
    }
}

void delete(List L) {
    if (L->length > 0 && L->index >= 0) {
        Node N = L->cursor;
        if (L->cursor == L->front) {
            deleteFront(L);
        } else if (L->cursor == L->back) {
            deleteBack(L);
        } else {
            N->prev->next = N->next;
            N->next->prev = N->prev;
            free(N);
            L->length--;
        }
        L->cursor = NULL;
        L->index = -1;
    }
}

void printList(FILE* out, List L) {
    Node N = L->front;
    while (N != NULL) {
        fprintf(out, "%d ", N->data);
        N = N->next;
    }
}

List copyList(List L) {
    List C = newList();
    Node N = L->front;


    while (N != NULL) {
        append(C, N->data);
        N = N->next;
    }
    return C;
}

bool equals(List A, List B) {
    if (A->length != B->length) {
        return false;
    }

    Node N = A->front;
    Node M = B->front;

    while (N != NULL && M != NULL) {
        if (N->data != M->data) {
            return false;
        }
        N = N->next;
        M = M->next;
    }

    return true;
}
