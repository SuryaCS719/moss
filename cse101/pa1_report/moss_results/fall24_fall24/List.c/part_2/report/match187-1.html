#include <stdbool.h>

List newList(void) {
    List L = (List)malloc(sizeof(struct ListObj));

    if (L == NULL) {
        exit(1);
    }

    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->length = 0;
    L->index = -1;

    return L;
}

void freeList(List* pL) {


    if (pL != NULL && *pL != NULL) {
        clear(*pL);
        free(*pL);
        *pL = NULL;
    }
}

bool equals(List A, List B) {
    if (A == NULL || B == NULL) {
        exit(1);
    }

    if (A->length != B->length) {
        return false;
    }

    Node nodeA = A->front;
    Node nodeB = B->front;

    while (nodeA != NULL && nodeB != NULL) {
        if (nodeA->data != nodeB->data) {
            return false;
        }
        nodeA = nodeA->next;
        nodeB = nodeB->next;
    }

    return true;
}

void clear(List L) {
    if (L == NULL) {
        exit(1);
    }

    Node current = L->front;
    Node temp = NULL;

    while (current != NULL) {
        temp = current->next;
        free(current);
        current = temp;
    }

    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->length = 0;
    L->index = -1;
}

    if (L == NULL) {
        exit(1);
    }
    if (L->length <= 0) {
        exit(1);
    }
    if (L->index < 0) {


        exit(1);

    L->cursor->data = x;
}

void moveFront(List L) {
    if (L == NULL) {
        exit(1);
    }
    if (L->length > 0) {
        L->cursor = L->front;
        L->index = 0;
    }
}

void moveBack(List L) {
    if (L == NULL) {
        exit(1);
    }
    if (L->length > 0) {
        L->cursor = L->back;
        L->index = L->length - 1;
    }
}

void movePrev(List L) {
    if (L == NULL) {
        exit(1);
    }

    if (L->cursor != NULL && L->index > 0) {
        L->cursor = L->cursor->prev;
        L->index--;
    } else if (L->cursor != NULL && L->index == 0) {
        L->cursor = NULL;
        L->index = -1;
    }
}

void moveNext(List L) {
    if (L == NULL) {
        exit(1);
    }

    if (L->cursor != NULL && L->index < L->length - 1) {
        L->cursor = L->cursor->next;
        L->index++;
    } else if (L->cursor != NULL && L->index == L->length - 1) {
        L->cursor = NULL;
        L->index = -1;
    }
}

void prepend(List L, int x) {
    if (L == NULL) {
        exit(1);
    }

    Node newNode = malloc(sizeof(struct NodeObj));
    if (newNode == NULL) {
        exit(1);
    }
    newNode->data = x;
    newNode->prev = NULL;
    newNode->next = L->front;

    if (L->length == 0) {
        L->front = newNode;
        L->back = newNode;
    } else {
        L->front->prev = newNode;
        L->front = newNode;
    }



    L->length++;

    if (L->index != -1) {
        L->index++;
    }
}

void insertBefore(List L, int x) {
    if (L == NULL) {
        exit(1);
    }
    if (L->length <= 0) {
        exit(1);
    }
    if (L->index < 0) {
        exit(1);
    }

    Node newNode = malloc(sizeof(struct NodeObj));
    if (newNode == NULL) {
        exit(1);
    }
    newNode->data = x;

    if (L->cursor == L->front) {
        newNode->prev = NULL;
        newNode->next = L->front;
        L->front->prev = newNode;
        L->front = newNode;
    } else {
        newNode->next = L->cursor;
        newNode->prev = L->cursor->prev;
        L->cursor->prev->next = newNode;
        L->cursor->prev = newNode;
    }

    L->length++;
    L->index++;
}

void deleteFront(List L) {
    if (L == NULL) {
        exit(1);
    }
    if (L->length <= 0) {
        exit(1);
    }

    Node temp = L->front;

    if (L->length == 1) {
        L->front = NULL;
        L->back = NULL;
    } else {
        L->front = L->front->next;
        L->front->prev = NULL;
    }

    if (L->index == 0) {
        L->cursor = NULL;
        L->index = -1;
    } else if (L->index > 0) {
        L->index--;
    }

    free(temp);
    L->length--;
}

    if (L == NULL) {
        exit(1);

    if (L->length <= 0) {

    }

    Node temp = L->back;

    if (L->length == 1) {
        L->front = NULL;
        L->back = NULL;
    } else {
        L->back = L->back->prev;
        L->back->next = NULL;
    }

    if (L->index == L->length - 1) {

        L->cursor = NULL;
        L->index = -1;
    }

    free(temp);
    L->length--;
}

void delete(List L) {
    if (L == NULL) {
        exit(1);
    }
    if (L->length <= 0) {
        exit(1);
    }
    if (L->index < 0) {
        exit(1);
    }

    Node temp = L->cursor;

    if (L->cursor == L->front) {
        L->front = L->front->next;
        if (L->front != NULL) {
            L->front->prev = NULL;
        }
    }
    else if (L->cursor == L->back) {
        L->back = L->back->prev;
        if (L->back != NULL) {
            L->back->next = NULL;
        }
    }
    else {
        L->cursor->prev->next = L->cursor->next;
        L->cursor->next->prev = L->cursor->prev;
    }

    free(temp);
    L->cursor = NULL;
    L->index = -1;
    L->length--;
}



void printList(FILE* out, List L) {
    if (L == NULL) {
        exit(1);
    }

    Node temp = L->front;
    while (temp != NULL) {
        fprintf(out, "%d ", temp->data);
        temp = temp->next;
    }
}



List copyList(List L) {
    if (L == NULL) {
        exit(1);
    }

    List newL = newList();
    Node temp = L->front;

    while (temp != NULL) {
        append(newL, temp->data);
        temp = temp->next;
    }

    newL->cursor = NULL;
    newL->index = -1;

    return newL;
}
