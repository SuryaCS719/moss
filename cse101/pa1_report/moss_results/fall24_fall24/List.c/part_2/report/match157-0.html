#include<stdio.h>
#include<stdlib.h>
#include<assert.h>
#include<stdbool.h>
#include "List.h"


// structs --------------------------------------------------------------------

// private NodeObj type
typedef struct NodeObj{
   int data;
   struct NodeObj* previous;
   struct NodeObj* next;
} NodeObj;

// private ListObj type
typedef struct ListObj{
    Node front;
    Node back;
    Node cursor;
    int length;
    int index;
} ListObj;

// Constructors-Destructors ---------------------------------------------------

// newNode()
// Returns reference to new Node object. Initializes next and data fields.


Node newNode(int data){
    Node N = malloc(sizeof(NodeObj));
    assert( N!=NULL );
    N->data = data;
    N->next = NULL;
    return(N);
}

// freeNode()
// Frees heap memory pointed to by *pN, sets *pN to NULL.
void freeNode(Node* pN){
    if( pN!=NULL && *pN!=NULL ){
        free(*pN);
        *pN = NULL;
    }
}

// newList()
// Returns reference to new empty List object.
List newList(){
    List L;
    L = malloc(sizeof(ListObj));
    L->front = L->back = L->cursor = NULL;
    L->length = 0;
    L->index = -1;
    return(L);
}

// freeList()
// Frees all heap memory associated with List *pL, and sets *pL to NULL.
void freeList(List* pL) {
    if(pL!=NULL && *pL!=NULL) { 
        clear(*pL);
        free(*pL);
        *pL = NULL;
    }
}


// Access functions -----------------------------------------------------------

// length()
// Returns the length of L.
int length(List L) {
    return(L->length);
}

// index()
// Returns the current index of L.
int index(List L) {
    return(L->index);
}

// front()
// Returns the value at the front of L.
int front(List L) {
    return(L->front->data);
}

// back()
// Returns the value at the front of L.
int back(List L) {
    return(L->back->data);
}

// get()
// Returns the value at the cursor.
int get(List L) {


    return(L->cursor->data);
}

// equals()
// Returns true if A is same int sequence as B, false otherwise.
bool equals(List A, List B){
    bool eq = false;
    Node N = NULL;
    Node M = NULL;

    eq = (length(A) == length(B));
    N = A->front;
    M = B->front;
    while (eq && N != NULL) {
        eq = (N->data == M->data);
        N = N->next;
        M = M->next;
    }
    return eq;
}

// Manipulation procedures ----------------------------------------------------

// clear()
// Resets L to its original empty state.
void clear(List L) {
    if( L == NULL ) {
        fprintf(stderr, "clear() failed\n");
        exit(EXIT_FAILURE);
    }

    if (L->length == 0) return;

    Node curr = L->front;
    while (curr != NULL) {


        Node temp = curr->next;
        freeNode(&curr);
        curr = temp;
    }
    L->front = NULL;
    L->back = NULL;
    L->length = 0;
    L->index = -1;
}

// set()
// Overwrites the cursor elementâ€™s data with x.
// Pre: length() > 0, index() >= 0
void set(List L, int x) {
    L->cursor->data = x;
}

// moveFront()
// If L is non-empty, sets cursor under the front element, otherwise does nothing.
void moveFront(List L) {
    if (L->length > 0) {
        L->index = 0;
        L->cursor = L->front;
    }
}

// moveBack()
// If L is non-empty, sets cursor under the back element, otherwise does nothing.
void moveBack(List L) {
    if (L->length > 0) {
        L->cursor = L->back;
        L->index = L->length - 1;
    }
}

// movePrev()
// Moves cursor back one space
void movePrev(List L) {
    if (L->cursor == NULL) {
        return;
    }

    if (L->length != L->front) {
        L->index -= 1;
        L->cursor = L->cursor->previous;
    } else {
        L->cursor = NULL;
        L->index = -1;
    }
}

// moveNext()
// Moves cursor forward one space
void moveNext(List L) {
    if (L == NULL) {
        fprintf(stderr, "moveNext() failed\n");
        exit(EXIT_FAILURE);
    }
    if (L->cursor != NULL) {
        if (L->cursor == L->back) {
            L->cursor = NULL;
            L->index = -1;
        } else {
            L->cursor = L->cursor->next;
            L->index++;
        }
    }
}

// prepend()
// Insert new element into L. If L is non-empty, insertion takes place before front element.
void prepend(List L, int x) {


    Node New = newNode(x);
    if (L->length == 0) {
        L->front = New;
        L->back = New;
        L->cursor = New;
        L->index = 0;
    } else {
        New->next = L->front;
        L->front->previous = New;
        L->front = New;
        L->index++;
    }
    L->length++;
}

// append()
// Insert new element into L. If L is non-empty, insertion takes place after back element.
void append(List L, int x) {
    Node new = newNode(x);
    if( L->length <= 0 ) {
        L->front = L->back = new;
    } else {
        new->previous = L->back;
        L->back->next = new;
        L->back = new;
    }
    L->length += 1;
}

// insertBefore()
// Insert new element before cursor.
// Pre: length()>0, index()>=0
void insertBefore(List L, int x) {
    if( L->cursor == L->front ) {
        prepend(L, x);
        return;
    }

    Node new = newNode(x);

    Node tempCursor = L->cursor->previous;
    new->next = L->cursor;
    new->previous = tempCursor;
    L->cursor->previous = new;
    tempCursor->next = new;

    L->length += 1;
    L->index += 1;
}

// insertAfter()
// Insert new element after cursor.
// Pre: length()>0, index()>=0
void insertAfter(List L, int x) {
    if( L->cursor == L->back ) {
        append(L, x);
        return;
    }

    Node new = newNode(x);

    Node tempCursor = L->cursor->next;
    new->previous = L->cursor;
    new->next = tempCursor;
    L->cursor->next = new;
    tempCursor->previous = new;
    L->length += 1;
}

// deleteFront()
// Delete the front element. Pre: length()>0


void deleteFront(List L) {
    if (L == NULL) {
        fprintf(stderr, "calling deleteFront() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }

    if (L->length > 0) {
        Node n = L->front;
        if (L->cursor == L->front) {
            L->cursor = NULL;
            L->index = -1;
        }

        if (L->length > 1) {
            L->front = L->front->next;
            L->front->previous = NULL;
            if(L->index != -1){
                L->index -= 1;
            }    
        } else {
            L->front = L->back = NULL;
            if(L->index != -1){
                L->index -= 1;
            }
        }
        L->length -= 1;
        freeNode(&n);
    }
}

// deleteBack()
// Delete the back element. Pre: length()>0
void deleteBack(List L) {


    if (length(L) <= 0) {
        printf("calling deleteBack() on an empty List\n");
        exit(EXIT_FAILURE);
    }

    Node temp = L->back;
    if (L->cursor == L->back) {
        L->cursor = NULL;
        L->index = -1;
    }

    if (length(L) == 1) {
        L->front = NULL;
        L->back = NULL;
        L->length = 0;
        freeNode(&temp);
    } else {
        L->back = temp->previous;
        L->back->next = NULL;
        L->length--;
        freeNode(&temp);
    }
}

// delete()
// Delete cursor element, making cursor undefined.
// Pre: length()>0, index()>=0
void delete(List L) {
    if (L->cursor == L->front) {
        deleteFront(L);
        return;
    }

    if (L->cursor == L->back) {
        deleteBack(L);
        return;
    }

    Node oldCursor = L->cursor->previous;
    Node newCursor = L->cursor->next;

    if (oldCursor)
        oldCursor->next = newCursor;
    if (newCursor)
        newCursor->next = oldCursor;

    L->cursor = NULL;
    L->index = -1;
    freeNode(&L->cursor);
    L->length -= 1;
}

// Other Functions ------------------------------------------------------------

// printList()
// Prints a string representation of L consisting of a space separated list 
// of ints to stdout.
void printList(FILE* out, List L){
    Node N = NULL;

    if( L==NULL ){
        fprintf(stderr, "List Error: calling printList() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }

    for(N = L->front; N != NULL; N = N->next){
        fprintf(out, "%d", N->data);
    }
    printf("\n");
}

// Returns a new List representing the same integer sequence as L.
List copyList(List L) {
    List tempList = newList();
    int indexL = index(L);
    moveFront(L);
    while(index(L) >= 0) {
        int x = get(L);
        append(tempList, x);
        moveNext(L);
    }
    moveFront(L);
    while (index(L) != indexL) {
        moveNext(L);
    }
    return tempList;
}
