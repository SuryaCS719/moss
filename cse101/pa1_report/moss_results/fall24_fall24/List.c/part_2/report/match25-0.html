#include <stdio.h>
#include <stdlib.h>
#include <assert.h>
#include <stdbool.h>
#include "List.h"

//private node stuff



typedef struct NodeObj* Node;

typedef struct NodeObj {
   int data; //the actual data inside
   Node prev;
   Node next;
} NodeObj;


typedef struct ListObj {
    Node front;
    Node back;
    int length;
    Node cursor;
    int index;
} ListObj;

//actual methods

// Constructor-Destructors
Node newNode(int data) {
    Node N = malloc(sizeof(struct NodeObj));
    assert(N != NULL);
    N->prev = N->next = NULL;
    N->data = data;
    return(N);

}

void freeNode(Node* pN){
   if( pN!=NULL && *pN!=NULL ){
      free(*pN);
      *pN = NULL;
   }
}


List newList(void) {
   List L;
   L = malloc(sizeof(struct ListObj));
   assert( L!=NULL );
   L->front = L->back = NULL;
   L->length = 0;
   L->cursor = NULL;
   L->index = -1;
   return(L);
}

void freeList(List* pL) {
    if (pL != NULL && *pL != NULL) {
        
        while ((*pL)->length > 0) { //remember that * is for dereferencing!!
            deleteFront(*pL);
        }
        free(*pL);
        *pL = NULL;
    }
}

//access functions

int length(List L) {
    return(L->length);
}

int index(List L) {
    if (L->length <= 0) {
        return(-1);
    }
    else {
        return(L->index);
    }
}

int front(List L) {
    if (L->length <= 0) {
        fprintf(stderr, "List Error: calling front(List L) on an empty List");
        exit(EXIT_FAILURE);
    }
    return(L->front->data);
}

int back(List L) {
    if (L->length <= 0) {
        fprintf(stderr, "List Error: calling back(List L) on an empty List");
        exit(EXIT_FAILURE);
    }
    return(L->back->data);
}

int get(List L) {
    if (L->length <= 0) {
        fprintf(stderr, "List Error: calling get(List L) on an empty List");
        exit(EXIT_FAILURE);
    }
    else if (L->index < 0) {
        fprintf(stderr, "List Error: calling get(List L) on a list with an undefined cursor.");
        exit(EXIT_FAILURE);
    }
    return(L->cursor->data);
}

bool equals(List A, List B) {
    if( A==NULL || B==NULL ){
      printf("List Error: calling equals() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }

   bool eq;
   Node N, M;

   eq = ( A->length == B->length );
   N = A->front;
   M = B->front;
   while( eq && N!=NULL){


      eq = ( N->data==M->data );
      N = N->next;
      M = M->next;
   }
   return eq;
    
}


//manipulation procedures
void clear(List L) {
    
    while (L->front != NULL) {
        deleteFront(L);
    }
}

void set(List L, int x) {
    if (L->length <= 0) {
        fprintf(stderr, "List Error: calling set(List L) on an empty List");
        exit(EXIT_FAILURE);
    }
    else if (L->index < 0) {
        fprintf(stderr, "List Error: calling set(List L) with an undefined cursor");
        exit(EXIT_FAILURE);
    }
    L->cursor->data = x;
}

void moveFront(List L) {
    if (L->length > 0) {
        L->cursor = L->front;
        L->index =0;
    }
}

void moveBack(List L) {
    if (L->length > 0) {


        L->cursor = L->back;
        L->index = L->length - 1;
    }
}

void movePrev(List L) {
    if (L->cursor && L->cursor != L->front) {
        L->cursor = L->cursor->prev;
        L->index = L->index - 1;
    }
    else if (L->cursor && L->cursor == L->front){
        L->cursor = NULL;
        L->index = -1;
    }
}

void moveNext (List L) {
    if (L->cursor && L->cursor != L->back) {
        L->cursor = L->cursor->next;
        L->index = L->index + 1;
    }
    else if (L->cursor && L->cursor == L->back){
        L->cursor = NULL;
        L->index = -1;
    }
}

void prepend(List L, int x) {
    if (L->length != 0) {
        Node N = newNode(x);
        L->index++; //cursor should still point to right place
        Node oldFront = L->front;
        oldFront->prev = N;
        N->next = oldFront;
        L->front = N;
        L->length = L->length + 1;
    }
    else if (L && L->length == 0) {
        Node N = newNode(x);
        L->front = N;
        L->back = N;
        L->length = 1;
    }
    else {
        fprintf(stderr, "List Error: calling prepend(List L, int x) on an undefined List");
        exit(EXIT_FAILURE);
    }
}

void append(List L, int x) {
    if (L->length != 0) {
        Node N = newNode(x);
        Node oldBack = L->back;
        oldBack->next = N;
        N->prev = oldBack;
        L->back = N;
        L->length = L->length + 1;
    }
    else if (L && L->length == 0) {
        Node N = newNode(x);
        L->front = N;
        L->back = N;
        L->length = 1;
    }
    else {
        fprintf(stderr, "List Error: calling append(List L, int x) on an undefined List");
        exit(EXIT_FAILURE);
    }
}

void insertBefore(List L, int x) {
    if (L->length <= 0) {
        fprintf(stderr, "List Error: calling index(List L) on an empty List");
        exit(EXIT_FAILURE);
    }
    else if (L->index < 0) {
        fprintf(stderr, "List Error: calling get(List L) on a list with an undefined cursor.");
        exit(EXIT_FAILURE);
    }

    Node N = newNode(x);
    if (L->cursor == L->front) { //cursor is at the beginning
        N->next = L->front;
        N->prev = NULL;
        L->front->prev = N;
        L->front = N;
    }
    else {
        L->cursor->prev->next = N;
        N->prev = L->cursor->prev;
        L->cursor->prev = N;
        N->next = L->cursor;
    }
    
    L->index = L->index + 1;
    L->length = L->length + 1;
}

void insertAfter(List L, int x) {
    if (L->length <= 0) {
        fprintf(stderr, "List Error: calling index(List L) on an empty List");
        exit(EXIT_FAILURE);
    }
    else if (L->index < 0) {
        fprintf(stderr, "List Error: calling get(List L) on a list with an undefined cursor.");
        exit(EXIT_FAILURE);
    }
    Node N = newNode(x);

    if (L->cursor == L->back) {
        N->prev = L->back;
        N->next = NULL;
        L->back->next = N;
        L->back = N;
    }
    else {
        L->cursor->next->prev = N;
        N->next = L->cursor->next;
        L->cursor->next = N;
        N->prev = L->cursor;
    }
    

    L->length = L->length + 1;
}

void deleteFront(List L) {
    if (L->length <= 0) {
        fprintf(stderr, "List Error: calling deleteFront(List L) on an empty List");
        exit(EXIT_FAILURE);
    }
    

    Node oldFront = L->front;
    if (L->length == 1) { //make sure to empty fields if only one node
        L->front = NULL;
        L->back = NULL;
        
    } else {
        if (L->front == L->cursor) {
            L->cursor = NULL;
            
        }

        L->front = L->front->next;
        L->front->prev = NULL;
        L->index--;
    }

    //check the cursor is not on this position
    
    L->length--;
    freeNode(&oldFront);
}

void deleteBack(List L) {
    if (L->length <= 0) {
        fprintf(stderr, "List Error: calling deleteBack(List L) on an empty List");
        exit(EXIT_FAILURE);
    }

    Node oldBack = L->back;

    if (L->length == 1) {
        L->front = NULL;
        L->back = NULL;
        
    }
    else {

        if (L->back == L->cursor) {
            L->cursor = NULL;
            L->index = -1;
        }

        L->back = L->back->prev;
        L->back->next = NULL;
        
    }
    L->length--;
    freeNode(&oldBack);
}

void delete(List L) {
    if (L->length <= 0) {
        fprintf(stderr, "List Error: calling delete(List L) on an empty List");
        exit(EXIT_FAILURE);
    }
    else if (L->index < 0) {
        fprintf(stderr, "List Error: calling delete(List L) on a list with an undefined cursor.");
        exit(EXIT_FAILURE);
    }
    Node oldCursor = L->cursor;
    if (L->front == oldCursor) {
        L->front = oldCursor->next;
    }
    else if (L->back == oldCursor) {
        L->back = oldCursor->prev;
    }
    else {
        L->cursor->prev->next = L->cursor->next;
        L->cursor->next->prev = L->cursor->prev;
    }
    
    L->cursor = NULL;


    L->index = -1;
    L->length--;
    freeNode(&oldCursor);
    
}

void printList (FILE* out, List L) {
   
   if(out==NULL ){
      fprintf(stderr, "Unable to open output file for writing");
      exit(1);
   }
   Node current = L->front;
   while (current != NULL) {
      fprintf(out, "%d", current->data);
      if (current != L->back) {
        fprintf(out, " ");
      }
      current = current->next;
    }

   // this is for printing to the file: fprintf(fileout, "print here")
}

List copyList(List L) {
    List nL = newList();
    Node current = L->front;
    while (current != NULL) {
        append(nL, current->data);
        current = current->next;
    }
    return(nL);
}
