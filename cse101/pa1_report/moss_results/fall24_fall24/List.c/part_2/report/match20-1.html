#include "List.h"

// structs --------------------------------------------------------------------

// private Node type


typedef struct NodeObj* Node;
// private ListObj type

typedef struct ListObj {
  Node cursor;
  Node front;
  Node back;
  int length;
  int index;
} ListObj;

// private NodeObj type
typedef struct NodeObj {
  Node next;
  Node previous;
  int data;
} NodeObj;

// Constructors-Destructors ---------------------------------------------------

// newNode()
// Returns reference to new Node object. Initializes next and data fields.
Node newNode(int data) {
  Node n = malloc(sizeof(struct NodeObj));
  assert(n != NULL);
  n->data = data;
  n->next = NULL;
  n->previous = NULL;
  return (n);
}

// freeNode()
// Frees heap memory pointed to by *pN, sets *pN to NULL.
void freeNode(Node *pN) {
  if (pN != NULL && *pN != NULL) {
    free(*pN);
    *pN = NULL;
  }
}

// newList()
// Creates and returns a new empty List.
List newList(void) {
  List L = malloc(sizeof(struct ListObj));
  assert(L != NULL);
  L->front = L->back = L->cursor = NULL;
  L->length = 0;
  L->index = -1;
  return (L);
}

// freeList()
// Frees all heap memory associated with *pL, and sets *pL to NULL.
void freeList(List *pL) {
  if (pL != NULL && *pL != NULL) {
    while (length(*pL) > 0) {
      deleteFront(*pL);
    }
    free(*pL);
    *pL = NULL;
  }
}

// Access functions -----------------------------------------------------------

// length()
// Returns the number of elements in L.
// Pre: L!=NULL
int length(List L) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling length on NULL List reference");
    exit(EXIT_FAILURE);
  }
  return (L->length);
}

// index()
// Returns index of cursor element if defined, -1 otherwise.
// Pre: L!=NULL
int index(List L) {
  if (L == NULL) {


    fprintf(stderr, "List Error: calling index on NULL List reference");
    exit(EXIT_FAILURE);

  }
  return (L->index);
}

// front()
// Returns front element of L.
// Pre: length()>0, L!=NULL
int front(List L) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling front on NULL List reference");
    exit(EXIT_FAILURE);

  }
  if (length(L) <= 0) {
    fprintf(stderr, "List Error: calling front on empty List");
    exit(EXIT_FAILURE);
  }

  return (L->front->data);
}

// back()
// Returns back element of L.
// Pre: length()>0, L!=NULL
int back(List L) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling back on NULL List reference");
    exit(EXIT_FAILURE);
  }
  if (length(L) <= 0) {
    fprintf(stderr, "List Error: calling back on empty List");
    exit(EXIT_FAILURE);
  }
  return (L->back->data);
}

// get()
// Returns cursor element of L.
// Pre: length()>0, index()>=0, L!=NULL
int get(List L) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling get on NULL List reference");
    exit(EXIT_FAILURE);
  }
  if (length(L) <= 0) {
    fprintf(stderr, "List Error: calling get on an empty List");
    exit(EXIT_FAILURE);
  }
  if (index(L) < 0) {


    fprintf(stderr, "List Error: calling get when cursor is not defined");
    exit(EXIT_FAILURE);
  }
  return (L->cursor->data);
}

// equals()
// Returns true iff Lists A and B contain the same sequence of elements, returns false otherwise.
// Pre: A!= NULL, B!=NULL
bool equals(List A, List B) {
  if (A == NULL) {
    fprintf(stderr, "List Error: calling equals on NULL List reference");
    exit(EXIT_FAILURE);
  }
  if (B == NULL) {
    fprintf(stderr, "List Error: calling equals on NULL List reference");
    exit(EXIT_FAILURE);
  }
  if (length(A) != length(B)) {
    return false;
  }
  Node currNodeA = A->front;
  Node currNodeB = B->front;
  for (int i = 0; i < length(A); i++) {
    if (currNodeA->data != currNodeB->data) {
      return false;
    }
    currNodeA = currNodeA->next;
    currNodeB = currNodeB->next;
  }
  return true;
}

// Manipulation procedures ----------------------------------------------------

// clear()
// Resets L to its original empty state.
// Pre: L!=NULL
void clear(List L) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling clear on NULL List reference");
    exit(EXIT_FAILURE);
  }
  while (length(L) > 0) {
    deleteFront(L);
  }
}

// set()
// Overwrites the cursor elementâ€™s data with x.
// Pre: length()>0, index()>=0, L!=NULL
void set(List L, int x) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling set on NULL List reference");
    exit(EXIT_FAILURE);
  }
  if (length(L) <= 0) {
    fprintf(stderr, "List Error: calling set on an empty List");
    exit(EXIT_FAILURE);
  }
  if (index(L) < 0) {
    fprintf(stderr, "List Error: calling set when cursor is not defined");
    exit(EXIT_FAILURE);
  }
  L->cursor->data = x;
}

// moveFront()
//If L is non-empty, sets cursor under the front element, otherwise does nothing.
// Pre: L!=NULL
void moveFront(List L) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling moveFront on NULL List reference");
    exit(EXIT_FAILURE);
  }
  if (length(L) != 0) {
    L->cursor = L->front;
    L->index = 0;
  }
}

// moveBack()
// If L is non-empty, sets cursor under the back element, otherwise does nothing.
// Pre: L!=NULL
void moveBack(List L) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling moveBack on NULL List reference");
    exit(EXIT_FAILURE);
  }
  if (length(L) != 0) {
    L->cursor = L->back;
    L->index = length(L) - 1;
  }
}

// movePrev()
// If cursor is defined and not at front, move cursor one step toward the front of L; if cursor is defined and at front, cursor becomes undefined; if cursor is undefined do nothing
// Pre: L!=NULL
void movePrev(List L) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling movePrev on NULL List reference");
    exit(EXIT_FAILURE);
  }
  if (L->cursor == NULL) {
    return;
  }
  if (L->front == L->cursor) {
    L->cursor = NULL;
    L->index = -1;
    return;
  }
  L->cursor = L->cursor->previous;
  L->index -= 1;
}

// moveNext()
// If cursor is defined and not at back, move cursor one step toward the back of L; if cursor is defined and at back, cursor becomes undefined; if cursor is undefined do nothing
// Pre: L!=NULL
void moveNext(List L) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling moveNext on NULL List reference");
    exit(EXIT_FAILURE);
  }
  if (L->cursor == NULL) {
    return;
  }
  if (L->back == L->cursor) {
    L->cursor = NULL;
    L->index = -1;
    return;
  }
  L->cursor = L->cursor->next;
  L->index += 1;
}

// prepend()
// Insert new element into L. If L is non-empty, insertion takes place before front element.
// Pre: L!=NULL
void prepend(List L, int x) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling prepend on NULL List reference");
    exit(EXIT_FAILURE);
  }
  Node n = newNode(x);
  n->next = L->front;
  n->previous = NULL;
  if (length(L) == 0) {
    L->front = n;
    L->back = n;
    L->length = 1;
    return;
  }
  L->front->previous = n;
  L->front = n;
  L->length += 1;
  if (index(L) >= 0) {
    L->index += 1;
  }
}

// append()
// Insert new element into L. If L is non-empty,insertion takes place after back element.
// Pre: L!=NULL
void append(List L, int x) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling append on NULL List reference");
    exit(EXIT_FAILURE);
  }
  Node n = newNode(x);
  n->next = NULL;
  if (length(L) == 0) {
    n->previous = NULL;
    L->front = n;
    L->back = n;
    L->length = 1;
    return;
  }
  L->back->next = n;
  n->previous = L->back;
  L->back = n;
  L->length += 1;
}

// insertBefore()
// Insert new element before cursor.
// Pre: length()>0, index()>=0, L!=NULL
void insertBefore(List L, int x) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling insertBefore on NULL List reference");
    exit(EXIT_FAILURE);
  }
  if (length(L) <= 0) {
    fprintf(stderr, "List Error: calling insertBefore on an empty List");
    exit(EXIT_FAILURE);
  }
  if (index(L) < 0) {
    fprintf(stderr,
            "List Error: calling insertBefore when cursor is not defined");
    exit(EXIT_FAILURE);
  }
  if (L->front == L->cursor) {
    prepend(L, x);
    return;
  }
  Node n = newNode(x);
  n->next = L->cursor;
  n->previous = L->cursor->previous;
  L->cursor->previous->next = n;
  L->cursor->previous = n;
  L->index += 1;


  L->length += 1;
}

// insertAfter()
// Insert new element after cursor
// Pre: length()>0, index()>=0, L!=NULL
void insertAfter(List L, int x) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling insertAfter on NULL List reference");
    exit(EXIT_FAILURE);
  }
  if (length(L) <= 0) {
    fprintf(stderr, "List Error: calling insertAfter on an empty List");
    exit(EXIT_FAILURE);
  }
  if (index(L) < 0) {
    fprintf(stderr, "List Error: calling insertAfter when cursor is not defined");
    exit(EXIT_FAILURE);
  }
  if (L->back == L->cursor) {
    append(L, x);
    return;
  }
  Node n = newNode(x);
  n->next = L->cursor->next;
  n->previous = L->cursor;
  Node temp = L->cursor->next;
  L->cursor->next = n;
  temp->previous = n;
  L->length += 1;
}

// deleteFront()
// Delete the front element.
// Pre: length()>0, L!=NULL
void deleteFront(List L) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling deleteFront on NULL List reference");
    exit(EXIT_FAILURE);
  }
  if (length(L) <= 0) {
    fprintf(stderr, "List Error: calling deleteFront on empty List");
    exit(EXIT_FAILURE);
  }
  Node temp = L->front;
  if (temp == L->cursor || length(L) == 1) {
    L->cursor = NULL;
    L->index = -1;
  } else {
    L->index -= 1;
  }
  L->front = L->front->next;
  freeNode(&temp);
  L->length -= 1;
}

// deleteBack()
// Delete the back element.
// Pre: length()>0, L!=NULL
void deleteBack(List L) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling deleteBack on NULL List reference");
    exit(EXIT_FAILURE);
  }
  if (length(L) <= 0) {
    fprintf(stderr, "List Error: calling deleteBack on empty List");
    exit(EXIT_FAILURE);
  }
  Node temp = L->back;
  if (temp == L->cursor) {
    L->cursor = NULL;
    L->index = -1;
  }
  L->back = L->back->previous;
  freeNode(&temp);
  L->length -= 1;
}

// delete()
// Delete cursor element, making cursor undefined
// Pre: length()>0, index()>=0, L!=NULL
void delete(List L) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling delete on NULL List reference");
    exit(EXIT_FAILURE);
  }
  if (length(L) <= 0) {
    fprintf(stderr, "List Error: calling delete on empty List");
    exit(EXIT_FAILURE);
  }
  if (index(L) < 0) {
    fprintf(stderr, "List Error: calling delete when cursor is not defined");
    exit(EXIT_FAILURE);
  }
  Node temp = L->cursor;
  if (L->cursor == L->front) {
    deleteFront(L);
  } else if (L->cursor == L->back) {
    deleteBack(L);
  } else {
    temp->previous->next = temp->next;
    temp->next->previous = temp->previous;
    L->cursor = NULL;
    L->length -= 1;
    freeNode(&temp);
  }
  L->index = -1;
  L->cursor = NULL;
}

// Other operations -----------------------------------------------------------

// printList()
// Prints to the file pointed to by out, a string representation of L consisting of a space separated sequence of integers, with front on left.
// Pre: L!=NULL
void printList(FILE *out, List L) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling printList on NULL List reference");
    exit(EXIT_FAILURE);
  }
  Node curr = L->front;
  for (int i = 0; i < length(L); i++) {
    fprintf(out, "%d ", curr->data);
    curr = curr->next;
  }
}

// copyList()
// Returns a new List representing the same integer sequence as L. The cursor in the new list is undefined, regardless of the state of the cursor in L. The state of L is unchanged.
// Pre: L!=NULL
List copyList(List L) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling copyList on NULL List reference");
    exit(EXIT_FAILURE);
  }
  List copy = newList();
  Node curr = L->front;
  for (int i = 0; i < length(L); i++) {
    append(copy, curr->data);
    curr = curr->next;
  }
  return (copy);
}

// concatList()
// Returns a new List which is the concatenation of A and B. The cursor in the new List is undefined, regardless of the states of the cursors in A and B. The states of A and B are unchanged.
// Pre: A!=NULL, B!=NULL
List concatList(List A, List B) {
  if (A == NULL) {
    fprintf(stderr, "List Error: calling concatList on NULL List reference");
    exit(EXIT_FAILURE);
  }
  if (B == NULL) {
    fprintf(stderr, "List Error: calling concatList on NULL List reference");
    exit(EXIT_FAILURE);
  }
  List cat = newList();
  Node curr = A->front;
  for (int i = 0; i < length(A); i++) {
    append(cat, curr->data);
    curr = curr->next;
  }
  curr = B->front;
  for (int i = 0; i < length(B); i++) {
    append(cat, curr->data);
    curr = curr->next;
  }
  return cat;
}
