#include <stdio.h>
#include <stdlib.h>
#include "List.h"



typedef struct NodeObj {
    int data;
    struct NodeObj* next;
    struct NodeObj* prev;
} NodeObj;

typedef struct ListObj {
    NodeObj* front;
    NodeObj* back;
    NodeObj* cursor;
    int length;
    int index;
} ListObj;

List newList(void) {
    List L = malloc(sizeof(ListObj));
    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->length = 0;
    L->index = -1;
    return L;
}

void freeList(List* pL) {
    if (pL != NULL && *pL != NULL) {
        while ((*pL)->length > 0) {
            deleteFront(*pL);
        }
        free(*pL);
        *pL = NULL;
    }
}

int length(List L) {
    return L->length;
}

int index(List L) {
    return L->index;
}

int front(List L) {
    if (L->length > 0) {
        return L->front->data;
    }
    fprintf(stderr, "List Error: calling front() on empty list\n");
    exit(EXIT_FAILURE);
}

int back(List L) {
    if (L->length > 0) {
        return L->back->data;
    }
    fprintf(stderr, "List Error: calling back() on empty list\n");
    exit(EXIT_FAILURE);
}

int get(List L) {
    if (L->length > 0 && L->index >= 0) {
        return L->cursor->data;
    }
    fprintf(stderr, "List Error: calling get() on empty list or undefined cursor\n");
    exit(EXIT_FAILURE);
}



bool equals(List A, List B) {
    if (A->length != B->length) {
        return false;
    }

    NodeObj* N = A->front;
    NodeObj* M = B->front;
    while (N != NULL && M != NULL) {
        if (N->data != M->data) {
            return false;
        }
        N = N->next;
        M = M->next;
    }

    return true;
}

void clear(List L) {
    while (L->length > 0) {
        deleteFront(L);
    }
    L->front = L->back = L->cursor = NULL;
    L->length = 0;
    L->index = -1;
}

void set(List L, int x) {
    if (L->length > 0 && L->index >= 0) {
        L->cursor->data = x;
    } else {
        fprintf(stderr, "List Error: calling set() on undefined cursor or empty list\n");
        exit(EXIT_FAILURE);
    }
}

void moveFront(List L) {
    if (L->length > 0) {
        L->cursor = L->front;
        L->index = 0;
    }
}

void moveBack(List L) {
    if (L->length > 0) {
        L->cursor = L->back;
        L->index = L->length - 1;
    }
}

void movePrev(List L) {
    if (L->cursor != NULL && L->index > 0) {
        L->cursor = L->cursor->prev;
        L->index--;
    } else if (L->cursor != NULL && L->index == 0) {
        L->cursor = NULL;
        L->index = -1;
    }
}

void moveNext(List L) {
    if (L->cursor != NULL && L->index < L->length - 1) {
        L->cursor = L->cursor->next;
        L->index++;
    } else if (L->cursor != NULL && L->index == L->length - 1) {


        L->cursor = NULL;
        L->index = -1;
    }
}

void prepend(List L, int x) {
    NodeObj* N = malloc(sizeof(NodeObj));
    N->data = x;
    N->next = L->front;
    N->prev = NULL;

    if (L->length == 0) {
        L->front = L->back = N;
    } else {
        L->front->prev = N;
        L->front = N;
    }

    if (L->index >= 0) {
        L->index++;
    }


    L->length++;
}

void append(List L, int x) {
    NodeObj* N = malloc(sizeof(NodeObj));
    N->data = x;
    N->next = NULL;
    N->prev = L->back;

    if (L->length == 0) {
        L->front = L->back = N;
    } else {
        L->back->next = N;


        L->back = N;
    }
    L->length++;
}

void insertBefore(List L, int x) {
    if (L->length > 0 && L->index >= 0) {
        if (L->cursor == L->front) {
            prepend(L, x);
        } else {
            NodeObj* N = malloc(sizeof(NodeObj));
            N->data = x;
            N->next = L->cursor;
            N->prev = L->cursor->prev;
            L->cursor->prev->next = N;
            L->cursor->prev = N;
            L->index++;
            L->length++;
        }
    } else {
        fprintf(stderr, "List Error: calling insertBefore() on undefined cursor or empty list\n");
        exit(EXIT_FAILURE);
    }
}

void insertAfter(List L, int x) {


    if (L->length > 0 && L->index >= 0) {
        if (L->cursor == L->back) {
            append(L, x);
        } else {
            NodeObj* N = malloc(sizeof(NodeObj));
            N->data = x;
            N->prev = L->cursor;
            N->next = L->cursor->next;
            L->cursor->next->prev = N;
            L->cursor->next = N;
            L->length++;
        }
    } else {
        fprintf(stderr, "List Error: calling insertAfter() on undefined cursor or empty list\n");
        exit(EXIT_FAILURE);
    }
}

void deleteFront(List L) {
    if (L->length > 0) {
        NodeObj* N = L->front;
        if (L->length == 1) {
            L->front = L->back = NULL;
        } else {
            L->front = L->front->next;
            L->front->prev = NULL;
        }

        if (L->index == 0) {
            L->cursor = NULL;
            L->index = -1;
        } else if (L->index > 0) {
            L->index--;
        }

        L->length--;
        free(N);
    } else {
        fprintf(stderr, "List Error: calling deleteFront() on an empty list\n");
        exit(EXIT_FAILURE);
    }
}

void deleteBack(List L) {
    if (L->length > 0) {
        NodeObj* N = L->back;
        if (L->length == 1) {
            L->front = L->back = NULL;
        } else {
            L->back = L->back->prev;
            L->back->next = NULL;
        }

        if (L->index == L->length - 1) {
            L->cursor = NULL;
            L->index = -1;
        }

        L->length--;
        free(N);
    } else {
        fprintf(stderr, "List Error: calling deleteBack() on an empty list\n");
        exit(EXIT_FAILURE);
    }
}



void delete(List L) {
    if (L->length > 0 && L->index >= 0) {
        if (L->cursor == L->front) {
            deleteFront(L);
        } else if (L->cursor == L->back) {
            deleteBack(L);
        } else {
            NodeObj* N = L->cursor;
            N->prev->next = N->next;
            N->next->prev = N->prev;
            L->cursor = NULL;
            L->index = -1;
            L->length--;
            free(N);
        }
    } else {
        fprintf(stderr, "List Error: calling delete() on undefined cursor or empty list\n");
        exit(EXIT_FAILURE);
    }
}

void printList(FILE* out, List L) {
    NodeObj* N = L->front;
    while (N != NULL) {
        fprintf(out, "%d ", N->data);
        N = N->next;
    }
}



List copyList(List L) {
    List copy = newList();
    NodeObj* N = L->front;
    while (N != NULL) {
        append(copy, N->data);
        N = N->next;
    }
    return copy;
}
