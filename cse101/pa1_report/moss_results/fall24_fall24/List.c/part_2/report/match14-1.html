#include "List.h"
#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>
#include <assert.h>

// Structs --------------------------------------------------------------------

// private Node type


typedef struct NodeObj *Node;
// private NodeObj type
typedef struct NodeObj {
    int data;
    Node prev;
    Node next;
} NodeObj;
// private ListObj type
typedef struct ListObj {
    Node front;
    Node back;
    int cursor;
    int length;
} ListObj;


// Constructors-Destructors ---------------------------------------------------

// Returns reference to new Node object with empty linkage fields.
Node newNode(int data) {
    Node N = malloc(sizeof(NodeObj));
    assert(N != NULL);

    N->data = data;
    N->prev = NULL;
    N->next = NULL;

    return (N);
}

// Frees heap memory pointed to by *pN, sets *pN to NULL.
void freeNode(Node *pN) {
    if (pN != NULL && *pN != NULL) {
        free(*pN);
        *pN = NULL;
    }
}

// Returns reference to new empty list object.
List newList() {
    List L;


    L = malloc(sizeof(ListObj));
    assert(L != NULL);

    L->front = L->back = NULL;
    L->length = 0;
    L->cursor = -1;

    return (L);
}

// Frees all heap memory associated with List *pL, and sets *pL to NULL.
void freeList(List *pL) {
    if (pL != NULL && *pL != NULL) {
        // check if the list content is empty. If not empty, clear the list first
        if (length(*pL) > 0) {
            clear(*pL);
        }

        free(*pL);
        *pL = NULL;
    }
}

// Access functions -----------------------------------------------------------

int length(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling length() on NULL List reference\n");
        exit(1);
    }
    return (L->length);
}

int index(List L) {
    if (L == NULL) {
        return -1;
    }
    return (L->cursor < 0 || L->cursor > length(L) - 1) ? -1 : L->cursor;
}

int front(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling front() on NULL List reference\n");
        exit(1);
    }
    if (length(L) == 0) {
        fprintf(stderr, "List Error: calling front() on empty List\n");
        exit(1);
    }
    return L->front->data;
}

int back(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling back() on NULL List reference\n");
        exit(1);
    }
    if (length(L) == 0) {
        fprintf(stderr, "List Error: calling back() on empty List\n");
        exit(1);
    }
    return L->back->data;
}

int get(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling get() on NULL List reference\n");
        exit(1);
    }
    // Pre: length()>0, index()>=0
    if (length(L) == 0) {
        fprintf(stderr, "List Error: calling get() on empty List\n");
        exit(1);
    }
    if (index(L) < 0) {
        fprintf(stderr, "List Error: calling get() on undefined cursor\n");
        exit(1);
    }

    // iterate to the cursor
    Node N = L->front;
    for (int i = 0; i < L->cursor; i++) {
        N = N->next;
        if (N == NULL) { // corrupted list
            fprintf(stderr, "List Error: list is corrupted\n");
            exit(1);
        }
    }
    return N->data;
}

bool equals(List A, List B) {
    // check if A and B are defined
    if (A == NULL || B == NULL) {
        fprintf(stderr, "List Error: calling equals() on NULL List reference\n");
        exit(1);
    }

    // check if length is same
    if (length(A) != length(B)) return false;
    // check individual elements
    Node N = A->front;
    Node M = B->front;
    while (N != NULL && M != NULL) { // TODO: might need to change to || instead of &&
        if (N->data != M->data) return false;


        N = N->next;
        M = M->next;
    }
    return true;
}

// Manipulation procedures ----------------------------------------------------

void clear(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling clear() on NULL List reference\n");
        exit(1);
    }
    // iterate through the list and free all nodes
    while (L->front != NULL) {
        Node N = L->front;
        L->front = L->front->next;
        freeNode(&N);
    }
    // reset all fields
    L->front = L->back = NULL;
    L->cursor = -1;
    L->length = 0;
}

void set(List L, int x) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling set() on NULL List reference\n");
        exit(1);
    }
    // Pre: length()>0, index()>=0
    if (length(L) == 0) {
        fprintf(stderr, "List Error: calling set() on empty List\n");
        exit(1);
    }
    if (index(L) < 0) {
        fprintf(stderr, "List Error: calling set() on undefined cursor\n");
        exit(1);
    }

    // iterate to the cursor
    Node N = L->front;
    for (int i = 0; i < L->cursor; i++) {
        N = N->next;
        if (N == NULL) { // corrupted list
            fprintf(stderr, "List Error: list is corrupted\n");
            exit(1);
        }
    }
    N->data = x;
}

void moveFront(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling moveFront() on NULL List reference\n");
        exit(1);
    }
    if (length(L) > 0) {
        L->cursor = 0;
    }
}

void moveBack(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling moveBack() on NULL List reference\n");
        exit(1);
    }
    if (length(L) > 0) {
        L->cursor = L->length - 1;
    }
}

void movePrev(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling movePrev() on NULL List reference\n");
        exit(1);
    }

    if (L->cursor >= 0) { // do nothing if cursor is undefined
        L->cursor--; // move cursor one step toward the front of L. If cursor is defined and at front, cursor becomes undefined (-1)
    }
}

void moveNext(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling moveNext() on NULL List reference\n");
        exit(1);
    }

    if (L->cursor >= 0) { // do nothing if cursor is undefined
        L->cursor++; // move cursor one step toward the back of L. If cursor is defined and at back, cursor becomes undefined (-1)
    }
}

void prepend(List L, int x) { // Changes the list length
    if (L == NULL) {
        fprintf(stderr, "List Error: calling prepend() on NULL List reference\n");
        exit(1);
    }

    Node N = newNode(x);
    if (length(L) == 0) {
        L->front = L->back = N;
    } else {
        N->next = L->front;
        L->front->prev = N;


        L->front = N;
    }

    L->length++;
}

void append(List L, int x) { // Changes the list length
    if (L == NULL) {
        fprintf(stderr, "List Error: calling append() on NULL List reference\n");
        exit(1);
    }

    Node N = newNode(x);
    if (length(L) == 0) {
        L->front = L->back = N;
    } else {
        N->prev = L->back;
        L->back->next = N;


        L->back = N;
    }

    L->length++;
}

void insertBefore(List L, int x) { // Changes the list length
    if (L == NULL) {
        fprintf(stderr, "List Error: calling insertBefore() on NULL List reference\n");
        exit(1);
    }
    // Pre: length()>0, index()>=0
    if (length(L) == 0) {
        fprintf(stderr, "List Error: calling insertBefore() on empty List\n");
        exit(1);
    }
    if (index(L) < 0) {
        fprintf(stderr, "List Error: calling insertBefore() on undefined cursor\n");
        exit(1);
    }

    if (L->cursor == 0) { // if cursor is at the beginning, prepend
        prepend(L, x);
    } else { // insert before cursor
        Node N = newNode(x);
        Node C = L->front;
        for (int i = 0; i < L->cursor; i++) {
            C = C->next;
            if (C == NULL) { // corrupted list
                fprintf(stderr, "List Error: list is corrupted\n");
                exit(1);
            }
        }
        N->prev = C->prev;
        N->next = C;
        C->prev->next = N;
        C->prev = N;
        L->length++;
    }

    // shift the cursor forward by 1
    L->cursor++;
}

void insertAfter(List L, int x) { // Changes the list length
    if (L == NULL) {
        fprintf(stderr, "List Error: calling insertAfter() on NULL List reference\n");
        exit(1);
    }
    // Pre: length()>0, index()>=0
    if (length(L) == 0) {
        fprintf(stderr, "List Error: calling insertAfter() on empty List\n");
        exit(1);
    }
    if (index(L) < 0) {
        fprintf(stderr, "List Error: calling insertAfter() on undefined cursor\n");
        exit(1);
    }

    if (L->cursor == length(L) - 1) { // if cursor is at the end, append
        append(L, x);
    } else { // insert after cursor
        Node N = newNode(x);
        Node C = L->front;
        for (int i = 0; i < L->cursor; i++) {
            C = C->next;
            if (C == NULL) { // corrupted list
                fprintf(stderr, "List Error: list is corrupted\n");
                exit(1);
            }
        }
        N->prev = C;
        N->next = C->next;
        C->next->prev = N;
        C->next = N;
        L->length++;
    }
}

void deleteFront(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling deleteFront() on NULL List reference\n");
        exit(1);
    }
    // Pre: length() > 0
    if (length(L) == 0) {
        fprintf(stderr, "List Error: calling deleteFront() on empty List\n");
        exit(1);
    }

    Node N = L->front;
    if (length(L) == 1) { // if this is the only node
        L->front = L->back = NULL;
        L->cursor = -1;
    } else {
        L->front = L->front->next;
        L->front->prev = NULL; // switched to the new front

        if (L->cursor == 0) { // if cursor is at the front
            L->cursor = -1;
        } else {
            L->cursor--;
        }
    }
    freeNode(&N);
    L->length--;
}

void deleteBack(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling deleteBack() on NULL List reference\n");
        exit(1);
    }
    // Pre: length() > 0
    if (length(L) == 0) {
        fprintf(stderr, "List Error: calling deleteBack() on empty List\n");
        exit(1);
    }

    Node N = L->back;
    if (length(L) == 1) { // if this is the only node
        L->front = L->back = NULL;
        L->cursor = -1;
    } else {
        L->back = L->back->prev;
        L->back->next = NULL; // switched to the new back

        if (L->cursor == length(L) - 1) { // if cursor is at the back
            L->cursor = -1;
        }
    }
    freeNode(&N);
    L->length--;
}

void delete(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling delete() on NULL List reference\n");
        exit(1);
    }
    // Pre: length()>0, index()>=0
    if (length(L) == 0) {
        fprintf(stderr, "List Error: calling delete() on empty List\n");
        exit(1);
    }
    if (index(L) < 0) {
        fprintf(stderr, "List Error: calling delete() on undefined cursor\n");
        exit(1);
    }

    // delete element at the cursor
    Node N = L->front;
    for (int i = 0; i < L->cursor; i++) {
        N = N->next;
        if (N == NULL) { // corrupted list
            fprintf(stderr, "List Error: list is corrupted\n");
            exit(1);
        }
    }
    if (N == L->front) { // edge cases
        deleteFront(L);
    } else if (N == L->back) {
        deleteBack(L);
    } else {
        N->prev->next = N->next;
        N->next->prev = N->prev;
        freeNode(&N);
        L->length--;
    }

    // cursor becomes undefined
    L->cursor = -1;
}

// Other operations -----------------------------------------------------------

void printList(FILE *out, List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling printList() on NULL List reference\n");
        exit(1);
    }

    Node N = L->front;
    while (N != NULL) {
        fprintf(out, "%d ", N->data);
        N = N->next;
    }
}

List copyList(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling copyList() on NULL List reference\n");
        exit(1);
    }

    List C = newList();
    Node N = L->front;
    while (N != NULL) {
        append(C, N->data);
        N = N->next;
    }
    C->cursor = -1; // cursor is undefined regardless of the state of the cursor in L
    return C;
}
