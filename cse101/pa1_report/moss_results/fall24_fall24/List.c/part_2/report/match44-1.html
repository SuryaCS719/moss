#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>
#include "List.h"

typedef struct NodeObj *Node;



typedef struct NodeObj {
    int data;
    Node next;
    Node prev;
} NodeObj;

typedef struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
    int index;
} ListObj;

void* safeMalloc(size_t size) {
    void *ptr = malloc(size);
    if (!ptr) {
        fprintf(stderr, "Memory allocation failed\n");
        exit(EXIT_FAILURE);
    }
    return ptr;
}

Node newNode(int data) {
    Node N = safeMalloc(sizeof(NodeObj));
    N->data = data;
    N->next = NULL;
    N->prev = NULL;
    return N;
}

void freeNode(Node *pN) {
    if (pN && *pN) {
        free(*pN);
        *pN = NULL;
    }
}

List newList() {
    List L = safeMalloc(sizeof(ListObj));
    L->front = L->back = L->cursor = NULL;
    L->length = 0;
    L->index = -1;
    return L;
}

void freeList(List *pL) {
    if (pL && *pL) {
        while ((*pL)->length != 0) {
            deleteFront(*pL);
        }
        free(*pL);
        *pL = NULL;
    }
}

int length(List L) {
    return L ? L->length : -1;
}

int index(List L) {
    return L ? L->index : -1;
}

int front(List L) {
    if (!L || L->length == 0) exit(EXIT_FAILURE);
    return L->front->data;
}

int back(List L) {
    if (!L || L->length == 0) exit(EXIT_FAILURE);
    return L->back->data;
}

int get(List L) {
    if (!L || L->length == 0 || L->index < 0) exit(EXIT_FAILURE);
    return L->cursor->data;
}

bool equals(List A, List B) {
    if (!A || !B || A->length != B->length) return false;

    for (Node N = A->front, M = B->front; N && M; N = N->next, M = M->next) {
        if (N->data != M->data) return false;
    }
    return true;
}

void clear(List L) {
    if (!L) exit(EXIT_FAILURE);
    Node current = L->front;
    while (current) {
        Node nextNode = current->next;
        free(current);
        current = nextNode;
    }
    L->front = L->back = L->cursor = NULL;
    L->length = 0;
    L->index = -1;
}

void set(List L, int x) {
    if (!L || L->length == 0 || L->index < 0) exit(EXIT_FAILURE);
    L->cursor->data = x;
}

void moveFront(List L) {
    if (L && L->length != 0) {
        L->cursor = L->front;
        L->index = 0;
    }
}

void moveBack(List L) {
    if (L && L->length != 0) {
        L->cursor = L->back;
        L->index = L->length - 1;
    }
}

void movePrev(List L) {
    if (L && L->cursor) {
        L->cursor = L->cursor->prev;
        L->index = (L->cursor == NULL) ? -1 : L->index - 1;
    }
}

void moveNext(List L) {
    if (L && L->cursor) {
        L->cursor = L->cursor->next;
        L->index = (L->cursor == NULL) ? -1 : L->index + 1;
    }
}

void prepend(List L, int x) {
    if (!L) exit(EXIT_FAILURE);
    Node N = newNode(x);
    if (L->length == 0) {
        L->front = L->back = N;
    } else {
        N->next = L->front;
        L->front->prev = N;
        L->front = N;
    }
    if (L->index != -1) L->index++;
    L->length++;
}

void append(List L, int x) {
    if (!L) exit(EXIT_FAILURE);
    Node N = newNode(x);
    if (L->length == 0) {
        L->front = L->back = N;
    } else {
        N->prev = L->back;
        L->back->next = N;
        L->back = N;
    }
    L->length++;
}

void insertBefore(List L, int x) {
    if (!L || L->index < 0) exit(EXIT_FAILURE);
    if (L->cursor == L->front) {
        prepend(L, x);
    } else {
        Node N = newNode(x);
        Node cursorPrev = L->cursor->prev;
        N->next = L->cursor;
        N->prev = cursorPrev;
        cursorPrev->next = N;
        L->cursor->prev = N;
        L->index++;
        L->length++;
    }
}

void insertAfter(List L, int x) {
    if (!L || L->index < 0) exit(EXIT_FAILURE);
    if (L->cursor == L->back) {
        append(L, x);
    } else {
        Node N = newNode(x);
        Node cursorNext = L->cursor->next;
        N->prev = L->cursor;
        N->next = cursorNext;
        L->cursor->next = N;
        if (cursorNext) {
            cursorNext->prev = N;
        }
        L->length++;
    }
}

void deleteFront(List L) {
    if (!L || L->length == 0) exit(EXIT_FAILURE);
    Node oldFront = L->front;
    L->front = L->front->next;
    if (L->front) {
        L->front->prev = NULL;
    } else {
        L->back = NULL;  // List is now empty
    }
    L->length--;
    if (L->index == 0) {
        L->cursor = NULL;
        L->index = -1;
    } else if (L->index != -1) {
        L->index--;
    }
    freeNode(&oldFront);
}

void deleteBack(List L) {
    if (!L || L->length == 0) exit(EXIT_FAILURE);
    Node oldBack = L->back;
    L->back = L->back->prev;
    if (L->back) {
        L->back->next = NULL;
    } else {
        L->front = NULL;  // List is now empty
    }
    L->length--;
    if (L->index == L->length) {
        L->cursor = NULL;
        L->index = -1;
    }
    freeNode(&oldBack);
}

void delete(List L) {
    if (!L || L->length == 0 || L->index < 0) exit(EXIT_FAILURE);
    Node cursorNode = L->cursor;
    if (cursorNode == L->front) {
        deleteFront(L);
    } else if (cursorNode == L->back) {
        deleteBack(L);
    } else {
        Node prevNode = cursorNode->prev;
        Node nextNode = cursorNode->next;
        prevNode->next = nextNode;
        nextNode->prev = prevNode;
        L->cursor = NULL;
        L->index = -1;
        L->length--;
        freeNode(&cursorNode);
    }
}

void printList(FILE *out, List L) {
    if (!L) exit(EXIT_FAILURE);
    for (Node N = L->front; N != NULL; N = N->next) {
        fprintf(out, "%d ", N->data);
    }
}

List copyList(List L) {
    if (!L) exit(EXIT_FAILURE);
    List duplicate = newList();
    for (Node N = L->front; N != NULL; N = N->next) {
        append(duplicate, N->data);
    }
    return duplicate;
}
