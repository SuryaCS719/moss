#include "List.h"
#include <stdio.h>
#include <stdlib.h>
#include <assert.h>
#include <stdbool.h>

typedef struct NodeObj* Node;


typedef struct NodeObj {
    ListElement data;
    Node next;
    Node prev;
} NodeObj;

typedef struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
    int cursorIndex;
} ListObj;

void nullChecker(List L, const char* funcName) {
    if (L == NULL) {
        printf("List Error: calling %s on NULL List reference\n", funcName);
        exit(EXIT_FAILURE);
    }
}

Node newNode(int data) {
    Node N = malloc(sizeof(NodeObj));
    assert(N != NULL);
    N->prev = N->next = NULL;
    N->data = data;
    return N;
}

//--Constructors-Deconstructors-------------
List newList(void) {
    List L = malloc(sizeof(ListObj));
    assert(L != NULL);
    L->front = L->back = L->cursor = NULL;
    L->length = 0;
    L->cursorIndex = -1;
    return L;
}

void freeList(List* pL) {


    if (pL != NULL && *pL != NULL) {
        clear(*pL);
        free(*pL);
        *pL = NULL;
    }
}

//--Access Functions----------------------------
int length(List L) {
    nullChecker(L, "length()");
    return L->length;
}

int index(List L) {
    nullChecker(L, "index()");
    return L->cursorIndex;
}

int front(List L) {
    nullChecker(L, "front()");
    if (L->front == NULL) {
        fprintf(stderr, "List Error: front() called on empty List\n");
        exit(EXIT_FAILURE);
    }
    return L->front->data;
}

int back(List L) {
    nullChecker(L, "back()");
    if (L->back == NULL) {
        fprintf(stderr, "List Error: back() called on empty List\n");
        exit(EXIT_FAILURE);
    }
    return L->back->data;
}

int get(List L) {
    nullChecker(L, "get()");
    if (L->cursor == NULL) {
        fprintf(stderr, "List Error: get() called with undefined cursor\n");
        exit(EXIT_FAILURE);
    }
    return L->cursor->data;
}

bool equals(List A, List B) {
    nullChecker(A, "equals()");
    nullChecker(B, "equals()");
    if (length(A) != length(B)) {
        return false;
    }
    Node tempA = A->front;
    Node tempB = B->front;
    while (tempA != NULL && tempB != NULL) {


        if (tempA->data != tempB->data) {
            return false;
        }
        tempA = tempA->next;
        tempB = tempB->next;
    }
    return true;
}

//--Manipulation procedures----------------------
void clear(List L) {
    nullChecker(L, "clear()");
    while (L->length > 0) {
        deleteFront(L);
    }
    L->cursorIndex = -1;
    L->cursor = NULL;
}

void set(List L, int x) {
    nullChecker(L, "set()");
    if (L->cursor == NULL) {
        fprintf(stderr, "List Error: set() called with undefined cursor\n");
        exit(EXIT_FAILURE);
    }
    L->cursor->data = x;
}

void moveFront(List L) {
    nullChecker(L, "moveFront()");
    if (L->length > 0) {
        L->cursor = L->front;
        L->cursorIndex = 0;
    }
}

void moveBack(List L) {
    nullChecker(L, "moveBack()");
    if (L->length > 0) {
        L->cursor = L->back;
        L->cursorIndex = L->length - 1;
    }
}

void movePrev(List L) {
    nullChecker(L, "movePrev()");
    if (L->cursor != NULL) {
        if (L->cursor == L->front) {
            L->cursor = NULL;
            L->cursorIndex = -1;
        } else {
            L->cursor = L->cursor->prev;
            L->cursorIndex--;
        }
    }
}

void moveNext(List L) {
    nullChecker(L, "moveNext()");
    if (L->cursor != NULL) {
        if (L->cursor == L->back) {
            L->cursor = NULL;
            L->cursorIndex = -1;
        } else {
            L->cursor = L->cursor->next;
            L->cursorIndex++;
        }
    }
}

void prepend(List L, int x) {
    nullChecker(L, "prepend()");
    Node new = newNode(x);
    if (new == NULL) {
        fprintf(stderr, "Memory allocation failed\n");
        exit(EXIT_FAILURE);
    }

    if (L->front == NULL) {
        new->prev = NULL;
        new->next = NULL;
        L->front = L->back = new;
    } else {
        new->next = L->front;
        new->prev = NULL;
        L->front->prev = new;
        L->front = new;
    }
    L->length++;
    if (L->cursorIndex != -1) {
        L->cursorIndex++;
    }
}

void append(List L, int x) {
    nullChecker(L, "append()");
    Node new = newNode(x);
    if (L->back == NULL) {
        L->front = L->back = new;
    } else {
        new->prev = L->back;
        L->back->next = new;
        L->back = new;
    }
    L->length++;
}

void insertBefore(List L, int x) {
    nullChecker(L, "insertBefore()");
    if (L->length == 0 || L->cursor == NULL) {
        fprintf(stderr, "List Error: insertBefore() called with undefined cursor or empty list\n");
        exit(EXIT_FAILURE);
    }
    Node new = newNode(x);
    if (L->cursor == L->front) {
        new->next = L->front;
        L->front->prev = new;
        L->front = new;
    } else {
        new->prev = L->cursor->prev;
        new->next = L->cursor;
        L->cursor->prev->next = new;


        L->cursor->prev = new;
    }
    L->length++;
    L->cursorIndex++;
}

void insertAfter(List L, int x) {
    nullChecker(L, "insertAfter()");
    if (L->length == 0 || L->cursor == NULL) {
        fprintf(stderr, "List Error: insertAfter() called with undefined cursor or empty list\n");
        exit(EXIT_FAILURE);
    }
    Node new = newNode(x);
    if (L->cursor == L->back) {
        new->prev = L->back;
        L->back->next = new;
        L->back = new;
    } else {
        new->next = L->cursor->next;
        new->prev = L->cursor;


        L->cursor->next->prev = new;
        L->cursor->next = new;
    }
    L->length++;
}

void deleteFront(List L) {
    nullChecker(L, "deleteFront()");
    if (L->length == 0) {
        fprintf(stderr, "List Error: deleteFront() called on empty list\n");
        exit(EXIT_FAILURE);
    }
    Node temp = L->front;
    if (L->length == 1) {
        L->front = L->back = NULL;
    } else {
        L->front = L->front->next;


        L->front->prev = NULL;
    }
    if (L->cursor == temp) {
        L->cursor = NULL;
        L->cursorIndex = -1;
    } else if (L->cursorIndex > 0) {
        L->cursorIndex--;
    }
    free(temp);
    L->length--;
}

void deleteBack(List L) {
    nullChecker(L, "deleteBack()");
    if (L->length == 0) {
        fprintf(stderr, "List Error: deleteBack() called on empty list\n");
        exit(EXIT_FAILURE);
    }
    Node temp = L->back;
    if (L->length == 1) {
        L->front = L->back = NULL;
    } else {
        L->back = L->back->prev;
        L->back->next = NULL;
    }


    if (L->cursor == temp) {
        L->cursor = NULL;
        L->cursorIndex = -1;
    }
    free(temp);
    L->length--;
}

void delete(List L) {
    nullChecker(L, "delete()");
    if (L->cursor == NULL) {
        fprintf(stderr, "List Error: delete() called with undefined cursor\n");
        exit(EXIT_FAILURE);
    }
    if (L->cursor == L->front) {
        deleteFront(L);
    } else if (L->cursor == L->back) {
        deleteBack(L);
    } else {
        Node temp = L->cursor;
        temp->prev->next = temp->next;
        temp->next->prev = temp->prev;
        free(temp);


        L->length--;
    }
    L->cursor = NULL;
    L->cursorIndex = -1;
}

void printList(FILE* out, List L){
	nullChecker(L,"printList()");
	Node current = L->front;
	while(current!=NULL){
		fprintf(out, "%d", current->data);
		current = current->next;
	}
	fprintf(out, "\n");
}

List copyList(List L){
	nullChecker(L, "copyList()");
	List copied = newList();
	assert(copied!=NULL);
	Node current = L->front;
	while(current!=NULL){
		append(copied, current->data);
		current = current->next;
	}
	return copied;

}
