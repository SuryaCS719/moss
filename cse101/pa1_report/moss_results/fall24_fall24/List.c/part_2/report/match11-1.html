#include<stdio.h>
#include<stdlib.h>
#include<assert.h>
#include<stdbool.h>

// --- NODE DEFINITIONS AND METHODS



typedef struct NodeObj {
    int data;
    struct NodeObj* prev;
    struct NodeObj* next;
} NodeObj;
typedef NodeObj* Node;

Node newNode(int data) {
    Node N = malloc(sizeof(NodeObj));
    assert(N != NULL);
    N->data = data;
    N->prev = NULL;
    N->next = NULL;
    return N;
}

void freeNode(Node* pN) {
    if (pN != NULL && *pN != NULL) {
        free(*pN);
        *pN = NULL;
    }
}

// --- LIST DEFINITIONS AND METHODS

typedef struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int index;
    int length;
} ListObj;
typedef ListObj* List;

// Constructors/Destructors

List newList() {
    List L = malloc(sizeof(ListObj));
    assert(L != NULL);
    L->front = L->back = L->cursor = NULL;
    L->index = -1;
    L->length = 0;
    return L;
}

void freeList(List* pL) {
    if (pL != NULL && *pL != NULL) {
        Node active_node = (*pL)->front;
        while (active_node != NULL) {
            Node next_node = active_node->next;
            freeNode(&active_node);
            active_node = next_node;
        }

        free(*pL);
        *pL = NULL;
    }
}

// Access methods

int length(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: list is NULL");
        exit(EXIT_FAILURE);
    }
    return L->length;
}

int index(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: list is NULL");
        exit(EXIT_FAILURE);
    }
    return L->index;
}

int front(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: list is NULL");
        exit(EXIT_FAILURE);
    }
    if (L->length <= 0) {
        fprintf(stderr,"List Error: list is empty");
        exit(EXIT_FAILURE);
    }
    return L->front->data;
}

int back(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: list is NULL");
        exit(EXIT_FAILURE);
    }
    if (L->length <= 0) {
        fprintf(stderr,"List Error: list is empty");
        exit(EXIT_FAILURE);
    }
    return L->back->data;
}

int get(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: list is NULL");
        exit(EXIT_FAILURE);
    }
    if (L->length <= 0) {
        fprintf(stderr,"List Error: list is empty");
        exit(EXIT_FAILURE);
    }
    if (L->index < 0) {
        fprintf(stderr,"List Error: no selected item");
        exit(EXIT_FAILURE);
    }


    return L->cursor->data;
}

bool equals(List A, List B) {
    if (A == NULL || B == NULL) {
        fprintf(stderr, "List Error: list is NULL");
        exit(EXIT_FAILURE);
    }

    if (length(A) == 0 && length(B) == 0) {
        return true;
    } else if (length(A) != length(B)) {
        return false;
    } else {
        int range = length(A);
        Node A_Node = A->front;
        Node B_Node = B->front;
        for (int i = 0; i < range; ++i) {
            if (A_Node->data != B_Node->data) {
                return false;
            }
            A_Node = A_Node->next;
            B_Node = B_Node->next;
        }
    }

    return true;
}

// Manipulation methods

void clear(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: list is NULL");
        exit(EXIT_FAILURE);
    }

    Node active_node = L->front;
    while (active_node != NULL) {
        Node next_node = active_node->next;
        freeNode(&active_node);
        active_node = next_node;
    }


    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->length = 0;
    L->index = -1;
}

void set(List L,int x) {
    if (L == NULL) {
        fprintf(stderr, "List Error: list is NULL");
        exit(EXIT_FAILURE);
    }

    if (L->length <= 0) {
        fprintf(stderr,"List Error: list is empty");
        exit(EXIT_FAILURE);
    }

    if (L->index < 0) {
        fprintf(stderr,"List Error: no selected item");
        exit(EXIT_FAILURE);
    }

    L->cursor->data = x;
}

void moveFront(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: list is NULL");
        exit(EXIT_FAILURE);
    }

    if (L->length > 0) {
        L->cursor = L->front;
        L->index = 0;
    }
}

void moveBack(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: list is NULL");
        exit(EXIT_FAILURE);
    }

    if (L->length > 0) {
        L->cursor = L->back;


        L->index = L->length - 1;
    }
}

void movePrev(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: list is NULL");
        exit(EXIT_FAILURE);
    }

    if (L->index > 0) {
        L->cursor = L->cursor->prev;
        L->index--;
    } else if (L->index == 0) {
        L->cursor = NULL;
        L->index = -1;
    }
}

void moveNext(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: list is NULL");
        exit(EXIT_FAILURE);
    }

    if (L->index != -1 && L->index < L->length - 1) {
        L->cursor = L->cursor->next;
        L->index++;
    } else if (L->index != -1 && L->index == L->length - 1) {
        L->cursor = NULL;
        L->index = -1;
    }
}

void prepend(List L, int x) {
    if (L == NULL) {
        fprintf(stderr, "List Error: list is NULL");
        exit(EXIT_FAILURE);
    }

    Node new_node = newNode(x);

    if (L->length > 0) {
        new_node->next = L->front;
        L->front->prev = new_node;

        L->front = new_node;
        if (L->index != -1) {
            L->index++;
        }
    } else {
        L->front = new_node;
        L->back = new_node;
    }

    L->length++;
}

void append(List L, int x) {
    if (L == NULL) {
        fprintf(stderr, "List Error: list is NULL");
        exit(EXIT_FAILURE);
    }

    Node new_node = newNode(x);

    if (L->length > 0) {
        L->back->next = new_node;
        new_node->prev = L->back;

        L->back = new_node;
    } else {
        L->front = new_node;
        L->back = new_node;
    }

    L->length++;
}

void insertBefore(List L, int x) {
    if (L == NULL) {
        fprintf(stderr, "List Error: list is NULL");
        exit(EXIT_FAILURE);
    }

    if (L->length <= 0) {
        fprintf(stderr,"List Error: list is empty");


        exit(EXIT_FAILURE);
    }

    if (L->index < 0) {
        fprintf(stderr,"List Error: no selected item");
        exit(EXIT_FAILURE);
    }

    Node new_node = newNode(x);

    if (L->index == 0) {
        new_node->next = L->cursor;
        L->cursor->prev = new_node;

        L->front = new_node;
    } else {
        Node back_node = L->cursor->prev;
        
        new_node->prev = back_node;
        new_node->next = L->cursor;

        back_node->next = new_node;
        L->cursor->prev = new_node;
    }

    if (L->index != -1) {
        L->index++;
    }
    L->length++;
}

void insertAfter(List L, int x) {
    if (L == NULL) {
        fprintf(stderr, "List Error: list is NULL");
        exit(EXIT_FAILURE);
    }

    if (L->length <= 0) {
        fprintf(stderr,"List Error: list is empty");
        exit(EXIT_FAILURE);
    }



    if (L->index < 0) {
        fprintf(stderr,"List Error: no selected item");
        exit(EXIT_FAILURE);
    }

    Node new_node = newNode(x);

    if (L->index == L->length - 1) {
        new_node->prev = L->cursor;
        L->cursor->next = new_node;

        L->back = new_node;
    } else {
        Node front_node = L->cursor->next;

        new_node->prev = L->cursor;
        new_node->next = front_node;

        L->cursor->next = new_node;
        front_node->prev = new_node;
    }

    L->length++;
}

void deleteFront(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: list is NULL");
        exit(EXIT_FAILURE);
    }

    if (L->length <= 0) {
        fprintf(stderr,"List Error: list is empty");
        exit(EXIT_FAILURE);
    }

    if (L->length == 1) {
        freeNode(&(L->front));

        L->front = NULL;
        L->back = NULL;
        L->cursor = NULL;
        L->index = -1;
    } else if (L->index == 0) {
        Node new_front = L->front->next;
        freeNode(&(L->front));
        new_front->prev = NULL;

        L->front = new_front;
        L->cursor = NULL;
        L->index = -1;
    } else {
        Node new_front = L->front->next;
        freeNode(&(L->front));
        new_front->prev = NULL;

        L->front = new_front;
        if (L->index != -1) {
            L->index--;
        }
    }

    L->length--;
}

void deleteBack(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: list is NULL");
        exit(EXIT_FAILURE);
    }

    if (L->length <= 0) {
        fprintf(stderr,"List Error: list is empty");
        exit(EXIT_FAILURE);
    }

    if (L->length == 1) {
        freeNode(&(L->back));

        L->front = NULL;
        L->back = NULL;
        L->cursor = NULL;
        L->index = -1;
    } else if (L->index == L->length - 1) {
        Node new_back = L->back->prev;
        freeNode(&(L->back));
        new_back->next = NULL;

        L->back = new_back;
        L->cursor = NULL;
        L->index = -1;
    } else {
        Node new_back = L->back->prev;
        freeNode(&(L->back));
        new_back->next = NULL;

        L->back = new_back;
    }

    L->length--;
}

void delete(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: list is NULL");
        exit(EXIT_FAILURE);
    }

    if (L->length <= 0) {
        fprintf(stderr,"List Error: list is empty");
        exit(EXIT_FAILURE);
    }

    if (L->index < 0) {
        fprintf(stderr,"List Error: no selected item");
        exit(EXIT_FAILURE);
    }

    if (L->length == 1) {
        freeNode(&(L->cursor));
        L->front = NULL;
        L->back = NULL;
        L->cursor = NULL;
        L->index = -1;

        L->length--;
    } else if (L->index == 0) {
        deleteFront(L);
    } else if (L->index == L->length - 1) {
        deleteBack(L);
    } else {
        L->cursor->prev->next = L->cursor->next;
        L->cursor->next->prev = L->cursor->prev;
        freeNode(&(L->cursor));
        
        L->cursor = NULL;
        L->index = -1;

        L->length--;
    }
}

// Other operations

void printList(FILE* out, List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: list is NULL");
        exit(EXIT_FAILURE);
    }

    if (L->length > 0) {
        fprintf(out,"%d",L->front->data);
        Node active_node = L->front->next;
        while (active_node != NULL) {
            fprintf(out," %d",active_node->data);
            active_node = active_node->next;
        }
    }
}

List copyList(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: list is NULL");
        exit(EXIT_FAILURE);
    }

    List new_list = newList();

    Node active_source_node = L->front;
    while (active_source_node != NULL) {
        append(new_list,active_source_node->data);
        active_source_node = active_source_node->next;
    }

    return new_list;
}
