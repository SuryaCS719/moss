#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>
#include <assert.h>
#include "List.h"

typedef struct NodeObj* Node;



typedef struct NodeObj {
    int data;
    Node next;
    Node prev;
} NodeObj;

typedef struct ListObj {
    Node front;
    Node cursor;
    Node back;
    int index;
    int length;
} ListObj;

//list constructor
List newList(void) {
    List L = malloc(sizeof(ListObj));
    assert(L != NULL);
    L->front = NULL;
    L->cursor = NULL;
    L->back = NULL;
    L->index = -1;
    L->length = 0;

    return(L); 
}

//node constructor
Node newNode(int x) {
    Node N = malloc(sizeof(NodeObj));
    assert(N != NULL);
    N->data = x;
    N->next = NULL;
    N->prev = NULL;

    return(N);
}

//list deconstructor
void freeList(List* pL) {
    if( pL!=NULL && *pL!=NULL ){
        clear(*pL);
        free(*pL);
      *pL = NULL;
    }
}

//node deconstructor
void freeNode(Node* pN){
   if( pN!=NULL && *pN!=NULL ){
      free(*pN);
      *pN = NULL;
   }
}

//access functions--------------------------------------------------------------
// Returns the number of elements in L.
int length(List L) {
    if (L == NULL){
        fprintf(stderr, "Can't check length, list is NULL\n");
        exit(EXIT_FAILURE);
    }

    return(L->length);
}

// Returns index of cursor element if defined, -1 otherwise.
int index(List L) {
    if (L == NULL){
        fprintf(stderr, "Can't check index, list is NULL\n");
        exit(EXIT_FAILURE);
    }

    if (L->cursor == NULL){
        return(-1);
    }

    return(L->index);
}

// Returns front element of L. Pre: length()>0
int front(List L){
    if (L == NULL){
        fprintf(stderr, "Can't check front, list is NULL \n");
        exit(EXIT_FAILURE);
    }

    if (L->length == 0){
        fprintf(stderr, "Can't check front, list is empty\n");
        exit(EXIT_FAILURE);
    }

    return(L->front->data);
}

// Returns back element of L. Pre: length()>0
int back(List L){
    if (L == NULL){
        fprintf(stderr, "Can't check back, list is NULL \n");
        exit(EXIT_FAILURE);
    }

    if (L->length == 0){
        fprintf(stderr, "Can't check back, list is empty\n");
        exit(EXIT_FAILURE);
    }

    return(L->back->data);
}

// Returns cursor element of L. Pre: length()>0, index()>=0
int get(List L){
    if (L == NULL){


        fprintf(stderr, "Can't check cursor, list is NULL \n");
        exit(EXIT_FAILURE);
    }

    if (L->length == 0){
        fprintf(stderr, "Can't check cursor, list is empty\n");
        exit(EXIT_FAILURE);
    }

    return(L->cursor->data);
}

// Returns true iff Lists A and B contain the same sequence of elements, returns false otherwise.
bool equals(List A, List B){
    if (A == NULL || B == NULL){
        fprintf(stderr, "Can't check if equal, one or both lists are NULL \n");
        exit(EXIT_FAILURE);
    }

    if (A->length != B->length) {
        return false;
    }

    Node checkA = A->front;
    Node checkB = B->front;
    //to be honest i think i only need to check one list since we already confirmed they are the same length 
    while (checkA != NULL) {
        if (checkA->data != checkB->data) {
            return false; 
        }
        checkA = checkA->next;


        checkB = checkB->next;
    }

    return true;
}

//manipulation procedures-------------------------------------------------------

// Resets L to its original empty state.
void clear(List L){
    if (L == NULL){
        fprintf(stderr, "Can't clear, list is NULL \n");
        exit(EXIT_FAILURE);
    }

    while (L->length > 0)
        deleteFront(L);
}

// Overwrites the cursor elementâ€™s data with x. Pre: length()>0, index()>=0
void set(List L, int x){
    if (L == NULL){
        fprintf(stderr, "Can't set cursor, list is NULL \n");
        exit(EXIT_FAILURE);
    }

    if (L->length==0){
        fprintf(stderr, "Can't set cursor, list is empty (length) \n");
        exit(EXIT_FAILURE);
    }

    if (L->index<0){
        fprintf(stderr, "Can't set cursor, list is empty (index) \n");
        exit(EXIT_FAILURE);
    }

    L->cursor->data = x;
}

// If L is non-empty, sets cursor under the front element, otherwise does nothing.
void moveFront(List L){
    if (L == NULL){
        fprintf(stderr, "Can't move cursor to front, list is NULL \n");
        exit(EXIT_FAILURE);
    }

    if (L->length > 0){
        L->cursor = L->front;
        L->index = 0;
    }
}

//If L is non-empty, sets cursor under the back element, otherwise does nothing.
void moveBack(List L){
    if (L == NULL){
        fprintf(stderr, "Can't move cursor to back, list is NULL \n");
        exit(EXIT_FAILURE);
    }

     if (L->length > 0){
        L->cursor = L->back;
        L->index = L->length-1;
    }
}

// If cursor is defined and not at front, move cursor one step toward the front of L; 
// if cursor is defined and at front, cursor becomes undefined; if cursor is undefined do nothing
void movePrev(List L){
    if (L == NULL){
        fprintf(stderr, "Can't move cursor to previous location, list is NULL \n");
        exit(EXIT_FAILURE);
    }



    if (L->cursor != NULL && L->cursor != L->front){
        L->cursor = L->cursor->prev;
        L->index--;
    }

    if (L->cursor != NULL && L->cursor == L->front){
        L->cursor = NULL;
        L->index = -1;
    }

    if(L->cursor == NULL) {
        return;
    }
}

// If cursor is defined and not at back, move cursor one step toward the back of L; 
// if cursor is defined and at back, cursor becomes undefined; if cursor is undefined do nothing
void moveNext(List L){
    if (L == NULL){
        fprintf(stderr, "Can't move cursor to next location, list is NULL \n");
        exit(EXIT_FAILURE);
    }

    if (L->cursor == NULL) {
        return; 
    }
    else if (L->cursor != L->back){
        L->cursor = L->cursor->next;
        L->index++;
    }

    else if (L->cursor == L->back){
        L->cursor = NULL;
        L->index =  -1;
    } 

}

// Insert new element into L. If L is non-empty, insertion takes place before front element.
void prepend(List L, int x){
    if (L == NULL){
        fprintf(stderr, "Can't prepend, list is NULL \n");
        exit(EXIT_FAILURE);
    }

    Node newN = newNode(x);

    if (L->length == 0){
        L->front = newN;
        L->back = newN;
    } else {
        //prev already null
        newN->next = L->front;
        L->front->prev = newN;
        L->front = newN;


        if (L->cursor != NULL)
            L->index++;
    }
    L->length++;
}

// Insert new element into L. If L is non-empty, insertion takes place after back element.
void append(List L, int x){
    if (L == NULL){
        fprintf(stderr, "Can't append, list is NULL \n");
        exit(EXIT_FAILURE);
    }

    Node newN = newNode(x);

    if (L->length == 0){
        L->front = newN;
        L->back = newN;
    } else {
        //next already null
        newN->prev = L->back;
        L->back->next = newN;
        L->back = newN;
    }
    L->length++;
}

// Insert new element before cursor. Pre: length()>0, index()>=0
void insertBefore(List L, int x){
    if (L == NULL){


        fprintf(stderr, "Can't insertBefore, list is NULL \n");
        exit(EXIT_FAILURE);
    }

    if (L->length == 0){
        fprintf(stderr, "Can't insertBefore, list is empty \n");
        exit(EXIT_FAILURE);
    }

    if (L->index == -1){
        fprintf(stderr, "Can't insertBefore, cursor is undefined \n");
        exit(EXIT_FAILURE);
    }

    Node newN = newNode(x);

    if (L->index == 0){
        newN->next = L->front;
        L->front->prev = newN;
        L->front = newN;
    } else {
        newN->next = L->cursor;
        newN->prev = L->cursor->prev;
        L->cursor->prev->next = newN; 
        L->cursor->prev = newN; 
    }
    L->length++;
    L->index++;

}

//Insert new element after cursor. Pre: length()>0, index()>=0
void insertAfter(List L, int x){

        newN->prev = L->back;

        //L->cursor = L->back->prev;
    } else {
        newN->prev = L->cursor;
        newN->next = L->cursor->next;
        L->cursor->next->prev = newN; 
        L->cursor->next = newN; 
        //L->index++;
    }

}

// Delete the front element. Pre: length()>0
void deleteFront(List L){
    if (L == NULL){
        fprintf(stderr, "Can't deleteFront, list is NULL \n");


        exit(EXIT_FAILURE);
    }

    if (L->length == 0){
        fprintf(stderr, "Can't deleteFront, list is empty \n");
        exit(EXIT_FAILURE);
    }

    Node oldN = L->front;
    if (L->length == 1){
        L->front = NULL;
        L->cursor = NULL;
        L->back = NULL;
        L->index = -1; 
    } else {
        if (L->cursor == L->front){
            L->cursor = NULL;
            L->index = -1;
        }

        L->front = L->front->next;  
        L->front->prev = NULL;  
        L->index--;

    }
    freeNode(&oldN); 
    L->length--; 
}   

// Delete the back element. Pre: length()>0
void deleteBack(List L){
    if (L == NULL){
        fprintf(stderr, "Can't deleteBack, list is NULL \n");
        exit(EXIT_FAILURE);
    }

    if (L->length == 0){
        fprintf(stderr, "Can't deleteBack, list is empty \n");
        exit(EXIT_FAILURE);
    }

    Node oldN = L->back;
    if (L->length == 1){
        L->front = NULL;
        L->cursor = NULL;
        L->back = NULL;
        L->index = -1; 
    } else {
        if (L->cursor == L->back){
            L->cursor = NULL;
            L->index = -1;
        }

        L->back = L->back->prev;  
        L->back->next = NULL;          
    }
    freeNode(&oldN); 
    L->length--; 
}   

// Delete cursor element, making cursor undefined. Pre: length()>0, index()>=0
void delete(List L){
    if (L == NULL){
        fprintf(stderr, "Can't delete, list is NULL \n");
        exit(EXIT_FAILURE);
    }

    if (L->length == 0){
        fprintf(stderr, "Can't delete, list is empty \n");
        exit(EXIT_FAILURE);
    }

    if (L->index == -1){
        fprintf(stderr, "Can't delete, cursor is undefined \n");
        exit(EXIT_FAILURE);
    }

    Node oldN = L->cursor;

    if (L->cursor == L->front){
        deleteFront(L);
        return;
    }

    if (L->cursor == L->back){
        deleteBack(L);
        return;
    }

    oldN->next->prev = oldN->prev;
    oldN->prev->next = oldN->next;
    L->cursor = NULL;
    L->index = -1;

    freeNode(&oldN); 
    L->length--;
}

// Prints to the file pointed to by out, a string representation of L consisting of a space separated sequence of integers, with front on left.
void printList(FILE* out, List L){
    if (L == NULL){
        fprintf(stderr, "Can't printList, list is NULL \n");
        exit(EXIT_FAILURE);
    }

    Node currN = L->front;

    while (currN != NULL) {
        fprintf(out, "%d ", currN->data);
        currN = currN->next;
    }
    fprintf(out, "\n");
    
}

// Returns a new List representing the same integer sequence as L. The cursor in the new list is undefined, regardless of the state of the cursor in L. The state of L is unchanged.
List copyList(List L){
    if (L == NULL){
        fprintf(stderr, "Can't copyList, list is NULL \n");
        exit(EXIT_FAILURE);
    }

    List newL = newList();
    Node currN = L->front;

    while (currN != NULL){
        append(newL, currN->data);
        currN = currN->next;
    }

    return(newL);
}
