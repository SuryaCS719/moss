#include<stdio.h>

// structs --------------------------------------------------------------------

// private Node type
typedef struct NodeObj *Node;

// private NodeObj type

typedef struct NodeObj {

    ListElement data;


    Node next;
    Node prev;
} NodeObj;

// private ListObj type
typedef struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
    int cursorIndex;
} ListObj;

// Constructors-Destructors ---------------------------------------------------

// newList()
// Returns reference to new empty Queue object.

List newList() {
    List L = malloc(sizeof(ListObj));
    assert(L!=NULL);
    L->front = L->back = NULL;
    L->cursor = NULL;
    L->length = 0;
    L->cursorIndex = -1;
    return (L);
}

// freeList()
// Frees all heap memory associated with  *pL, and sets *pL to NULL.

void freeList(List *pL) {
    if (pL != NULL && *pL != NULL) {
            while(length(*pL) !=0 ){
               deleteFront(*pL);
           }
        free(*pL);
        *pL = NULL;
    }
}

// newNode()
// Returns reference to new Node object. Initializes next and data fields.

Node newNode(ListElement data) {
    Node N = malloc(sizeof(NodeObj));
    assert(N!=NULL);
    N->data = data;
    N->next = NULL;
    N->prev = NULL;
    return (N);
}

// freeNode()
// Frees heap memory pointed to by *pN, sets *pN to NULL.
void freeNode(Node *pN) {


    if (pN != NULL && *pN != NULL) {
        free(*pN);
        *pN = NULL;
    }
}

// Access functions -----------------------------------------------------------

//length()
//returns the number of elements in L;

int length(List L) {

    if (L == NULL) {
        return 0;
    } else
        return L->length;
}

//index()
//returns index of cursor, -1 if not defined.
int index(List L) {

    if (L == NULL || L->cursor == NULL) {
        return -1;
    } else
        return L->cursorIndex;
}

// front()
// Returns the value at the front of L.
// Pre: lengh() >0
ListElement front(List L) {

    if (L == NULL) {
        fprintf(stderr, "List Error: calling front() on NULL Queue reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->length < 0) {

        fprintf(stderr, "List Error: calling front() on an empty Queue\n");
        exit(EXIT_FAILURE);
    }
    return (L->front->data);
}

// back()
// returns back element of L.
//Pre: length(L) > 0

ListElement back(List L) {

    if (L == NULL || L->length == 0) { // add more later
        fprintf(stderr, "Error calling back() on empty list");
        exit(EXIT_FAILURE);
    }
    return L->back->data;
}

//get()
// reutnrs cursor element of L;
//pre: lenght(L) > 0, index (L) >= 0

ListElement get(List L) {

    if (L == NULL) {

        fprintf(stderr, "Error calling get() on empty list");
        exit(EXIT_FAILURE);

    }
    return L->cursor->data;
}
//equals()
// Returns true iff Lists A and B contain the same
// sequence of elements, returns false otherwise.

bool equals(List A, List B) {

    if (A == NULL || B == NULL) {
           fprintf(stderr, "List Error: calling equals() on NULL list.\n");
           exit(EXIT_FAILURE);
       }

    if (A->length != B->length) {
           return false;
       }



    moveFront(A);
    moveFront(B);

    while (A->cursor != NULL && B->cursor != NULL) {
            if (A->cursor->data != B->cursor->data) {
                return false;
            }
            moveNext(A);
            moveNext(B);
        }

        return true;

}

// Manipulation procedures ----------------------------------------------------

//clear()
//Resets L to its original empty state.

void clear(List L) {
    while (L->length > 0) {
        deleteFront(L);
    }
    L->cursor = NULL;
    L->cursorIndex = -1;
}
//set()
// overwrites the cursor element's data with x

void set(List L, ListElement x) {

    if (L == NULL) {
        fprintf(stderr, "List Error: calling set() on an empty List \n");
        exit(EXIT_FAILURE);
    }
    if (L->cursor == NULL) {
        fprintf(stderr, "List Error: calling set() on an undefined cursor.\n");
        exit(EXIT_FAILURE);
    }
    L->cursor->data = x;
}

// moveFront()
//If L is non-empty, sets cursor under the front element, otherwise does nothing.
void moveFront(List L) {
    if (L != NULL && L->length > 0) {

        L->cursor = L->front;
        L->cursorIndex = 0;

    } else {
        L->cursor = NULL;      //
        L->cursorIndex = -1;

    }
}

// movBack()
//If L is non-empty, sets cursor under the back element,
// If cursor is defined and at front, cursor becomes undefined.
void moveBack(List L) {
    if (L != NULL && L->length > 0) {

        L->cursor = L->back;
        L->cursorIndex = L->length - 1;

    } else {
        L->cursor = NULL;      //
        L->cursorIndex = -1;

    }
}

// movePrev()
// If cursor is defined and not at front, move cursor one step toward the front of L.
// If cursor is defined and at front, cursor becomes undefined.
// If cursor is undefined do nothing.
void movePrev(List L) {
    if (L->cursor != NULL && L->cursorIndex != -1) {
        if (L->cursor->prev != NULL) {
            L->cursor = L->cursor->prev;
            L->cursorIndex--;
        } else {
            //if front set undefined
            L->cursor = NULL;
            L->cursorIndex = -1;
        }
    } else {
            //do nothing
        return;
    }
}
// moveNext()
// If cursor is defined and not at back, move cursor one step toward the back of L.
// If cursor is defined and at back, cursor becomes undefined.
// If cursor is undefined do nothing.
void moveNext(List L) {
    if (L->cursor != NULL && L->cursor->next != NULL) {


        L->cursor = L->cursor->next;
        L->cursorIndex++;
    } else {

        L->cursor = NULL;
        L->cursorIndex = -1;
    }
}

// prepend()
// Insert new element into L. If L is non-empty, insertion takes place before front element.

void prepend(List L, ListElement x) {
    Node N = newNode(x);

    if (L->length == 0) {
        L->front = L->back = N;
    } else {

        N->next = L->front;
        L->front->prev = N;


        L->front = N;
    }
    L->length++;
    if (L->cursorIndex != -1) {
        L->cursorIndex++;
    }
}

//append()
//Insert new element into L.  If L is non-empty,
// insertion takes place after back element.

void append(List L, ListElement x) {
    Node N = newNode(x);  //
    if (L->length == 0) {
        L->front = L->back = N;
    } else {
        L->back->next = N;  //
        N->prev = L->back;
        L->back = N;        //
    }
    L->length++;

}

//inserBefore()
// Insert new element before cursor. Pre: length()>0, index()>=0

void insertBefore(List L, ListElement x) {
    if (L == NULL || L->cursor == NULL) {

        fprintf(stderr, "Error, calling insertBefore() on empty list.\n");
        exit(EXIT_FAILURE);
    }
    Node N = newNode(x);

    if (L->cursor == L->front) {

        N->next = L->cursor;
        L->cursor->prev = N;
        L->front = N;

        L->front = N;
    } else {
        N->prev = L->cursor->prev;
        N->next = L->cursor;


        L->cursor->prev->next = N;
        L->cursor->prev = N;
    }
    L->length++;
    L->cursorIndex++;
}

//insertAfter()
// Insert new element after cursor.
// Pre: length()>0, index()>=0
void insertAfter(List L, ListElement x) {
    if (L == NULL || L->cursor == NULL) {

        fprintf(stderr, "Error, calling insertAfter() on empty list.\n");
        exit(EXIT_FAILURE);
    }
    Node N = newNode(x);

    if (L->cursor == L->back) {

        N->prev = L->cursor;
        L->cursor->next = N;

        L->back = N;
    } else {
        N->next = L->cursor->next;
        N->prev = L->cursor;
        L->cursor->next->prev = N;
        L->cursor->next = N;
    }
    L->length++;
}

//deleteFront()
//// Delete the front element. Pre: length()>0

void deleteFront(List L) {
    if (L == NULL) {

        fprintf(stderr, "List Error\n");


        exit(EXIT_FAILURE);
    }
    if (L->length == 0) {
        fprintf(stderr, "List Error\n");
        exit(EXIT_FAILURE);
    }

    Node N = L->front;

    if (L->cursor != NULL && L->cursorIndex > 0) {
        L->cursorIndex--;}

        if (L->cursor == L->front) {
            L->cursor = NULL;
            L->cursorIndex = -1;
        }

        if (L->length > 1) {
            L->front = L->front->next;


            L->front->prev = NULL;
        } else {
            L->front = L->back = NULL;
        }

        freeNode(&N);
        L->length--;
    }

// deleteBack()
// Delete the back element. Pre: length()>0
void deleteBack(List L) {

    if (L == NULL) {
        fprintf(stderr, "List Error\n");
        exit(EXIT_FAILURE);
    }

    if (L->length == 0) {
        fprintf(stderr, "List Error\n");
        exit(EXIT_FAILURE);
    }

    Node N = L->back;

        if (L->length == 1) {
            L->front = L->back = NULL;
        } else {  //
            L->back = L->back->prev;
            L->back->next = NULL;
        }

        if (L->cursor == N) {


            L->cursor = NULL;
            L->cursorIndex = -1;
        }

        freeNode(&N);
        L->length--;
    }


//delete()
// Delete cursor element, making cursor undefined
// Pre: length()>0, index()>=0

void delete(List L) {

    if (L->cursor == NULL || L->cursorIndex == -1) {
        fprintf(stderr, "Error, calling delete() on an undefined cursor.\n");
        exit(EXIT_FAILURE);
    }

    if (L->cursorIndex == -1) {
        fprintf(stderr, "Error, calling delete() undefined cursor.\n");
        exit(EXIT_FAILURE);

    }

    Node N = L->cursor;
    //one node
    if (L->length == 1) {
        L->front = L->back = NULL;
    } else if (L->cursor == L->front) {
        L->front = L->cursor->next;
        L->front->prev = NULL;

    } else if (L->cursor == L->back) {

        L->back = L->back->prev;
        L->back->next = NULL;
        //cursor in middle now
    } else {
        L->cursor->prev->next = L->cursor->next;
        L->cursor->next->prev = L->cursor->prev;

    }
    freeNode(&N);
    L->cursor = NULL;
    L->cursorIndex = -1;
    L->length--;

}

//printList()
// Prints to the file pointed to by out, a
// string representation of L consisting
// of a space separated sequence of integers,
// with front on left.
void printList(FILE *out, List L) {
    Node N = L->front;
    while (N != NULL) {
        fprintf(out, FORMAT" ", N->data);
        N = N->next;
    }
    fprintf(out, "\n");
}

// copyList()
// Returns a new List representing the same integer
// sequence as L. The cursor in the new list is undefined,
// regardless of the state of the cursor in L. The state
// of L is unchanged.

List copyList(List L) {

    if (L == NULL) {
        fprintf(stderr, "List Error: calling copyList() on empty list.\n");
        exit(EXIT_FAILURE);
    }

    Node curr = L->front;
    List copiedList = newList();
    while (curr != NULL) {
        append(copiedList, curr->data);
        curr = curr->next;

    }
    return copiedList;

}

//concatList();
// Returns a new List which is the concatenation of
// A and B.  The cursor in the new List is undefined,
// regardless of the states of the cursors in A and B.
// The states of A and B are unchanged

List concatList(List A, List B) {

    List concatedList = newList();

    Node currB = B->front;

    int len = B->length;
    concatedList = copyList(A);

    moveBack(concatedList);

    for (int i = 0; i < len; i++) {

        append(concatedList, currB->data);
        currB = currB->next;

    }
    return concatedList;

}
