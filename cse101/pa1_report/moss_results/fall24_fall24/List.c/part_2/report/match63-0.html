// structs
#include "List.h"
#include <stdlib.h>
#include <stdio.h>
#include <assert.h>

// private Node type
typedef struct NodeObj* Node;

// private NodeObj type
typedef struct NodeObj {
    ListElement data;
    Node next;
    Node prev;
} NobeObj;

// private ListObj type
typedef struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
    int cursorIndex;
} ListObj;


// Constructors-Destructors

Node newNode(ListElement data) {
    Node N = malloc(sizeof(NobeObj));
    assert(N != NULL);
    N->data = data;
    N->next = NULL;
    N->prev = NULL;
    return N;
}

void freeNode(Node* pN) {
    if (pN != NULL  && *pN != NULL) {


        free(*pN);
        *pN = NULL;
    }
}
 
List newList() {
    List L;
    L = malloc(sizeof(ListObj));
    assert(L != NULL);
    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->length = 0;
    L->cursorIndex = -1;
    return L;
}

void freeList(List* pL) {
    if (pL == NULL || *pL == NULL) {
        return;
    }
    clear(*pL);
    free(*pL);
    *pL = NULL;
}

// Access functions

bool isEmpty(List L){
   if(L == NULL) {
      printf("List Error: calling isEmpty() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   return(L->length==0);
}

int length(List L) {
    return L->length;
}

int index(List L) {
    return L->cursorIndex;
}

int front(List L) {
    if (L->length == 0) {
        printf("no length in front()");
        exit(EXIT_FAILURE);
    }
    return L->front->data;
}

int back(List L) {
    if (L->length == 0 ) {
        printf("no length in back()");
        exit(EXIT_FAILURE);
    }
    return L->back->data;
}

int get(List L) {
    if (L->length == 0) {
        printf("get() no length");
        exit(EXIT_FAILURE);
    }
    if (L->cursorIndex < 0) {
        printf("get() undefined cursor");
        exit(EXIT_FAILURE);
    }
    return L->cursor->data;
}

bool equals(List A, List B) {
    if (A->length != B->length) {
        return false;
    }

    Node aCurrent = A->front;
    Node bCurrent = B->front;

    while (aCurrent != NULL && bCurrent != NULL) {


        if (aCurrent->data != bCurrent->data) {
            return false;
        }

        aCurrent = aCurrent->next;
        bCurrent = bCurrent->next;
    }
    return true;
}

// Manipulation procedures
void clear(List L) {
    Node current = L->front;
    Node nextN = NULL;
    while (current != NULL) {
        nextN = current->next;
        free(current);
        current = nextN;
    }
    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->length = 0;
    L->cursorIndex = -1;
}

void set(List L, int x) {
    if (L->length == 0) {
        printf("set() empty");
        exit(EXIT_FAILURE);
    }
    if (L->cursorIndex < 0) {
        printf("set() undefined cursor index");
        exit(EXIT_FAILURE);
    }
    L->cursor->data = x;
}

void moveFront(List L) {
    if (L->length == 0) {
        return;
    }
    L->cursor = L->front;
    L->cursorIndex = 0;
}

void moveBack(List L) {
    if (L == NULL) {
        printf("List Error: calling moveBack() on NULL List reference \n");
        exit(EXIT_FAILURE);
    }
    if (isEmpty(L)) {
        printf("List Error: calling moveBack() on empty List \n");
        exit(EXIT_FAILURE);
    }

    L->cursor = L->back;
    L->cursorIndex = L->length - 1;
}

void movePrev(List L) {
    if (L == NULL) {
        printf("List Error: calling movePrev() on NULL List reference \n");
        exit(EXIT_FAILURE);
    }
    if (L->cursor != NULL && L->cursorIndex > 0) {
        L->cursor = L->cursor->prev;
        L->cursorIndex--;
    }
    else {
        L->cursor = NULL;
        L->cursorIndex = -1;
    }
}

void moveNext(List L) {
    if (L == NULL) {
        printf("List Error: calling movenext() on NULL List reference \n");
        exit(EXIT_FAILURE);
    }
    if (L->cursorIndex >= 0 && L->cursorIndex < length(L) - 1) {
        L->cursor = L->cursor->next;
        L->cursorIndex++;
    }
    else {
        L->cursor = NULL;
        L->cursorIndex = -1;
    }
}

void prepend(List L, int x) {
    if (L == NULL) {
        printf("List Error: calling prepend() on NULL List reference \n");
        exit(EXIT_FAILURE);
    }
    Node N = newNode(x);
    if (length(L) == 0) {
        L->front = N;
        L->back = N;
    }
    else {
        N->next = L->front;
        L->front->prev = N;
        L->front = N;
    }
    L->cursorIndex++;
    L->length++;
}

void append(List L, int x) {
    
    if (L == NULL) {
        printf("List Error: calling append() on NULL List reference \n");
        exit(EXIT_FAILURE);
    }
    Node N = newNode(x);
    
    if (length(L) == 0) {
        L->front = N;
        L->back = N;
    }
    else {
        L->back->next = N;
        N->prev = L->back;
        L->back = N;
    }

    L->length++;
}

void insertBefore(List L, int x) {
    if (L == NULL) {
        printf("List Error: calling insertBefore() on NULL List reference \n");
        exit(EXIT_FAILURE);
    }
    if (isEmpty(L)) {
        printf("empty");
        exit(EXIT_FAILURE);
    }
    if (index(L) < 0) {
        printf("no index");
        exit(EXIT_FAILURE);
    }
    Node N = newNode(x);
    if (L->cursor == L->front) {
        N->next = L->front;
        L->front->prev = N;
        L->front = N;
    } else {
        N->prev = L->cursor->prev;
        N->next = L->cursor;
        L->cursor->prev->next = N;
        L->cursor->prev = N;
    }


    L->length++;
    L->cursorIndex++;
    
}

void insertAfter(List L, int x) {
    if (L == NULL) {
        printf("List Error: calling insertAfter() on NULL List reference \n");
        exit(EXIT_FAILURE);
    }
    if (isEmpty(L)) {
        printf("empty insertAfter");
        exit(EXIT_FAILURE);
    }
    if (index(L) < 0) {
        printf("no index insertAfter");
        exit(EXIT_FAILURE);
    }
    Node N = newNode(x);
    if (L->cursor == L->back) {
        N->prev = L->back;
        L->back->next = N;
        L->back = N;
    } else {
        N->prev = L->cursor;
        N->next = L->cursor->next;
        L->cursor->next->prev = N;


        L->cursor->next = N;
    }
    L->length++;
}

void deleteFront(List L) {
    if (L == NULL) {
        printf("List Error: calling deleteFront() on NULL List reference \n");
        exit(EXIT_FAILURE);
    }
    if (isEmpty(L)) {
        printf("empty deleteFront");
        exit(EXIT_FAILURE);
    }

    Node oldFront = L->front;
    if (L->front == L->back) {
        L->front = NULL;
        L->back = NULL;
    } else {
        L->front = L->front->next;
        L->front->prev = NULL;
    }
    free(oldFront);
    L->cursorIndex--;
    L->length--;

}

void deleteBack(List L) {
    if (L == NULL) {
        printf("List Error: calling deleteBack() on NULL List reference \n");
        exit(EXIT_FAILURE);
    }
    if (isEmpty(L)) {
        printf("deleteBack empty");
        exit(EXIT_FAILURE);
    }
    
    Node oldBack = L->back;
    if (L->length == 1) {
        L->front = NULL;
        L->back = NULL;
    }
    else {
        L->back = L->back->prev;
        L->back->next = NULL;
    }
    if (L->cursor == oldBack) {


        L->cursor = NULL;
        L->cursorIndex = -1;
    }
    free(oldBack);
    L->length--;
}

void delete(List L) {
     if (L == NULL) {
        printf("List Error: calling delete() on NULL List reference \n");
        exit(EXIT_FAILURE);
    }
    if (isEmpty(L)) {
        printf("delete empty");
        exit(EXIT_FAILURE);
    }
    if (index(L) < 0) {
        printf("delete undefined cursor");
        exit(EXIT_FAILURE);
    }
    Node cursorNode = L->cursor;
    if (cursorNode == L->front) {
        deleteFront(L);
    }
    else if (cursorNode == L->back) {
        deleteBack(L);
    }
    else {
        L->cursor->prev->next = L->cursor->next;
        L->cursor->next->prev = L->cursor->prev;
        free(L->cursor);
        L->cursor = NULL;
        L->cursorIndex = -1;
        L->length--;
    }
}
