#include <stdio.h>
#include <stdlib.h>
#include <assert.h>
#include <stdbool.h>
#include "List.h"

// Structs --------------------------------------------------------------------

// private Node type
typedef struct NodeObj* Node;

// private NodeObj type


typedef struct NodeObj{
    ListElement data;
    Node next;
    Node prev;
} NodeObj;

// private ListObj type
typedef struct ListObj{
    Node front;
    Node cursor;
    Node back;
    int index;
    int length;
} ListObj;


// Constructors-Destructors ---------------------------------------------------

// newList()
// Creates and returns a new empty List.
List newList(void){
    List L = malloc(sizeof(ListObj));
    assert(L != NULL);	// Check for successful memory allocation

    L->front = NULL;
    L->cursor = NULL;
    L->back = NULL;


    L->index = -1;  // When cursor is undefined, set index to -1
    L->length = 0;

    return L;
}

// freeList(List* pL)
// Frees all heap memory associated with *pL, and sets *pL to NULL.
void freeList(List* pL){
    if(pL != NULL && *pL != NULL){
        while((*pL)->length > 0){
	    deleteFront(*pL);
	}
	free(*pL);


	*pL = NULL;
    }
}

// Access Functions -----------------------------------------------------------

// length(List L)
// Returns the number of elements in L.
int length(List L){return L->length;}

// index(List L)
// Returns the index of cursor element if defined, -1 otherwise.
int index(List L){return L->index;}

// front(List L)
// Returns front element of L. Pre: length() > 0
int front(List L){
    assert(L->length > 0);  // Ensure List is not empty
    return L->front->data;
}

// back(List L)
// Returns back element of L. Pre: length() > 0
int back(List L){
    assert(L->length > 0);  // Ensure List is not empty
    return L->back->data;
}

// get(List L)
// Returns cursor element of L. Pre: length() > 0, index() >= 0
int get(List L){
    assert(L->length > 0);  // Ensure List is not empty
    assert(L->index >= 0);  // Ensure cursor is defined
    return L->cursor->data;
}

// equals(List A, List B)
// Returns true iff Lists A and B contain the same sequence of elements,
// returns false otherwise.
bool equals(List A, List B){
    if(A->length != B->length){return false;}

    Node N = A->front;
    Node M = B->front;
    while(N != NULL && M != NULL){
        if(N->data != M->data){return false;}
	N = N->next;
	M = M->next;
    }
    return true;
}

// Manipulation Procedures ----------------------------------------------------

// clear(List L)
// Resets L to its original empty state.
void clear(List L){


    while(L->length > 0){
        deleteFront(L);
    }
    L->cursor = NULL;
    L->index = -1;
}

// set(List L, int x)
// Overwrites the cursor element's data with x. Pre: length() > 0, index() >= 0
void set(List L, int x){
    assert(L->length > 0);	// Ensure List not empty
    assert(L->index >= 0);	// Ensure cursor is defined
    L->cursor->data = x;
}

// moveFront(List L)
// If L is non-empty, sets cursor under the front element,
// otherwise does nothing.
void moveFront(List L){
    if(L->length > 0){
        L->cursor = L->front;
	L->index = 0;
    }
}

// moveBack(List L)
// If L is non-empty, sets cursor under the back element,
// otherwise does nothing.
void moveBack(List L){
    if(L->length > 0){
        L->cursor = L->back;
	L->index = L->length - 1;
    }
}

// movePrev(List L)
// If cursor is defined and not at front,
// move cursor one step toward the front of L;
// if cursor is defined and at front, cursor becomes undefined;
// if cursor is undefined do nothing
void movePrev(List L){


    if(L->cursor != NULL && L->index > 0){
        L->cursor = L->cursor->prev;
	L->index--;
    }else if(L->index == 0){
        L->cursor = NULL;
	L->index = -1;
    }
}

// moveNext(List L)
// If cursor is defined and not at back,
// move cursor one step toward the back of L;
// if cursor is defined and at back, cursor becomes undefined;
// if cursor is undefined do nothing
void moveNext(List L){


    if(L->cursor != NULL && L->index < L->length - 1){
        L->cursor = L->cursor->next;
	L->index++;
    }else if(L->index == L->length - 1){
        L->cursor = NULL;
	L->index = -1;
    }
}

// prepend(List L, int x)
// Insert new element into L. If L is non-empty,
// insertion takes place before front element.
void prepend(List L, int x){
    Node N = (Node)malloc(sizeof(NodeObj));
    assert(N != NULL);		// Check for successful memory allocation
    N->data = x;
    N->next = L->front;		// Set node before front
    N->prev = NULL;
    if(L->front != NULL){
        L->front->prev = N;	// Set prev for next node
    }else{
        L->back = N;		// If the list was empty, N is also back
    }
    L->front = N;		// Set N to front
    L->length++;
    if(L->index != -1){		// If the list wasn't empty,
        L->index++;		// the index of the cursor moves
    }
}

// append(List L, int x)
// Insert new element into L. If L is non-empty,
// insertion takes place after back element.
void append(List L, int x){
    Node N = (Node)malloc(sizeof(NodeObj));
    assert(N != NULL);          // Check for successful memory allocation
    N->data = x;
    N->next = NULL;
    N->prev = L->back;
    if(L->back != NULL){
        L->back->next = N;
    }else{
        L->front = N;
    }
    L->back = N;
    L->length++;
}

// insertBefore(List L, int x)
// Insert new element before cursor. Pre: length() > 0, index() >= 0
void insertBefore(List L, int x){
    assert(L->length > 0);	// Ensure List is not empty
    assert(L->index >= 0);	// Ensure cursor is defined
    if(L->cursor == L->front){
        prepend(L, x);
    }else{
        Node N = (Node)malloc(sizeof(NodeObj));
	assert(N != NULL);	// Check for successful memory allocation
	N->data = x;
	N->prev = L->cursor->prev;
	N->next = L->cursor;
	L->cursor->prev->next = N;
	L->cursor->prev = N;
	L->length++;
	L->index++;
    }
}

// insertAfter(List L, int x)
// Insert new element after cursor. Pre: length() > 0, index() >= 0
void insertAfter(List L, int x){
    assert(L->length > 0);      // Ensure List is not empty
    assert(L->index >= 0);      // Ensure cursor is defined
    if(L->cursor == L->back){
        append(L, x);
    }else{
        Node N = (Node)malloc(sizeof(NodeObj));
	assert(N != NULL);      // Check for successful memory allocation
        N->data = x;
	N->next = L->cursor->next;
	N->prev = L->cursor;
	L->cursor->next->prev = N;
	L->cursor->next = N;
	L->length++;
    }
}

// deleteFront(List L)
// Delete the front element. Pre: length() > 0
void deleteFront(List L){
    assert(L->length > 0);	// Ensure List is not empty
    Node N = L->front;
    if(L->cursor == L->front){
        L->cursor = NULL;
	L->index = -1;
    } else {
	    L->index--;
    }
    if(L->front == L->back){
        L->front = L->back = NULL;
	
    }else{
	L->front = L->front->next;
	L->front->prev = NULL;
    }
    free(N);
    L->length--;
}

// deleteBack(List L)
// Delete the back element. Pre: length() > 0
void deleteBack(List L){
    assert(L->length > 0);      // Ensure List is not empty
    Node N = L->back;
    if(L->cursor == L->back){
        L->cursor = NULL;
	L->index = -1;
    }
    if(L->front == L->back){
        L->front = L->back = NULL;
    }else{
        L->back = L->back->prev;
	L->back->next = NULL;
    }
    free(N);
    L->length--;
}

// delete(List L)
// Delete cursor element, making cursor undefined.
// Pre: length() > 0, index() >= 0
void delete(List L){
    assert(L->length > 0);	// Ensure List is not empty
    assert(L->index >= 0);	// Ensure cursor is defined
    if(L->cursor == L->front){
        deleteFront(L);
    }else if(L->cursor == L->back){
        deleteBack(L);
    }else{
        Node N = L->cursor;
	N->prev->next = N->next;
	N->next->prev = N->prev;
	free(N);
	L->length--;
    }
    L->cursor = NULL;
    L->index = -1;
}

// Other Operations -----------------------------------------------------------

// printlist()
// Prints to the file pointed to by out, a string representation of L
// consisting of a space separated sequence of integers, with front on left.
void printList(FILE* out, List L){
    Node N = L->front;
    while(N != NULL){
        fprintf(out, "%d ", N->data);
	N = N->next;
    }
}

// copyList(List L)
// Returns a new List representing the same integer sequence as L. The cursor
// in the new list is undefined, regardless of the state of the cursor in L.
// The state of L is unchanged.
List copyList(List L){
    List copy = newList();
    Node N = L->front;
    while(N != NULL){


        append(copy, N->data);
	N = N->next;
    }
    return copy;
}

// concatList(List A, List B)
// Returns a new List which is the concatenation of A and B. The cursor in the
// new list is undefined, regardless of the states of the cursors in A and B.
// The states of A and B are unchanged.
List concatList(List A, List B){
    List C = newList();
    Node N = A->front;
    Node M = B->front;
    while(N != NULL){
        append(C, N->data);
	N = N->next;
    }
    while(M != NULL){
        append(C, M->data);
	M = M->next;
    }
    return C;
}
