#include "List.h"
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// Node object definition
typedef struct NodeObj {
    int data;
    struct NodeObj* next;
    struct NodeObj* prev;
} NodeObj;



typedef NodeObj* Node;

// List object definition
typedef struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
    int index;
} ListObj;

// Constructors-Destructors ---------------------------------------------------

List newList(void) {
    // Memory allocation for List
    List L = malloc(sizeof(ListObj));
    
    // Allocation failure handling
    if (L == NULL) {
        fprintf(stderr, "Error: Failed to allocate memory for new List\n");
        exit(EXIT_FAILURE);
    }

    // Initialize list fields
    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->length = 0;
    L->index = -1;

    return L;
}

void freeList(List* pL) {
    if (pL != NULL && *pL != NULL) {
        // Clear the list and release memory
        clear(*pL);
        free(*pL);
        *pL = NULL;
    }
}

// Access functions -----------------------------------------------------------

int length(List L) {
    if (L == NULL) {
        fprintf(stderr, "Error: List is NULL in length()\n");
        exit(EXIT_FAILURE);
    }
    return L->length;
}

int index(List L) {


    if (L == NULL) {
        fprintf(stderr, "Error: List is NULL in index()\n");
        exit(EXIT_FAILURE);
    }
    return L->index;
}

int front(List L) {
    if (L == NULL || L->length == 0) {
        fprintf(stderr, "Error: Invalid List or List is empty in front()\n");
        exit(EXIT_FAILURE);
    }
    return L->front->data;
}

int back(List L) {
    if (L == NULL || L->length == 0) {
        fprintf(stderr, "Error: Invalid List or List is empty in back()\n");
        exit(EXIT_FAILURE);
    }
    return L->back->data;
}

int get(List L) {
    if (L == NULL || L->length == 0 || L->index == -1) {
        fprintf(stderr, "Error: Invalid List or cursor is undefined in get()\n");
        exit(EXIT_FAILURE);
    }
    return L->cursor->data;
}

bool equals(List A, List B) {
    if (A == NULL || B == NULL) {
        fprintf(stderr, "Error: One or both Lists are NULL in equals()\n");
        exit(EXIT_FAILURE);
    }

    if (A->length != B->length) return false;

    Node N1 = A->front;
    Node N2 = B->front;

    while (N1 != NULL && N2 != NULL) {
        if (N1->data != N2->data) return false;
        N1 = N1->next;
        N2 = N2->next;
    }

    return true;
}

// Manipulation procedures ----------------------------------------------------

void clear(List L) {
    if (L == NULL) {
        fprintf(stderr, "Error: List is NULL in clear()\n");
        exit(EXIT_FAILURE);
    }

    // Free each node in the list
    Node current = L->front;
    while (current != NULL) {
        Node temp = current;
        current = current->next;
        free(temp);
    }



    L->front = L->back = L->cursor = NULL;
    L->length = 0;
    L->index = -1;
}

void set(List L, int x) {
    if (L == NULL || L->length == 0 || L->index == -1) {
        fprintf(stderr, "Error: Invalid List or cursor is undefined in set()\n");
        exit(EXIT_FAILURE);
    }
    L->cursor->data = x;
}

void moveFront(List L) {
    if (L != NULL && L->length > 0) {
        L->cursor = L->front;
        L->index = 0;
    }
}

void moveBack(List L) {
    if (L != NULL && L->length > 0) {
        L->cursor = L->back;
        L->index = L->length - 1;
    }
}

void movePrev(List L) {
    if (L != NULL && L->length > 0 && L->cursor != NULL) {
        if (L->cursor == L->front) {
            L->cursor = NULL;
            L->index = -1;
        } else {
            L->cursor = L->cursor->prev;
            L->index--;
        }
    }
}

void moveNext(List L) {
    if (L != NULL && L->length > 0 && L->cursor != NULL) {
        if (L->cursor == L->back) {
            L->cursor = NULL;
            L->index = -1;
        } else {
            L->cursor = L->cursor->next;
            L->index++;
        }
    }
}

void prepend(List L, int x) {
    if (L == NULL) {
        fprintf(stderr, "Error: List is NULL in prepend()\n");
        exit(EXIT_FAILURE);
    }

    Node newNode = malloc(sizeof(NodeObj));


    if (newNode == NULL) {
        fprintf(stderr, "Error: Failed to allocate memory in prepend()\n");
        exit(EXIT_FAILURE);
    }

    newNode->data = x;
    newNode->next = L->front;
    newNode->prev = NULL;

    if (L->length == 0) {
        L->front = L->back = newNode;
    } else {
        L->front->prev = newNode;
        L->front = newNode;
    }

    if (L->cursor != NULL) {
        L->index++;
    }
    L->length++;
}

void append(List L, int x) {
    if (L == NULL) {
        fprintf(stderr, "Error: List is NULL in append()\n");
        exit(EXIT_FAILURE);
    }

    Node newNode = malloc(sizeof(NodeObj));
    if (newNode == NULL) {
        fprintf(stderr, "Error: Memory allocation failed in append()\n");
        exit(EXIT_FAILURE);
    }

    newNode->data = x;
    newNode->next = NULL;
    newNode->prev = L->back;

    if (L->length == 0) {
        L->front = L->back = newNode;
    } else {
        L->back->next = newNode;
        L->back = newNode;
    }

    L->length++;
}

void insertBefore(List L, int x) {
    if (L == NULL || L->length == 0 || L->index == -1) {
        fprintf(stderr, "Error: Invalid List or cursor in insertBefore()\n");
        exit(EXIT_FAILURE);
    }



    Node newNode = malloc(sizeof(NodeObj));
    if (newNode == NULL) {
        fprintf(stderr, "Error: Memory allocation failed in insertBefore()\n");
        exit(EXIT_FAILURE);
    }

    newNode->data = x;
    newNode->prev = L->cursor->prev;
    newNode->next = L->cursor;

    if (L->cursor->prev != NULL) {
        L->cursor->prev->next = newNode;
    } else {
        L->front = newNode;
    }

    L->cursor->prev = newNode;
    L->length++;
    L->index++;
}

void insertAfter(List L, int x) {
    if (L == NULL || L->length == 0 || L->index == -1) {
        fprintf(stderr, "Error: Invalid List or cursor in insertAfter()\n");
        exit(EXIT_FAILURE);
    }

    Node newNode = malloc(sizeof(NodeObj));
    if (newNode == NULL) {
        fprintf(stderr, "Error: Memory allocation failed in insertAfter()\n");
        exit(EXIT_FAILURE);
    }

    newNode->data = x;
    newNode->next = L->cursor->next;
    newNode->prev = L->cursor;

    if (L->cursor->next != NULL) {
        L->cursor->next->prev = newNode;
    } else {
        L->back = newNode;
    }

    L->cursor->next = newNode;


    L->length++;
}

void deleteFront(List L) {
    if (L == NULL || L->length == 0) {
        fprintf(stderr, "Error: Invalid List or List is empty in deleteFront()\n");
        exit(EXIT_FAILURE);
    }

    Node temp = L->front;

    if (L->length == 1) {
        L->front = L->back = L->cursor = NULL;
        L->index = -1;
    } else {
        L->front = L->front->next;
        L->front->prev = NULL;
        if (L->cursor != NULL) {
            L->index--;
        }
    }

    free(temp);
    L->length--;
}

void deleteBack(List L) {
    if (L == NULL || L->length == 0) {
        fprintf(stderr, "Error: Invalid List or List is empty in deleteBack()\n");
        exit(EXIT_FAILURE);
    }

    Node temp = L->back;

    if (L->length == 1) {
        L->front = L->back = L->cursor = NULL;
        L->index = -1;
    } else {
        L->back = L->back->prev;
        L->back->next = NULL;
        if (L->cursor != NULL && L->index == L->length - 1) {
            L->cursor = NULL;
            L->index = -1;
        }
    }

    free(temp);
    L->length--;
}

void delete(List L) {
    if (L == NULL || L->length == 0 || L->index == -1) {
        fprintf(stderr, "Error: Invalid List or cursor in delete()\n");
        exit(EXIT_FAILURE);
    }

    Node temp = L->cursor;

    if (L->cursor->prev != NULL) {
        L->cursor->prev->next = L->cursor->next;
    } else {
        L->front = L->cursor->next;
    }

    if (L->cursor->next != NULL) {
        L->cursor->next->prev = L->cursor->prev;
    } else {
        L->back = L->cursor->prev;
    }

    free(temp);
    L->cursor = NULL;
    L->index = -1;
    L->length--;
}

// Other operations -----------------------------------------------------------

void printList(FILE* out, List L) {
    if (out == NULL || L == NULL) {
        fprintf(stderr, "Error: Invalid output or List in printList()\n");
        exit(EXIT_FAILURE);
    }

    Node N = L->front;
    while (N != NULL) {
        fprintf(out, "%d ", N->data);
        N = N->next;
    }
}

List copyList(List L) {
    if (L == NULL) {
        fprintf(stderr, "Error: List is NULL in copyList()\n");
        exit(EXIT_FAILURE);
    }

    List copy = newList();
    Node N = L->front;

    while (N != NULL) {
        append(copy, N->data);
        N = N->next;
    }

    return copy;
}

List concatList(List A, List B) {
    if (A == NULL || B == NULL) {
        fprintf(stderr, "Error: One or both Lists are NULL in concatList()\n");
        exit(EXIT_FAILURE);
    }

    List C = copyList(A);
    Node N = B->front;

    while (N != NULL) {
        append(C, N->data);
        N = N->next;
    }

    return C;
}
