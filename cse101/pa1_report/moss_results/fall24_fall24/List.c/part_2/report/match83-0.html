#include "List.h"
#include <stdlib.h>
#include <stdio.h>
#include <string.h>

typedef struct NodeObj {
	int data;
	struct NodeObj* prev;
	struct NodeObj* next;
} NodeObj;

typedef NodeObj* Node;

typedef struct ListObj {
	Node cursor;
	Node front;
	Node back;
	int length;
	int cursorIndex;
} ListObj;

List newList(void) {
	List L = malloc(sizeof(ListObj));
	L->cursor = NULL;
	L->front = NULL;
	L->back = NULL;
	L->length = 0;
	L->cursorIndex = -1;
	return L;
}



void freeList(List* pL) {
	if (pL != NULL && *pL != NULL) {
		clear(*pL);
		free(*pL);
		*pL = NULL;
	}
}

void deleteFront(List L) {
        if (L == NULL || L->length <= 0) {
                return;
        }

        Node temp = L->front;
        if (L->cursor == L->front) {
                L->cursor = NULL;
                L->cursorIndex = -1;
        } else if (L->cursor != NULL) {
		L->cursorIndex--;
	}

        if (L->front == L->back) {
                L->front = L->back = NULL;
        } else {
                L->front = L->front->next;
                L->front->prev = NULL;
        }

        free(temp);
        L->length--;
}


void clear(List L) {
	if (L == NULL) {
		return;
	}

	while (L->front != NULL) {
		deleteFront(L); // moved up, remember to delete commented out chuck below
	}

	L->length = 0;
	L->cursor =NULL;
	L->cursorIndex = -1;

}

void set(List L, int x) {
	if (L == NULL || L->cursor == NULL) {
		return;
	}

	L->cursor->data = x;
}



void prepend(List L, int x) {

	if (L == NULL) {
		return;
	}

	Node newNode = malloc(sizeof(NodeObj));
	if (newNode == NULL) {
		fprintf(stderr, "Memory allocation failed\n");
		exit(EXIT_FAILURE);
	}

	newNode->data = x;
	newNode->prev = NULL;
	newNode->next = L->front;

	if (L->front != NULL) {


		L->front->prev = newNode;
	} else {
		L->back = newNode;
	}

	L->front = newNode;
	L->length++;
	if (L->cursorIndex >= 0 ) {
		L->cursorIndex++;
	}
}

void append(List L, int x) {
	Node newNode = malloc(sizeof(NodeObj));
	newNode->data = x;
	newNode->next = NULL;
	newNode->prev = L->back;

	if (L->back != NULL) {


		L->back->next = newNode;
	} else {
		L->front = newNode;
	}

	L->back = newNode;
	L->length++;
}

void moveFront(List L) {
	if (L->length > 0) {
		L->cursor = L->front;
		L->cursorIndex = 0;
	}
}

void moveBack(List L) {
	if (L->length > 0) {
		L->cursor = L->back;
		L->cursorIndex = L->length - 1;
	}
}

void moveNext(List L) {
	if (L->cursor != NULL && L->cursor != L->back) {
		L->cursor = L->cursor->next;
		L->cursorIndex++;
	} else if (L->cursor == L->back) {
		L->cursor = NULL;
		L->cursorIndex = -1;
	}
}

void movePrev(List L) {
	if (L->cursor != NULL && L->cursor != L->front) {
		L->cursor = L->cursor->prev;


		L->cursorIndex--;
	} else if (L->cursor == L->front) {
		L->cursor = NULL;
		L->cursorIndex = -1;
	}
}

int length(List L) {
	return L->length;
}

int index(List L) {
	return L->cursorIndex;
}

int front(List L) {
	if (L->length > 0) {
		return L->front->data;
	}
	return -1;
}

int back (List L) {
	if (L->length > 0) {
		return L->back->data;
	}
	return -1;
}

int get(List L) {
	if (L->length > 0 && L->cursor != NULL) {
		return L->cursor->data;
	}
	return -1;
}

bool equals(List A, List B) {
	if (A == NULL || B == NULL) {
		return false;
	}

	if (A->length != B->length) {
		return false;
	}

	Node nodeA = A->front;
	Node nodeB = B->front;

	while (nodeA != NULL && nodeB != NULL) {


	       if (nodeA->data != nodeB->data) {	
		       return false;
		}
	       nodeA = nodeA->next;
	       nodeB = nodeB->next;
	}
	return true;
}

void insertBefore(List L, int x) {

	if (L == NULL || L->cursor == NULL || L->length <= 0 || L->cursorIndex < 0) {
                return;
        }


	Node newNode = malloc(sizeof(NodeObj));
	if (newNode == NULL) {
		fprintf(stderr, "Memory allocation failed uh oh\n");
		exit(EXIT_FAILURE);
	}
        newNode->data = x;


	if (L->cursor == L->front) {
		free(newNode);
		prepend(L,x);
	} else {
		newNode->prev = L->cursor->prev;
		newNode->next = L->cursor;
		if (L->cursor->prev != NULL) {
			L->cursor->prev->next = newNode;
		}

		L->cursor->prev = newNode;
		L->length++;
		L->cursorIndex++;
	}
}

void insertAfter(List L, int x) {
	if (L == NULL || L->cursor == NULL || L->length <= 0 || L->cursorIndex < 0) {
		return;
	}

	Node newNode = malloc(sizeof(NodeObj));
	if (newNode == NULL) {
		fprintf(stderr, "Failed allocating memory nooo\n");
		exit(EXIT_FAILURE);
	}
	newNode->data = x;

	if (L->cursor == L->back) {
		free(newNode);
		append(L, x);
	} else {
		newNode->next = L->cursor->next;
		newNode->prev = L->cursor;
		L->cursor->next->prev = newNode;
		L->cursor->next = newNode;
		L->length++;
	}
}

void deleteBack(List L) {
	if (L == NULL || L->length <= 0) {
		return;
	}

	Node temp = L->back;
	if (L->cursor == L->back) {
		L->cursor = NULL;
		L->cursorIndex = -1;
	}

	if (L->front == L->back) {
		L->front = L->back = NULL;
	} else {
		L->back = L->back->prev;
		L->back->next = NULL;
	}

	free(temp);
	L->length--;
}


void delete(List L) {
	if (L == NULL || L->cursor == NULL || L->length <= 0 || L->cursorIndex < 0) {
		return;
	}

	if (L->cursor == L->front) {
		deleteFront(L);
	} else if (L->cursor == L->back) {
		deleteBack(L);
	} else {
		Node temp = L->cursor;
		temp->prev->next = temp->next;
		temp->next->prev = temp->prev;
		free(temp);
		L->length--;


		L->cursor = NULL;
		L->cursorIndex  = -1;
	}
}


void printList(FILE* out, List L) {
	if (L == NULL || out == NULL) {
		return;
	}

	Node current = L->front;
		while (current != NULL) {
			fprintf(out, "%d", current->data);
			current = current->next;
			if (current != NULL) {
				fprintf(out, " ");
			}
		}
	}

List copyList(List L) {
	if (L == NULL) {
		return NULL;
	}

	List newL = newList();

	Node current = L->front;
	while (current != NULL) {
		append(newL, current->data);
		current = current->next;
	}

	return newL;
}
