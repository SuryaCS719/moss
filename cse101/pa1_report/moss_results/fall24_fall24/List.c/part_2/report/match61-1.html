#include "List.h"
// structs --------------------------------------------------------------------
// private NodeObj definition


typedef struct NodeObj{
    int data;
    struct NodeObj* next;
    struct NodeObj* prev;
} NodeObj;
// private NodeObj pointer
typedef struct NodeObj* Node;
// private ListObj definition
typedef struct ListObj{
    Node front;
    Node back;
    Node cursor;
    int length;
    int index;
} ListObj;
// Constructors-Destructors ---------------------------------------------------
// List constructor, sets all fields to default values
List newList(void){
    List L = malloc(sizeof(ListObj));
    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->length = 0;
    L->index = -1;
    return L;
}
// List destructor, deletes all nodes and frees all associated list memory if not undefined
void freeList(List* list){


    if(list != NULL && *list != NULL){
        while(length(*list) > 0){
            deleteFront(*list);
        }
        free(*list);
        *list = NULL;
    }
}
// Node constructor, sets fields to default values
Node newNode(int data){
    Node newnode = malloc(sizeof(NodeObj));
    newnode->data = data;
    newnode->next = NULL;
    newnode->prev = NULL;
    return newnode;
}
// Node destructor, frees allocated memory if node is not already undefined


void freeNode(Node* node){
    if(node != NULL && *node != NULL){
        free(*node);
        *node = NULL;
    }
}
// Access functions -----------------------------------------------------------
// Length access, returns number of elements
int length(List L){
    return L->length;
}
// Index access, returns current cursor index. -1 if undefined
int index(List L){
    return L->index;
}
// Front of list access, returns first element of list
// Exceptions: Empty list
int front(List L){
    if(L->length <= 0){
        fprintf(stderr, "List Error: calling front() on empty List\n");
        exit(1);
    }
    return L->front->data;
}
// End of list access, returns last element of list
// Exceptions: Empty list
int back(List L){
    if(L->length <= 0){
        fprintf(stderr, "List Error: calling back() on empty List\n");
        exit(1);
    }
    return L->back->data;
}
// Cursor element access, returns element at cursor
// Exceptions: Empty list, undefined cursor
int get(List L){
    if(length(L) <= 0){
        fprintf(stderr, "List Error: calling get() on empty List\n");
        exit(1);
    }
    if(index(L) < 0){
        fprintf(stderr, "List Error: calling get() on undefined cursor\n");
        exit(1);
    }
    return L->cursor->data;
}
// List equivalency access, returns if list arguments are equal or not
bool equals(List A, List B){
    if(A->length != B->length){
        return false;
    }
    Node newnode = A->front;
    Node nodeB = B->front;
    while(newnode != NULL && nodeB != NULL){
        if(newnode->data != nodeB->data){
            return false;
        }
        newnode = newnode->next;


        nodeB = nodeB->next;
    }
    return true;
}
// Manipulation procedures ----------------------------------------------------
// Clear function, deletes all nodes and resets all fields to default values
void clear(List L){
    while(length(L) > 0){
        deleteFront(L);
    }
    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->length = 0;
    L->index = -1;
}
// Set function, overwrites cursor element's data with x
// Exceptions: Empty list, undefined cursor
void set(List L, int x){
    if(length(L) <= 0){
        fprintf(stderr, "List Error: calling set() on empty List\n");
        exit(1);
    }
    if(index(L) < 0){
        fprintf(stderr, "List Error: calling set() on undefined cursor\n");
        exit(1);
    }
    L->cursor->data = x;
}
// Move front cursor function, sets cursor to first list element. Otherwise, do nothing
void moveFront(List L){
    if(length(L) > 0){
        L->cursor = L->front;
        L->index = 0;
    }
}
// Move back cursor function, sets cursor to last element. Otherwise, do nothing
void moveBack(List L){
    if(length(L) > 0){
        L->cursor = L->back;
        L->index = L->length - 1;
    }
}
// Move previous cursor function, sets cursor to previous node. If cursor is at front, it becomes undefined
// If already undefined, do nothing
void movePrev(List L){
    if(L->cursor != NULL && L->cursor->prev != NULL){
        L->cursor = L->cursor->prev;
        L->index--;
    }else if(L->cursor != NULL && L->cursor->prev == NULL){
        L->cursor = NULL;
        L->index = -1;
    }else{
        return;
    }
}
// Move next cursor function, sets cursor to next node. If cursor is at back, it becomes undefined
// If already undefined, do nothing
void moveNext(List L){
    if(L->cursor != NULL && L->cursor->next != NULL){
        L->cursor = L->cursor->next;
        L->index++;
    }else if(L->cursor != NULL && L->cursor->next == NULL){
        L->cursor = NULL;
        L->index = -1;
    }else{
        return;
    }
}
// Prepend add function, inserts a new element x to list at the front. If empty list, add first element
void prepend(List L, int x){
    Node newnode = newNode(x);
    if(length(L) <= 0){
        L->front = newnode;
        L->back = newnode;
    }else{
        newnode->next = L->front;
        L->front->prev = newnode;
        L->front = newnode;
    }
    if(L->index >= 0){
        L->index++;
    }
    L->length++;
}
// Append add function, adds new element x to end of list. If empty list, add first element
void append(List L, int x){
    Node newnode = newNode(x);
    if(length(L) <= 0){
        L->front = newnode;
        L->back = newnode;
    }else{
        newnode->prev = L->back;
        L->back->next = newnode;
        L->back = newnode;
    }
    L->length++;
}
// InsertBefore add function, adds new element x before cursor location
// Excpetions: Empty list, undefined cursor
void insertBefore(List L, int x){
    if(length(L) <= 0){
        fprintf(stderr, "List Error: calling insertBefore() on empty List\n");
        exit(1);
    }
    if(index(L) < 0){
        fprintf(stderr, "List Error: calling insertBefore() on undefined cursor\n");
        exit(1);
    }
    Node newnode = newNode(x);
    if(L->cursor->prev != NULL){
        newnode->prev = L->cursor->prev;
        L->cursor->prev->next = newnode;
    }else{
        L->front = newnode;
    }
    newnode->next = L->cursor;
    L->cursor->prev = newnode;
    L->index++;
    L->length++;
}
// InsertAfter add function, adds new element x after cursor location
// Exceptions: Empty list, undefined cursor
void insertAfter(List L, int x){
    if(length(L) <= 0){
        fprintf(stderr, "List Error: calling insertAfter() on empty List\n");
        exit(1);
    }
    if(index(L) < 0){
        fprintf(stderr, "List Error: calling insertAfter() on undefined cursor\n");
        exit(1);
    }
    Node newnode = newNode(x);
    if(L->cursor->next != NULL){
        newnode->next = L->cursor->next;
        L->cursor->next->prev = newnode;
    }else{
        L->back = newnode;
    }
    newnode->prev = L->cursor;
    L->cursor->next = newnode;
    L->length++;
}
// DeleteFront delete function, deletes first element in list
// Exceptions: Empty list
void deleteFront(List L){
    if(length(L) <= 0){
        fprintf(stderr, "List Error: calling deleteFront() on empty List\n");
        exit(1);
    }
    Node temp = L->front;
    if(L->cursor == L->front){
        L->cursor = NULL;
        L->index = -1;
    }
    if(length(L) > 1){
        L->front = L->front->next;
        L->front->prev = NULL;
    }else{
        L->front = NULL;
        L->back = NULL;
    }
    freeNode(&temp);
    L->length--;
    if(L->index >= 0){
        L->index--;
    }
}

// DeleteBack delete function, deletes last element in list
// Exceptions: Empty list
void deleteBack(List L){
    if(length(L) <= 0){
        fprintf(stderr, "List Error: calling deleteBack() on empty List\n");
        exit(1);
    }
    Node temp = L->back;
    if(L->cursor == L->back){
        L->cursor = NULL;
        L->index = -1;
    }
    if(length(L) > 1){
        L->back = L->back->prev;
        L->back->next = NULL;
    }else{
        L->front = NULL;
        L->back = NULL;
    }
    freeNode(&temp);
    L->length--;
}

// Delete function, deletes element at cursor location
// Exceptions: Empty list, undefined cursor
void delete(List L){
    if(length(L) <= 0){
        fprintf(stderr, "List Error: calling delete() on empty List\n");
        exit(1);
    }
    if(index(L) < 0){
        fprintf(stderr, "List Error: calling delete() on undefined cursor\n");
        exit(1);
    }
    Node temp = L->cursor;
    if(L->cursor->prev != NULL){
        L->cursor->prev->next = L->cursor->next;
    } else {
        L->front = L->cursor->next;
    }


    if(L->cursor->next != NULL){
        L->cursor->next->prev = L->cursor->prev;
    } else {
        L->back = L->cursor->prev;
    }
    // if(L->cursor->prev != NULL){
    //     L->cursor->prev->next = L->cursor->next;
    //     L->cursor->next->prev = L->cursor->prev;
    // }else{
    //     L->front = L->cursor->next;
    // }
    L->cursor = NULL;
    L->index = -1;
    freeNode(&temp);
    L->length--;
}

// Other operations -----------------------------------------------------------
// PrintList accessory function, prints list contents separated by spaces
void printList(FILE* out, List L){
    Node iterator = L->front;
    while(iterator != NULL){
        fprintf(out, "%d ", iterator->data);
        iterator = iterator->next;
    }
}

// CopyList accessory function, copies L in its entirety to a new list
List copyList(List L){
    List newlist = newList();
    Node iterator = L->front;
    while(iterator != NULL){
        append(newlist, iterator->data);
        iterator = iterator->next;
    }
    newlist->cursor = NULL;
    newlist->index = -1;
    return newlist;
}
