#include <stdio.h>
#include <stdlib.h>
#include "List.h"

typedef struct NodeObj {
    int data;
    struct NodeObj* next;
    struct NodeObj* prev;
} NodeObj;

typedef NodeObj* Node;

typedef struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
    int index;
} ListObj;

// Constructors and Destructors
List newList(void) {
    List L = malloc(sizeof(ListObj));
    L->front = L->back = L->cursor = NULL;
    L->length = 0;
    L->index = -1;
    return L;
}

void freeList(List* pL) {


    if (pL != NULL && *pL != NULL) {
        while (length(*pL) > 0) {
            deleteFront(*pL);  // Safely delete front element
        }
        free(*pL);  // Free the List structure itself
        *pL = NULL;
    }
}

void printError(const char* message) {
    fprintf(stderr, "%s\n", message);
}

// Access functions
int length(List L) {
    return L->length;
}

int index(List L) {
    return L->index;
}

int front(List L) {
    if (L == NULL) {
        printError("List is NULL");
        exit(EXIT_FAILURE); // You can also handle this differently
    }
    
    if (L->length == 0) {
        printError("List is empty. Cannot access front.");
        return -1; // Or handle this case as needed
    }
    
    return L->front->data; // Return the front element
}

// Get the back element of the list
int back(List L) {
    if (L == NULL) {
        printError("List is NULL");
        exit(EXIT_FAILURE);
    }
    
    if (L->length == 0) {
        printError("List is empty. Cannot access back.");
        return -1;
    }
    
    return L->back->data; // Return the back element
}

// Get the cursor element of the list
int get(List L) {
    if (L == NULL) {
        printError("List is NULL");
        exit(EXIT_FAILURE);
    }
    
    if (L->length == 0 || L->index < 0) {
        printError("List is empty or cursor is undefined. Cannot access element.");
        return -1;
    }
    
    return L->cursor->data; // Return the cursor element
}

// Manipulation procedures
void clear(List L) {
    while (L->length > 0) {
        deleteFront(L);
    }
    L->cursor = NULL;
    L->index = -1;
}

void moveFront(List L) {
    if(L->length > 0) {
        L->cursor = L->front;
        L->index = 0;
    }
}

void moveBack(List L) {
    if(L->length > 0) {
        L->cursor = L->back;
        L->index = L->length - 1;
    }
}

void movePrev(List L) {
    if(L->cursor != NULL) {
        if(L->cursor == L->front) {
            L->cursor = NULL;
            L->index = -1;
        } else {
            L->cursor = L->cursor->prev;
            L->index--;
        }
    }
}

void moveNext(List L) {
    if(L->cursor != NULL) {
        if(L->cursor == L->back) {
            L->cursor = NULL;
            L->index = -1;
        } else {
            L->cursor = L->cursor->next;
            L->index++;
        }
    }
}

void prepend(List L, int x) {
    Node N = malloc(sizeof(NodeObj));
    N->data = x;
    N->prev = NULL;
    N->next = L->front;

    if (L->front != NULL) {


        L->front->prev = N;
    }
    L->front = N;

    if (L->length == 0) {
        L->back = N;
        L->cursor = NULL;
        L->index = -1;   
    } else {
        if (L->index >= 0) {
            L->index++;
        }
    }
    L->length++;
}

void append(List L, int x) {
    Node N = malloc(sizeof(NodeObj));
    N->data = x;
    N->next = NULL;
    N->prev = L->back;
    if(L->back != NULL) L->back->next = N;


    L->back = N;
    if(L->length == 0) L->front = N;
    L->length++;
}

void insertBefore(List L, int x) {
    if (L == NULL || L->length == 0 || L->index < 0) {
        fprintf(stderr, "Cannot insert before: List is empty or cursor is undefined.\n");
        return;
    }

    Node N = malloc(sizeof(NodeObj));
    if (N == NULL) {
        fprintf(stderr, "Memory allocation failed\n");
        exit(EXIT_FAILURE);
    }
    N->data = x;

    if (L->cursor == L->front) {
        N->next = L->front;
        N->prev = NULL;
        L->front->prev = N;
        L->front = N;
    } else {
        N->prev = L->cursor->prev;
        N->next = L->cursor;

        if (L->cursor->prev != NULL) {
            L->cursor->prev->next = N;
        }
        L->cursor->prev = N;

        if (L->cursor == L->back) {
            L->back = N;
        }
    }

    L->length++;
    L->index++;
}

void insertAfter(List L, int x) {
    if(L->length > 0 && L->index >= 0) {
        Node N = malloc(sizeof(NodeObj));
        N->data = x;
        N->prev = L->cursor;
        N->next = L->cursor->next;


        if(L->cursor->next != NULL) L->cursor->next->prev = N;
        L->cursor->next = N;
        if(L->cursor == L->back) L->back = N;
        L->length++;
    }
}

void deleteFront(List L) {
    if (L->length > 0) {
        printf("Before deleteFront: front = %d\n", front(L)); // Debug print

        Node temp = L->front;
        if (L->length == 1) {
            L->front = L->back = NULL;
            L->cursor = NULL;
            L->index = -1;
        } else {
            L->front = L->front->next;
            L->front->prev = NULL;
        }
        free(temp);
        L->length--;
        if (L->index == 0) {
            L->cursor = NULL;
            L->index = -1;
        } else if (L->index != -1) {
            L->index--;
        }

        printf("After deleteFront: front = %d, length = %d\n", (L->front ? L->front->data : -1), L->length); // Debug print
    }
}
void deleteBack(List L) {
    if(L->length > 0) {
        Node temp = L->back;
        if(L->length == 1) {
            L->front = L->back = NULL;
        } else {


            L->back = L->back->prev;
            L->back->next = NULL;
        }
        free(temp);
        L->length--;
        if(L->index == L->length) {
            L->cursor = NULL;
            L->index = -1;
        }
    }
}

void delete(List L) {
    if (L->length > 0 && L->index >= 0) {
        if (L->cursor == L->front) {
            deleteFront(L); // No need to set cursor to NULL here; handled in deleteFront
            return;
        } 
        if (L->cursor == L->back) {
            deleteBack(L);  // Handled in deleteBack
            return;
        } 
        // For deleting a node in the middle
        Node temp = L->cursor;
        L->cursor->prev->next = L->cursor->next;
        L->cursor->next->prev = L->cursor->prev;
        free(temp);
        L->length--;
        L->cursor = NULL;
        L->index = -1;
    }
}

// Other operations
bool equals(List A, List B) {
    if(A->length != B->length) return false;
    Node nA = A->front;
    Node nB = B->front;
    while(nA != NULL && nB != NULL) {


        if(nA->data != nB->data) return false;
        nA = nA->next;
        nB = nB->next;
    }
    return true;
}

List copyList(List L) {
    List C = newList();
    Node N = L->front;
    while(N != NULL) {
        append(C, N->data);
        N = N->next;
    }
    return C;
}

void printList(FILE* out, List L) {
    Node N = L->front;
    while(N != NULL) {
        fprintf(out, "%d ", N->data);
        N = N->next;
    }
    fprintf(out, "\n");
}
