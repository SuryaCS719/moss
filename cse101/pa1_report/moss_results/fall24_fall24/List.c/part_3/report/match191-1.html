#include<stdio.h>
#include<stdlib.h>
#include<assert.h>
#include<stdbool.h>
#include "List.h"

// structs --------------------------------------------------------------------

// private Node type
typedef struct NodeObj* Node;

// private NodeObj type
typedef struct NodeObj{
   ListElement data;
   Node next;
   Node prev;
} NodeObj;

// private ListObj type
typedef struct ListObj{
   Node front;
   Node back;
   int length;
   Node cursor;
   int cursorIndex;
} ListObj;


// Constructors-Destructors ---------------------------------------------------

// newNode()
// Returns reference to new Node object. Initializes next and data fields.
Node newNode(ListElement data){
   Node N = malloc(sizeof(NodeObj));
   assert( N != NULL );
   N->data = data;
   N->next = NULL;
   N->prev = NULL;
   return(N);
}

// freeNode()
// Frees heap memory pointed to by *pN, sets *pN to NULL.
void freeNode(Node* pN){
   if(pN != NULL && *pN != NULL){
      free(*pN);
      *pN = NULL;
   }
}

// newList()
// Returns reference to new empty List object.
List newList(){
   List L = malloc(sizeof(ListObj));
   assert(L != NULL);
   L->front = L->back = NULL;
   L->length = 0;
   L->cursor = NULL;
   L->cursorIndex = -1;
   return L;
}

// freeList()
// Frees all heap memory associated with List *pL, and sets *pQ to NULL.
void freeList(List* pL){
   if(pL != NULL && *pL != NULL){
      clear(*pL);
      free(*pL);
      *pL = NULL;
   }
}

// Access functions -----------------------------------------------------------
int length(List L){
    if (L == NULL) {
      fprintf(stderr, "List error: calling length() on NULL List reference\n");
    }
    return L -> length;
}

int index(List L) {
    if (L == NULL) {
      fprintf(stderr, "List error: calling index() on NULL List reference\n");
    }
    return L->cursorIndex;
}

int front(List L) {
    if (L == NULL || L->length == 0) {
      fprintf(stderr, "List error: calling front() on NULL or empty List reference\n");
    }
    return L->front->data;
}

int back(List L) {
    if (L == NULL || L->length == 0) {
      fprintf(stderr, "List error: calling back() on NULL or empty List reference\n");
    }
    return L->back->data;
}

int get(List L) {
    if (L == NULL || L->cursor == NULL) {
      printf("stderr: calling get() on NULL or undefined cursor\n");
    }
    return L->cursor->data;
}

bool equals(List A, List B) {
    if (A == NULL || B == NULL) {
      fprintf(stderr, "List Error: calling equals() on NULL List reference\n");
    }
    if (A->length != B->length) {
        return false;
    }

    Node N = A->front;
    Node M = B->front;
    while (N != NULL && M != NULL) {
        if (N->data != M->data) {
            return false;
        }
        N = N->next;
        M = M->next;
    }
    return true;
}

// Manipulation procedures ----------------------------------------------------
void clear(List L){
    if (L == NULL){
      fprintf(stderr, "List error: calling clear() on NULL List reference\n");
   }
   while (L->length > 0){


      deleteFront(L);
   }
   L->cursor = NULL;
   L->cursorIndex = -1;
}

void set(List L, int x) {
    if (L == NULL || L->cursor == NULL) {
      fprintf(stderr, "List error: calling set() on NULL List or undefined cursor\n"); 
   }
   L->cursor->data = x;
}

void moveFront(List L) {
    if (L == NULL || L->length == 0) {
        fprintf(stderr, "List Error: calling moveFront() on NULL or empty List\n");


        return;
    }
    L->cursor = L->front;
    L->cursorIndex = 0;
}

void moveBack(List L) {
    if (L == NULL || L->length == 0) {
        fprintf(stderr, "List Error: calling moveBack() on NULL or empty List\n");
        return;
    }


    L->cursor = L->back;
    L->cursorIndex = L->length - 1;
}

void movePrev(List L) {
    if (L == NULL || L->cursor == NULL || L->cursorIndex == 0) {
        L->cursor = NULL;
        L->cursorIndex = -1;
        return;
    }
    L->cursor = L->cursor->prev;
    L->cursorIndex--;
}

void moveNext(List L) {
    if (L == NULL || L->cursor == NULL || L->cursorIndex == L->length - 1) {
        L->cursor = NULL;
        L->cursorIndex = -1;
        return;
    }
    L->cursor = L->cursor->next;
    L->cursorIndex++;
}

// prepend()
void prepend(List L, int x) {
    Node N = newNode(x);
    if (L->length == 0) {
        L->front = L->back = N;
    } else {
        N->next = L->front;
        L->front->prev = N;
        L->front = N;
    }
    L->length++;
    if (L->cursorIndex != -1) {
        L->cursorIndex++;
    }
}

// Insert new element into L. If L is non-empty, insertion takes place after back element.
void append(List L, int x) {
    Node N = newNode(x);
    if (L->length == 0) {
        L->front = L->back = N;
    } else {
        N->prev = L->back;


        L->back->next = N;
        L->back = N;
    }
    L->length++;
}

void insertBefore(List L, int x) {
    if (L == NULL || L->cursor == NULL) {
        fprintf(stderr, "List Error: calling insertBefore() on NULL List or undefined cursor\n");

    }
    if (L->cursor == L->front) {
        prepend(L, x);
    } else {
        Node N = newNode(x);
        N->prev = L->cursor->prev;
        N->next = L->cursor;
        L->cursor->prev->next = N;
        L->cursor->prev = N;


        L->length++;
        L->cursorIndex++;
    }
}
void insertAfter(List L, int x) {
    if (L == NULL || L->cursor == NULL) {
        fprintf(stderr, "List error: calling insertAfter() on NULL List or undefined cursor\n");

    }
    if (L->cursor == L->back) {
        append(L, x);
    } else {
        Node N = newNode(x);
        N->next = L->cursor->next;
        N->prev = L->cursor;
        L->cursor->next->prev = N;


        L->cursor->next = N;
        L->length++;
    }
}

void deleteFront(List L) {
    if (L == NULL || L->length == 0) {
        fprintf(stderr, "List Error: calling deleteFront() on NULL or empty List\n");

    }
    Node N = L->front;
    if (L->length > 1) {
        L->front = L->front->next;
        L->front->prev = NULL;
    } else {
        L->front = L->back = NULL;
    }
    L->length--;
    if (L->cursorIndex == 0) {
        L->cursor = NULL;
        L->cursorIndex = -1;
    } else if (L->cursorIndex != -1) {
        L->cursorIndex--;
    }
    freeNode(&N);
}

void deleteBack(List L) {
    if (L == NULL || L->length == 0) {
        fprintf(stderr, "List Error: calling deleteBack() on NULL or empty List\n");

    }
    Node N = L->back;
    if (L->length > 1) {
        L->back = L->back->prev;
        L->back->next = NULL;
    } else {
        L->front = L->back = NULL;
    }
    L->length--;
    if (L->cursorIndex == L->length) {
        L->cursor = NULL;
        L->cursorIndex = -1;
    }
    freeNode(&N);
}

void delete(List L) {
    if (L == NULL || L->cursor == NULL) {
        fprintf(stderr, "List Error: calling delete() on NULL List or undefined cursor\n");

    }
    if (L->cursor == L->front) {
        deleteFront(L);
    } else if (L->cursor == L->back) {
        deleteBack(L);
    } else {
        Node N = L->cursor;
        N->prev->next = N->next;
        N->next->prev = N->prev;
        freeNode(&N);
        L->length--;
        L->cursor = NULL;
        L->cursorIndex = -1;
    }
}

// Other Functions ------------------------------------------------------------
void printList(FILE* out, List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling printList() on NULL List reference\n");

    }
    Node N = L->front;
    while (N != NULL) {


        fprintf(out, "%d ", N->data);
        N = N->next;
    }
    fprintf(out, "\n");
}

List copyList(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling copyList() on NULL List reference\n");

    }
    List C = newList();
    Node N = L->front;
    while (N != NULL) {
        append(C, N->data);
        N = N->next;
    }
    return C;
}
