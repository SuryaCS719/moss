#include <stdio.h>
#include <stdlib.h>
#include "List.h"

// Private for Node
typedef struct NodeObj {
    int data;
    struct NodeObj* next;
    struct NodeObj* prev;
} NodeObj;

typedef NodeObj* Node;

// Private for List
struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
    int index;
};

// Construct/Destruct

List newList(void) {
    List L = malloc(sizeof(struct ListObj));
    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->length = 0;
    L->index = -1;
    return L;
}

void freeList(List* pL) {
    if (pL != NULL && *pL != NULL) {
        clear(*pL);
        free(*pL);
        *pL = NULL;
    }
}

// Access

int length(List L) {
    return L->length;
}

int index(List L) {
    return L->index;
}

int front(List L) {
    if (L->length > 0) {
        return L->front->data;
    }
    fprintf(stderr, "Error: List is empty. Can't access front.\n");
    return -1;
}

int back(List L) {
    if (L->length > 0) {
        return L->back->data;
    }
    fprintf(stderr, "Error: List is empty. Can't access back.\n");
    return -1;
}

int get(List L) {
    if (L->length > 0 && L->index >= 0) {
        return L->cursor->data;
    }
    fprintf(stderr, "Error: List is either empty or cursor is undefined. Can't use get.\n");
    return -1;
}

bool equals(List A, List B) {
    if (A == NULL || B == NULL || A->length != B->length) return false;
    Node N = A->front;
    Node M = B->front;
    while (N != NULL && M != NULL) {
        if (N->data != M->data) return false;
        N = N->next;
        M = M->next;
    }
    return true;
}

// Manipulations

void clear(List L) {
    while (L->length > 0) {
        deleteFront(L);
    }
}



void set(List L, int x) {
    if (L->length > 0 && L->index >= 0) {
        L->cursor->data = x;
    }
}

void moveFront(List L) {
    if (L->length > 0) {
        L->cursor = L->front;
        L->index = 0;
    }
}

void moveBack(List L) {
    if (L->length > 0) {
        L->cursor = L->back;
        L->index = L->length - 1;
    }
}

void movePrev(List L) {
    if (L->cursor != NULL && L->index > 0) {
        L->cursor = L->cursor->prev;
        L->index--;
    } else if (L->cursor != NULL && L->index == 0) {
        L->cursor = NULL;
        L->index = -1;
    }
}

void moveNext(List L) {
    if (L->cursor != NULL && L->index < L->length - 1) {
        L->cursor = L->cursor->next;
        L->index++;
    } else if (L->cursor != NULL && L->index == L->length - 1) {
        L->cursor = NULL;
        L->index = -1;
    }
}

void prepend(List L, int x) {
    Node N = malloc(sizeof(NodeObj));
    N->data = x;
    N->next = L->front;


    N->prev = NULL;
    if (L->front != NULL) L->front->prev = N;
    L->front = N;
    if (L->back == NULL) L->back = N;
    L->length++;
    if (L->index != -1) L->index++;
}

void append(List L, int x) {
    Node N = malloc(sizeof(NodeObj));
    N->data = x;
    N->next = NULL;
    N->prev = L->back;
    if (L->back != NULL) L->back->next = N;
    L->back = N;
    if (L->front == NULL) L->front = N;


    L->length++;
}

void insertBefore(List L, int x) {
    if (L->length > 0 && L->index >= 0) {
        Node N = malloc(sizeof(NodeObj));
        N->data = x;
        N->prev = L->cursor->prev;
        N->next = L->cursor;
        if (L->cursor->prev != NULL) L->cursor->prev->next = N;
        L->cursor->prev = N;
        if (L->index == 0) L->front = N;
        L->index++;
        L->length++;
    }
}

void insertAfter(List L, int x) {


    if (L->length > 0 && L->index >= 0) {
        Node N = malloc(sizeof(NodeObj));
        N->data = x;
        N->next = L->cursor->next;
        N->prev = L->cursor;
        if (L->cursor->next != NULL) L->cursor->next->prev = N;
        L->cursor->next = N;
        if (L->index == L->length - 1) L->back = N;
        L->length++;
    }
}

void deleteFront(List L) {
    if (L->length > 0) {
        Node N = L->front;
        if (L->length > 1) {
            L->front = L->front->next;
            L->front->prev = NULL;
        } else {
            L->front = L->back = NULL;
        }
        free(N);
        L->length--;
        if (L->index == 0) {
            L->cursor = NULL;
            L->index = -1;
        } else if (L->index != -1) {
            L->index--;
        }
    }
}

void deleteBack(List L) {
    if (L->length > 0) {
        Node N = L->back;
        if (L->length > 1) {
            L->back = L->back->prev;
            L->back->next = NULL;
        } else {
            L->front = L->back = NULL;
        }
        free(N);
        L->length--;
        if (L->index == L->length) {
            L->cursor = NULL;
            L->index = -1;
        }
    }
}



void delete(List L) {
    if (L->length > 0 && L->index >= 0) {
        Node N = L->cursor;
        if (N->prev != NULL) N->prev->next = N->next;
        if (N->next != NULL) N->next->prev = N->prev;
        if (N == L->front) L->front = N->next;
        if (N == L->back) L->back = N->prev;
        L->cursor = NULL;
        L->index = -1;
        free(N);
        L->length--;
    }
}

// print & other functions

void printList(FILE* out, List L) {
    Node N = L->front;
    while (N != NULL) {
        fprintf(out, "%d ", N->data);
        N = N->next;
    }
}

List copyList(List L) {
    List C = newList();
    Node N = L->front;
    while (N != NULL) {
        append(C, N->data);
        N = N->next;
    }
    return C;
}

List concatList(List A, List B) {
    List C = copyList(A);
    Node N = B->front;
    while (N != NULL) {
        append(C, N->data);
        N = N->next;
    }
    return C;
}
