#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>
#include "List.h"



Node newNode(int data) {
	Node N = malloc(sizeof(NodeObj));
	if (N == NULL) {
		printf("Error : malloc failed in newNode()\n");
		exit(1);
	}
	N->data = data;
	N->next = NULL;
	N->prev = NULL;
	return N;
}
void freeNode(Node* pN) {
    if (pN != NULL && *pN != NULL) {
        free(*pN);
        *pN = NULL;
    }
}


List newList(void) {
    List L = malloc(sizeof(ListObj));
    if (L == NULL) {
        printf("Error: malloc failed in newList()\n");
        exit(1);
    }
    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->length = 0;
    L->index = -1;
    return L;
}

// Free an existing list
void freeList(List* pL) {
    if (pL != NULL && *pL != NULL) {


        while (length(*pL) > 0) {
            deleteFront(*pL);
        }
        free(*pL);
        *pL = NULL;
    }
}

int length(List L) {
	if (L == NULL) {
		printf("List Error: calling length() on NULL List reference\n");
		exit(1);
	}
	return L->length;
}

int index(List L) {
	if (L->cursor == NULL) {
		return -1;
	}
	return L->index;
}

int front(List L) {
    if (L == NULL || L->length == 0) {
        printf("List Error: calling front() on empty List\n");
        exit(1);
    }
    return L->front->data;
}

int back(List L) {
    if (L == NULL || L->length == 0) {
        printf("List Error: calling back() on empty List\n");
        exit(1);
    }
    return L->back->data;
}

int get(List L) {
    if (L == NULL || L->cursor == NULL) {
        printf("List Error: calling get() with an undefined cursor\n");
        exit(1);
    }
    return L->cursor->data;
}

void clear(List L) {
    while (L->length > 0) {
        deleteFront(L);
    }
    L->cursor = NULL;
    L->index = -1;
}

void prepend(List L, int x) {
    Node N = newNode(x);
    if (L->length == 0) {
        L->front = L->back = N;
    } else {
        N->next = L->front;
        L->front->prev = N;
        L->front = N;
    }


    L->length++;
    if (L->index != -1) {
        L->index++;
    }
}

void append(List L, int x) {
    Node N = newNode(x);
    if (L->length == 0) {
        L->front = L->back = N;
    } else {
        L->back->next = N;
        N->prev = L->back;
        L->back = N;
    }
    L->length++;
}

void deleteFront(List L) {
    if (L->length == 0) {
        printf("List Error: calling deleteFront() on an empty List\n");
        exit(1);
    }
    Node temp = L->front;
    if (L->length > 1) {
        L->front = L->front->next;
        L->front->prev = NULL;
    } else {
        L->front = L->back = NULL;
    }
    freeNode(&temp);
    L->length--;
    if (L->index != -1) {
        L->index--;
    }
}

void deleteBack(List L) {
	if (L == NULL || L->length == 0) {
        printf("List Error: calling deleteBack() on an empty List\n");
        exit(1);
    }

    Node temp = L->back;

     if (L->cursor == L->back) {
        L->cursor = NULL;
        L->index = -1;
    }

     if (L->length > 1) {
        L->back = L->back->prev;


        L->back->next = NULL;
    } else {
        L->front = L->back = NULL;
    }

    freeNode(&temp);
    L->length--;
}


void delete(List L) {
    if (L == NULL || L->cursor == NULL) {
        printf("List Error: calling delete() on NULL List reference or with undefined cursor\n");
        exit(1);
    }
    Node temp = L->cursor;

    if (L->cursor == L->front) {
        deleteFront(L);
    }
    else if (L->cursor == L->back) {
        deleteBack(L);
    }
    else {
        L->cursor->prev->next = L->cursor->next;


        L->cursor->next->prev = L->cursor->prev;
        freeNode(&temp);
        L->length--;
    }

    L->cursor = NULL;
    L->index = -1;
}


void moveFront(List L) {
    if (L == NULL) {
        printf("List Error: calling moveFront() on NULL List reference\n");
        exit(1);
    }
    if (L->length > 0) {
        L->cursor = L->front;
        L->index = 0;
    }
}

void moveBack(List L) {
    if (L == NULL) {
        printf("List Error: calling moveBack() on NULL List reference\n");
        exit(1);
    }
    if (L->length > 0) {
        L->cursor = L->back;
        L->index = L->length - 1;
    }
}

void moveNext(List L) {
    if (L == NULL) {
        printf("List Error: calling moveNext() on NULL List reference\n");
        exit(1);
    }
    if (L->cursor != NULL) {
        if (L->cursor == L->back) { 
            L->cursor = NULL;
            L->index = -1;
        } else { 
            L->cursor = L->cursor->next;
            L->index++;
        }
    }
}

void insertBefore(List L, int x) {
    if (L == NULL || L->cursor == NULL) {
        printf("List is empty or cursor is not set.\n");
        return;
    }

    
    NodeObj* newNode = malloc(sizeof(NodeObj));
    if (newNode == NULL) {
        printf("Memory allocation failed.\n");
        return;
    }
    newNode->data = x;

    NodeObj* current = L->cursor;

    if (current == L->front) {
        newNode->next = L->front;  
        newNode->prev = NULL;       
        L->front->prev = newNode;   
        L->front = newNode;         
    } else {
        newNode->next = current;     
        newNode->prev = current->prev;

        current->prev->next = newNode;
        current->prev = newNode;
    }

    L->length++;
    L->index++; 
}

void insertAfter(List L, int x) {
	if (L == NULL || L->cursor == NULL) {
        printf("List Error: calling insertAfter() on NULL List reference or with undefined cursor\n");
        exit(1);
    }
    Node N = newNode(x);
    Node temp = L->cursor;
    if (temp == L->back) {  // Insert at the back
        L->back->next = N;
        N->prev = L->back;
        L->back = N;
    } else {  // Insert in the middle
        N->next = temp->next;
        N->prev = temp;
        temp->next->prev = N;
        temp->next = N;
    }
    L->length++;
}

void movePrev(List L) {
    if (L == NULL) {
        printf("List Error: calling movePrev() on NULL List reference\n");
        exit(1);
    }
    if (L->cursor != NULL) {
        if (L->cursor == L->front) { 
            L->cursor = NULL;
            L->index = -1;
        } else {  
            L->cursor = L->cursor->prev;
            L->index--;
        }
    }
}

void printList(FILE* out, List L) {
    if (L == NULL) {
        printf("List Error: calling printList() on NULL List reference\n");
        exit(1);
    }
    Node temp = L->front;
    while (temp != NULL) {
        fprintf(out, "%d ", temp->data);
        temp = temp->next;
    }
}

List copyList(List L) {
    if (L == NULL) {
        printf("List Error: calling copyList() on NULL List reference\n");
        exit(1);
    }
    List newL = newList();  
    Node temp = L->front;
    while (temp != NULL) {
        append(newL, temp->data);
        temp = temp->next;
    }
    return newL;
}

bool equals(List A, List B) {
	if (A == NULL || B == NULL) {
        printf("List Error: calling equals() on NULL List reference\n");
        exit(1);
    }
    if (A->length != B->length) return 0;
    Node nodeA = A->front;
    Node nodeB = B->front;
    while (nodeA != NULL && nodeB != NULL) {
        if (nodeA->data != nodeB->data) return 0;
        nodeA = nodeA->next;
        nodeB = nodeB->next;
    }
    return 1;
}
