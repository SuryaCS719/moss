#include<stdio.h>
#include<stdlib.h>
#include<assert.h>
#include<stdbool.h>
#include "List.h"



typedef struct NodeObj* Node;

typedef struct ListObj* List;

typedef struct NodeObj {
   ElementKey data;
   Node next;
   Node prev;
} NodeObj;

typedef struct ListObj {
   Node front;
   Node back;
   Node cursor;
   int length;
   int Listindex;
} ListObj;

Node newNode(ElementKey data) {
   Node N;
   N = malloc(sizeof(NodeObj));
   if (N == NULL) {
      return NULL;
   }
   N->data = data;
   N->next = NULL;
   N->prev = NULL;
   return(N);
}

void freeNode(Node* pN) {
   if (pN != NULL && *pN != NULL) {
      free(*pN);
      *pN = NULL;
   }
}

List newList(void){
    List L;
    L = malloc(sizeof(ListObj));
    if (L == NULL) {
      return NULL;
   }
    L->front = NULL;
    L->back = NULL;
    L->length = 0;
    return(L);
}


void freeList(List* pL) {
    if (pL != NULL && *pL != NULL) {
        Node current = (*pL)->front;
        Node nextNode;

        while (current != NULL) {
            nextNode = current->next;
            free(current);
            current = nextNode;
        }

        free(*pL);
        *pL = NULL;
    }
}


int length(List L) {
    if (L == NULL) {
        return -1;
    }
    return L->length;
}

int index(List L) {
    if (L != NULL) {
        int idx = 0;
        Node current = L->front;
        
        while (current != NULL) {
            if (current == L->cursor) {
                return idx;
            }
            current = current->next;
            idx++;
        }
    }
    return -1; 
}

int front(List L) {
    if (L == NULL || L->length <= 0) {
        return -1;
    }
    return L->front->data;
}

int back(List L) {
    if (L == NULL || L->length <= 0) {
        return -1;
    }
    return L->back->data;
}

int get(List L) {
    if (length(L) > 0 && index(L) >= 0) {
        Node current = L->front;

        for (int i = 0; i < index(L); i++) {
            current = current->next;
        }

        return current->data;
    } else {
        return -1;
    }
}

bool equals(List A, List B){
    if( A==NULL || B==NULL ){
      return false;
   }

   bool eq;
   Node N, M;

   eq = ( A->length == B->length );
   N = A->front;
   M = B->front;
   while( eq && N!=NULL){


      eq = ( N->data==M->data );
      N = N->next;
      M = M->next;
   }
   return eq;
}

void clear(List L) {
    if (L == NULL) {
        return;
    }

    Node current = L->front;
    Node nextNode;

    while (current != NULL) {
        nextNode = current->next;
        freeNode(&current);  
        current = nextNode;
    }

    L->front = L->back = NULL;
    L->length = 0;
    L->Listindex = -1;
    L->cursor = NULL;
}

void set(List L, int x){
    if (length(L) > 0 && index(L) >= 0) {
        L->cursor->data = x;
    }
}

void moveFront(List L) {
    if (length(L) > 0) {
        L->cursor = L->front;
        L->Listindex = 0; 
    } else {
        printf("List is empty. No action performed.\n");
    }
}

void moveBack(List L) {
    if (length(L) > 0) {
        
        L->cursor = L->back;
        L->Listindex = length(L) - 1; 
    }
    
}

void movePrev(List L) {
    if (L == NULL) {
        fprintf(stderr, "Error: calling movePrev() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }

    if (L->cursor != NULL && L->cursor != L->front) {
        
        L->cursor = L->cursor->prev;
        L->Listindex--; 


    } else if (L->cursor == L->front) {
        
        L->cursor = NULL;
        L->Listindex = -1;
    }
    
}

void moveNext(List L) {
    if (L == NULL) {
        fprintf(stderr, "Error: calling moveNext() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }

    if (L->cursor != NULL && L->cursor != L->back) {
        
        L->cursor = L->cursor->next;
        L->Listindex++; 


    } else if (L->cursor == L->back) {
        
        L->cursor = NULL;
        L->Listindex = -1;
    }
    
}

void append(List L, ElementKey x) {
    Node newNodePtr = newNode(x);
    newNodePtr->prev = L->back;
    if (L->back != NULL) {
        L->back->next = newNodePtr;
    } else {
        L->front = newNodePtr;
    }
    L->back = newNodePtr;
    L->length++;
}

void prepend(List L, ElementKey x) {
    Node newNodePtr = newNode(x);
    newNodePtr->next = L->front;
    if (L->front != NULL) {
        L->front->prev = newNodePtr;
    } else {
        L->back = newNodePtr;
    }
    L->front = newNodePtr;
    L->length++;
}


void insertBefore(List L, int x) {
    if (L->length < 0 || L->Listindex < 0) {
        fprintf(stderr, "Error: calling insertBefore() with invalid List state\n");
        exit(EXIT_FAILURE);
    }

    Node newNodePtr = newNode(x);

    if (L->cursor == L->front) {
        newNodePtr->next = L->front;
        newNodePtr->prev = NULL;
        if (L->front != NULL) {
            L->front->prev = newNodePtr;
        }
        L->front = newNodePtr;
    } else {
        newNodePtr->next = L->cursor;
        newNodePtr->prev = L->cursor->prev;
        L->cursor->prev->next = newNodePtr;


        L->cursor->prev = newNodePtr;
        L->Listindex++;
    }

    L->length++;
}

void insertAfter(List L, int x) {
    if (L->length < 0 || L->Listindex < 0 || L->Listindex >= L->length) {
        fprintf(stderr, "Error: calling insertAfter() with invalid List state\n");
        exit(EXIT_FAILURE);
    }

    Node newNodePtr = newNode(x);

    if (L->cursor == L->back) {
        newNodePtr->next = NULL;
        newNodePtr->prev = L->cursor;
        L->cursor->next = newNodePtr;
        L->back = newNodePtr;
        
    } else {
        newNodePtr->next = L->cursor->next;
        newNodePtr->prev = L->cursor;


        L->cursor->next->prev = newNodePtr;
        L->cursor->next = newNodePtr;
        L->Listindex++;
    }

    L->length++;

}

void deleteFront(List L) {
    if (length(L) > 0) {
        Node temp = L->front;
        if (length(L) == 1) {
            
            L->front = L->back = NULL;
        } else {
            
            L->front = L->front->next;
            L->front->prev = NULL;
        }
        free(temp);
        L->length--;
    } else {
        fprintf(stderr, "Error: calling deleteFront() on an empty list\n");
        exit(EXIT_FAILURE);
    }
}

void deleteBack(List L) {
    if (length(L) > 0) {
        Node temp = L->back;
        if (length(L) == 1) {
            L->front = L->back = NULL;
        } else {
            L->back = L->back->prev;
            L->back->next = NULL;
        }
        free(temp);
        L->length--;
    } else {
        fprintf(stderr, "Error: calling deleteBack() on an empty list\n");
        exit(EXIT_FAILURE);
    }
}

void delete(List L) {
    if (length(L) > 0 && index(L) >= 0) {
        Node current = L->front;

        
        for (int i = 0; i < index(L); i++) {
            current = current->next;
        }

        
        if (current->prev != NULL) {
            current->prev->next = current->next;
        } else {
            
            L->front = current->next;
        }

        if (current->next != NULL) {
            current->next->prev = current->prev;
        } else {
            
            L->back = current->prev;
        }

        
        freeNode(&current);

        
        L->length--;
        L->Listindex = -1;
        L->cursor = NULL;
    } else {
        fprintf(stderr, "Error: calling delete() with invalid list state\n");
        exit(EXIT_FAILURE);
    }
}
List copyList(List L) {
    if (L == NULL) {
        fprintf(stderr, "Error: calling copyList() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }

    
    List CopynewList = newList();

    
    Node current = L->front;

    while (current != NULL) {
        
        append(CopynewList, current->data);
        current = current->next;
    }

    
    CopynewList->cursor = NULL;

    return CopynewList;
}

void printList(FILE* out, List L) {
    Node current = L->front;
    
    
    if (current == NULL) {
        fprintf(out, "Empty List\n");
        return;
    }

    while (current != NULL) {
        fprintf(out, "%d ", current->data); 
        current = current->next;
    }
    fprintf(out, "\n");
}
