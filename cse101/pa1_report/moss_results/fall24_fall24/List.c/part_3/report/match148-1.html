// structs --------------------------------------------------------------------

// private Node type
typedef struct NodeObj* Node;

// private NodeObj type
typedef struct NodeObj{
   ListElement data;
   int ind;
   Node next;
   Node prev;
} NodeObj;

// private ListObj type
typedef struct ListObj{
   Node front;
   Node back;
   Node cursor;
   int length;
} ListObj;


// Constructors-Destructors ---------------------------------------------------

// newNode()
// Returns reference to new Node object. Initializes next and data fields.
Node newNode(ListElement data){
   Node N = malloc(sizeof(NodeObj));
   assert( N!=NULL );
   N->data = data;
   N->ind = -1;
   N->next = NULL;
   N->prev = NULL;
   return(N);
}

// freeNode()
// Frees heap memory pointed to by *pN, sets *pN to NULL.
void freeNode(Node* pN){
   if( pN!=NULL && *pN!=NULL ){
      free(*pN);
      *pN = NULL;
   }
}

// newList()
// Returns reference to new empty List object.
List newList(){
   List L;
   L = malloc(sizeof(ListObj));
   assert( L!=NULL );
   L->front = L->back = L->cursor = NULL;
   L->length = 0;
   return(L);
}

// freeList()
// Frees all heap memory associated with List *pL, and sets *pL to NULL.
void freeList(List* pL){
   if(pL!=NULL && *pL!=NULL) { 
      while( !isEmpty(*pL) ) { 
         clear(*pL); 
      }
      free(*pL);
      *pL = NULL;
   }
}


// Access functions -----------------------------------------------------------

// length()
// Returns the length of L.
// Pre: Existing List.
int length(List L){
   if(L == NULL){
      fprintf(stderr, "List Error: calling length() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   return(L->length);
}

// index()
// Returns index of cursor element if defined, -1 otherwise.
// Pre: List existing.
int index(List L){
   
   if(L == NULL){
      fprintf(stderr, "List Error: calling index() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   
   if(L->cursor == NULL){
      return -1;
   }
   
   return L->cursor->ind;
}

// front()
// Returns the value at the front of L.
// Pre: !isEmpty(L)


ListElement front(List L){
   if(L == NULL){
      fprintf(stderr, "List Error: calling front() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if(isEmpty(L)){
      fprintf(stderr, "List Error: calling front() on an empty List\n");
      exit(EXIT_FAILURE);
   }
   return L->front->data;
}

// back()
// Returns value at back of L.
// Pre: !isEmpty(L)
ListElement back(List L){
   if(L == NULL){
      fprintf(stderr, "List Error: calling back() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if( isEmpty(L) ){
      fprintf(stderr, "List Error: calling back() on an empty List\n");
      exit(EXIT_FAILURE);
   }
   return L->back->data;
}

// get()
// Returns cursor element of L. 
// Pre: length()>0, index()>=0
ListElement get(List L){
   //printf("Maybe Null L");
   if(L == NULL){
      fprintf(stderr, "List Error: calling get() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   
   if(isEmpty(L)){
      fprintf(stderr, "List Error: calling get() on empty List reference\n");
      exit(EXIT_FAILURE);
   }
   
   if(index(L) <= -1){
      fprintf(stderr, "List Error: calling get() on cursor element with undefined index\n");
      exit(EXIT_FAILURE);
   }
   
   return L->cursor->data;
}

// equals()
// Returns true if A is same int sequence as B, false otherwise.
bool equals(List A, List B){
   if( A==NULL || B==NULL ){
      printf("List Error: calling equals() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }

   bool eq;
   Node N, M;

   eq = ( A->length == B->length );
   N = A->front;
   M = B->front;
   while( eq && N != NULL){
      eq = ( N->data == M->data );
      N = N->next;
      M = M->next;
   }
   return eq;
}

// isEmpty()
// Returns true if L is empty, otherwise returns false.
bool isEmpty(List L){
   if(L == NULL){
      fprintf(stderr, "List Error: calling isEmpty() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if(L->length <= 0){
      return true;
   }
   return false;
}

//clear()
// Resets L to its original empty state.
void clear(List L){
   if(L == NULL){
      fprintf(stderr, "List Error: calling clear() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if(isEmpty(L)){
      return;
   }

   Node M = L->front;
   
   while(M != NULL) {
      Node N = M;
      M = M->next;
      
      //L->length--;
      freeNode(&N);
   }

   L->front = NULL;
   L->back = NULL;
   L->cursor = NULL;


   L->length = 0;
   
}

//moveFront

      fprintf(stderr, "List Error: calling moveFront() on NULL List reference\n");

   if(isEmpty(L)){

      fprintf(stderr, "List Error: calling moveBack() on NULL List reference\n");

//movePrev()
// If cursor is defined and not at front, move cursor one
// step toward the front of L; if cursor is defined and at
// front, cursor becomes undefined; if cursor is undefined
// do nothing
void movePrev(List L){
   //printf("Started stepback prev");
   if(L == NULL){
      fprintf(stderr, "List Error: calling movePrev() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   //printf("Before ind?");
   if (index(L) <= -1){
      return;
   }
   //printf("Before iind?");
   if (index(L) == L->front->ind){
      L->cursor = NULL;
      return;
   }
   L->cursor = L->cursor->prev;
}

//moveNext()
// If cursor is defined and not at back, move cursor one
// step toward the back of L; if cursor is defined and at
// back, cursor becomes undefined; if cursor is undefined
// do nothing
void moveNext(List L){
   if(L == NULL){
      fprintf(stderr, "List Error: calling moveNext() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   //printList(stdout, L);
   //printf("\n");
   if(index(L) <= -1){
      return;
   }
   if(index(L) == L->back->ind){
      L->cursor = NULL;
      return;
   }

   L->cursor = L->cursor->next;
}


//prepend()
// Insert new element into L. If L is non-empty,
// insertion takes place before front element.
void prepend(List L, int x){
   Node N = newNode(x);
   if(L == NULL){
      fprintf(stderr, "List Error: calling prepend() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }

   if(isEmpty(L)) { 
      L->front = L->back = N;
      N->ind = 0;
      L->length++;
      return;
   }

   L->front->prev = N;
   N->next = L->front;
   L->front = N;
   Node M = L->front;
   int count = 0;
   while(M != NULL) {
      M->ind = count;
      count++;
      M = M->next;
   }
   L->length++;
}

// append()
// Places new data at the back of L.
void append(List L, int x){
   Node N = newNode(x);
   if(L == NULL){
      fprintf(stderr, "List Error: calling append() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }

   if( isEmpty(L) ) { 
      L->front = L->back = N;
      N->ind = 0;
      L->length++;
      return;
   }
   
   L->back->next = N;
   N->prev = L->back;
   N->ind = L->length;
   L->back = N;


   L->length++;
}

//insertBefore()
// Insert new element before cursor.
// Pre: length()>0, index()>=0
void insertBefore(List L, int x){
   if(L == NULL){
      fprintf(stderr, "List Error: calling insertBefore() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if(isEmpty(L)){
      fprintf(stderr, "List Error: calling insertBefore() on empty List reference\n");
      exit(EXIT_FAILURE);
   }
   if(index(L) <= -1){
      fprintf(stderr, "List Error: calling insertBefore() on cursor element with undefined index\n");
      exit(EXIT_FAILURE);
   }

   if(index(L) == 0){
      prepend(L, x);
      return;
   }
   
   Node N = newNode(x);
   L->cursor->prev->next = N;
   N->prev = L->cursor->prev;
   L->cursor->prev = N;
   N->next = L->cursor;

   Node M = L->front;
   int count = 0;
   while(M != NULL) {
      M->ind = count;
      count++;
      M = M->next;
   }


   L->length++;

}

//insertAfter()
// Insert new element after cursor.
// Pre: length()>0, index()>=0
void insertAfter(List L, int x){
   if(L == NULL){
      fprintf(stderr, "List Error: calling insertAfter() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if(isEmpty(L)){
      fprintf(stderr, "List Error: calling insertAfter() on empty List reference\n");

   }
   
   if(index(L) <= -1){
      fprintf(stderr, "List Error: calling insertAfter() on cursor element with undefined index\n");
      exit(EXIT_FAILURE);
   }

   if(index(L) == L->back->ind){
      append(L, x);
      return;
   }

   Node N = newNode(x);
   L->cursor->next->prev = N;
   N->next = L->cursor->next;
   L->cursor->next = N;
   N->prev = L->cursor;

   Node M = L->front;
   int count = 0;
   while(M != NULL) {
      M->ind = count;
      count++;
      M = M->next;
   }


   L->length++;
   
}

//deleteFront()
// Delete the front element. Pre: length()>0
void deleteFront(List L){
   if(L == NULL){
      fprintf(stderr, "List Error: calling deleteFront() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if(isEmpty(L)){
      fprintf(stderr, "List Error: calling deleteFront() on empty List reference\n");
      exit(EXIT_FAILURE);
   }

   Node df = L->front;

   if(L->cursor == L->front) {
      L->cursor = NULL;
   }
   if(L->length == 1){
      L->front = L->back = NULL;
   } else {
      L->front = L->front->next;
      L->front->prev = NULL;
   }

   freeNode(&df);
   L->length--;

   Node M = L->front;
   int count = 0;
   while(M != NULL) {
      M->ind = count;
      count++;


      M = M->next;
   }
   
}

//deleteBack()
// Delete the back element. Pre: length()>0
void deleteBack(List L){
   if(L == NULL){
      fprintf(stderr, "List Error: calling deleteBack() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if(isEmpty(L)){
      fprintf(stderr, "List Error: calling deleteBack() on empty List reference\n");
      exit(EXIT_FAILURE);
   }

   Node db = L->back;
   if(L->cursor == L->back) {
      L->cursor = NULL;
   }
   if(L->length == 1){
      L->front = L->back = NULL;
   } else {
      L->back->prev->next = NULL;
      L->back = L->back->prev;
   }
   
   freeNode(&db);
   L->length--;

   Node M = L->front;
   int count = 0;
   while(M != NULL) {
      M->ind = count;
      count++;


      M = M->next;
   }
}

//delete()
// Delete cursor element, making cursor undefined.
// Pre: length()>0, index()>=0
void delete(List L){
   if(L == NULL){
      fprintf(stderr, "List Error: calling delete() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if(isEmpty(L)){
      fprintf(stderr, "List Error: calling delete() on empty List reference\n");
      exit(EXIT_FAILURE);
   }
   if(index(L) <= -1){
      fprintf(stderr, "List Error: calling delete() on cursor element with undefined index\n");
      exit(EXIT_FAILURE);
   }

   if(index(L) == 0){
      deleteFront(L);
      L->cursor = NULL;
      return;
   }
   if(index(L) == L->back->ind){
      deleteBack(L);
      L->cursor = NULL;
      return;
   }

   Node dc = L->cursor;

   if(length(L)==1){
      L->cursor->next = NULL;
      L->cursor->prev = NULL;
   } else {
      L->cursor->prev->next = L->cursor->next;
      L->cursor->next->prev = L->cursor->prev;
   }

   freeNode(&dc);
   L->length--;
   L->cursor = NULL;

   Node M = L->front;
   int count = 0;
   while(M != NULL) {
      M->ind = count;
      count++;
      M = M->next;
   }
}

//copyList()
// Returns a new List representing the same integer
// sequence as L. The cursor in the new list is undefined,
// regardless of the state of the cursor in L. The state
// of L is unchanged.
List copyList(List L) {
   if(L == NULL){
      printf("List Error: calling copyList() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   List nL = newList();
   Node N = L->front;
   while(N != NULL) {
      append(nL, N->data);
      N = N->next;
   }
   return nL;
}
