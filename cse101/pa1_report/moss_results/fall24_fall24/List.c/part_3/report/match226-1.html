#include "List.h"
#include <stdio.h>
#include <stdlib.h>
#include <assert.h>
#include <stdbool.h>

// redefine node and list structures to improve modularity and avoid redundancy
typedef struct NodeObj {
    int data;
    struct NodeObj* prev;
    struct NodeObj* next;
} NodeObj;

typedef struct NodeObj* Node;

typedef struct ListObj {
    Node head;
    Node tail;
    Node current;
    int size;
    int cursorIndex;
} ListObj;

// create and return a new empty list
List newList(void) {


    List L = (List)malloc(sizeof(ListObj));
    if (L == NULL) {
        fprintf(stderr, "memory allocation failed\n");
        exit(EXIT_FAILURE);
    }
    L->head = NULL;
    L->tail = NULL;
    L->current = NULL;
    L->size = 0;
    L->cursorIndex = -1;
    return L;
}

// free all memory associated with *pL and set *pL to null
void freeList(List* pL) {
    if (pL != NULL && *pL != NULL) {
        Node temp;
        while ((*pL)->head != NULL) {
            temp = (*pL)->head;
            (*pL)->head = (*pL)->head->next;
            free(temp);
        }
        free(*pL);
        *pL = NULL;
    }
}

// return the number of elements in L
int length(List L) {
    return L->size;
}

// return the index of cursor element if defined, -1 otherwise
int index(List L) {
    return L->current != NULL ? L->cursorIndex : -1;
}

// return front element of L. pre: length() > 0
int front(List L) {
    if (length(L) > 0 && L->head != NULL) {
        return L->head->data;
    }
    fprintf(stderr, "list is empty\n");
    exit(EXIT_FAILURE);
}

// return back element of L. pre: length() > 0
int back(List L) {
    if (length(L) > 0 && L->tail != NULL) {
        return L->tail->data;
    }
    fprintf(stderr, "list is empty\n");
    exit(EXIT_FAILURE);
}

// return cursor element of L. pre: length() > 0, index() >= 0
int get(List L) {
    if (length(L) > 0 && index(L) >= 0 && L->current != NULL) {
        return L->current->data;
    }
    fprintf(stderr, "cursor is undefined\n");
    exit(EXIT_FAILURE);
}

// check if lists A and B are equal by comparing elements
bool equals(List A, List B) {
    if (length(A) != length(B)) {
        return false;
    }
    Node nodeA = A->head;
    Node nodeB = B->head;
    while (nodeA != NULL && nodeB != NULL) {
        if (nodeA->data != nodeB->data) {
            return false;
        }
        nodeA = nodeA->next;
        nodeB = nodeB->next;
    }
    return true;
}

// reset L to its original empty state
void clear(List L) {
    while (L->head != NULL) {
        Node temp = L->head;
        L->head = L->head->next;
        free(temp);
    }
    L->tail = NULL;
    L->current = NULL;
    L->size = 0;
    L->cursorIndex = -1;
}

// overwrite the cursor element's data with x. pre: length() > 0, index() >= 0
void set(List L, int x) {
    if (length(L) > 0 && index(L) >= 0) {
        L->current->data = x;
    } else {
        fprintf(stderr, "preconditions not met for set\n");
        exit(EXIT_FAILURE);
    }
}

// set cursor to the front element
void moveFront(List L) {
    if (length(L) > 0) {
        L->current = L->head;
        L->cursorIndex = 0;
    }
}

// set cursor to the back element
void moveBack(List L) {
    if (length(L) > 0) {
        L->current = L->tail;
        L->cursorIndex = L->size - 1;
    }
}

// move cursor one step toward the front
void movePrev(List L) {
    if (L->current != NULL) {
        if (L->current->prev != NULL) {
            L->current = L->current->prev;
            L->cursorIndex--;
        } else {
            L->current = NULL;
            L->cursorIndex = -1;
        }
    }
}

// move cursor one step toward the back
void moveNext(List L) {
    if (L->current != NULL) {
        if (L->current->next != NULL) {
            L->current = L->current->next;
            L->cursorIndex++;
        } else {
            L->current = NULL;
            L->cursorIndex = -1;
        }
    }
}

// insert new element into L before the front element


void prepend(List L, int x) {
    Node newNode = (Node)malloc(sizeof(NodeObj));
    if (newNode == NULL) {
        fprintf(stderr, "memory allocation failed\n");
        exit(EXIT_FAILURE);
    }
    newNode->data = x;
    newNode->prev = NULL;
    newNode->next = L->head;
    if (L->head != NULL) {
        L->head->prev = newNode;
    }
    L->head = newNode;
    if (L->size == 0) {
        L->tail = newNode;
    }
    L->size++;
    if (L->cursorIndex != -1) {
        L->cursorIndex++;
    }
}

// insert new element into L after the back element


void append(List L, int x) {
    Node newNode = (Node)malloc(sizeof(NodeObj));
    if (newNode == NULL) {
        fprintf(stderr, "memory allocation failed\n");
        exit(EXIT_FAILURE);
    }
    newNode->data = x;
    newNode->next = NULL;
    newNode->prev = L->tail;
    if (L->tail != NULL) {
        L->tail->next = newNode;
    }
    L->tail = newNode;
    if (L->size == 0) {
        L->head = newNode;
    }
    L->size++;
}

// insert new element before cursor. pre: length() > 0, index() >= 0
void insertBefore(List L, int x) {
    if (length(L) <= 0 || index(L) < 0) {
        fprintf(stderr, "preconditions not met for insertBefore\n");


        exit(EXIT_FAILURE);
    }
    Node newNode = (Node)malloc(sizeof(NodeObj));
    if (newNode == NULL) {
        fprintf(stderr, "memory allocation failed\n");
        exit(EXIT_FAILURE);
    }
    newNode->data = x;
    newNode->next = L->current;
    newNode->prev = L->current->prev;
    if (L->current->prev != NULL) {
        L->current->prev->next = newNode;
    } else {
        L->head = newNode;
    }
    L->current->prev = newNode;
    L->size++;
    L->cursorIndex++;
}

// insert new element after cursor. pre: length() > 0, index() >= 0
void insertAfter(List L, int x) {
    if (length(L) <= 0 || index(L) < 0) {
        fprintf(stderr, "preconditions not met for insertAfter\n");


        exit(EXIT_FAILURE);
    }
    Node newNode = (Node)malloc(sizeof(NodeObj));
    if (newNode == NULL) {
        fprintf(stderr, "memory allocation failed\n");
        exit(EXIT_FAILURE);
    }
    newNode->data = x;
    newNode->prev = L->current;
    newNode->next = L->current->next;
    if (L->current->next != NULL) {
        L->current->next->prev = newNode;
    } else {
        L->tail = newNode;
    }
    L->current->next = newNode;
    L->size++;
}

// delete the front element. pre: length() > 0
void deleteFront(List L) {
    if (length(L) <= 0) {
        fprintf(stderr, "cannot delete front element: list is empty\n");
        exit(EXIT_FAILURE);
    }
    Node temp = L->head;
    L->head = L->head->next;
    if (L->head != NULL) {
        L->head->prev = NULL;
    } else {
        L->tail = NULL;
    }
    if (L->current == temp) {
        L->current = NULL;
        L->cursorIndex = -1;
    } else if (L->cursorIndex != -1) {
        L->cursorIndex--;
    }
    free(temp);
    L->size--;
}

// delete the back element. pre: length() > 0
void deleteBack(List L) {
    if (length(L) <= 0) {
        fprintf(stderr, "cannot delete back element: list is empty\n");
        exit(EXIT_FAILURE);
    }
    Node temp = L->tail;
    L->tail = L->tail->prev;
    if (L->tail != NULL) {
        L->tail->next = NULL;
    } else {
        L->head = NULL;
    }
    if (L->current == temp) {
        L->current = NULL;
        L->cursorIndex = -1;
    }
    free(temp);
    L->size--;
}

// delete cursor element, making cursor undefined. pre: length() > 0, index() >= 0


void delete(List L) {
    if (length(L) <= 0 || index(L) < 0) {
        fprintf(stderr, "preconditions not met for delete\n");
        exit(EXIT_FAILURE);
    }
    Node temp = L->current;
    if (L->current == L->head) {
        L->head = L->current->next;
        if (L->head != NULL) {
            L->head->prev = NULL;
        } else {
            L->tail = NULL;
        }
    } else if (L->current == L->tail) {
        L->tail = L->current->prev;
        if (L->tail != NULL) {
            L->tail->next = NULL;
        } else {
            L->head = NULL;
        }
    } else {
        L->current->prev->next = L->current->next;
        L->current->next->prev = L->current->prev;
    }
    free(temp);
    L->size--;
    L->current = NULL;
    L->cursorIndex = -1;
}

// print list to file
void printList(FILE* out, List L) {
    Node temp = L->head;
    while (temp != NULL) {
        fprintf(out, "%d ", temp->data);
        temp = temp->next;
    }
}

// return a copy of list L
List copyList(List L) {
    List copy = newList();
    Node temp = L->head;
    while (temp != NULL) {
        append(copy, temp->data);
        temp = temp->next;
    }
    return copy;
}
