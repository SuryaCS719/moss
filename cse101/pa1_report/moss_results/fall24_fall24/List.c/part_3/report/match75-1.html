#include<stdio.h>
#include<stdlib.h>
#include<assert.h>
#include<stdbool.h>
#include "List.h"

typedef struct NodeObj{
    int data;
    struct NodeObj* next;


    struct NodeObj* prev;   
} NodeObj;

typedef NodeObj* Node;

typedef struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
    int index; 
} ListObj;

// Constructors-Destructors -------------

//Returns reference to a new Node object. Initializes next, prev and data fields
Node newNode(int data){
    Node newN = malloc(sizeof(NodeObj));
    assert( newN != NULL);
    newN -> data = data;
    newN -> next = NULL;
    newN -> prev = NULL;
    return newN;
}

//Frees heapy memory pointed to by the *pN, sets *pN to NULL
void freeNode(Node* pN){
    if (pN != NULL && *pN!= NULL){
        free(*pN);
        *pN = NULL;
    }
}

//Returns reference to a new empty list object
List newList(void){
    List newL = malloc(sizeof(ListObj));
    assert(newL != NULL);
    newL -> front = NULL;
    newL -> back = NULL;
    newL -> cursor = NULL;
    newL -> length = 0;
    newL -> index = -1;
    return newL;
}

//Frees all heap memory associated with List *pL, and sets *pL to NULL
void freeList(List *pL){
    if(pL != NULL && *pL != NULL){
        clear(*pL);
        free(*pL);
        *pL = NULL;
    }
}

// Access functions --------------------

int length(List L){
    return L -> length;
}

int index(List L){
    return L -> index;
}

int front(List L){
    if (L -> length > 0){
        return L -> front -> data; 
    }
    printf("List Error: calling front() on an empty List\n");
    exit(EXIT_FAILURE);
}

int back(List L){
    if (L -> length > 0){
        return L -> back -> data;
    }
    printf("List Error: calling back() on an empty List\n");
    exit(EXIT_FAILURE);
}

int get(List L){
    if (L-> length > 0 && L->index >= 0 && L->cursor != NULL){
        return L->cursor-> data;
    }
    printf("List Error: calling get() when the cursor is undefined or the List is empty \n");
    exit(EXIT_FAILURE);
}

bool equals(List A, List B){
    if(A -> length != B -> length){
        return false;
    }

    Node N1 = A -> front;
    Node N2 = B -> front;

    while (N1 != NULL){
        if (N1 -> data != N2 -> data){
            return false;
        }
        N1 = N1 -> next;
        N2 = N2 -> next;
    }
    return true;

}

// Manipulation Procedures --------------------
void clear(List L){
    Node temp;

    while (L -> front != NULL){
        temp = L -> front;
        L -> front = L -> front -> next;
        free(temp);
    }

    L -> back = NULL;
    L -> cursor = NULL;
    L -> length = 0;
    L -> index = -1; 
}



void set(List L, int x){
    if(L -> length > 0 && L -> index >= 0){
        L -> cursor -> data = x;
    }
    else { 
        printf("List Error: calling get() when the cursor is undefined or the List is empty \n");
    }
}

void moveFront(List L){
    if (L -> length > 0){
        L -> cursor = L -> front;
        L -> index = 0;
    }
}

void moveBack(List L){
    if (L -> length){


        L -> cursor = L -> back;
        L -> index = L ->length - 1 ;
    }
}

void movePrev(List L){
    if (L -> cursor != NULL && L -> index > 0){
        L -> cursor = L -> cursor -> prev;
        L -> index --;
    }
    else if (L -> cursor != NULL && L -> index == 0){
        L -> cursor = NULL;
        L -> index = -1;
    }
}

void moveNext(List L){
    if (L -> cursor != NULL && (L->index < L->length - 1)){
        L -> cursor = L -> cursor -> next;
        L -> index ++;
    }
    else if (L -> cursor != NULL && (L->index == L->length - 1)){
        L -> cursor = NULL;
        L -> index = -1;
    }
}

void prepend(List L, int x){
    Node newN = newNode(x);

    if (L -> length == 0){
        L -> front = newN;
        L -> back = newN;
    } else {
        newN -> next = L -> front;
        L -> front -> prev = newN;
        L -> front = newN;
    }

    L -> length ++;

    if (L -> cursor != NULL){
        L-> index ++;
    }
}

void append(List L, int x){
    Node newN = newNode(x);

    if (L -> length == 0){
        L -> front = newN;
        L -> back = newN;
    } else {
        newN -> prev = L -> back;
        L -> back -> next = newN;
        L -> back = newN;
    }



    L -> length ++;
}

void insertBefore(List L, int x){

    if (L -> length > 0 && L -> index >= 0){
        Node newN = newNode(x); 
    
        if (L -> cursor == L -> front){
            newN -> next = L -> front;
            L -> front -> prev = newN;
            L -> front = newN;
        }
        else{
            newN -> next = L -> cursor;


            newN -> prev = L -> cursor -> prev;
            L -> cursor -> prev -> next = newN;
            L -> cursor -> prev = newN;
        }
        L -> length ++;
        L -> index ++;
    }

    else{
        printf("List Error: calling insertBefore() on an empty list or with an undefined cursor\n");
        exit(EXIT_FAILURE);
    }
}



void insertAfter(List L, int x){
    if (L -> length > 0 && L -> index >= 0){
        Node newN = newNode(x); 
    
        if (L -> cursor == L -> back){
            newN -> prev = L -> cursor;
            L -> cursor -> next = newN;
            L -> back = newN;
        }
        else{
            newN-> next = L -> cursor -> next;
            L -> cursor -> next -> prev = newN;
            L -> cursor -> next = newN;
            newN -> prev = L -> cursor;
            
        }
        L -> length ++;
    }
        
    else{
        printf("List Error: calling insertBefore() on an empty list or with an undefined cursor\n");
    }
}

void deleteFront(List L){
    if (L -> length > 0){
        Node temp = L -> front; //creates a temp pointer that points to the same node as L -> front

        if (L -> length == 1){
            L -> front = NULL;
            L -> back = NULL;
            L -> cursor = NULL;
            L->index = -1;
        }
        else{
            L -> front = L -> front -> next;
            L -> front -> prev = NULL;

            if (L -> cursor != NULL){
                if (L -> index == 0){
                    L-> cursor = NULL;
                    L -> index = -1;
                }
                else{
                    L -> index--;
                }
            }
        }

    free(temp);
    L -> length --;

    } else {
        printf("List Error: calling deleteFront() on an empty list\n");
    }
}

void deleteBack(List L){
    if (L -> length > 0){
        Node temp = L -> back; //creates a temp pointer that points to the same node as L -> front

        if (L -> length == 1){
            L -> front = NULL;
            L -> back = NULL;
            L -> cursor = NULL;
            L->index = -1;
        }
        else{
            L -> back = L -> back -> prev;
            L -> back -> next = NULL;

            if (L -> cursor != NULL){
                if (L -> index == L -> length - 1){
                    L-> cursor = NULL;
                    L -> index = -1;
                }
            }
        }

    free(temp);
    L -> length--;

    } else {
        printf("List Error: calling deleteBack() on an empty list\n");
    }

}

void delete(List L){
    if (L -> length > 0 && L -> index >= 0){
        Node temp = L -> cursor;

        if(L -> length == 1){
            L -> front = NULL;
            L -> back = NULL;
            L -> cursor = NULL;
            L->index = -1;
        }
        //deleting front cursor
        else if(L -> cursor == L -> front){
            L -> front = L -> front -> next;
            L -> front -> prev = NULL;

        }
        //deleting a back cursor 
        else if(L -> cursor == L -> back){
            L -> back = L -> back -> prev;
            L -> back -> next = NULL;
        }
        // deleting a node in the middle 
        else{
            L -> cursor -> prev -> next = L -> cursor -> next;
            L -> cursor -> next -> prev = L -> cursor -> prev;
        }

        free(temp);

        L -> cursor = NULL;
        L -> index = -1;
        L -> length--;
    } else {
        printf("List Error: calling delete() on an empty list\n");
    }
}

void printList(FILE *out, List L){
    Node current = L -> front;

    while (current != NULL){
        fprintf(out, "%d ", current -> data);
        current = current -> next;
    }
}

List copyList(List L){
    List newL = newList();

    if (L == NULL || L->length == 0) {
        return newL;
    }

    Node current = L -> front;
    while (current != NULL){
        append(newL, current -> data);
        current = current -> next;
    }

    return newL;

}
