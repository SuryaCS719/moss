#include "List.h"

typedef int ListElement;

typedef struct NodeObj* Node;
typedef struct NodeObj{
   ListElement data;
   Node next;
   Node prev;
} NodeObj;

typedef struct ListObj{
    int length;
    Node front;
    Node back;
    Node cursor;
    int index;
}List_Obj;
// Constructors-Destructors ---------------------------------------------------
// Creates and returns a new empty Node
Node newNode(ListElement data){
    Node node = calloc(1, sizeof(NodeObj));
    assert(node != NULL);
    node->data = data;
    node->next = NULL;
    node->prev = NULL;
    return(node);
}
//Frees all heap memory associated with the node
void freeNode(Node* pNode){
    free(*pNode);
    *pNode = NULL;
}
// Creates and returns a new empty List.
List newList(void){
    List list = calloc(1, sizeof(List_Obj));
    assert(list != NULL);
    list->front = list->back = list->cursor = NULL;
    list->index = -1;
    list->length = 0;
    return(list);
}
// Frees all heap memory associated with *pL, and sets pL to NULL.
void freeList(List* pL){
    clear(*pL);
    free(*pL);


    *pL = NULL;
} 
// Access functions -----------------------------------------------------------
// Returns the number of elements in L.
int length(List L){
    return L->length;
}
// Returns index of cursor element if defined, -1 otherwise.
int index(List L){
    return L->index;
}
// Returns front element of L. Pre: length()>0
int front(List L){
    if(length(L) == 0){
        fprintf(stderr, "List Error: Attempting to use front() method on empty list.\n");
        exit(EXIT_FAILURE);
    }
    return L->front->data;
}
// Returns back element of L. Pre: length()>0
int back(List L){
    if(length(L) == 0){
        fprintf(stderr, "List Error: Attempting to use back() method on empty list.\n");
        exit(EXIT_FAILURE);
    }
    return L->back->data;
}
// Returns cursor element of L. Pre: length()>0, index()>=0
int get(List L){
    if(length(L) == 0){
        fprintf(stderr, "List Error: Attempting to use get() method on empty list,\n");
        exit(EXIT_FAILURE);
    }
    if(index(L) < 0){
        fprintf(stderr, "List Error: Attempting to use get() method on undefined cursor.\n");
        exit(EXIT_FAILURE);
    }
    return L->cursor->data;
}
// Returns true iff Lists A and B contain the same
 // sequence of elements, returns false otherwise.
bool equals(List A, List B){
    if(length(A) == length(B)){
        moveFront(A);
        moveFront(B);
        for(int i = 0; i < length(A); i++){
            if(get(A) != get(B)){
                return false;
            }
            moveNext(A);
            moveNext(B);
        }
        return true;
    }
    return false;
}
// Manipulation procedures ----------------------------------------------------
// Resets L to its original empty state.
void clear(List L){
    while(length(L)){
        deleteFront(L);
    }
    L->length = 0;
} 
// Overwrites the cursor elementâ€™s data with x.
 // Pre: length()>0, index()>=0


void set(List L, int x){
    if(length(L) <= 0){
        fprintf(stderr, "List Error: Attempting to set() value in an empty list.\n");
        exit(EXIT_FAILURE);
    }
    if(index(L) < 0){
        fprintf(stderr, "List Error: Attempting to set() value of undefined cursor.\n");
        exit(EXIT_FAILURE);
    }
    L->cursor->data = x;
}
// If L is non-empty, sets cursor under the front element,
 // otherwise does nothing.
void moveFront(List L){
    if(L->length == 0){
        return;
    }
    L->index = 0;
    L->cursor = L->front;
} 
// If L is non-empty, sets cursor under the back element,
 // otherwise does nothing.
void moveBack(List L){
    if(L->length == 0){
        return;
    }
    L->index = L->length - 1;
    L->cursor = L->back;
}
// If cursor is defined and not at front, move cursor one
 // step toward the front of L; if cursor is defined and at
 // front, cursor becomes undefined; if cursor is undefined
 // do nothing
void movePrev(List L){
    if(index(L) > -1){
        L->index -= 1;
        L->cursor = L->cursor->prev;
    }
}
// If cursor is defined and not at back, move cursor one
 // step toward the back of L; if cursor is defined and at
 // back, cursor becomes undefined; if cursor is undefined
 // do nothing
void moveNext(List L){
    if(index(L) > -1){
        if(L->cursor->next == NULL){
            L->index = -1;
        }
        else{
            L->index += 1;
        }
        L->cursor = L->cursor->next;
    }
}
// Insert new element into L. If L is non-empty,
 // insertion takes place before front element.
void prepend(List L, int x){
    Node item = newNode(x);
    if(length(L) == 0){
        L->back = item;
    }
    else{
        item->next = L->front;
        L->front->prev = item;
    }


    L->front = item;
    L->length += 1;
    if(index(L) != -1){
        L->index += 1;
    }
} 
// Insert new element into L. If L is non-empty,
 // insertion takes place after back element.
void append(List L, int x){
    Node item = newNode(x);
    if(length(L) == 0){
        L->front = item;
    }
    else{
        item->prev = L->back;
        L->back->next = item;
    }


    L->back = item;
    L->length += 1;
}
// Insert new element before cursor.
 // Pre: length()>0, index()>=0
void insertBefore(List L, int x){
    if(length(L) <= 0){
        fprintf(stderr, "List Error: Attempting to insertBefore() on a list of length 0.\n");
        exit(EXIT_FAILURE);
    }
    if(length(L) < 0){
        fprintf(stderr, "List Error: Attempting to insertBefore() with an undefined cursor.\n");
        exit(EXIT_FAILURE);
    }
    Node newData = newNode(x);
    newData->prev = L->cursor->prev;
    newData->next = L->cursor;
    if(index(L) == 0){


        L->front = newData;
    }
    else{
        L->cursor->prev->next = newData;
    }
    L->cursor->prev = newData;
    L->length += 1;
    L->index += 1;
}
// Insert new element after cursor.
 // Pre: length()>0, index()>=0
void insertAfter(List L, int x){
if(length(L) <= 0){
        fprintf(stderr, "List Error: Attempting to insertAfter() on a list of length 0.\n");
        exit(EXIT_FAILURE);
    }
    if(length(L) < 0){
        fprintf(stderr, "List Error: Attempting to insertAfter() with an undefined cursor.\n");
        exit(EXIT_FAILURE);
    }
    Node newData = newNode(x);
    newData->next = L->cursor->next;
    newData->prev = L->cursor;
    if(index(L) == length(L) - 1){


        L->back = newData;
    }
    else{
        L->cursor->next->prev = newData;
    }
    L->cursor->next = newData;
    L->length += 1;
}
// Delete the front element. Pre: length()>0
void deleteFront(List L){
    if(L->length == 0){
        fprintf(stderr, "List Error: Attempting to deleteFront() on list of length 0.\n");
        exit(EXIT_FAILURE);
    }
    Node newFront = L->front->next;
    freeNode(&(L->front));
    L->front = newFront;
    if(index(L) == 0){
        L->index = -1;
        L->cursor = NULL;
    }
    else if(index(L) != -1){
        L->index -= 1;
    }
    L->length -= 1;
}
// Delete the back element. Pre: length()>0
void deleteBack(List L){
    if(L->length == 0){
        fprintf(stderr, "List Error: Attempting to deleteBack() on list of length 0.\n");
        exit(EXIT_FAILURE);
    }
    Node newBack = L->back->prev;
    freeNode(&(L->back));
    L->back = newBack;
    if(L->index == length(L) - 1){
        L->index = -1;


        L->cursor = NULL;
    }
    L->length -= 1;
}
// Delete cursor element, making cursor undefined.
// Pre: length()>0, index()>=0
void delete(List L){
    if(length(L) <= 0){
        fprintf(stderr, "List Error: Attempting to delete() element from empty list.\n");
        exit(EXIT_FAILURE);
    }
    if(index(L) < 0){
        fprintf(stderr, "List Error: Attempting to delete() element using undefined cursor.\n");
        exit(EXIT_FAILURE);
    }
    if(index(L) == 0){
        L->front = L->cursor->next;
    }
    if(index(L) != 0){
        L->cursor->prev->next = L->cursor->next;
    }
    if(index(L) == length(L) - 1){
        L->back = L->cursor->prev;
    }
    if(index(L) != length(L) - 1){
        L->cursor->next->prev = L->cursor->prev;
    }
    freeNode(&(L->cursor));
    L->index = -1;
    L->length -= 1;
    L->cursor = NULL;
}
// Other operations -----------------------------------------------------------
// Prints to the file pointed to by out, a
 // string representation of L consisting
// of a space separated sequence of integers,
// with front on left.
void printList(FILE* out, List L){
    Node current = L->front;
    for(int i = 0; i < length(L); i++){
        if(i != 0){
            fprintf(out, " ");
        }
        fprintf(out, PRINT_FORMAT, current->data);
        current = current->next;
    }
    fprintf(stdout, "\n");
}
// Returns a new List representing the same integer
 // sequence as L. The cursor in the new list is undefined,
// regardless of the state of the cursor in L. The state
// of L is unchanged.
List copyList(List L){
    List copy = newList();
    Node current = L->front;
    for(int i = 0; i < length(L); i++){
        append(copy, current->data);
        current = current->next;
    }
    return copy;
}
