#include "List.h"
#include <stdlib.h>



typedef struct NodeObj {
    int data;
    struct NodeObj *prev, *next;
} *Node;

struct ListObj {
    int length, cursor_index;
    Node front, back, cursor;
};

List newList(void) {
    List L = malloc(sizeof(struct ListObj));
    L->length = 0;
    L->cursor_index = -1;
    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    return L;
}

void freeList(List* pL) {
    clear(*pL);
    free(*pL);
    *pL = NULL;
}

int length(List L) {
    return L->length;
}

int index(List L) {
    return L->cursor_index;
}

int front(List L) {


    return L->front->data;
}

int back(List L) {
    return L->back->data;
}

int get(List L) {
    return L->cursor->data;
}

void clear(List L) {
    // Free all elements of List
    Node head = L->front;
    while (head != NULL) {
        Node n = head->next;
        free(head);
        head = n;
    }

    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->length = 0;
    L->cursor_index = -1;
}

void set(List L, int x) {
    L->cursor->data = x;
}

void moveFront(List L) {
    if (L->length > 0) {
        L->cursor = L->front;
        L->cursor_index = 0;
    }
}

void moveBack(List L) {
    if (L->length > 0) {


        L->cursor = L->back;
        L->cursor_index = L->length - 1;
    }
}

void movePrev(List L) {
    if (L->cursor_index == -1) return;
    if (L->cursor_index == 0) {
        // at front
        L->cursor_index = -1;
        L->cursor = NULL;
    } else {
        L->cursor = L->cursor->prev;
        L->cursor_index--;
    }
}

void moveNext(List L) {
    if (L->cursor_index == -1) return;
    if (L->cursor_index == L->length -1) {
        // at back
        L->cursor_index = -1;
        L->cursor = NULL;
    } else {
        L->cursor = L->cursor->next;
        L->cursor_index++;
    }
}

void prepend(List L, int x) {
    Node n = malloc(sizeof(struct NodeObj));
    n->data = x;
    n->next = NULL;
    n->prev = NULL;
    if (L->length == 0) {
        L->front = n;
        L->back = n;
    } else {
        n->next = L->front;
        L->front->prev = n;
        L->front = n;
    }

    if (L->cursor_index != -1) {
        L->cursor_index++;
    }

    L->length++;
}

void append(List L, int x) {
    Node n = malloc(sizeof(struct NodeObj));
    n->data = x;
    n->next = NULL;
    n->prev = NULL;
    if (L->length == 0) {
        L->front = n;
        L->back = n;
    } else {
        n->prev = L->back;
        L->back->next = n;
        L->back = n;
    }

    L->length++;
}

void insertBefore(List L, int x) {
    Node n = malloc(sizeof(struct NodeObj));
    n->data = x;
    n->next = NULL;
    n->prev = NULL;

    Node cur_prev = L->cursor->prev;
    if (cur_prev == NULL) {
        // at front
        L->front = n;
    } else {
        cur_prev->next = n;
        n->prev = cur_prev;
    }

    n->next = L->cursor;

    L->cursor->prev = n;
    L->cursor_index++;
    L->length++;
}

void insertAfter(List L, int x) {
    Node n = malloc(sizeof(struct NodeObj));
    n->data = x;
    n->next = NULL;
    n->prev = NULL;

    Node cur_next = L->cursor->next;
    if (cur_next == NULL) {
        // at back
        L->back = n;
    } else {
        cur_next->prev = n;
        n->next = cur_next;
    }

    n->prev = L->cursor;

    L->cursor->next = n;
    L->length++;
}

void deleteFront(List L) {
    Node n = L->front->next;
    free(L->front);
    L->front = NULL;
    if (n == NULL) {
        L->back = NULL;
    } else {
        L->front = n;
        n->prev = NULL;
    }

    if (L->cursor_index == 0) {
        L->cursor_index = -1;
        L->cursor = NULL;
    }

    L->length--;
}

void deleteBack(List L) {
    Node n = L->back->prev;
    free(L->back);
    L->back = NULL;
    if (n == NULL) {
        L->front = NULL;
    } else {
        L->back = n;
        n->next = NULL;
    }

    if (L->cursor_index == L->length - 1) {
        L->cursor_index = -1;
        L->cursor = NULL;
    }
    
    L->length--;
}

void delete(List L) {
    Node np = L->cursor->prev;
    Node nn = L->cursor->next;
    free(L->cursor);

    if (np != NULL) {
        np->next = nn;
    } else {
        L->front = nn;
    }

    if (nn != NULL) {
        nn->prev = np;
    } else {
        L->back = np;
    }

    L->cursor = NULL;
    L->cursor_index = -1;
    L->length--;
}

void printList(FILE* out, List L) {
    Node n = L->front;
    if (n != NULL) {
        fprintf(out, "%d", n->data);
        n = n->next;
        while (n != NULL) {
            fprintf(out, " %d", n->data);
            n = n->next;
        }
    }
}

List copyList(List L) {
    List new = newList();
    Node n = L->front;
    while (n != NULL) {
        append(new, n->data);
        n = n->next;
    }

    return new;
}

List concatList(List A, List B) {
    List C = copyList(A);
    
    Node n = B->front;
    while (n != NULL) {
        append(C, n->data);
        n = n->next;
    }

    return C;
}
