#include "List.h"
#include <assert.h>
#include <stdio.h>
#include <stdlib.h>

typedef struct ListElement {
    int value;
    struct ListElement *prev;
    struct ListElement *next;
} ListElement;

typedef struct ListObj {
    ListElement *front;
    ListElement *back;
    ListElement *cursor;
    int index;
} ListObj;

typedef ListObj *List;

List newList() {
    List list = malloc(sizeof(ListObj));
    assert(list != NULL);

    list->front = NULL;
    list->back = NULL;
    list->cursor = NULL;
    return list;
}

void freeList(List *pL) {
    if (*pL == NULL)
        return;
    clear(*pL);
    free(*pL);
    *pL = NULL;
}

int length(List L) {
    if (L->front == NULL)
        return 0;
    int n = 0;
    for (ListElement *curr = L->front; curr != NULL; curr = curr->next)
        n++;
    return n;
}

int index(List L) {
    if (L->cursor == NULL)
        return -1;


    return L->index;
}

int front(List L) {
    if (length(L) < 1) {
        fprintf(stderr, "Cannot retrieve front of an empty List");
        exit(EXIT_FAILURE);
    }
    return L->front->value;
}

int back(List L) {
    if (length(L) < 1) {
        fprintf(stderr, "Cannot retrieve front of an empty List");
        exit(EXIT_FAILURE);
    }
    return L->back->value;
}

int get(List L) {
    if (length(L) < 1) {
        fprintf(stderr, "Cannot get element of empty list");
        exit(EXIT_FAILURE);
    }
    if (index(L) < 0) {
        fprintf(stderr, "Cannot get value of undefined cursor");
        exit(EXIT_FAILURE);
    }
    return L->cursor->value;
}

bool equals(List A, List B) {
    ListElement *AElement = A->front;
    ListElement *BElement = B->front;

    while (AElement != NULL && BElement != NULL) {
        if (BElement->value != AElement->value)
            return false;
        AElement = AElement->next;
        BElement = BElement->next;
    }
    return AElement == NULL && BElement == NULL;
}

void clear(List L) {
    if (length(L) < 1)
        return;
    while (length(L) > 0) {


        deleteBack(L);
    }
}

void set(List L, int x) {
    if (length(L) < 1) {
        fprintf(stderr, "Cannot set element of empty list");
        exit(EXIT_FAILURE);
    }
    if (index(L) < 0) {
        fprintf(stderr, "Cannot set value of undefined cursor");
        exit(EXIT_FAILURE);
    }

    L->cursor->value = x;
}

void moveFront(List L) {
    if (L->front == NULL)
        return;
    L->cursor = L->front;
    L->index = 0;
}

void moveBack(List L) {
    if (L->back == NULL)
        return;
    L->index = length(L) - 1;
    L->cursor = L->back;
}

void movePrev(List L) {
    if (L->cursor == NULL)
        return;
    L->index--;
    L->cursor = L->cursor->prev;
}

void moveNext(List L) {
    if (L->cursor == NULL)
        return;
    L->index++;
    L->cursor = L->cursor->next;
}

void prepend(List L, int x) {
    ListElement *newElement = malloc(sizeof(ListElement));
    newElement->value = x;
    newElement->prev = NULL;
    newElement->next = L->front;
    if (L->front != NULL) {
        L->front->prev = newElement;
    }
    L->front = newElement;
    if (L->back == NULL) {
        L->back = L->front;
    }
    if (L->cursor != NULL && L->cursor != L->front) {
        L->index++;
    }
}

void append(List L, int x) {
    ListElement *newElement = malloc(sizeof(ListElement));
    newElement->value = x;
    newElement->prev = L->back;
    newElement->next = NULL;
    if (L->back != NULL) {
        L->back->next = newElement;
    }
    L->back = newElement;
    if (L->front == NULL) {
        L->front = L->back;
    }
}



void insertBefore(List L, int x) {
    if (length(L) < 1) {
        fprintf(stderr, "Cannot insert before in empty list\n");
        exit(EXIT_FAILURE);
    }
    if (index(L) < 0) {
        fprintf(stderr, "Cannot insert before undefined cursor\n");
        exit(EXIT_FAILURE);
    }

    ListElement *newElement = malloc(sizeof(ListElement));
    newElement->value = x;
    newElement->prev = L->cursor->prev;
    newElement->next = L->cursor;
    L->cursor->prev = newElement;

    L->index++;

    if (L->cursor == L->front) {
        L->front = newElement;
        return;
    }
    newElement->prev->next = newElement;
}



void insertAfter(List L, int x) {
    if (length(L) < 1) {
        fprintf(stderr, "Cannot insert after in empty list\n");
        exit(EXIT_FAILURE);
    }
    if (index(L) < 0) {
        fprintf(stderr, "Cannot insert after undefined cursor\n");
        exit(EXIT_FAILURE);
    }

    ListElement *newElement = malloc(sizeof(ListElement));
    newElement->value = x;
    newElement->prev = L->cursor;
    newElement->next = L->cursor->next;
    L->cursor->next = newElement;

    if (L->cursor == L->back) {
        L->back = newElement;
        return;
    }
    newElement->next->prev = newElement;
}

void deleteFront(List L) {
    if (length(L) < 1) {
        fprintf(stderr, "Cannot delete front element of list with no elements\n");
        exit(EXIT_FAILURE);
    }
    ListElement *newFront = L->front->next;
    if (L->cursor == L->front) {
        L->cursor = NULL;
    }
    free(L->front);
    L->front = newFront;
    if (L->front == NULL) {
        L->back = NULL;
        L->cursor = NULL;
        return;
    }
    L->index--;
    L->front->prev = NULL;
}

void deleteBack(List L) {
    if (length(L) < 1) {
        fprintf(stderr, "Cannot delete back element of list with no elements\n");
        exit(EXIT_FAILURE);
    }
    if (L->cursor == L->back) {
        L->cursor = NULL;
    }
    ListElement *newBack = L->back->prev;
    free(L->back);
    L->back = newBack;
    if (L->back == NULL) {
        L->front = NULL;
        L->cursor = NULL;
        return;
    }
    L->back->next = NULL;
}



void delete(List L) {
    if (length(L) < 1) {
        fprintf(stderr, "Cannot delete element of list with no elements\n");
        exit(EXIT_FAILURE);
    }
    if (index(L) < 0) {
        fprintf(stderr, "Cannot delete element at undefined cursor\n");
        exit(EXIT_FAILURE);
    }


    if (L->front == L->cursor)
        L->front = L->front->next;

    if (L->back == L->cursor)
        L->back = L->back->prev;

    if (L->cursor->prev != NULL)
        L->cursor->prev->next = L->cursor->next;
    if (L->cursor->next != NULL)
        L->cursor->next->prev = L->cursor->prev;

    free(L->cursor);
    L->cursor = NULL;
}

void printList(FILE *out, List L) {
    for (ListElement *element = L->front; element->next != NULL;
         element = element->next) {
        fprintf(out, "%d, ", element->value);
    }
    fprintf(out, "%d", L->back->value);
}

List copyList(List L) {
    List list = newList();
    for (ListElement *element = L->front; element != NULL;
         element = element->next) {
        append(list, element->value);
    }
    return list;
}

List concatList(List A, List B) {
    List list = newList();

    for (ListElement *element = A->front; element != NULL;
         element = element->next) {
        append(list, element->value);
    }
    for (ListElement *element = B->front; element != NULL;
         element = element->next) {
        append(list, element->value);
    }
    return list;
}
