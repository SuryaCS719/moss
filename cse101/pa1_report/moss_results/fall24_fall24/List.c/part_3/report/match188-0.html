#include "List.h"

// structs ------------------------------------------------------------

// private Node type
typedef struct NodeObj* Node;

// private NodeObj type
typedef struct NodeObj{ 
   ListElement data; // data stored at NodeObj
   Node next; // pointer to next node: NULL if last node in list
   Node prev; // pointer to previous node: NULL if first node in list
} NodeObj;

// private ListObj type
typedef struct ListObj{
   Node front; // first element in list
   Node back; // last element in list
   Node cursor; // where cursor is pointing to
   int length; // number of NodeObjs in list
   int index; // index of cursor: -1 if undefined
} ListObj;

// Constructors-Destructors ---------------------------------------------------

// newNode()
// returns refernece to new Node object. Initializes, data, next, and prev fields
Node newNode(ListElement data){
   Node N = malloc(sizeof(NodeObj));
   assert( N!=NULL );
   N -> data = data;
   N -> next = NULL;
   N -> prev = NULL;
   return(N);
}

//freeNode()
// frees heap memory pointed to by *pN, sets *pN to NULL
void freeNode(Node* pN){ 
   if(pN!=NULL && *pN!=NULL ){
      free(*pN);
      *pN = NULL;
   }
}

// newList()
// Creates and returns a new empty List
List newList(void){
   List L;
   L = malloc(sizeof(ListObj));
   assert( L!=NULL );
   L -> front = L -> back = L -> cursor = NULL;
   L -> length = 0;
   L -> index = -1;
   return(L);
}

// freeList()
// Frees all heap memory associated with *pL, and sets
// *pL to NULL.
void freeList(List* pL){
   if (pL!=NULL && *pL!=NULL){
      while( length(*pL) != 0){
         deleteFront(*pL);
      }
      free(*pL);
      *pL = NULL;
   }
}    
                            
// Access functions -----------------------------------------------------------

// length()
// Returns the number of elements in L.
int length(List L){
   if(L == NULL){
      fprintf(stderr, "List error: calling length() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   return(L -> length);
}     

// index()
// Returns index of cursor element if defined, -1 otherwise.
int index(List L){
   if(L == NULL){
      fprintf(stderr,"List error: calling index() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   return(L -> index);
}      

// front()
// Returns front element of L. Pre: length()>0
int front(List L){
   if(L == NULL){
      fprintf(stderr,"List error: calling front() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if(length(L)==0){
      fprintf(stderr,"List error: calling front() on an empty List\n");
      exit(EXIT_FAILURE);
   }
   return(L -> front -> data);
}     

// back()
// Returns back element of L. Pre: length()>0
int back(List L){
   if(L == NULL){
      fprintf(stderr,"List error: calling back() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if(length(L)==0){
      fprintf(stderr,"List error: calling back() on an empty List\n");
      exit(EXIT_FAILURE);
   }
   return(L -> back -> data);
} 

// get()
// Returns cursor element of L. Pre: length()>0, index()>=0
int get(List L){
   if(L == NULL){
      fprintf(stderr,"List error: calling get() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if(length(L)==0){
      fprintf(stderr,"List error: calling get() on an empty List\n");
      exit(EXIT_FAILURE);
   }
   if(L -> index < 0){
      fprintf(stderr,"List error: calling get() on list with undefined cursor\n");
      exit(EXIT_FAILURE);
   }
   return(L -> cursor -> data);
}        

// equals()
// Returns true iff Lists A and B contain the same
// sequence of elements, returns false otherwise.
bool equals(List A, List B){
   if( A==NULL || B == NULL){
      fprintf(stderr,"List error: calling equals() on a NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   bool eq;
   Node N, M;
   eq = (A->length == B->length);
   N = A->front;
   M = B->front;
   while( eq && N!=NULL && M!=NULL){
      eq = ( N->data==M->data);
      N = N->next;


      M = M->next;
   }
   return eq;
}    
                                
// Manipulation procedures ----------------------------------------------------

// clear()
// Resets L to its original empty state.
void clear(List L){
   if(L == NULL){
      fprintf(stderr,"List error: calling clear() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   while(length(L) > 0){
      deleteFront(L);
   }
}         

// set()
// Overwrites the cursor elementâ€™s data with x.
// Pre: length()>0, index()>=0
void set(List L, int x){
   if(L == NULL){
      fprintf(stderr,"List error: calling set() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if(length(L)==0){
      fprintf(stderr,"List error: calling set() on an empty List\n");
      exit(EXIT_FAILURE);
   }
   if(L -> index < 0){
      fprintf(stderr,"List error: calling set() on list with undefined cursor\n");
      exit(EXIT_FAILURE);
   }
   L -> cursor -> data = x;
}    
                            
// moveFront()
// If L is non-empty, sets cursor under the front element,
// otherwise does nothing.                          
void moveFront(List L){
   if(L == NULL){
      fprintf(stderr,"List error: calling moveFront() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if(length(L) > 0){
      L -> index = 0;
      L -> cursor = L -> front;
   }
}   

// moveBack()
// If L is non-empty, sets cursor under the back element,
// otherwise does nothing.
void moveBack(List L){
   if(L == NULL){
      fprintf(stderr,"List error: calling moveBack() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if(length(L) > 0){
      L -> index = length(L) - 1;
      L -> cursor = L -> back;
   }
}      
                            
// movePrev()
// If cursor is defined and not at front, move cursor one
// step toward the front of L; if cursor is defined and at
// front, cursor becomes undefined; if cursor is undefined
// do nothing
void movePrev(List L){
   if(L == NULL){
      fprintf(stderr,"List error: calling movePrev() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if (L -> index == 0){ // cursor is at front
      L -> index -=1;
      L -> cursor = NULL;
   }
   if(L -> index > 0){


      L -> cursor = L -> cursor -> prev;
      L -> index -= 1;
   }
    // if index < 0 do nothing: no code needed
}
       
// moveNext()
// If cursor is defined and not at back, move cursor one
// step toward the back of L; if cursor is defined and at
// back, cursor becomes undefined; if cursor is undefined
// do nothing
void moveNext(List L){
   if(L == NULL){
      fprintf(stderr,"List error: calling moveNext() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if (L -> index == length(L) -1){ // cursor is at back
      L -> index = -1;
      L -> cursor = NULL;
   }
   if(0 <= L -> index){ 
      if( L -> index <= length(L) -1){ // defined and not at back


         L -> cursor = L -> cursor -> next;
         L -> index += 1;
      }    
   }
   // if index < 0 do nothing: no code needed
}      

// prepend()
// Insert new element into L. If L is non-empty,
// insertion takes place before front element.
void prepend(List L, int x) {
   if(L == NULL){
      fprintf(stderr,"List error: calling prepend() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   Node N = newNode(x);
   if(length(L) == 0){
      L -> front =N;
      L -> back = N;
      // no need to update prev and next in N because they are already set to NULL
    }
    else { // list is not empty
      // set prev/next relationships between L -> front and N
      L -> front -> prev = N;
      N -> next = L -> front;
      // update L -> front
      L -> front = N;
      // update index, if necessary
      if (L -> index >= 0){
         L -> index += 1;
      }
   }


   L -> length += 1;
}    
 
// append()
// Insert new element into L. If L is non-empty,
// insertion takes place after back element.
void append(List L, int x){
   if(L == NULL){
      fprintf(stderr,"List error: calling append() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   Node N = newNode(x);
   if(length(L) == 0){
      L -> front = L -> back = N;
      L-> length +=1;
      // no need to update prev and next in N because they are already set to NULL
   }
   else {
      // set prev/next relationships between L-> back and N


      L -> back -> next = N;
      N -> prev = L -> back;
      // update L -> back
      L -> back = N;
      L -> length +=1;
   }
}  
                                
// insertBefore()
// Insert new element before cursor.
// Pre: length()>0, index()>=0
void insertBefore(List L, int x){
   if(L == NULL){
      fprintf(stderr,"List error: calling insertBefore() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if(length(L)==0){
      fprintf(stderr,"List error: calling insertBefore() on an empty List\n");
      exit(EXIT_FAILURE);
   }
   if(L -> index < 0){
      fprintf(stderr,"List error: calling insertBefore() on list with undefined cursor\n");
      exit(EXIT_FAILURE);
   }
   Node N = newNode(x);
    N -> next = L -> cursor;
    if(L->index>0){
      L -> cursor -> prev ->  next = N;
      N -> prev = L -> cursor -> prev;
   }
   else{
      L ->front = N;
   }


   L -> cursor -> prev = N;
   L -> length += 1;
   L -> index +=1;
}   

// insertAfter()
// Insert new element after cursor.
// Pre: length()>0, index()>=0
void insertAfter(List L, int x){
   if(L == NULL){
      fprintf(stderr,"List error: calling insertAfter() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if(length(L)==0){
      fprintf(stderr,"List error: calling insertAfter() on an empty List\n");
      exit(EXIT_FAILURE);
   }
   if(L -> index < 0){
      fprintf(stderr,"List error: calling insertAfter() on list with undefined cursor\n");
      exit(EXIT_FAILURE);
   }
   Node N = newNode(x);
   N -> prev = L -> cursor;
   if(L->back != L->cursor){
      L -> cursor -> next -> prev = N;
      N -> next = L -> cursor -> next;
   }
   else{
      L -> back = N;
   }
   L-> cursor -> next = N;
   L -> length +=1;
}       

// deleteFront()
// Delete the front element. Pre: length()>0
void deleteFront(List L){
   if(L == NULL){
      fprintf(stderr,"List error: calling deleteFront() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if(length(L)==0){
      fprintf(stderr,"List error: calling deleteFront() on an empty List\n");
      exit(EXIT_FAILURE);
   }
   Node N_delete = L -> front; //save pointer to node to be freed
   L -> front = L -> front -> next; // update pointers for removal of current front
   if(L->length > 1){
      L -> front -> prev = NULL;
   }
   else{
      L->front = L->back = NULL;  
   } 
   L -> length -= 1;
   freeNode(&N_delete); // free memory of now unneeded node
   if(L -> index >=0){
      L -> index -= 1;
   }
}   

// deleteBack()
// Delete the back element. Pre: length()>0
void deleteBack(List L){    
   if(L == NULL){
      fprintf(stderr,"List error: calling deleteFront() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if(length(L)==0){
      fprintf(stderr,"List error: calling deleteFront() on an empty List\n");
      exit(EXIT_FAILURE);
   }
   Node N_delete = L->back;
   if(L->length > 1){
      L -> back -> prev -> next = NULL;
      L -> back = L -> back -> prev;
   }
   else{ // length == 1
      L->front = L->back = NULL;
   }
   if(L->index == L->length-1){ // cursor becomes undefined if cursor is deleted element
      L->index = -1;
      L->cursor = NULL;
   }
   L->length -=1;
   freeNode(&N_delete);
}

// delete() 
// Delete cursor element, making cursor undefined.
// Pre: length()>0, index()>=0
void delete(List L){
   Node N_delete = L->cursor;
   if(L == NULL){
      fprintf(stderr,"List error: calling delete() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if(length(L)==0){
      fprintf(stderr,"List error: calling delete() on an empty List\n");
      exit(EXIT_FAILURE);
   }
   if(L -> index < 0){
      fprintf(stderr,"List error: calling delete() on list with undefined cursor\n");
      exit(EXIT_FAILURE);
   }
   if(L->index!=L->length-1){
      L->cursor->next->prev = L->cursor->prev;
   }
   else{
      L->back = L->back->prev;
   }
   if(L->index>0){
      L->cursor->prev->next = L->cursor->next;
   }
   else{
      L->front = L->front->next;
   }
   L->length -=1;
   L->index = -1;
   L->cursor = NULL;
   freeNode(&N_delete);
} 

// Other operations -----------------------------------------------------------

// printList() 
// Prints to the file pointed to by out, a
// string representation of L consisting
// of a space separated sequence of integers,
// with front on left.
void printList(FILE* out, List L){
   if(L == NULL){
      fprintf(stderr,"List error: calling printList() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   Node N = NULL;
   for(N= L-> front;N != NULL; N = N -> next){
      fprintf(out,"%d ", N -> data);
   }
   printf("\n");
}
                                                             
// copyList()
// Returns a new List representing the same integer
// sequence as L. The cursor in the new list is undefined,
// regardless of the state of the cursor in L. The state
// of L is unchanged.
// make sure to not overwrite a list with this function, ALWAYS make a new list
List copyList(List L){
   if(L == NULL){
      fprintf(stderr,"List error: calling copyList() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   List M = newList();
   Node N = NULL;
   for(N = L->front; N != NULL; N = N->next){
      append(M, N->data);
   }
   return(M);
}

// concatList()
// Returns a new List which is the concatenation of
// A and B. The cursor in the new List is undefined,
// regardless of the states of the cursors in A and B.
// The states of A and B are unchanged.
// make sure to not overwrite a list with this function, ALWAYS make a new list
List concatList(List A, List B){
   if(A == NULL || B==NULL){
      fprintf(stderr,"List error: calling concatList() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   List M = copyList(A);
   Node N = NULL;
   for(N = B->front; N!=NULL; N = N->next){
      append(M, N->data);
   }
   return(M);
}
