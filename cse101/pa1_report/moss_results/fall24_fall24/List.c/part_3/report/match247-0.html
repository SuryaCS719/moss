#include "List.h"

// Structs -----
typedef struct new_node *Node;

typedef struct new_node
{
    Node next;
    Node prev;
    int data;
} new_node;

typedef struct nlist
{
    Node head;
    Node back;
    Node cursor;
    int index;
    int length;
} nlist;

// Constructors and Destructors -----
Node newNode(int data)
{
    Node N = malloc(sizeof(new_node));
    N->prev = NULL;
    N->next = NULL;
    N->data = data;
    return N;
}

void freeNode(Node *pN)
{


    if (pN != NULL && *pN != NULL)
    {
        free(*pN);
        *pN = NULL;
    }
}

List newList(void)
{

    List L = malloc(sizeof(nlist));
    L->head = NULL;
    L->cursor = NULL;
    L->back = NULL;
    L->index = -1;
    L->length = 0;

    return L;
}

void freeList(List *pL)
{
    if (pL != NULL && *pL != NULL)
    {
        List L = *pL;

        while (L->length > 0)
        {
            deleteFront(L);
        }

        free(L);

        *pL = NULL;
    }
}

// Access Functions
int length(List L)
{
    if (L == NULL)
    {
        fprintf(stderr, "Length cannot exist on a null list");
        exit(EXIT_FAILURE);
    }
    return L->length;
}

int index(List L)
{
    if (L == NULL)
    {
        fprintf(stderr, "Index cannot exist on a null list");
        exit(EXIT_FAILURE);
    }
    return L->index;
}

int front(List L)
{
    if (L->length == 0)
    {
        fprintf(stderr, "Front cannot exist on a list length = 0");
        exit(EXIT_FAILURE);
    }
    if (L == NULL)
    {
        fprintf(stderr, "Front cannot exist on a null list length");
        exit(EXIT_FAILURE);
    }

    return L->head->data;
}

int back(List L)
{
    if (L->length == 0)
    {
        fprintf(stderr, "Back cannot exist on a list length = 0");
        exit(EXIT_FAILURE);
    }
    if (L == NULL)
    {
        fprintf(stderr, "Back cannot exist on a null list length");
        exit(EXIT_FAILURE);
    }

    return L->back->data;
}



int get(List L)
{
    if (L->length == 0)
    {
        fprintf(stderr, "Cannot get object when list length = 0");
        exit(EXIT_FAILURE);
    }
    if (L->index < 0)
    {
        fprintf(stderr, "Cannot get object when list length < 0");
        exit(EXIT_FAILURE);
    }
    if (L == NULL)
    {
        fprintf(stderr, "Cannot get object with null list");
        exit(EXIT_FAILURE);
    }
    if (L->cursor == NULL)
    {
        fprintf(stderr, "Cannot get object with null list cursor");
        exit(EXIT_FAILURE);
    }

    return L->cursor->data;
}

bool equals(List A, List B)
{
    if (A == NULL || B == NULL)
    {
        fprintf(stderr, "Cannot find equal with null list ");
        exit(EXIT_FAILURE);
    }

    if (A == NULL && B == NULL)
    {
        return true;
    }

    if (A->length != B->length)
    {
        return false;
    }

    Node currentA = A->head;
    Node currentB = B->head;
    while (currentA != NULL)
    {
        if (currentA->data != currentB->data)
        {
            return false;
        }
        currentA = currentA->next;
        currentB = currentB->next;
    }
    return true;
}

// Manipulation Procedures -----
void clear(List L)
{
    while (length(L) > 0)
    {
        deleteFront(L);
    }

    if (L == NULL)
    {
        fprintf(stderr, "Cannot clear null list");
        exit(EXIT_FAILURE);
    }

    L->head = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->length = 0;
    L->index = -1;
}



void set(List L, int n)
{
    if (L->length == 0)
    {
        fprintf(stderr, "Cannot set list with length 0");
        exit(EXIT_FAILURE);
    }

    if (L->index < 0)
    {
        fprintf(stderr, "Cannot set list with index less than 0");
        exit(EXIT_FAILURE);
    }

    if (L == NULL)
    {
        fprintf(stderr, "Cannot set null list");
        exit(EXIT_FAILURE);
    }
    L->cursor->data = n;
}

void moveFront(List L)
{
    if (L->length > 0)
    {
        L->index = 0;
        L->cursor = L->head;
    }

    if (L == NULL)
    {
        fprintf(stderr, "Cannot moveFront to null list");
        exit(EXIT_FAILURE);
    }
}

void moveBack(List L)
{
    if (L->length > 0)
    {
        L->cursor = L->back;
        L->index = L->length - 1;
    }

    if (L == NULL)
    {
        fprintf(stderr, "Cannot moveBack to null list");
        exit(EXIT_FAILURE);
    }
}

void movePrev(List L) {
    if (!L) {
        fprintf(stderr, "List Error: movePrev() called on a NULL List.\n");
        exit(EXIT_FAILURE);
    }

    if (L->cursor) {
        if (L->cursor == L->head) {
            L->cursor = NULL;
            L->index = -1;
        } else {
            L->cursor = L->cursor->prev;
            L->index--;
        }
    }
}


void moveNext(List L)
{

    if (L->cursor == L->back)
    {
        L->cursor = NULL;
        L->index = -1;
    }

    if (L == NULL)
    {
        fprintf(stderr, "Cannot moveNext on null list");
        exit(EXIT_FAILURE);
    }



    else if (L->cursor != NULL)
    {
        L->cursor = L->cursor->next;
        L->index++;
    }
}

void prepend(List L, int x) {
    if (!L) {
        fprintf(stderr, "List Error: prepend() called on a NULL List.\n");
        exit(EXIT_FAILURE);
    }

    Node new = newNode(x);

    if (L->length == 0) {
        L->head = L->back = new;
    } else {
        new->next = L->head;
        L->head->prev = new;
        L->head = new;

        if (L->cursor) {
            L->index++;
        }
    }

    L->length++;
}


void append(List L, int x)
{
    Node new_node = newNode(x);

    if (L->head == NULL)
    {
        L->head = new_node;
        L->back = new_node;
    }
    else
    {
        L->back->next = new_node;
        new_node->prev = L->back;
        L->back = new_node;
    }

    L->length++;
}

void insertBefore(List L, int x) {
    if (!L || L->length == 0 || L->index < 0) {
        fprintf(stderr, "List Error: insertBefore() called with invalid conditions.\n");
        exit(EXIT_FAILURE);
    }

    if (L->cursor == NULL) return;

    Node N = newNode(x);

    if (L->cursor == L->head) {
        N->next = L->head;
        L->head->prev = N;
        L->head = N;
    } else {
        N->next = L->cursor;
        N->prev = L->cursor->prev;
        N->prev->next = N;
        L->cursor->prev = N;
    }

    L->index++;
    L->length++;
}


void insertAfter(List L, int x)
{
    if (L->length <= 0 || L->index < 0)
    {
        fprintf(stderr, "List Error: insertAfter() called on an empty list or with an undefined cursor.\n");
        exit(EXIT_FAILURE);
    }

    Node new_node = newNode(x);

    if (L->cursor == L->back)
    {
        new_node->prev = L->back;
        L->back->next = new_node;
        L->back = new_node;
    }
    else
    {
        new_node->next = L->cursor->next;
        new_node->prev = L->cursor;
        if (L->cursor->next != NULL)
        {
            L->cursor->next->prev = new_node;
        }
        L->cursor->next = new_node;
    }

    L->length++;
}

void deleteFront(List L) {
    if (!L || L->length == 0) {
        fprintf(stderr, "List Error: deleteFront() called with invalid conditions.\n");
        exit(EXIT_FAILURE);
    }

    Node N = L->head;

    if (L->length == 1) {
        L->head = L->back = L->cursor = NULL;
        L->index = -1;
    } else {
        L->head = L->head->next;
        L->head->prev = NULL;
        if (L->index == 0) {
            L->cursor = NULL;
            L->index = -1;
        } else {
            L->index--;
        }
    }

    freeNode(&N);


    L->length--;
}

void deleteBack(List L)
{
    if (L == NULL)
    {
        fprintf(stderr, "List Error: deleteBack() called on a NULL List.\n");
        exit(EXIT_FAILURE);
    }
    if (L->length == 0)
    {
        fprintf(stderr, "List Error: deleteBack() called on an empty list.\n");
        exit(EXIT_FAILURE);
    }

    Node N = L->back;

    if (L->length == 1)
    {
        L->head = L->cursor = L->back = NULL;
        L->index = -1;
    }
    else
    {
        if (L->cursor == N)
        {
            L->cursor = NULL;
            L->index = -1;
        }
        L->back = N->prev;
        L->back->next = NULL;
    }

    freeNode(&N);
    L->length--;
}

void delete(List L)
{
    if (L == NULL)
    {
        fprintf(stderr, "List Error: delete() called on a NULL List.\n");
        exit(EXIT_FAILURE);
    }


    if (L->length == 0)
    {
        fprintf(stderr, "List Error: delete() called on an empty list.\n");
        exit(EXIT_FAILURE);
    }
    if (L->index < 0)
    {
        fprintf(stderr, "List Error: delete() called with an undefined cursor.\n");
        exit(EXIT_FAILURE);
    }

    Node N = L->cursor;
    L->cursor = NULL;
    L->index = -1;

    if (L->length == 1)
    {
        L->head = L->back = NULL;
    }
    else if (L->head == N)
    {
        L->head = L->head->next;
        L->head->prev = NULL;
    }
    else if (L->back == N)
    {
        L->back = L->back->prev;


        L->back->next = NULL;
    }
    else
    {
        N->prev->next = N->next;
        N->next->prev = N->prev;
    }

    freeNode(&N);
    L->length--;
}

// Other Operations -----
void printList(FILE *out, List L)
{
    if (!L)
    {
        fprintf(stderr, "List Error: printList() called on a NULL List.\n");
        exit(EXIT_FAILURE);
    }

    for (Node N = L->head; N; N = N->next)
    {
        fprintf(out, "%d%s", N->data, N->next ? " " : "");
    }
    fprintf(out, "\n");
}

List copyList(List L)
{
    if (!L)
    {
        fprintf(stderr, "List Error: copyList() called on a NULL List.\n");
        exit(EXIT_FAILURE);
    }

    List copy = newList();
    for (Node N = L->head; N; N = N->next)
        append(copy, N->data);

    return copy;
}
