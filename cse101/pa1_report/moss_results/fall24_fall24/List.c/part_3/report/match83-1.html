#include <stdio.h>

List newList(void) {
    List L = (List) malloc(sizeof(struct ListObj));
    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->length = 0;
    L->index = -1;
    return L;
}

void freeList(List* pL) {
    if (*pL != NULL && pL != NULL) {
        clear(*pL);
        free(*pL);
        *pL = NULL;
    }
}

Node newNode(int data) {
    Node node = (Node) malloc(sizeof(struct NodeObj));
    node->data = data;
    node->next = NULL;
    node->previous = NULL;
    return node;
}

void freeNode(Node* node) {
    if (*node != NULL && node != NULL) {
	    free(*node);
        *node = NULL;
    }
}

int length(List L) {
	if (L == NULL) {
		fprintf(stderr, "Calling length() on NULL list reference.\n");
		exit(EXIT_FAILURE);
	}
	return L->length;
}

	if (L == NULL) {

bool equals(List A, List B) {
	if (A == NULL || B == NULL) {
		fprintf(stderr, "Calling equals() on NULL list reference.\n");
		exit(EXIT_FAILURE);
	}

	if (length(A) != length(B)) {return false;}
    	Node a = A->front;
	Node b = B->front;
	while (a && b) {
	    	if (a->data != b->data) {return false;}
		a = a->next;


		b = b->next;
    }
    return true;
}

   	 while(length(L) > 0) {
		deleteFront(L);

		fprintf(stderr, "Calling moveFront() on empty list.\n");
		exit(EXIT_FAILURE);

	if (length(L) > 0) {
		L->cursor = L->front;
        	L->index = 0;
    	}

void movePrev(List L) {
	if (L == NULL) {
		fprintf(stderr, "Calling movePrev() on NULL list reference.\n");

		fprintf(stderr, "Calling movePrev() on undefined cursor.\n");

	if (length(L) <= 0) {
		fprintf(stderr, "Calling movePrev() on empty list.\n");
		exit(EXIT_FAILURE);
	}
	if (L->cursor && L->cursor != L->front) {
        	L->cursor = L->cursor->previous;
   	  	L->index = L->index - 1;
    	}
    	else {


    		L->cursor = NULL;
    		L->index = -1;
	}
}

void moveNext(List L) {
	if (L == NULL) {
		fprintf(stderr, "Calling moveNext() on NULL list reference.\n");
		exit(EXIT_FAILURE);
	}
	if (index(L) < 0) {
		fprintf(stderr, "Calling moveNext() on undefined cursor.\n");
		exit(EXIT_FAILURE);
	}
	if (length(L) <= 0) {
		fprintf(stderr, "Calling moveNext() on empty list.\n");
		exit(EXIT_FAILURE);
	}
    	if (L->cursor != NULL && L->cursor != L->back) {
        	L->cursor = L->cursor->next;
        	L->index++;
    	}
    	else {
	    	L->cursor = NULL;
	    	L->index = -1;
    	}
}

void prepend(List L, int x) {

	if (L == NULL) {
		fprintf(stderr, "Calling prepend() on NULL list reference.\n");
		exit(EXIT_FAILURE);
	}
	Node node = newNode(x);
	if (length(L) == 0) {
		L->front = node;
		L->back = node;
	}
	else {
		node->next = L->front;
		L->front->previous = node;
		L->front = node;
		L->index++;	
	}


	L->length++;  
}

void append(List L, int x) {
	if (L == NULL) {
		fprintf(stderr, "Calling append() on NULL list reference.\n");
		exit(EXIT_FAILURE);
	}
	Node node = newNode(x);
	if (length(L) == 0) {
		L->front = node;
		L->back = node;
	}
	else {
		L->back->next = node;
		node->previous = L->back;
		L->back = node;     
	}
	L->length++;
}

void insertBefore(List L, int x) {
	if (L == NULL) {
		fprintf(stderr, "Calling insertBefore() on NULL list reference.\n");
		exit(EXIT_FAILURE);
	}


	if (length(L) <= 0) {
		fprintf(stderr, "Calling insertBefore() on empty list.\n");
		exit(EXIT_FAILURE);
	}
	if (index(L) < 0) {
		fprintf(stderr, "Calling insertBefore() on undefined cursor/\n");
		exit(EXIT_FAILURE);
	}
	Node node = newNode(x);
	if (index(L) == 0) {
		prepend(L, x);
		freeNode(&node);
	}
	else {
		if (L->cursor == L->front) {
			L->front = node;
		}
		else {
			L->cursor->previous->next = node;
			node->previous = L->cursor->previous;
		}
		node->next = L->cursor;
		L->cursor->previous = node;
		L->length++;
		L->index++;
	}
}


void insertAfter(List L, int x) {
	if (L == NULL) {
		fprintf(stderr, "Calling insertAfter() on NULL list reference.\n");
		exit(EXIT_FAILURE);
	}


	if (length(L) <= 0) {
		fprintf(stderr, "Calling insertAfter() on empty list.\n");
		exit(EXIT_FAILURE);
	}
	if (index(L) < 0) {
		fprintf(stderr, "Calling insertAfter() on undefined cursor/\n");
		exit(EXIT_FAILURE);
	}
	Node node = newNode(x);
	if (length(L) == index(L) + 1) {
		append(L, x);
		freeNode(&node);	
	}
	else {
		if (L->cursor == L->back) {
			L->back = node;
		}
		else {
			L->cursor->next->previous = node;
			node->next = L->cursor->next;
		}
	node->previous = L->cursor;


	L->cursor->next = node;
	L->length++;
	}
}

void deleteFront(List L) {
	if (L == NULL) {
		fprintf(stderr, "Calling deleteFront() on NULL list reference.\n");
		exit(EXIT_FAILURE);
	}
	if (length(L) <= 0) {
		fprintf(stderr, "Calling deleteFront() on empty list.\n");
		exit(EXIT_FAILURE);
	}
	if (length(L) > 1) {
		if (L->cursor == L->front) {
			L->cursor = NULL;
			L->index = -1;
		}
		Node temp = L->front;
		L->front = L->front->next;
		L->front->previous = NULL;
		freeNode(&temp);
	}
	else {
		freeNode(&(L->front));
	}
	L->length--;
	L->index -= L->index == -1 ? 0 : 1;
}

void deleteBack(List L) {
	if (L == NULL) {
		fprintf(stderr, "Calling deleteFront() on NULL list reference.\n");
		exit(EXIT_FAILURE);
	}
	if (length(L) <= 0) {
		fprintf(stderr, "Calling deleteFront() on empty list.\n");
		exit(EXIT_FAILURE);
	}
    	if (index(L) == length(L) - 1) {
      		L->cursor = NULL;
  	     	L->index = -1;
   	 }

	if (length(L) > 1) {
		L->back = L->back->previous;
		freeNode(&(L->back->next));
		L->back->next = NULL;
	}
	else {
		freeNode(&(L->back));
		L->front = NULL;
		L->back = NULL;
	}
	L->length--; 
}

void delete(List L) {
    
	if (L == NULL) {
		fprintf(stderr, "Calling delete() on NULL list reference.\n");
		exit(EXIT_FAILURE);
	}
	if (length(L) <= 0) {
		fprintf(stderr, "Calling delete() on empty list.\n");
		exit(EXIT_FAILURE);
	}
	if (index(L) < 0) {
		fprintf(stderr, "Calling delete() on undefined cursor.\n");
		exit(EXIT_FAILURE);
	}
	if (length(L) > 0 && index(L) >= 0) {
        if (index(L) == 0) {deleteFront(L);}
        else if (index(L) == length(L) - 1) {deleteBack(L);}
        else {
            L->cursor->previous->next = L->cursor->next;
            L->cursor->next->previous = L->cursor->previous;
            freeNode(&(L->cursor));
	    L->cursor = NULL;
            L->length--;
            L->index = -1;
        }
    }
}

void printList(FILE* out, List L) {
    
	if (L == NULL) {
		fprintf(stderr, "Calling printList() on NULL list reference.\n");
		exit(EXIT_FAILURE);
	}
	Node node = L->front;
	while (node) {
        	fprintf(out, "%d ", node->data);
        	node = node->next;
    	}
}

	List copy = newList();
