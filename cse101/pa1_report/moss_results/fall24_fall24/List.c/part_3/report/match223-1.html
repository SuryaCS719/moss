#include "List.h"
#include<stdio.h>
#include<stdlib.h>
#include<assert.h>
#include<stdbool.h>

// Structs ------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
typedef struct NodeObj* Node;

typedef struct NodeObj{
   ListElement data;
   Node next;
   Node prev;
} NodeObj;

typedef struct ListObj{
   Node front;
   Node cursor;
   Node back;
   int index;
   int length;
} ListObj;

// Constructors-Destructors -----------------------------------------------------------------------------------------------------------------------------------------------------------------
Node newNode(ListElement data) {
   // Allocating just enough space on memory
   Node N = malloc(sizeof(NodeObj));
   assert( N!=NULL );
   N->data = data;
   N->next = NULL;
   N->prev = NULL;
   return(N);
}

void freeNode(Node* pN) {
   if( pN!=NULL && *pN!=NULL ){
      free(*pN);
      *pN = NULL;
   }
}

List newList(void) {
    List L;
    L = malloc(sizeof(ListObj));
    assert(L != NULL);
    L->front = L->back = L->cursor = NULL;
    L->length = 0;
    L->index = -1;
    return L;
}

void freeList(List* lQ) {
    if(lQ != NULL && *lQ != NULL) {
        while(length(*lQ) > 0) {
            deleteFront(*lQ);
        }
    }
    clear(*lQ);
    free(*lQ);
    *lQ = NULL;
}


// Access Functions --------------------------------------------------------------------------------------------------------------------------------------------------------------------------
int length(List L) {

    if(L==NULL){
      printf("Queue Error: calling getLength() on NULL Queue reference\n");
      exit(EXIT_FAILURE);
    } else if(L->length < 0) {
      printf("Queue Error: calling getLength() on 0-length Queue reference\n");
      exit(EXIT_FAILURE);
    }

    return L->length;
   
}

int index(List L) {
     if(L == NULL){
      printf("Queue Error: calling getIndex() on NULL Queue reference\n");
      exit(EXIT_FAILURE);
    }

    if(L->cursor == NULL) {
        return -1;
    }



    return L->index;
}

int front(List L) {

    if(L==NULL){
      printf("Queue Error: calling getFront() on NULL Queue reference\n");
      exit(EXIT_FAILURE);
    }

    if(length(L) <= 0) {
        printf("List Error: calling front() on empty List\n");
        exit(EXIT_FAILURE);
    }

    return L->front->data;

}

int back(List L) {

    if(L==NULL){
      printf("Queue Error: calling getFront() on NULL Queue reference\n");
      exit(EXIT_FAILURE);
    }

    if(length(L) <= 0) {
        printf("List Error: calling back() on empty List\n");
        exit(EXIT_FAILURE);
    }

    return L->back->data;
}

int get(List L) {

    if(L==NULL) {
      printf("Queue Error: calling get() on NULL Queue reference\n");
      exit(EXIT_FAILURE);
    }

    if(length(L) <= 0 && index(L) < 0) {
        return -1;
    }

    return(L->cursor->data);
}

bool equals(List A, List B) {

    if(A == NULL || B == NULL) {
        return false;
    }

    if(length(A) != length(B)) {
        return false;
    }

    Node aStart = A->front;
    Node bStart = B->front;

    while(aStart != NULL && bStart != NULL) {
        if(aStart->data != bStart->data) {
            return false;
        } else {


            aStart = aStart->next;
            bStart = bStart->next;
        }     
    }

    return true;

}

// Manipulation procedures --------------------------------------------------------------------------------------------------------------------------------------------------------------------------
void clear(List L) {

    if(L == NULL){
        printf("List Error: calling clear() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }

    while (length(L) > 0) {
        deleteFront(L);
    }
    
    L->cursor = NULL;
    L->length = 0;
    L->index = -1;
    
}

void set(List L, int x) {
    if(L == NULL){
        printf("List Error: calling set() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }

    if((length(L) > 0 || index(L) >= 0)){
        L->cursor->data = x;
    }
}

void moveFront(List L) {
    if(L == NULL){
        printf("List Error: calling moveFront() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }

    if((length(L) > 0)) {
        L->cursor = L->front;
        L->index = 0;
    }
}

void moveBack(List L) {


    if(L == NULL){
        printf("List Error: calling moveBack() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }

    if((length(L) > 0)) {
        L->cursor = L->back;
        L->index = length(L) - 1;
    }
}

void movePrev(List L) {

    if(L == NULL){
        printf("List Error: calling moveBack() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }

    if(L->cursor != NULL && L->cursor != L->front) {
        L->cursor = L->cursor->prev;
        L->index = L->index - 1;
    } else {
        L->cursor = NULL;
        L->index = -1;
    }
}

void moveNext(List L) {

    if(L == NULL){
        printf("List Error: calling moveNext() on NULLx List reference\n");
        exit(EXIT_FAILURE);
    }

    if(L->cursor != NULL && L->cursor != L->back) {
        L->cursor = L->cursor->next;
        L->index = L->index + 1;
    } else {
        L->cursor = NULL;


        L->index = -1;
    }
}

void prepend(List L, int x) {
    if(L == NULL){
        printf("List Error: calling prepend() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }

    Node n = newNode(x);
    
    if(length(L) == 0) {
        L->front = L->back = n;
    } else {
        n->next = L->front;
        L->front->prev = n;
        L->front = n;
    }

    L->length = L->length + 1;

    if(L->index != -1) {
        L->index = L->index + 1;
    }

}

void append(List L, int x) {

    if(L == NULL){


        printf("List Error: calling append() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }

    Node n = newNode(x);

    if(length(L) == 0) {
        L->front = L->back = n;
    } else {
        n->prev = L->back;
        L->back->next = n;
        L->back = n;
    }

    L->length = L->length + 1;

}

void insertBefore(List L, int x) {

    if(L == NULL){
        printf("List Error: calling insertBefore() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }

    if(length(L) > 0 && index(L) >= 0) {
        Node n = newNode(x);

        if(L->cursor == L->front) {
            n->next = L->front;  

            if (L->front != NULL) { 
                L->front->prev = n; 
            }

            L->front = n;

            if(L->back == NULL) { 
                L->back = n;
            }

        } else {
            n->prev = L->cursor->prev;
            n->next = L->cursor;
            L->cursor->prev->next = n;
            L->cursor->prev = n;  
        }

        L->length++;
        L->index++;
    
    } else {  
      fprintf(stderr, "List Error: calling insertBefore on empty list or undefined cursor\n");
      exit(EXIT_FAILURE);

    }

}

void insertAfter(List L, int x) {

    if(L == NULL){
        printf("List Error: calling insertBefore() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }

    if(length(L) > 0 && index(L) >= 0){

        if(L->cursor == L->back) {
            append(L, x);
        } else {
            Node n = newNode(x);
            n->next = L->cursor->next;
            n->prev = L->cursor;
            L->cursor->next->prev = n;
            L->cursor->next = n;   
            L->length = L->length + 1;
        }
    }
    
}

void deleteFront(List L) {

    if(L == NULL){
        printf("List Error: calling deleteFront() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if(length(L) > 0) {
        Node temp = L->front;

        if(length(L) > 1) {
            L->front = L->front->next;
            L->front->prev = NULL;
        } else if(length(L) == 1) {
            L->front = L->back = NULL;
        }

        freeNode(&temp);
        L->length = L->length - 1;
        if(L->index != -1 ) {
            L->index = L->index - 1;
        }
    }
    
}

void deleteBack(List L) {

    if(L == NULL){
        printf("List Error: calling deleteBack() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }

    if(length(L) > 0){


        Node temp = L->back;
        if(length(L) > 1) {
            L->back = L->back->prev;
            L->back->next = NULL;

            if(L->cursor == temp) {
                L->cursor = NULL;
                L->index = -1;
            }

        } else if(length(L) == 1) {
            L->front = L->back = NULL;
            L->index = -1;
        }

        freeNode(&temp);
        L->length = L->length - 1;
    }
    
}

void delete(List L) {

    if(L == NULL){
        printf("List Error: calling delete() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }

    if(length(L) > 0 && index(L) >= 0){
        Node temp = L->cursor;
        if(L->cursor == L->front) {
            deleteFront(L);
        } else if(L->cursor == L->back) {
            deleteBack(L);
        } else {
            L->cursor->prev->next = L->cursor->next;
            L->cursor->next->prev = L->cursor->prev;
            freeNode(&temp);
            L->length = L->length - 1;
            L->index = -1;
            L->cursor = NULL;
        }
    }

   
}

// Other-operations ---------------------------------------------------------------------------------------------------------------------------------------------------------------------------
void printList(FILE* out, List L) {

    if (L == NULL) {
        printf("List Error: calling printList() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }

    Node currentNode = L->front;

    while(currentNode != NULL) {
        fprintf(out, "%d ", currentNode->data);
        currentNode = currentNode->next;
    }

    fprintf(out, "\n");
}

List copyList(List L) {

    if (L == NULL) {
        printf("List Error: calling copyList() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }

    List newList;
    newList = malloc(sizeof(ListObj));
    assert(newList != NULL);
    newList->front = newList->back = newList->cursor = NULL;
    newList->length = 0;
    newList->index = -1;

    Node current = L->front;
    while ((current != NULL))
    {
        append(newList, current->data);
        current = current->next;
    }

    return newList;
    
}
