#include<stdio.h>
#include<stdlib.h>
#include "List.h"

// private struct, not exported
typedef struct NodeObj{
	int data;
	struct NodeObj* next;
    struct NodeObj* previous;
} NodeObj;



// private reference type, not exported
typedef NodeObj* Node;
Node newNode(int data) {
	Node N = malloc(sizeof(NodeObj));
	N->data = data;
	N->next = N->previous = NULL;
	return(N);
}

void freeNode(Node* pN) {
	if(pN != NULL && *pN != NULL){
		free(*pN);
		*pN = NULL;
	}
}

typedef struct ListObj{
    int length;
    int index;
    Node front;
    Node back;
    Node cursor;
} ListObj;

// Constructors-Destructors ---------------------------------------------------
List newList(void){
    List L = malloc(sizeof(ListObj));
    L->length = 0;
    L->index = -1;
    L->front = L->back = L->cursor = NULL;
    return L;
} // Creates and returns a new empty List.

void freeList(List* pL){
    if(pL != NULL && *pL != NULL){
        clear(*pL);
        free(*pL);
        *pL = NULL;
    }
} // Frees all heap memory associated with *pL, and sets *pL to NULL.



// Access functions -----------------------------------------------------------
int length(List L){
    if(L == NULL){
        printf("List Error: calling length() with NULL list\n");
        exit(EXIT_FAILURE);
    }
    
    return L->length;
} // Returns the number of elements in L.

int index(List L){
    if(L == NULL){
        printf("List Error: calling index() with NULL list\n");
        exit(EXIT_FAILURE);
    }
    return L->index;
} // Returns index of cursor element if defined, -1 otherwise.

int front(List L){
    if(L == NULL){
        printf("List Error: calling front() with NULL list\n");
        exit(EXIT_FAILURE);
    }
    if(length(L) <= 0){
        printf("List Error: calling front() when length is 0\n");
        exit(EXIT_FAILURE);
    }
    
    return L->front->data;
} // Returns front element of L. Pre: length()>0

int back(List L){
    if(L == NULL){
        printf("List Error: calling back() with NULL list\n");
        exit(EXIT_FAILURE);
    }
    if(length(L) <= 0){
        printf("List Error: calling back() when length is 0\n");
        exit(EXIT_FAILURE);
    }


    return L->back->data;
} // Returns back element of L. Pre: length()>0

int get(List L){
    if(L == NULL){
        printf("List Error: calling get() with NULL list\n");
        exit(EXIT_FAILURE);
    }
    if(length(L) <= 0){
        printf("List Error: calling get() when length is 0\n");
        exit(EXIT_FAILURE);
    }
    if(index(L) < 0){
        printf("List Error: calling get() when index is less than 0\n");
        exit(EXIT_FAILURE);
    }

    return L->cursor->data;
} // Returns cursor element of L. Pre: length()>0, index()>=0

int equals(List A, List B){
    if( A==NULL || B==NULL ){
      printf("List Error: calling equals() on NULL Queue reference\n");
      exit(EXIT_FAILURE);
   }

   int eq;
   Node N, M;

   eq = ( A->length == B->length );
   N = A->front;
   M = B->front;
   while( eq && N!=NULL){
      eq = ( N->data==M->data );


      N = N->next;
      M = M->next;
   }
   return eq;
} // Returns true iff Lists A and B contain the same sequence of elements, returns false otherwise.



// Manipulation procedures ----------------------------------------------------
void clear(List L){
    if(L == NULL){
        printf("List Error: calling clear() with NULL list\n");
        exit(EXIT_FAILURE);
    }
    while(length(L) > 0){
		deleteFront(L);
	}
} // Resets L to its original empty state.

void set(List L, int x){
    if(L == NULL){


        printf("List Error: calling set() with NULL list\n");
        exit(EXIT_FAILURE);
    }
    if(length(L) <= 0){
        printf("List Error: calling set() when length is 0\n");
        exit(EXIT_FAILURE);
    }
    if(index(L) < 0){
        printf("List Error: calling set() when index is less than 0\n");
        exit(EXIT_FAILURE);
    }

    L->cursor->data = x;
} // Overwrites the cursor elementâ€™s data with x. Pre: length()>0, index()>=0

void moveFront(List L){
    if(L == NULL){
        printf("List Error: calling moveFront() with NULL list\n");
        exit(EXIT_FAILURE);
    }
    if(length(L) != 0){
        L->cursor = L->front;
        L->index = 0;
    }    
} // If L is non-empty, sets cursor under the front element, otherwise does nothing.

void moveBack(List L){
    if(L == NULL){
        printf("List Error: calling moveBack() with NULL list\n");
        exit(EXIT_FAILURE);
    }
    if(length(L) != 0){
        L->cursor = L->back;
        L->index = length(L) - 1;
    }
} // If L is non-empty, sets cursor under the back element, otherwise does nothing.

void movePrev(List L){
    if(L == NULL){
        printf("List Error: calling movePrev() with NULL list\n");
        exit(EXIT_FAILURE);
    }
    if(index(L) > 0){
        L->cursor = L->cursor->previous;
        L->index -= 1;
    } else if(index(L) == 0){
        L->index = -1;
        L->cursor = NULL;
    }
} // If cursor is defined and not at front, move cursor one step toward the front of L; if cursor is defined and at front, cursor becomes undefined; if cursor is undefined do nothing

void moveNext(List L){
    if(L == NULL){
        printf("List Error: calling moveNext() with NULL list\n");
        exit(EXIT_FAILURE);
    }
    if(index(L) != -1){
        if(index(L) != length(L) - 1){
            L->cursor = L->cursor->next;
            L->index += 1;
        } else {
            L->cursor = NULL;
            L->index = -1;
        }
    }
} // If cursor is defined and not at back, move cursor one step toward the back of L; if cursor is defined and at back, cursor becomes undefined; if cursor is undefined do nothing

void prepend(List L, int x){
    if(L == NULL){
        printf("List Error: calling prepend() with NULL list\n");
        exit(EXIT_FAILURE);
    }
    if(length(L) != 0){
        Node N = newNode(x);
        L->front->previous = N;
        N->next = L->front;
        L->front = N;

        if(index(L) >= 0){
            L->index += 1;
        }
        L->length += 1;
    } else {
        L->front = L->back = newNode(x);
        L->length = 1;
    }
} // Insert new element into L. If L is non-empty, insertion takes place before front element.

void append(List L, int x){
    if(L == NULL){
        printf("List Error: calling append() with NULL list\n");
        exit(EXIT_FAILURE);
    }
    if(length(L) != 0){
        Node N = newNode(x);
        L->back->next = N;
        N->previous = L->back;
        L->back = N;

        L->length += 1;
    } else {
        L->front = L->back = newNode(x);
        L->length = 1;
    }
} // Insert new element into L. If L is non-empty, insertion takes place after back element.

void insertBefore(List L, int x){
    if(L == NULL){


        printf("List Error: calling insertBefore() with NULL list\n");
        exit(EXIT_FAILURE);
    }
    if(length(L) <= 0){
        printf("List Error: calling insertBefore() when length is 0\n");
        exit(EXIT_FAILURE);
    }
    if(index(L) < 0){
        printf("List Error: calling insertBefore() when index is less than 0\n");
        exit(EXIT_FAILURE);
    }
    Node N = newNode(x);
    N->next = L->cursor;
    if(L->cursor->previous != NULL){
        L->cursor->previous->next = N;
        N->previous = L->cursor->previous;
    } else {
        L->front = N;
    }
    L->cursor->previous = N;

    L->index++;


    L->length++;
} // Insert new element before cursor. Pre: length()>0, index()>=0

void insertAfter(List L, int x){
    if(L == NULL){
        printf("List Error: calling insertAfter() with NULL list\n");
        exit(EXIT_FAILURE);
    }
    if(length(L) <= 0){
        printf("List Error: calling insertAfter() when length is 0\n");
        exit(EXIT_FAILURE);
    }
    if(index(L) < 0){
        printf("List Error: calling insertAfter() when index is less than 0\n");
        exit(EXIT_FAILURE);
    }

    Node N = newNode(x);
    N->previous = L->cursor;
    if(L->cursor->next != NULL){
        L->cursor->next->previous = N;
        N->next = L->cursor->next;
    } else {
        L->back = N;
    }
    L->cursor->next = N;



    L->length++;
} // Insert new element after cursor. Pre: length()>0, index()>=0

void deleteFront(List L){
    if(L == NULL){
        printf("List Error: calling deleteFront() with NULL list\n");
        exit(EXIT_FAILURE);
    }
    if(length(L) <= 0){
        printf("List Error: calling deleteFront() when length is 0\n");
        exit(EXIT_FAILURE);
    }

    if(L->cursor == L->front){
        L->index = -1;
        L->cursor = NULL;
    }

    
    if(length(L) == 1){
        free(L->front);
        L->front = L->back = NULL;
    } else {
        Node N = L->front->next;
        free(L->front);
        L->front = N;
        L->front->previous = NULL;
        if(L->index >= 0){
            L->index-=1;
        }
    }

    L->length--;
} // Delete the front element. Pre: length()>0



void deleteBack(List L){
    if(L == NULL){
        printf("List Error: calling deleteBack() with NULL list\n");
        exit(EXIT_FAILURE);
    }
    if(length(L) <= 0){
        printf("List Error: calling deleteBack() when length is 0\n");
        exit(EXIT_FAILURE);
    }

    if(L->cursor == L->back){
        L->index = -1;
        L->cursor = NULL;
    }

    if(length(L) == 1){
        free(L->back);
        L->back = L->front = NULL;
    } else {
        Node N = L->back->previous;
        free(L->back);
        L->back = N;
        L->back->next = NULL;
    }

    L->length--;
} // Delete the back element. Pre: length()>0

void delete(List L){
    if(L == NULL){
        printf("List Error: calling delete() with NULL list\n");
        exit(EXIT_FAILURE);
    }
    if(length(L) <= 0){
        printf("List Error: calling delete() when length is 0\n");
        exit(EXIT_FAILURE);
    }
    if(index(L) < 0){
        printf("List Error: calling delete() when index is less than 0\n");
        exit(EXIT_FAILURE);
    }

    
    if(L->cursor == L->front){
        deleteFront(L);
    } else if(L->cursor == L->back){
        deleteBack(L);
    } else {
        L->index = -1;
        Node prev = L->cursor->previous;
        Node nex = L->cursor->next;
        free(L->cursor);
        L->cursor = NULL;
        prev->next = nex;
        nex->previous = prev;
        L->length--;
    }

} // Delete cursor element, making cursor undefined. Pre: length()>0, index()>=0


// Other operations -----------------------------------------------------------
void printList(FILE* out, List L){
    if(L == NULL){
        printf("List Error: calling printList() with NULL list\n");
        exit(EXIT_FAILURE);
    }
    Node current = L->front;
	while(1){
		fprintf(out, "%d", current->data);
		current = current->next;
        if(current == NULL) break;
        fprintf(out, " ");
	}
	// fprintf(out, "\n");
} // Prints to the file pointed to by out, a string representation of L consisting of a space separated sequence of integers, with front on left.

List copyList(List L){
    if(L == NULL){
        printf("List Error: calling copyList() with NULL list\n");
        exit(EXIT_FAILURE);
    }

    List l = newList();
    Node current = L->front;
	while(current != NULL){
		append(l, current->data);
		current = current->next;
	}
    return l;

} // Returns a new List representing the same integer sequence as L. The cursor in the new list is undefined, regardless of the state of the cursor in L. The state of L is unchanged.

List concatList(List A, List B){
    if(A == NULL || B == NULL){
        printf("List Error: calling concatList() with NULL list\n");
        exit(EXIT_FAILURE);
    }

    List L = newList();

    Node current = A->front;
	while(current != NULL){
		append(L, current->data);
		current = current->next;
	}

    current = B->front;
    while(current != NULL){
        append(L, current->data);
        current = current->next;
    }
    return L;
} // Returns a new List which is the concatenation of A and B. The cursor in the new List is undefined, regardless of the states of the cursors in A and B. The states of A and B are unchanged.
