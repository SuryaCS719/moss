#include "List.h"

#include <stdlib.h>

// structs --------------------------------------------------------------------

// private Node type
typedef struct NodeObj* Node;

// private NodeObj type
typedef struct NodeObj
{
    ListElement data;
    Node next;
    Node previous;
} NodeObj;

// private ListObj type
typedef struct ListObj
{
    Node front;
    Node back;
    Node cursor;
    int index;
    int length;
} ListObj;

// Constructors-Destructors ---------------------------------------------------

// NewNode()
// Returns reference to new Node object. Initializes next and previous data
// fields.
Node newNode(ListElement data)
{
    Node N = malloc(sizeof(NodeObj));
    assert(N != NULL);
    N->data = data;
    N->next = NULL;
    N->previous = NULL;
    return (N);
}

// freeNode()
//  Frees heap memory pointed to by *pN, sets *pN to NULL.
void freeNode(Node *pN)
{
    if (pN != NULL && *pN != NULL)
    {
        free(*pN);
        *pN = NULL;
    }
}

// newList()
List newList()
{
    List L;
    L = malloc(sizeof(ListObj));
    assert(L != NULL);
    L->front = L->back = L->cursor = NULL;
    L->length = 0;
    L->index = -1;
    return (L);
}

// freeList()
void freeList(List *pL)
{
    if (pL != NULL && *pL != NULL)
    {
        while ((*pL)->length > 0)
        {
            deleteFront(*pL);
        }
        free(*pL);
        *pL = NULL;
    }
}

// Access functions -----------------------------------------------------------

// length()
// Returns the number of elements in L.
int length(List L)
{
    if (L == NULL)
    {
        printf("List Error: calling length() on NULL Queue reference\n");
        exit(EXIT_FAILURE);
    }

    return (L->length);
}

// index()
// Returns index of cursor element if defined, -1 otherwise.
int index(List L)
{
    if (L == NULL)
    {
        printf("List Error: calling index() on NULL Queue reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->cursor == NULL)
    {
        return -1;
    }
    return (L->index);
}

// int front()
// Returns front element of L. Pre: length()>0
int front(List L)
{
    if (L == NULL){
        printf("List Error: Calling front() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }

    if (length(L) == 0)
    {
        printf("List Error: calling front() on an EMPTY List\n");
        exit(EXIT_FAILURE);
    }

    return (L->front->data);
}

// int back()
// Returns back element of L. Pre: length()>0
int back(List L)
{
    if (L == NULL){
        printf("Calling back() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->length == 0)
    {
        printf("List Error: calling back() on an EMPTY List\n");
        exit(EXIT_FAILURE);
    }

    return (L->back->data);
}

// int get()
// Returns cursor element of L. Pre: length()>0, index()>=0
int get(List L)
{
    if (L == NULL){
        printf("Calling back() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->length == 0 || L->index < 0)
    {
        printf("List Error: calling get() on an EMPTY List\n");
        exit(EXIT_FAILURE);
    }
    return (L->cursor->data);
}

// bool equals()
/* Returns true iff Lists A and B contain the same sequence of elements,
returns false otherwise.*/
bool equals(List A, List B)
{
    if (A == NULL || B == NULL)
    {
        printf("List Error: calling equals() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (A->length != B->length)
    {
        return false;
    }
    Node N, M;
    N = A->front;
    M = B->front;

    while (N != NULL && M != NULL)
    {
        if (N->data != M->data)
        {
            return false;
        }
        N = N->next;
        M = M->next;
    }
    return true;
}

// Manipulation procedures ----------------------------------------------------

// clear()
// Resets L to its original empty state.
void clear(List L)
{

    if (L == NULL)
    {
        printf("List Error: calling clear() on NULL List reference\n");
    }


    while (length(L) > 0)
    {
        deleteFront(L); // because deletefront will iterate the next node to be the
                        // front so i dont have to
    }
    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->index = -1;
    L->length = 0;
}

// set()
// Overwrites the cursor elementâ€™s data with x.
// Pre: length()>0, index()>=0
void set(List L, int x)
{
    if (L == NULL){
        printf("Calling set() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->length == 0)
    {
        printf("List Error: calling set() on an EMPTY list\n");
        exit(EXIT_FAILURE);
    }
    if (index(L) < 0)
    {
        printf("List error: calling set() on undefined cursor\n");
        exit(EXIT_FAILURE);
    }


    L->cursor->data = x;
}

// movefront()
// If L is non-empty, sets cursor under the front element, otherwise does
// nothing.
void moveFront(List L)
{
    if (L == NULL){
        printf("Calling moveFront() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->length > 0)
    {
        L->cursor = L->front;
        L->index = 0; // can't do L->index = L->cursor because cursor is a pointer
    }
}

// moveBack
// If L is non-empty, sets cursor under the back element, otherwise does
// nothing.
void moveBack(List L)
{
    if (L == NULL){
        printf("Calling moveBack() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->length > 0)
    {
        L->cursor = L->back;
        L->index = L->length - 1;
    }
}

// movePrev()
/* If cursor is defined and not at front, move cursor one
  step toward the front of L; if cursor is defined and at
  front, cursor becomes undefined; if cursor is undefined
  do nothing*/
void movePrev(List L)
{
    if (L == NULL)
    {
        printf("List Error: calling movePrev() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->cursor != NULL && L->cursor == L->front)
    {
        L->cursor = NULL;
        L->index = -1;
    }
    if (L->cursor != NULL && L->cursor != L->front)
    {
        L->cursor = L->cursor->previous;
        L->index--;
    }
}

// moveNext()
/* If cursor is defined and not at back, move cursor one
  step toward the back of L; if cursor is defined and at
  back, cursor becomes undefined; if cursor is undefined
  do nothing */
void moveNext(List L)
{
    if (L == NULL){
        printf("Calling moveNext() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->cursor != NULL && L->cursor == L->back)
    {
        L->cursor = NULL;
        L->index = -1;
    }


    if (L->cursor != NULL && L->cursor != L->back)
    {
        L->cursor = L->cursor->next;
        L->index++;
    }
}

// prepend()
// Insert new element into L. If L is non-empty,
// insertion takes place before front element.
void prepend(List L, int x)
{
    if (L == NULL)
    {
        printf("List Error: calling prepend() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    Node N = newNode(x);
    if (L->length ==0)
    { // if N is the only element then it's the front and back pointer
        L->front = L->back = N;
    }
    else
    {
        N->next = L->front; // New Node's next pointer will point to the front node.
        L->front->previous =
            N;        // The front node's previous pointer will point to the new Node
        L->front = N; // The front Node now is the New node
    }
    L->length++;
    if (L->cursor != NULL)
    { // because the cursor doesn't change to a diff node,
      // but its position # will shift
        if (L->index != -1)
        {
            L->index++;
        }
    }
}

// append()
// Insert new element into L. If L is non-empty,
// insertion takes place after back element.
void append(List L, int x)
{
    if (L == NULL)
    {
        printf("List Error: calling append() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }

    Node N = newNode(x);

    if (L->length ==
        0)
    { // if N is the only element then it's the front and back pointer
        L->front = L->back = N;
    }
    else
    {
        N->previous = L->back; // New Node's previous pointer will point to the
                               // current back node
        L->back->next = N;     // The current back node's previous pointer will point to
                               // the new Node
        L->back = N;           // The back Node now is now set to be the New node
    }
    L->length++;

    if (L->cursor == NULL)
    {
        L->index = -1;
    }
}

// insertBefore()
// Insert new element before cursor. Pre: length()>0, index()>=0
void insertBefore(List L, int x)
{
    if (L == NULL)
    {
        printf("List Error: calling insertBefore() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->index < 0)
    {
        printf("List Error: calling insertBefore() on undefined cursor position\n");
        exit(EXIT_FAILURE);
    }
    if (L->length > 0 && L->index >= 0)
    {

        if (L->index == 0)
        { // if cursor is at front
            prepend(L, x);
        }

        else
        { // cursor defined but not at front
            Node N = newNode(x);
            N->previous = L->cursor->previous;
            L->cursor->previous->next = N;


            L->cursor->previous = N;
            N->next = L->cursor;
            L->index++;
            L->length++;
        }
    }
}

// insertAfter()
// Insert new element after cursor. Pre: length()>0, index()>=0
void insertAfter(List L, int x)
{
    if (L == NULL)
    {
        printf("List Error: calling insertAfter() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->index < 0)
    {
        printf("List Error: calling insertAfter() on undefined cursor position\n");
        exit(EXIT_FAILURE);
    }
    if (L->length > 0 && L->index >= 0)
    {

        if (L->index == L->length - 1)
        {
            append(L, x);
        }
        else
        {
            Node N = newNode(x);
            L->cursor->next->previous = N;
            N->next = L->cursor->next;


            L->cursor->next = N;
            N->previous = L->cursor;
            L->length++;
        }
    }
}

// deleteFront()
// Delete the front element. Pre: length()>0
void deleteFront(List L)
{
    if (L == NULL)
    {
        printf("List Error: calling deleteFront() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->length == 0)
    {
        printf("List Error: calling deleteFront() on an empty List reference\n");
        exit(EXIT_FAILURE);
    }

    Node N = L->front;

    if (L->index == 0) //if cursor at front
    {
        L->cursor = NULL;
        L->index = -1;
    }else if (L->index > 0) // cursor defined & not at front
    {
        L->index--;
    }

    if (L->length > 1) // if theres more than 1 (else only 1)
    {
        L->front = L->front->next;
        L->front->previous = NULL;
        L->length--;
    } else{ // only 1 node in list
        L->front = L->back = L->cursor = NULL;
        L->index = -1;
        L->length = 0;
    }


    freeNode(&N);
}

// deleteBack()
// Delete the back element. Pre: length()>0
void deleteBack(List L)
{
    if (L == NULL)
    {
        printf("List Error: calling deleteBack() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->length == 0)
    {
        printf("List Error: calling deleteBack() on an empty List reference\n");
        exit(EXIT_FAILURE);
    }

    Node N = L->back;

// if cursor is at back node
    if (L->index == L->length-1)
    { 
        L->cursor = NULL;
        L->index = -1;
    }

    if (L->length > 1) // if theres more than 1 (else only 1)
    {
        L->back = L->back->previous;
        L->back->next = NULL;
        L->length--;
    } else{ // only 1 node in list
        L->front = L->back = L->cursor = NULL;
        L->index = -1;
        L->length = 0;
    }

    freeNode(&N);
}

// delete()
// Delete cursor element, making cursor undefined. // Pre: length()>0, index()>=0
void delete(List L)
{
    if (L == NULL)
    {
        printf("List Error: calling delete() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->length == 0)
    {
        printf("List Error: calling delete() on an empty List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->index < 0)
    {
        printf("List Error: calling delete() on NULL cursor\n");
        exit(EXIT_FAILURE);
    }

    // N is basically saying struct NodeObj* N so N is a pointer that is now set to cursor
    // because you are able to access N's contents with &N, and we dont want to touch cursor

    // edge case where cursor is at the front of the list
    if (L->index == 0){
        deleteFront(L);
    } 
    else if (L->index == L->length - 1) // if cursor is at back
    { 
        deleteBack(L);
    }
    else // if cursor is in middle of list now
    { 
        Node N = L->cursor;
        L->cursor->previous->next = L->cursor->next;
        L->cursor->next->previous = L->cursor->previous;
        L->cursor = NULL;
        L->index = -1;
        L->length--;
        freeNode(&N);
    }
}

// printList()
// Prints to the file pointed to by out, a string representation of L consisting
// of a space separated sequence of integers, with front on left.
void printList(FILE* out, List L){
    if (L == NULL){
        printf("List Error: Calling printList() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    Node N = L->front;
    while(N != NULL){
        fprintf(out, "%d ", N->data);
        N = N->next; 
    }
    fprintf(out, "\n"); // in case this fn is called multiple times, makes it so 1 line is 1 list
}

// copyList() 
// Returns a new List representing the same integer sequence as L.
  //The cursor in the new list is undefined,regardless of the state of the cursor in L. 
  //The state of L is unchanged.
List copyList(List L){ 
    if (L == NULL){
        printf("List Error: Calling copyList() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }

    List Copy = newList(); // makes an empty list to place data into, blank because its creating an empty one
    Node N = L->front; //makes a pointer that starts from the front node in existing list

    while(N != NULL){ // iterates through list
        append(Copy, N->data); // append takes the list it wants to place data into, and the data from the node
        N = N->next;
    }
    return Copy;
}
