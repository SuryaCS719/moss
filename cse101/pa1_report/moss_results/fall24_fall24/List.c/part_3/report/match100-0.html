#include <stdio.h>
#include "List.h"
#include <assert.h>




typedef struct NodeObj* Node;

typedef struct NodeObj {
        Node next;
        Node prev;
        int data;
} NodeObj;

typedef struct ListObj {
        Node front;
        Node back;
        Node cursor;
        int index;
        int length;
}ListObj;


Node createNode(int data) {
        Node N = malloc(sizeof(NodeObj));
        assert(N);
        N->data = data;
        N->prev = NULL;
        N->next = NULL;
        return N;
}

List newList(void) {


        List L = malloc(sizeof(ListObj));
        assert(L);
        L->front = NULL;
        L->back = NULL;
        L->cursor = NULL;
        L->index = -1;
        L->length = 0;
        return L;
}

bool isEmpty(List L) {
        if (L->length < 1) {
                return true;
        }
        return false;
}

int length(List L) {
        return L->length;
}

int index(List L) {
        return L->index;
}

int front(List L) {
        if (L == NULL) {


                fprintf(stderr, "Error: Trying to call front() using a NULL pointer");
                exit(EXIT_FAILURE);
        }
        if (isEmpty(L)) {
                fprintf(stderr, "Error: Trying to call front() on an empty list");
                exit(EXIT_FAILURE);
        }
        return L->front->data;
}


int back(List L) {
         if (L == NULL) {


                fprintf(stderr, "Error: Trying to call back() using a NULL pointer");
                exit(EXIT_FAILURE);
        }
        if (isEmpty(L)) {
                fprintf(stderr, "Error: Trying to call back() on an empty list");
                exit(EXIT_FAILURE);
        }
        return L->back->data;
}


int get(List L) {
        if (L == NULL) {
                fprintf(stderr, "Error: Trying to call get() using a NULL pointer");
                exit(EXIT_FAILURE);
        }
        if (isEmpty(L)) {
                fprintf(stderr, "Error: Trying to call get() on an empty list");
                exit(EXIT_FAILURE);
        }
        if (L->cursor == NULL || L->index < 0) {
                fprintf(stderr, "Error: Trying to call get() on an undefined cursor");
                exit(EXIT_FAILURE);
        }
        return L->cursor->data;
}

bool equals(List A, List B) {
        if (A == NULL || B == NULL) {
                fprintf(stderr, "Error: Trying to call equals() using a NULL pointer");
                exit(EXIT_FAILURE);
        }
        if(A->length != B->length) {
                return false;
        }
        Node nodeA = A->front;
        Node nodeB = B->front;

        while(nodeA != NULL && nodeB != NULL) {
                if(nodeA->data != nodeB->data) {
                        return false;
                }
                nodeA = nodeA->next;
                nodeB = nodeB->next;

        }
        return true;
}


void freeNode(Node* pN) {
        if( pN != NULL && *pN != NULL) {
                free(*pN);
                *pN = NULL;
        }
}


void freeList(List *pL) {
        if(pL != NULL && *pL != NULL) {
                List L = *pL;
                while(!isEmpty(L)) {
                        deleteFront(L);
                }
        }
        free(*pL);
        *pL = NULL;
}


void clear(List L) {
        assert(L);
        while(L->length > 0) {
                deleteFront(L);
        }
        L->front = NULL;
        L->back = NULL;
        L->cursor = NULL;
        L->index = -1;
        L->length = 0;
}

void set(List L, int x) {
        if (L == NULL) {
                fprintf(stderr, "Error: Trying to call set() using a NULL pointer");


                exit(EXIT_FAILURE);
        }
        if (isEmpty(L)) {
                fprintf(stderr, "Error: Trying to call set() on an empty list");
                exit(EXIT_FAILURE);
        }
        if (L->index < 0) {
                fprintf(stderr, "Error: Trying to call set() on an undefined cursor");
                exit(EXIT_FAILURE);
        }
        L->cursor->data = x;
}

void moveFront(List L) {
        if(isEmpty(L))
                return;
        L->cursor = L->front;
        L->index = 0;
}

void moveBack(List L) {
        if(isEmpty(L)) {
                return;
        }
        L->cursor = L->back;
        L->index = L->length - 1;
}

void movePrev(List L) {
        if(L->cursor == NULL) {
                return;
        }
        if(L->cursor == L->front) {
                L->cursor = NULL;
                L->index = -1;
        }else {
                L->cursor = L->cursor->prev;
                L->index--;
        }
}

void moveNext(List L) {
        if(L->cursor == NULL) {
                return;
        }
        if(L->cursor == L->back ) {
                L->cursor = NULL;
                L->index = -1;
        } else {
                L->cursor = L->cursor->next;
                L->index++;
        }
}


void prepend(List L, int x) {
        Node N = createNode(x);
        if(isEmpty(L)) {
                L->front = L->back = N;
                L->length++;
                L->index++;
                return;
        } else if(L->cursor != NULL) {
                L->front->prev = N;
                N->next = L->front;
                L->front = N;
                L->length++;
                L->index++;
                return;
        }
        L->front->prev = N;
        N->next = L->front;
        L->front = N;
        L->length++;

}

void append(List L, int x) {
        Node N = createNode(x);
        if(isEmpty(L)) {
                L->front = L->back = N;
                L->length++;

        } else {
        L->back->next = N;
        N->prev = L->back;
        L->back = N;


        L->length++;
        }
}


void insertBefore(List L, int x) {
        if (L == NULL) {
                fprintf(stderr, "Error: Trying to call insertBefore() using a NULL pointer");
                exit(EXIT_FAILURE);
        }
        if (isEmpty(L)) {
                fprintf(stderr, "Error: Trying to call insertBefore() on an empty list");
                exit(EXIT_FAILURE);
        }
        if (L->cursor == NULL) {
                fprintf(stderr, "Error: Trying to call insertBefore() on an undefined cursor");
                exit(EXIT_FAILURE);
        }
        Node N = createNode(x);
        if (L->cursor == L->front) {
                L->cursor->prev = N;
                N->next = L->cursor;
                N->prev = NULL;
                L->index++;
                L->length++;
                L->front = N;
                return;
        } else {
                N->prev = L->cursor->prev;
                N->next = L->cursor;
                L->cursor->prev->next = N;
                L->cursor->prev = N;
                L->index++;
                L->length++;
        }
}


void insertAfter(List L, int x) {
        if (L == NULL) {
                fprintf(stderr, "Error: Trying to call insertAfter() using a NULL pointer");
                exit(EXIT_FAILURE);
        }
        if (isEmpty(L)) {
                fprintf(stderr, "Error: Trying to call insertAfter() on an empty list");
                exit(EXIT_FAILURE);
        }
        if (L->cursor == NULL) {
                fprintf(stderr, "Error: Trying to call insertAfter() on an undefined cursor");
                exit(EXIT_FAILURE);
        }
        Node N = createNode(x);
        if(L->cursor->next == NULL) {
                L->cursor->next = N;
                N->prev = L->cursor;
                L->length++;
                L->back = N;
        } else {
                N->prev = L->cursor;
                N->next = L->cursor->next;
                L->cursor->next->prev = N;
                L->cursor->next = N;
                L->length++;

        }
}


void deleteFront(List L) {
    if (L == NULL) {


        fprintf(stderr, "Error: Trying to call deleteFront() using a NULL pointer\n");
        exit(EXIT_FAILURE);
    }
    if (isEmpty(L)) {
        fprintf(stderr, "Error: Trying to call deleteFront() on an empty list\n");
        exit(EXIT_FAILURE);
    }
    Node temp = L->front;
    if (L->length == 1) {
        L->front = NULL;
        L->back = NULL;
        L->cursor = NULL;
        L->index = -1;
    } else {
        L->front = L->front->next;
        L->front->prev = NULL;
        if (L->cursor == temp) {
            L->cursor = NULL;
            L->index = -1;
        } else if (L->index != -1) {
            L->index--;
        }
    }
    L->length--;
    temp->next = NULL;
    temp->prev = NULL;
    freeNode(&temp);
}

void deleteBack(List L) {
    if (L == NULL) {


        fprintf(stderr, "Error: Trying to call deleteBack() using a NULL pointer\n");
        exit(EXIT_FAILURE);
    }
    if (isEmpty(L)) {
        fprintf(stderr, "Error: Trying to call deleteBack() on an empty list\n");
        exit(EXIT_FAILURE);
    }
    Node temp = L->back;
    if (L->length == 1) {
        L->front = NULL;
        L->back = NULL;
        L->cursor = NULL;
        L->index = -1;
    } else {
        L->back = L->back->prev;
        L->back->next = NULL;
        if (L->cursor == temp) {
            L->cursor = NULL;
            L->index = -1;
        }
    }
    L->length--;
    temp->next = NULL;
    temp->prev = NULL;
    freeNode(&temp);
}


void delete(List L) {
        if (L == NULL) {
                fprintf(stderr, "Error: Trying to call delete() using a NULL pointer");
                exit(EXIT_FAILURE);
        }
        if (isEmpty(L)) {
                fprintf(stderr, "Error: Trying to call delete() on an empty list");
                exit(EXIT_FAILURE);
        }
        if (L->cursor == NULL) {
                fprintf(stderr, "Error: Trying to call delete() on an undefined cursor");
                exit(EXIT_FAILURE);
        }
        Node temp = L->cursor;
        if (L->length == 1) {
                L->front = NULL;
                L->back = NULL;
                L->cursor = NULL;
                L->index = -1;
        }
        else if(L->cursor == L->front && !(isEmpty(L))) {
                temp->next->prev = NULL;
                L->front = temp->next;
        }
        else if(L->cursor == L->back && !(isEmpty(L))) {
                temp->prev->next = NULL;
                L->back = temp->prev;
        }
        else {
                temp->next->prev = temp->prev;
                temp->prev->next = temp->next;
        }
        L->index = -1;
        L->length--;
        temp->next = NULL;
        temp->prev = NULL;
        freeNode(&temp);
}



void printList(FILE* out, List L) {
        Node temp = L->front;
        while(temp) {
                fprintf(out, "%d ", temp->data);
                temp = temp->next;
        }
}

List copyList(List L) {
        if (L == NULL) {
                fprintf(stderr, "Error: Trying to call delete() using a NULL pointer");
                exit(EXIT_FAILURE);
        }
        Node temp = L->front;
        List newL = newList();
        temp = L->front;
        while(temp) {
                append(newL, temp->data);
                temp = temp->next;
        }
        return newL;
}
