#include<stdio.h>
#include<stdlib.h>
#include<stdbool.h>
#include<assert.h>
#include "List.h"
//structs
typedef struct NodeObj{
    int data;
    struct NodeObj* prev;


    struct NodeObj* next;
}NodeObj;

typedef NodeObj* Node;

typedef struct ListObj{
    Node front;
    Node back;
    Node cursor;
    int cursor_index;
    int length;
}ListObj;

//Constructors and Deconstructors
Node newNode(int data, Node prev, Node next){
    Node N = malloc(sizeof(NodeObj));
    assert(N!=NULL);
    N->data = data;
    N->prev = prev;
    N->next = next;
    return N;
}

void freeNode(Node* pN){
    if(pN!=NULL && *pN!=NULL){


        free(*pN);
        *pN = NULL;
    }
}

List newList(void){
    List L = malloc(sizeof(ListObj));
    assert(L!=NULL);
    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->cursor_index = -1;
    L->length = 0;
    return L;
}

void freeList(List* pL){
    if(pL!=NULL && *pL!=NULL){
        while(length(*pL)!=0){
            deleteFront(*pL);
        }
        free(*pL);
        *pL = NULL;
    }
}

//Access functions
int length(List L){
    return L->length;
}

int index(List L){
    return L->cursor_index;
}

int front(List L){
    if(length(L)==0){
        fprintf(stderr,"List Error: calling front() on empty list\n");
        exit(EXIT_FAILURE);
    }
    return L->front->data;
}

int back(List L){
    if(length(L)==0){
        fprintf(stderr,"List Error: calling back() on empty list\n");
        exit(EXIT_FAILURE);
    }
    return L->back->data;
}

int get(List L){
    if(length(L)==0){
        fprintf(stderr,"List Error: calling get() on empty list\n");
        exit(EXIT_FAILURE);
    }
    if(index(L)==-1){
        fprintf(stderr,"List Error: calling get() on a NULL cursor\n");
        exit(EXIT_FAILURE);
    }
    return L->cursor->data;
}

bool equals(List A, List B){
    if(length(A)!=length(B)){
        return false;
    }
    moveFront(A);
    moveFront(B);
    while(index(A)!=-1){
        if(get(A)!=get(B)){
            return false;
        }
        moveNext(A);
        moveNext(B);
    }
    return true;
}

//Manipulation procedures
void clear(List L){
    if(L!=NULL){
        while(length(L)!=0){
            deleteFront(L);
        }
    }
}

void set(List L, int x){
    if(length(L)==0){
        fprintf(stderr,"List Error: calling set() on empty list\n");
        exit(EXIT_FAILURE);
    }
    if(index(L)==-1){
        fprintf(stderr,"List Error: calling set() on a NULL cursor\n");
        exit(EXIT_FAILURE);
    }
    L->cursor->data=x;
}

void moveFront(List L){
    if(length(L)!=0){
        L->cursor = L->front;
        L->cursor_index = 0;
    }
}

void moveBack(List L){
    if(length(L)!=0){
        L->cursor = L->back;
        L->cursor_index = length(L)-1;
    }
}

void movePrev(List L){
    if(index(L)!=-1){
        if(L->cursor != L->front){
            L->cursor=L->cursor->prev;
            L->cursor_index--;
        } else {
            L->cursor=NULL;
            L->cursor_index = -1;
        }
    }
}

void moveNext(List L){
    if(index(L)!=-1){
        if(L->cursor != L->back){
            L->cursor=L->cursor->next;
            L->cursor_index++;
        } else {
            L->cursor=NULL;
            L->cursor_index = -1;
        }
    }
}

void prepend(List L, int x){
    if(length(L)==0){
        Node new = newNode(x,NULL,NULL);
        L->front = new;
        L->back = new;
    }else{
        Node new = newNode(x,NULL,L->front);
        L->front->prev = new;
        L->front = new;
        if(L->cursor_index!=-1){
            L->cursor_index++;
        }
    }
    L->length++; 
}

void append(List L, int x){
    if(length(L)==0){
        Node new = newNode(x,NULL,NULL);
        L->front = new;
        L->back = new;
    }else{
        Node new = newNode(x,L->back,NULL);
        L->back->next = new;
        L->back = new;
    }
    L->length++;
}

void insertBefore(List L, int x){
    if(length(L)==0){
        fprintf(stderr,"List Error: calling insertBefore() on empty list\n");
        exit(EXIT_FAILURE);
    }
    if(index(L)==-1){
        fprintf(stderr,"List Error: calling insertBefore() on a NULL cursor\n");
        exit(EXIT_FAILURE);
    }
    if(index(L)==0){
        prepend(L,x);
    }else{
        Node new = newNode(x, L->cursor->prev, L->cursor);
        L->cursor->prev->next = new;
        L->cursor->prev = new;
        L->cursor_index++;
        L->length++;
    }
}

void insertAfter(List L, int x){
    if(length(L)==0){
        fprintf(stderr,"List Error: calling insertAfter() on empty list\n");
        exit(EXIT_FAILURE);
    }
    if(index(L)==-1){
        fprintf(stderr,"List Error: calling insertAfter() on a NULL cursor\n");
        exit(EXIT_FAILURE);
    }
    if(index(L)==length(L)-1){
        append(L,x);
    }else{
        Node new = newNode(x, L->cursor, L->cursor->next);
        L->cursor->next->prev = new;
        L->cursor->next = new;
        L->length++;
    }
}

void deleteFront(List L){
    if(length(L)==0){
        fprintf(stderr,"List Error: calling deleteFront() on empty list\n");
        exit(EXIT_FAILURE);
    }
    Node delete = L->front;
    if(length(L)>1){
        L->front->next->prev = NULL;
        L->front=L->front->next;
    }else{
        L->front=NULL;
        L->back=NULL;
        L->cursor=NULL;
    }
    if(index(L)==0){
        L->cursor_index=-1;
    }else if(index(L)!=-1){
        L->cursor_index--;
    }
    freeNode(&delete);
    L->length--;
}

void deleteBack(List L){
    if(length(L)==0){
        fprintf(stderr,"List Error: calling deleteBack() on empty list\n");
        exit(EXIT_FAILURE);
    }
    Node delete = L->back;
    if(length(L)>1){
        L->back->prev->next = NULL;
        L->back=L->back->prev;
    }else{
        L->front=NULL;
        L->back=NULL;
        L->cursor=NULL;
    }
    if(index(L)==length(L)-1){
        L->cursor_index=-1;
    }
    freeNode(&delete);
    L->length--;
}

void delete(List L){
    if(length(L)==0){
        fprintf(stderr,"List Error: calling delete() on empty list\n");
        exit(EXIT_FAILURE);
    }
    if(index(L)==-1){
        fprintf(stderr,"List Error: calling delete() on a NULL cursor\n");
        exit(EXIT_FAILURE);
    }
    if(index(L)==0){
        deleteFront(L);
    }else if(index(L)==length(L)-1){
        deleteBack(L);
    } else {
        Node delete = L->cursor;
        L->cursor->prev->next=L->cursor->next;
        L->cursor->next->prev=L->cursor->prev;
        freeNode(&delete);
        L->cursor_index=-1;
        L->length--;
    }
}

// Other operations
void printList(FILE* out, List L){
    Node temp_cursor = L->front;
    while(temp_cursor != NULL && temp_cursor->next != NULL){
        fprintf(out, "%d", temp_cursor->data);
        fprintf(out, " ");
        temp_cursor = temp_cursor->next;
    }
    fprintf(out, "%d", temp_cursor->data);
}

List copyList(List L){
    List new = newList();
    Node temp_cursor = L->front;
    while(temp_cursor != NULL){
        append(new, temp_cursor->data);
        temp_cursor = temp_cursor->next;
    }
    return new;
}

List concatList(List A, List B){
    List new = copyList(A);
    Node temp_cursor = B->front;
    while(temp_cursor!=NULL){
        append(new, temp_cursor->data);
        temp_cursor=temp_cursor->next;
    }
    return new;
}
