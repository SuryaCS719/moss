#include <stdio.h>
#include <stdlib.h>
#include "List.h"

List newList(void){
    List L = malloc(sizeof(struct ListObj));
    L->front = L->back = L->cursor = NULL;
    L->index = -1;
    L->length = 0;
    return L;
}

void freeList(List* pL){
    if (pL != NULL && *pL != NULL){


        while (length(*pL) > 0){
            deleteFront(*pL);
        }
        free(*pL);
        *pL = NULL;
    }
}

int length(List L){
    if (L == NULL){
        printf("Error: NULL\n");
        exit(EXIT_FAILURE);
    }
    return L->length;
}

int getIndex(List L){
    if (L == NULL){
        printf("Error: NULL\n");
        exit(EXIT_FAILURE);
    }
    return L->index;
}

int front(List L){
    if (L == NULL){
        printf("Error: NULL\n");
        exit(EXIT_FAILURE);
    }
    return L->front->data;
}

int back(List L){
    if (L == NULL){
        printf("Error: NULL\n");
        exit(EXIT_FAILURE);
    }
    if (L->length == 0) {
        printf("Error: NULL\n");
        exit(EXIT_FAILURE);
    }


    return L->back->data;
}

int get(List L){
    if (L==NULL || L->length<=0 || L->index<0){
        printf("Error: NULL\n");
        exit(EXIT_FAILURE);
    }
    return L->cursor->data;
}

bool equals(List A, List B){
    if (A==NULL || B == NULL){
        printf("Error: NULL\n");
        exit(EXIT_FAILURE);
    }
    if (A->length != B->length){
        return false;
    }
    Node nodeA = A->front;
    Node nodeB = B->front;
    while (nodeA!=NULL && nodeB!=NULL){
        if (nodeA->data != nodeB->data){
            return false;
        }
        nodeA = nodeA->next;
        nodeB = nodeB->next;
    }
    return true; 
}

void clear(List L){
    while (length(L)>0){
        deleteFront(L);
    }
    L->cursor = NULL;


    L->index = -1;
}

void set(List L, int x){
    if (L==NULL || L->length <= 0 || L->index <0){
        printf("Error: NULL\n");
        exit(EXIT_FAILURE);
    }
    L->cursor->data=x;
}

void moveFront(List L){
    if (L==NULL || L->length<=0){
        printf("Error: NULL\n");


        exit(EXIT_FAILURE);
    }
    L->cursor = L->front;
    L->index = 0;
}

void moveBack(List L){
    if (L==NULL || L->length<=0){
        printf("Error: NULL\n");
        exit(EXIT_FAILURE);
    }


    L->cursor = L->back;
    L->index = L->length - 1;
}

void movePrev(List L){
    if (L==NULL || L->cursor == NULL || L->index==0){
        L->cursor = NULL;
        L->index = -1;
        return;
    }
    if (L->cursor != NULL){


        L->cursor = L->cursor->prev;
        L->index--;
    }
}

void moveNext(List L){
    if (L==NULL || L->cursor == NULL || L->index == L->length -1){
        L->cursor = NULL;
        L->index = -1;
        return;
    }
    if (L->cursor != NULL){
        L->cursor = L->cursor->next;
        L->index++;
    }
}

void prepend(List L, int x){
    if (L==NULL){
        printf("Error: NULL\n");
        exit(EXIT_FAILURE);
    }
    Node N = malloc(sizeof(struct NodeObj));
    if (N==NULL){
        printf("Error: NULL\n");
        exit(EXIT_FAILURE);
    }
    N->data = x;
    //N->next = L->front;
    N->prev = NULL;

    if(L->length == 0){
        N->next = NULL;
        L->front = L->back = N;
    }
    else{
        N->next = L->front;
        L->front->prev=N;
        L->front=N;
    }
    L->length++;
}

void append(List L, int x){
    if (L==NULL){
         printf("Error: NULL\n");
        exit(EXIT_FAILURE);
    }
    Node N = malloc(sizeof(struct NodeObj));
    if (N==NULL){
         printf("Error: NULL\n");
        exit(EXIT_FAILURE);
    }
    N->data = x;
    //N->prev = L->back;
    N->next = NULL;

    if (L->length == 0){
        N->prev = NULL;
        L->front = L->back = N;
    }
    else{
        N->prev = L->back;
        L->back->next = N;
        L->back = N;
    }
    L->length++;
}

void insertBefore(List L, int x){
    if (L->index<0){
        printf("Error: NULL\n");
        exit(EXIT_FAILURE);
    }
    if (L->cursor == L->front){
        prepend(L,x);
    }
    else{
        Node N = malloc(sizeof(struct NodeObj));
        N->data = x;
        N->next = L->cursor;
        N->prev = L->cursor->prev;
        L->cursor->prev->next = N;
        L->cursor->prev = N;
        L->length++;
        L->index++;
    }
}

void insertAfter(List L, int x){
    if (L->index<0){
        printf("Error: NULL\n");
        exit(EXIT_FAILURE);
    }
    if (L->cursor == L->back){
        append(L,x);
    }
    else{
        Node N = malloc(sizeof(struct NodeObj));
        N->data = x;
        N->prev = L->cursor;
        N->next = L->cursor->next;
        L->cursor->next->prev = N;


        L->cursor->next = N;
        L->length++;
    }
}

void deleteFront(List L) {
    if (L == NULL || L->length <= 0) {
        printf("Error: NULL\n");
        exit(EXIT_FAILURE);
    }

    Node N = L->front;
    if (L->length == 1) {
        L->front = L->back = NULL;
    } else {
        L->front = L->front->next;
        L->front->prev = NULL;
    }

    if (L->index == 0) {
        L->cursor = NULL;
        L->index = -1;
    } else {
        L->index--;
    }

    free(N);
    L->length--;
}

void deleteBack(List L) {
    if (L == NULL || L->length <= 0) {
        printf("Error: NULL\n");
        exit(EXIT_FAILURE);
    }

    Node N = L->back;
    if (L->length == 1) {
        L->front = L->back = NULL;
    } else {
        L->back = L->back->prev;
        L->back->next = NULL;
    }

    if (L->index == L->length - 1) {
        L->cursor = NULL;
        L->index = -1;
    } 



    free(N);
    L->length--;
}

void delete(List L) {
    if (L == NULL || L->length <= 0 || L->index < 0) {
        printf("Error\n");
        exit(EXIT_FAILURE);
    }

    if (L->cursor == L->front) {
        deleteFront(L);
    } else if (L->cursor == L->back) {
        deleteBack(L);
    } else {
        Node N = L->cursor;
        L->cursor->prev->next = L->cursor->next;
        L->cursor->next->prev = L->cursor->prev;
        free(N);
        L->length--;
        L->cursor = NULL;
        L->index = -1;
    }
}

void printList(FILE* out, List L) {
    if (L==NULL){
        printf("Error\n");
        exit(EXIT_FAILURE);
    }
    Node N = L->front;
    while (N != NULL) {
        fprintf(out, "%d ", N->data);
        //append(copy, N->data);
        N = N->next;
    }
    fprintf(out, "\n");
}

List copyList(List L) {
    if (L==NULL){
        printf("Error\n");
        exit(EXIT_FAILURE);
    }

    List copy = newList();
    Node N = L->front;

    while (N != NULL) {
        append(copy, N->data);
        N = N->next;
    }
    return copy;
}
