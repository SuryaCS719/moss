#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>
#include <assert.h>

#include "List.h"

typedef struct NodeObj {
    int data;
    struct NodeObj* next;
    struct NodeObj* prev;
} NodeObj;

typedef NodeObj* Node; // shorthand for a pointer to NodeObj

typedef struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
    int index;
} ListObj;

typedef ListObj* List;

// create a new list
List newList(void) {
    List L = malloc(sizeof(ListObj));
    assert(L != NULL); 

    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->length = 0;
    L->index = -1;

    return L;
}

// free the list
void freeList(List* pL) {
    if (pL != NULL && *pL != NULL) {


        while (length(*pL) != 0) {
            deleteFront(*pL);
        }
        free(*pL);
        *pL = NULL;
    }
}

// access functions

int length(List L) {
    if (L == NULL) {
        printf("List error: calling length on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    return L->length;
}

int index(List L) {
    if (L == NULL) {
        printf("List error: calling index on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    return L->index;
}

int front(List L) {
    if (L == NULL) {
        printf("List error: calling front on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->length == 0) {
        printf("List error: calling front on an empty List\n");
        exit(EXIT_FAILURE);
    }
    return L->front->data;
}

int back(List L) {
    if (L == NULL) {
        printf("List error: calling back on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->length == 0) {
        printf("List error: calling back on an empty List\n");
        exit(EXIT_FAILURE);
    }
    return L->back->data;
}

int get(List L) {
    if (L == NULL) {
        printf("List error: calling get on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->length == 0) {
        printf("List error: calling get on an empty List\n");
        exit(EXIT_FAILURE);
    }
    if (L->index < 0) {
        printf("List error: calling get() when cursor is undefined\n");
        exit(EXIT_FAILURE);
    }

    return L->cursor->data; 
}

bool equals(List A, List B) {
    if (A == NULL && B == NULL) {
        return true;
    }
    if (A == NULL || B == NULL) {
        return false;
    }
    if (A->length != B->length) {
        return false;
    }

    // compare lists
    Node currentA = A->front;
    Node currentB = B->front;

    while (currentA != NULL && currentB != NULL) {
        if (currentA->data != currentB->data) {
            return false;
        }


        currentA = currentA->next;
        currentB = currentB->next;
    }
    return true;
}

// manipulator functions

void clear(List L) {
    if (L == NULL) {
        printf("List error: calling clear() on a NULL list\n");
        exit(EXIT_FAILURE);
    }
    while (length(L) > 0) {
        deleteFront(L);
    }

    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->index = -1;
    L->length = 0;
}

void set(List L, int x) {
    if (L == NULL) {
        printf("List error: calling set() on a NULL list\n");
        exit(EXIT_FAILURE);
    }
    if (L->length == 0) {
        printf("List error: calling set() while list length is 0\n");
        exit(EXIT_FAILURE);
    }
    if (L->index < 0) {
        printf("List error: calling set() while List index < 0\n");
        exit(EXIT_FAILURE);
    }

    L->cursor->data = x;
}

void moveFront(List L) {
    if (L == NULL) {
        printf("List error: calling moveFront() on a NULL list\n");
        exit(EXIT_FAILURE);
    }
    if (L->length > 0) { 
        L->cursor = L->front;
        L->index = 0;
    }
}

void moveBack(List L) {
    if (L == NULL) {
        printf("List error: calling moveBack() on a NULL list\n");
        exit(EXIT_FAILURE);
    }
    if (L->length > 0) {
        L->cursor = L->back; 
        L->index = L->length - 1;
    }
}

void movePrev(List L) {
    if (L == NULL) {
        printf("List error: calling movePrev() on a NULL list\n");
        exit(EXIT_FAILURE);
    }
    if (L->cursor == L->front) {
        L->cursor = NULL;
        L->index = -1;
    } else if (L->cursor != NULL) {


        L->cursor = L->cursor->prev;
        L->index--;
    }
}

void moveNext(List L) {
    if (L == NULL) {
        printf("List error: calling moveNext() on a NULL list\n");
        exit(EXIT_FAILURE);
    }
    if (L->cursor == L->back) {
        L->cursor = NULL;
        L->index = -1;
    } else if (L->cursor != NULL) {
        L->cursor = L->cursor->next;
        L->index++;
    }
}

void prepend(List L, int x) {
    if (L == NULL) {
        printf("List error: calling prepend() on a NULL list\n");
        exit(EXIT_FAILURE);
    }

    Node newNode = malloc(sizeof(NodeObj));
    newNode->data = x;
    newNode->next = NULL;
    newNode->prev = NULL;

    if (L->length == 0) {
        L->front = L->back = newNode;
    } else {
        newNode->next = L->front;
        L->front->prev = newNode;
        L->front = newNode;
    }

    if (L->cursor != NULL) {
        L->index++;
    }
    L->length++;
}

void append(List L, int x) {
    if (L == NULL) {
        printf("List error: calling append() on a NULL list\n");
        exit(EXIT_FAILURE);
    }

    Node newNode = malloc(sizeof(NodeObj));
    newNode->data = x;
    newNode->next = NULL;
    newNode->prev = NULL;

    if (L->length == 0) {
        L->front = L->back = newNode;
    } else {
        newNode->prev = L->back;
        L->back->next = newNode;
        L->back = newNode;
    }
    L->length++;
}

void insertBefore(List L, int x) {
    if (L == NULL) {
        printf("List error: insertBefore() on a NULL list\n");
        exit(EXIT_FAILURE);
    }
    if (L->length == 0) {
        printf("List error: insertBefore() called when list length = 0\n");
        exit(EXIT_FAILURE);
    }
    if (L->index < 0) {
        printf("List error: insertBefore() called when cursor index < 0\n");
        exit(EXIT_FAILURE);
    }

    Node newNode = malloc(sizeof(NodeObj));
    newNode->data = x;

    if (L->cursor == L->front) {
        newNode->next = L->front;
        L->front->prev = newNode;
        L->front = newNode;
    } else {
        newNode->prev = L->cursor->prev;
        newNode->next = L->cursor;
        L->cursor->prev->next = newNode;
        L->cursor->prev = newNode;
    }


    L->index++;
    L->length++;
}

void insertAfter(List L, int x) {
    if (L == NULL) {
        printf("List error: insertAfter() called on a NULL List\n");
        exit(EXIT_FAILURE);
    }
    if (L->length == 0) {
        printf("List error: insertAfter() called when list length = 0\n");

    }
    if (L->index < 0) {
        printf("List error: insertAfter() called when cursor index < 0\n");
        exit(EXIT_FAILURE);
    }

    Node newNode = malloc(sizeof(NodeObj));
    newNode->data = x;

    if (L->cursor == L->back) {
        newNode->prev = L->back;
        L->back->next = newNode;
        L->back = newNode;
    } else {
        newNode->next = L->cursor->next;
        newNode->prev = L->cursor;
        L->cursor->next->prev = newNode;
        L->cursor->next = newNode;
    }
    L->length++;
}

// delete functions

void deleteFront(List L) {
    if (L == NULL) {
        printf("List error: calling deleteFront() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->length == 0) {
        printf("List error: calling deleteFront() on an empty List\n");
        exit(EXIT_FAILURE);
    }

    Node temp = L->front;
    if (L->length > 1) {
        L->front = L->front->next;
        L->front->prev = NULL;
    } else {
        L->front = L->back = NULL;
    }
    free(temp);
    L->length--;

    if (L->cursor != NULL) {
        L->index--;
    }
}

void deleteBack(List L) {
    if (L == NULL) {
        printf("List error: calling deleteBack() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->length == 0) {
        printf("List error: calling deleteBack() on an empty List\n");
        exit(EXIT_FAILURE);
    }

    Node temp = L->back;
    if (L->length > 1) {
        L->back = L->back->prev;
        L->back->next = NULL;
    } else {
        L->front = L->back = NULL;
    }
    free(temp);
    L->length--;

    if (L->cursor == temp) {
        L->cursor = NULL;
        L->index = -1;
    }
}

void delete(List L) {
    if (L == NULL) {
        printf("List error: calling delete() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->length == 0) {
        printf("List error: calling delete() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    if (L->index < 0) {
        printf("List error: calling delete() when cursor index is undefined\n");
        exit(EXIT_FAILURE);
    }

    Node temp = L->cursor;

    if (L->cursor == L->front) {
        deleteFront(L);
    } else if (L->cursor == L->back) {
        deleteBack(L);
    } else {
        L->cursor->prev->next = L->cursor->next;
        L->cursor->next->prev = L->cursor->prev;
        free(temp);
        L->length--;
        L->index = -1;
        L->cursor = NULL;
    }
}
void printList(FILE* out, List L) {
    if (L == NULL) {
        printf("List Error: calling printList() on a NULL List\n");
        exit(EXIT_FAILURE);
    }

    Node N = L->front;
    while (N != NULL) {
        fprintf(out, "%d ", N->data);
        N = N->next;
    }
    fprintf(out, "\n"); 
}

List copyList(List L) {
    if (L == NULL) {
        printf("List Error: calling copyList() on a NULL List\n");
        exit(EXIT_FAILURE);
    }

    List copy = newList();
    Node N = L->front;

    while (N != NULL) {
        append(copy, N->data);
        N = N->next;
    }

    copy->cursor = NULL;
    copy->index = -1;

    return copy;
}

List concatList(List A, List B) {
    if (A == NULL || B == NULL) {
        printf("List Error: calling concatList() on a NULL List\n");
        exit(EXIT_FAILURE);
    }

    List concat = newList();
    Node N;
    
    N = A->front;
    while (N != NULL) {
        append(concat, N->data);
        N = N->next;
    }

    N = B->front;
    while (N != NULL) {
        append(concat, N->data);
        N = N->next;
    }

    concat->cursor = NULL;
    concat->index = -1;

    return concat;
}
