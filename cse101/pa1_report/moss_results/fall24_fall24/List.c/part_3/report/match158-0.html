#include <stdio.h>
#include <stdlib.h>
#include "List.h"

typedef struct NodeObj {
    int data;
    struct NodeObj* next;
    struct NodeObj* prev;
} NodeObj;

typedef NodeObj* Node;

typedef struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
    int index;
} ListObj;

//IMPLEMENTATION OF CONSTRUCTOR

List newList(void){
    List L = malloc(sizeof(ListObj));
    if (L == NULL) {
        fprintf(stderr, "Could not create new List.\n");
        exit(EXIT_FAILURE);
    }
    L->front = L->back = L->cursor = NULL;
    L->length = 0;
    L->index = -1;
    return L;
}

//IMPLEMENTATION OF DESTRUCTOR

void freeList(List* pL) {
    if (pL != NULL && *pL != NULL) {
        while (length(*pL) > 0) {
            deleteFront(*pL);
        }
        free (*pL); //freeing heap memory
        *pL = NULL; //setting pl to null
    }
}

int get(List L) {
    if (L == NULL) {
        fprintf(stderr, "ERROR: Called when NULL\n");
        exit(EXIT_FAILURE);
    }
    if (L->length == 0 || L->index == -1) {
        fprintf(stderr, "ERROR: Called on empty list or w/ an undefined cursor\n");
        exit(EXIT_FAILURE);
    }
    return L->cursor->data;
}


bool equals(List A, List B) {
    if (A == NULL || B == NULL) {
        fprintf(stderr, "ERROR: Called when NULL\n");
        exit(EXIT_FAILURE); //checking if null
    }

    // check if lengths are the same
    if (A->length != B->length) {
        return false;
    }

    //comparing the nodes
    Node nodeA = A->front;
    Node nodeB = B->front;
    
    while (nodeA != NULL && nodeB != NULL) {
        if (nodeA->data != nodeB->data) {
            return false;
        }
        nodeA = nodeA->next;
        nodeB = nodeB->next;
    }
    return true;
}

void clear(List L){
    if (L == NULL) {
        fprintf(stderr, "ERROR: Called when NULL\n");
        exit(EXIT_FAILURE);
    }
    while (L->length > 0) {


        deleteFront(L);
    }
    L->cursor = NULL;
    L->index = -1; //resetting to original empty statee
}

void set(List L, int x){
    if (L == NULL || L->cursor == NULL) { //length > 0 and index >= 0
        fprintf(stderr, "ERROR: Called when NULL or undefined cursor\n");
        exit(EXIT_FAILURE);
    }
    L->cursor->data = x; //overwrite with x
}

void moveFront(List L){
    if (L == NULL || L->length == 0) return; 
    L->cursor = L->front;
    L->index = 0;//if not empty then sets cursor under front element
}

void moveBack(List L){
    if (L == NULL || L->length == 0) return;


    L->cursor = L->back;
    L->index = L->length - 1;////if not empty then sets cursor under back element
}

void movePrev(List L){
    if (L == NULL || L->cursor == NULL) return;
    if (L->cursor == L->front) {
        L->cursor = NULL;
        L->index = -1;
    } else {
        L->cursor = L->cursor->prev;
        L->index--;
    }
}

void moveNext(List L) {
    if (L == NULL || L->cursor == NULL) return;
    if (L->cursor == L->back) {
        L->cursor = NULL;
        L->index = -1;
    } else {
        L->cursor = L->cursor->next;
        L->index++;
    }
}

void prepend(List L, int x) {
    if (L == NULL) {
        fprintf(stderr, "ERROR: Called when NULL\n");
        exit(EXIT_FAILURE);
    }
    Node N = malloc(sizeof(NodeObj));
    if (N == NULL) {
        fprintf(stderr, "ERROR: memory allocation failed\n");
        exit(EXIT_FAILURE);
    }
    N->data = x;
    N->next = L->front;
    N->prev = NULL;
    if (L->front != NULL) {
        L->front->prev = N;
    } else {
        L->back = N;
    }
    L->front = N;
    L->length++;
}

void deleteFront(List L) {
    if (L == NULL || L->length == 0) {
        fprintf(stderr, "ERROR: Called when NULL or empty List\n");
        exit(EXIT_FAILURE);
    }
    Node temp = L->front;
    if (L->front == L->back) { // Only one element
        L->front = L->back = NULL;
    } else {
        L->front = L->front->next;
        L->front->prev = NULL;
    }
    free(temp);
    L->length--;
    if (L->index == 0) {
        L->cursor = NULL;
        L->index = -1;
    } else if (L->index > 0) {
        L->index--;
    }
}

void deleteBack(List L) {
    if (L == NULL || L->length == 0) {
        fprintf(stderr, "ERROR: Called when NULL or empty List\n");
        exit(EXIT_FAILURE);
    }
    Node temp = L->back;
    if (L->front == L->back) { // Only one element
        L->front = L->back = NULL;
    } else {
        L->back = L->back->prev;
        L->back->next = NULL;
    }


    free(temp);
    L->length--;
    if (L->index == L->length) {
        L->cursor = NULL;
        L->index = -1;
    }
}

void delete(List L) {
    if (L == NULL || L->length == 0 || L->cursor == NULL) {
        fprintf(stderr, "ERROR: Called when list is NULL, empty, or with undefined cursor\n");
        exit(EXIT_FAILURE);
    }

    Node temp = L->cursor;

    // cursor at front
    if (L->cursor == L->front) {
        L->front = L->front->next;
        if (L->front != NULL) {
            L->front->prev = NULL;
        } else {
            L->back = NULL; //made list empty
        }
    }
    // cursr at back
    else if (L->cursor == L->back) {
        L->back = L->back->prev;
        if (L->back != NULL) {
            L->back->next = NULL;
        } else {
            L->front = NULL; // list empty
        }
    }
    // cursor in midddle
    else {
        L->cursor->prev->next = L->cursor->next;
        L->cursor->next->prev = L->cursor->prev;
    }

    free(temp);
    L->length--;
    L->cursor = NULL;
    L->index = -1;
}
