#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>
#include "List.h"

// NodeObj struct definition
typedef struct NodeObj {
    int data;
    struct NodeObj* prev;
    struct NodeObj* next;
} NodeObj;

// Node type
typedef NodeObj* Node;

// ListObj struct definition
typedef struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
    int index;
} ListObj;

// Constructors-Destructors ---------------------------------------------------

List newList(void) {
    List L = malloc(sizeof(ListObj));
    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->length = 0;
    L->index = -1;
    return L;
}

void freeList(List* pL) {
    if (pL != NULL && *pL != NULL) {
        clear(*pL);
        free(*pL);
        *pL = NULL;
    }
}

// Private constructor for a Node
Node newNode(int data) {
    Node N = malloc(sizeof(NodeObj));
    N->data = data;
    N->prev = NULL;
    N->next = NULL;
    return N;
}

// Destructor for a Node
void freeNode(Node* pN) {
    if (pN != NULL && *pN != NULL) {
        free(*pN);
        *pN = NULL;
    }
}

// Access functions -----------------------------------------------------------

int length(List L) {
    return L->length;
}

int index(List L) {
    return L->index;
}

int front(List L) {
    if (L->length > 0) {
        return L->front->data;
    }
    return -1;
}

int back(List L) {
    if (L->length > 0) {
        return L->back->data;
    }
    return -1;
}

int get(List L) {
    if (L->length > 0 && L->index >= 0) {
        return L->cursor->data;
    }
    return -1;
}

bool equals(List A, List B) {
    if (A->length != B->length) return false;

    Node N = A->front;
    Node M = B->front;
    while (N != NULL && M != NULL) {
        if (N->data != M->data) {
            return false;
        }
        N = N->next;
        M = M->next;
    }
    return true;
}

// Manipulation procedures ----------------------------------------------------

void clear(List L) {
    while (L->length > 0) {
        deleteFront(L);
    }
    L->cursor = NULL;
    L->index = -1;
}



void set(List L, int x) {
    if (L->length > 0 && L->index >= 0) {
        L->cursor->data = x;
    }
}

void moveFront(List L) {
    if (L->length > 0) {
        L->cursor = L->front;
        L->index = 0;
    }
}

void moveBack(List L) {
    if (L->length > 0) {


        L->cursor = L->back;
        L->index = L->length - 1;
    }
}

void movePrev(List L) {
    if (L->cursor != NULL && L->index > 0) {
        L->cursor = L->cursor->prev;
        L->index--;
    } else if (L->cursor != NULL && L->index == 0) {
        L->cursor = NULL;
        L->index = -1;
    }
}

void moveNext(List L) {
    if (L->cursor != NULL && L->index < L->length - 1) {
        L->cursor = L->cursor->next;
        L->index++;
    } else if (L->cursor != NULL && L->index == L->length - 1) {
        L->cursor = NULL;
        L->index = -1;
    }
}

void prepend(List L, int x) {
    Node N = newNode(x);
    if (L->length == 0) {
        L->front = L->back = N;
    } else {
        N->next = L->front;
        L->front->prev = N;
        L->front = N;
    }
    L->length++;
    if (L->index >= 0) {
        L->index++;
    }
}

void append(List L, int x) {
    Node N = newNode(x);
    if (L->length == 0) {
        L->front = L->back = N;
    } else {
        N->prev = L->back;
        L->back->next = N;
        L->back = N;
    }
    L->length++;
}

void insertBefore(List L, int x) {
    if (L->length > 0 && L->index >= 0) {
        if (L->cursor == L->front) {
            prepend(L, x);
        } else {
            Node N = newNode(x);
            N->next = L->cursor;
            N->prev = L->cursor->prev;
            L->cursor->prev->next = N;
            L->cursor->prev = N;
            L->length++;
            L->index++;
        }
    }
}

void insertAfter(List L, int x) {
    if (L->length > 0 && L->index >= 0) {
        if (L->cursor == L->back) {
            append(L, x);
        } else {
            Node N = newNode(x);
            N->prev = L->cursor;
            N->next = L->cursor->next;
            L->cursor->next->prev = N;
            L->cursor->next = N;


            L->length++;
        }
    }
}

void deleteFront(List L) {
    if (L->length > 0) {
        Node N = L->front;
        if (L->length > 1) {
            L->front = L->front->next;
            L->front->prev = NULL;
        } else {
            L->front = L->back = NULL;
        }
        freeNode(&N);
        L->length--;
        if (L->index > 0) {
            L->index--;
        } else if (L->index == 0) {
            L->cursor = NULL;
            L->index = -1;
        }
    }
}

void deleteBack(List L) {


    if (L->length > 0) {
        Node N = L->back;
        if (L->length > 1) {
            L->back = L->back->prev;
            L->back->next = NULL;
        } else {
            L->front = L->back = NULL;
        }


        freeNode(&N);
        L->length--;
        if (L->index == L->length) {
            L->cursor = NULL;
            L->index = -1;
        }
    }
}

void delete(List L) {
    if (L->length > 0 && L->index >= 0) {
        Node N = L->cursor;
        if (L->cursor == L->front) {
            deleteFront(L);
        } else if (L->cursor == L->back) {
            deleteBack(L);
        } else {
            L->cursor->prev->next = L->cursor->next;
            L->cursor->next->prev = L->cursor->prev;
            freeNode(&N);
            L->length--;
        }
        L->cursor = NULL;
        L->index = -1;
    }
}

// Other operations -----------------------------------------------------------

void printList(FILE* out, List L) {
    Node N = L->front;
    while (N != NULL) {
        fprintf(out, "%d ", N->data);
        N = N->next;
    }
    fprintf(out, "\n");
}

List copyList(List L) {
    List C = newList();
    Node N = L->front;
    while (N != NULL) {
        append(C, N->data);
        N = N->next;
    }
    return C;
}
