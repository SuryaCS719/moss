#include <stdio.h>
#include <stdlib.h>
#include "List.h" 

//Referenced/ inspo from Queue.c provided on webpage
//Node structure
typedef struct NodeObj {
    int data;
    struct NodeObj* prev;
    struct NodeObj* next;
} NodeObj;

typedef NodeObj* Node;

//List Structure
typedef struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
    int index;
} ListObj;

//Creating and returning a new empty list
List newList(void) {
    List L = malloc(sizeof(ListObj));
    L->front = L->back = L->cursor = NULL;
    L->length = 0;
    L-> index = -1;
    return L;
}


//Freeing memory
void freeList(List* pL) {
    if (pL != NULL && *pL != NULL) {


        clear(*pL);
        free(*pL);
        *pL = NULL;
    }
}


//number of elements in the list
int length(List L) {
    return L->length;
}


//index of cursor
int index(List L) {
    return L->cursor != NULL ? L->index : -1;
}


//front of list value
int front(List L) {
    if (L->length > 0) {
        return L->front->data;
    }
    printf("Error: empty list");
    exit(EXIT_FAILURE);
}


//back of list value
int back(List L) {
    if (L->length > 0 ) {
        return L->back->data;
    }
    printf("Error: empty list");
    exit(EXIT_FAILURE);
}


//value of cursor
int get(List L) {
    if (L == NULL) {
        printf("Error: NULL list\n");
        exit(EXIT_FAILURE);
    }
    if (L->cursor == NULL) {
        printf("Error: undefined\n");
        exit(EXIT_FAILURE);
    }
    
    return L->cursor->data;
}


//checking if lists are equal
bool equals(List A, List B) {
    if (A->length != B->length) return false;
    Node nodeA = A->front;
    Node nodeB = B->front;
    while (nodeA != NULL && nodeB != NULL) {
        if (nodeA->data != nodeB->data) return false;
        nodeA = nodeA->next;
        nodeB = nodeB->next;
    }
    return true;
}


//clearing list

void clear(List L) {
    while (L->length > 0) {
        deleteFront(L);
    }
    L->cursor = NULL;
    L->index = -1;
}


//value at cursor 
void set(List L, int x) {
    if (L->length > 0 && L->cursor != NULL) {


        L->cursor->data = x;
    } else {
        printf("Error: undefined\n");
        exit(EXIT_FAILURE);
    }
}


//move the cursor to front

void moveFront(List L) {
    if (L->length > 0) {
        L->cursor = L->front;
        L->index = 0;
        
    } else {
        L->cursor = NULL;
        L->index = -1;
        //printf("List is empty, set to NULL\n");
    }
}


//move cursor to back
void moveBack(List L) {
    if (L->length > 0) {
        L->cursor = L->back;
        L->index = L->length - 1;
        
    } else {
        L->cursor = NULL;
        L->index = -1;
        //printf("List is empty, set to NULL\n");
    }
}


//move cursor to prev step
void movePrev(List L) {
    if (L->cursor != NULL) {
        if (L->cursor == L->front) {
            L->cursor = NULL;
            L->index = -1;
        } else {
            L->cursor = L->cursor->prev;
            L->index--;
        }
    }
}


//move cursor to next step
void moveNext(List L) {
    if (L->cursor != NULL) {
        if (L->cursor == L->back) {
            L->cursor = NULL;
            L->index = -1;
        } else {
            L->cursor = L->cursor->next;
            L->index++;
        }
    }
}


//prepend new element in front
void prepend(List L, int x) {
    Node node = malloc(sizeof(NodeObj));
    node->data = x;
    node->prev = NULL;
    node->next = L->front;
    if (L->front != NULL) {
        L->front->prev = node;
    }
    L->front = node;
    if (L->length == 0) {
        L->back = L->front;
    }
    L->length++;
    if (L->index != -1) L->index++;
}


//append new element to back
void append(List L, int x) {
    Node node = malloc(sizeof(NodeObj));
    node->data = x;
    node->next = NULL;
    node->prev = L->back;
    if (L->back != NULL) {
        L->back->next = node;
    }
    L->back = node;
    if (L->length == 0) {
        L->front = L->back;
    }
    L->length++;
}


//insert element before cursor
void insertBefore(List L, int x) {
    if (L->length > 0 && L->cursor != NULL) {
        Node node = malloc(sizeof(NodeObj));
        node->data = x;
        node->prev = L->cursor->prev;
        node->next = L->cursor;
        if (L->cursor->prev != NULL) {
            L->cursor->prev->next = node;
        }
        L->cursor->prev = node;
        if (L->cursor == L->front) {


            L->front = node;
        }
        L->length++;
        L->index++;
    } else {
        printf("Error: empty list\n");
        exit(EXIT_FAILURE);
    }
}


//insert element after cursor
void insertAfter(List L, int x) {
    if (L->length > 0 && L->cursor != NULL) {
        Node node = malloc(sizeof(NodeObj));
        node->data = x;
        node->prev = L->cursor;
        node->next = L->cursor->next;
        if (L->cursor->next != NULL) {
            L->cursor->next->prev = node;
        }
        L->cursor->next = node;
        if (L->cursor == L->back) {


            L->back = node;
        }
        L->length++;
    } else {
        printf("Error: undefined cursor or empty list\n");
        exit(EXIT_FAILURE);
    }
}


//deletes front node
void deleteFront(List L) {
    if (L->length > 0) {
        Node node = L->front;
        if (L->length > 1) {
            L->front = L->front->next;
            L->front->prev = NULL;
        } else {


            L->front = L->back = NULL;
        }
        free(node);
        L->length--;
        if (L->index != -1) L->index--;
    } else {
        printf("Error: empty list\n");
        exit(EXIT_FAILURE);
    }
}


//deletes back node 
void deleteBack(List L) {
    if (L->length > 0) {
        Node node = L->back;
        if (L->length > 1) {
            L->back = L->back->prev;
            L->back->next = NULL;
        } else {
            L->back = L->front = NULL;
        }
        free(node);
        L->length--;
        if (L->cursor == node) {
            L->cursor = NULL;
            L->index = -1;
        }
    } else {
        printf("Error: empty list\n");
        exit(EXIT_FAILURE);
    }
}


//deletes cursor node
void delete(List L) {
    if (L->length > 0 && L->cursor != NULL) {
        Node node = L->cursor;
        if (L->cursor == L->front) {
            deleteFront(L);
        } else if (L->cursor == L->back) {
            deleteBack(L);
        } else {
            L->cursor->prev->next = L->cursor->next;


            L->cursor->next->prev = L->cursor->prev;
            free(node);
            L->length--;
        }
        L->cursor = NULL;
        L->index = -1;
    } else {
        printf("Error: empty list or undefined cursor\n");
        exit(EXIT_FAILURE);
    }
}



//printing list of data
void printList(FILE* out, List L) {
    Node node = L->front;
    while (node != NULL) {
        fprintf(out, "%d ", node->data);
        node = node->next;
    }
}


//creates a copy of list L
List copyList(List L) {
    List X = newList();
    Node node = L->front;
    while (node != NULL) {
        append(X, node->data);
        node = node->next;
    }
    return X;
}




List concatList(List A, List B) {
    List X = newList();
    Node node = A->front;
    while (node != NULL) {
        append(X, node->data);
        node = node->next;
    }
    node = B->front;
    while (node != NULL) {
        append(X, node->data);
        node = node->next;
    }
    return X;

}
