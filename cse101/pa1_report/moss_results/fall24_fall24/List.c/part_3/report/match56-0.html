#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>
#include <assert.h>
#include "List.h"

typedef struct nodeObj* Node;

typedef struct ListObj {
    int index;
    int length;
    
    Node front;    
    Node back;
    Node cursor;
} ListObj;

typedef struct nodeObj {
    int data;

    struct nodeObj* next;
    struct nodeObj* prev;
} nodeObj;

List newList(void) {
    List L = malloc(sizeof(ListObj));
    
    L->length = 0;
    L->index = -1;
    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    
    return L;
}

Node newNode(int data) {
    Node n = malloc(sizeof(nodeObj));
    if (n == NULL) {
        fprintf(stderr, "Memory allocation failed for new node.\n");
        exit(EXIT_FAILURE);
    }

    n->data = data;
    n->prev = NULL;
    n->next = NULL;
    return n;
}

void freeList(List* pL) {
    if ((pL != NULL) && (*pL != NULL)) {
        clear(*pL);
        free(*pL);
        *pL = NULL;
    } 
}

void freeNode(Node* pN) {
    if (pN != NULL && *pN != NULL) {
        (*pN)->next = NULL;
        (*pN)->prev = NULL;
        free(*pN);


        *pN = NULL;
    }
}

int length(List L) {
    if (L == NULL) {
        fprintf(stderr, "L = NULL\n");
        exit(EXIT_FAILURE);
    }

    return L->length;
}

int index(List L) {
    return L->index;
}

int front(List L) {
    if (L == NULL || L->length == 0) {
        fprintf(stderr, "List is NULL or empty when trying to access front.\n");
        exit(EXIT_FAILURE);
    }
    return L->front->data;
}

int back(List L) {
    if (L == NULL || L->length == 0) {
        fprintf(stderr, "List is NULL or empty when trying to access back.\n");


        exit(EXIT_FAILURE);
    }
    return L->back->data;
}

int get(List L) {
    if (length(L) == 0) {
        fprintf(stderr, "Error in get(), empty list!\n");
        exit(EXIT_FAILURE);
    }

    if (index(L) < 0) {
        fprintf(stderr, "Error in get(), there is no index!\n");
        exit(EXIT_FAILURE);
    }

    return L->cursor->data;
}

bool equals(List A, List B) {
    if (A == NULL || B == NULL) {
        return false; // If one is NULL, they are not equal
    }

    bool equal = (length(A) == length(B));
    if (!equal) {
        return false; // Different lengths
    }

    Node nodeA = A->front;
    Node nodeB = B->front;

    while (nodeA != NULL && nodeB != NULL) {
        if (nodeA->data != nodeB->data) {
            return false;  // Data not equal, lists are not equal
        }
        nodeA = nodeA->next;
        nodeB = nodeB->next;
    }

    return true;
}

void clear(List L) {
    if (L == NULL) {
        fprintf(stderr, "Attempted to clear a NULL List.\n");
        exit(EXIT_FAILURE);
    }

    while (L->front != NULL) {
        Node temp = L->front;
        L->front = L->front->next;
        freeNode(&temp);
    }

    L->back = NULL;
    L->cursor = NULL;
    L->index = -1;
    L->length = 0;
}

void set(List L, int x) {
    if (L == NULL) {
        fprintf(stderr, "Error, List is NULL and set() can't be called!\n");
        exit(EXIT_FAILURE);
    }

    if (index(L) < 0) {
        fprintf(stderr, "Error, no index in List so set() can't be called!\n");
        exit(EXIT_FAILURE);
    }

    if (length(L) == 0) {
        fprintf(stderr, "Error, List is empty so set() can't be called!\n");
        exit(EXIT_FAILURE);
    }
    
    L->cursor->data = x;
}

void prepend(List L, int x) {           


    Node node = newNode(x);
    
    if (L->length == 0) {
        L->front = node;
        L->back = node;
        L->cursor = node;
        L->index = 0;
    } else {
        node->next = L->front;
        L->front->prev = node;
        L->front = node;
        L->index++; 
    }
    
    L->length++;
}

void append(List L, int x) {            
    Node node = newNode(x);
    
    if (L->length == 0) {
        L->front = node;
        L->back = node;
    } else {
        node->prev = L->back;
        L->back->next = node; // Fix this line
        L->back = node;
    }
    
    L->length++;
}

void insertBefore(List L, int x) {      
    Node node = newNode(x);

    if (L == NULL) {
        fprintf(stderr, "Error, can't use insertBefore() on a List that is NULL!\n");
        exit(EXIT_FAILURE);
    }

    if (L->length <= 0) {
        fprintf(stderr, "Error, can't use insertBefore() in an empty List!\n");
        exit(EXIT_FAILURE);
    }

    if (L->index < 0) {
        fprintf(stderr, "Error, can't use insertBefore() doesn't have an index!\n");
        exit(EXIT_FAILURE);
    }



    if (index(L) == 0) {
        L->front->prev = node;
        node->next = L->front;
        L->front = node;
    } else {
        node->prev = L->cursor->prev;
        L->cursor->prev->next = node;
        node->next = L->cursor;
        L->cursor->prev = node;
    }
    
    L->length++;
    L->index++;
}

void insertAfter(List L, int x) {
    if (L == NULL || L->cursor == NULL) {
        fprintf(stderr, "Error, calling insertAfter() on NULL List or without a cursor!\n");
        exit(EXIT_FAILURE);
    }

    Node node = malloc(sizeof(nodeObj));
    if (node == NULL) {
        fprintf(stderr, "Error, memory allocation failed!\n");
        exit(EXIT_FAILURE);
    }
    node->data = x;

    if (L->cursor == L->back) {
        node->prev = L->back;
        node->next = NULL;
        
        if (L->back != NULL) {
            L->back->next = node;
        }

        L->back = node;

        if (L->front == NULL) {
            L->front = node;
        }
    } else {
        node->next = L->cursor->next;
        node->prev = L->cursor;

        if (L->cursor->next != NULL) {
            L->cursor->next->prev = node;
        }

        L->cursor->next = node;
    }

    L->length++;
}


void deleteFront(List L) {
    if (L == NULL) {
        fprintf(stderr, "Attempted to delete front from a NULL List.\n");
        exit(EXIT_FAILURE);
    }

    if (L->length == 0) {
        fprintf(stderr, "Attempted to delete front from an empty list.\n");
        exit(EXIT_FAILURE);
    }

    Node temp = L->front; 

    if (L->cursor == L->front) {
        L->cursor = NULL; 
        L->index = -1; 
    } else if (L->index > 0) {
        L->index--; 
    }


    if (L->length > 1) {
        L->front = L->front->next; 
        L->front->prev = NULL; 
    } else {

        L->front = NULL; 
        L->back = NULL;
    }

    L->length--; 
    freeNode(&temp);
}



void deleteBack(List L) {
    if (L == NULL || L->length == 0) {
        fprintf(stderr, "Error, can't use deleteBack() on a List that is NULL or empty.\n");
        exit(EXIT_FAILURE);
    }
    
    Node backNode = L->back;

    if (L->cursor == L->back) {
        L->cursor = NULL;
        L->index = -1;
    }

    if (L->length == 1) {
        L->front = NULL;
        L->back = NULL;
        freeNode(&backNode);
    } else {
        L->back = backNode->prev;
        L->back->next = NULL;
        freeNode(&backNode);
    }
    L->length--;
}



void delete(List L) {
    if (L == NULL) {
        fprintf(stderr, "Error, can't use delete() on a List that is NULL!\n");
        exit(EXIT_FAILURE);
    }

    if (length(L) == 0 || index(L) < 0) {
        fprintf(stderr, "Error, can't call delete() on an empty List or undefined cursor!\n");
        exit(EXIT_FAILURE);
    }
    
    Node cursorNode = L->cursor;
    
    if (cursorNode == L->front) {
        L->front = cursorNode->next;
        
        if (L->front != NULL) {
            L->front->prev = NULL;
        }
    } else if (cursorNode == L->back) {
        L->back = cursorNode->prev;
        
        if (L->back != NULL) {
            L->back->next = NULL;
        }
    } else {
        cursorNode->prev->next = cursorNode->next;
        cursorNode->next->prev = cursorNode->prev;
    }
    
    L->cursor = NULL;
    L->index = -1;
    L->length--;
    freeNode(&cursorNode);
}

void moveFront(List L) {
    if (L == NULL || L->length == 0) {
        fprintf(stderr, "Error, can't use moveFront() on a List that is NULL or empty!\n");
        exit(EXIT_FAILURE);
    }

    L->cursor = L->front;
    L->index = 0;
}

void moveBack(List L) {
    if (L == NULL || L->length == 0) {
        fprintf(stderr, "Error, can't use moveBack() on a List that is NULL or empty!\n");
        exit(EXIT_FAILURE);
    }

    L->cursor = L->back;
    L->index = L->length - 1;
}

void movePrev(List L) {
    if (L == NULL || L->length == 0) {
        fprintf(stderr, "Error, can't use movePrev() on a List that is NULL or empty!\n");
        exit(EXIT_FAILURE);
    }
    
    if (L->cursor != NULL) {
        if (L->cursor == L->front) {
            L->cursor = NULL;
            L->index = -1;
        } else {
            L->cursor = L->cursor->prev;
            L->index--;
        }
    }
}

void moveNext(List L) {
    if (L == NULL || L->length == 0) {
        fprintf(stderr, "Error, can't use moveNext() on a List that is NULL or empty!\n");
        exit(EXIT_FAILURE);
    }

    if (L->cursor != NULL) {
        if (L->cursor == L->back) {
            L->cursor = NULL;
            L->index = -1;
        } else {
            L->cursor = L->cursor->next;
            L->index++;
        }
    }
}

void printList(FILE* out, List L) {
    if (L == NULL || L->length == 0) {
        fprintf(stderr, "Error, can't printList() because List is NULL or empty.\n");
        exit(EXIT_FAILURE);
    }
    
    Node node;
    for (node = L->front; node != NULL; node = node->next) {
        fprintf(out, "%d ", node->data);
    }
    fprintf(out, "\n"); // Newline for better formatting
}


List copyList(List L) {
    if (L == NULL) {
        fprintf(stderr, "Error, copyList called on a NULL List.\n");
        exit(EXIT_FAILURE);
    }
    
    List C = newList(); 
    Node node;

    for (node = L->front; node != NULL; node = node->next) {
        append(C, node->data);
    }

    C->cursor = NULL; 
    C->index = -1; 

    return C; 
}
