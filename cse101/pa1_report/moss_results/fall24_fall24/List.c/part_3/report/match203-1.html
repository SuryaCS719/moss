#include<stdio.h>
#include<stdlib.h>
#include<assert.h>
#include<stdbool.h>
#include "List.h"

// private Node type
typedef struct NodeObj* Node;

// private NodeObj type
typedef struct NodeObj{
   ListElement data;
   Node next;
   Node prev;
} NodeObj;

// private QueueObj type
typedef struct ListObj{
   Node front;
   Node back;
   Node cursor;
   int index;
   int length;
} ListObj;

// Creates and returns a new empty List.
List newList(void){
    List L;
    L = malloc(sizeof(ListObj));
    assert(L != NULL);
    L->front = L->back= L->cursor = NULL;
    L->index = -1;
    L->length = 0;
    return (L);


} 
// Frees all heap memory associated with *pL, and sets
// *pL to NULL.
void freeList(List* pL){
    if (pL!= NULL && *pL != NULL){
        clear(*pL);
        free(*pL);
        *pL = NULL;
    }
} 
// Access functions -----------------------------------------------------------
// Returns the number of elements in L.
int length(List L){
    if (L == NULL){
        printf("List Error: calling length() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    return (L->length);

} 
// Returns index of cursor element if defined, -1 otherwise.
int index(List L){
    if (L == NULL){
        printf("List Error: calling index() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    return (L->index);

} 
// Returns front element of L. Pre: length()>0
int front(List L){
    if (L == NULL){
        printf("List Error: calling front() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if(length(L) <= 0){
        printf("List Error: calling front() on empty list\n");
        exit(EXIT_FAILURE);

    }
    return (L->front->data);

} 
// Returns back element of L. Pre: length()>0
int back(List L){
    if (L == NULL){
        printf("List Error: calling back() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if(length(L) <= 0){
        printf("List Error: calling back() on empty list\n");
        exit(EXIT_FAILURE);
    }
    return (L->back->data);

} 
// Returns cursor element of L. Pre: length()>0, index()>=0
int get(List L){
    if (L == NULL){
        printf("List Error: calling get() on empty list\n");
        exit(EXIT_FAILURE);
    }
    if(length(L) <= 0){
        printf("List Error: calling get() on empty list\n");
        exit(EXIT_FAILURE);

    }
    if(index(L) < 0){
        printf("List Error: calling get() on non existent index\n");
        exit(EXIT_FAILURE);
    }
    return L->cursor->data;
}
// Returns true iff Lists A and B contain the same
 // sequence of elements, returns false otherwise.
bool equals(List A, List B){
    if(A== NULL || B == NULL){
        return false;
    }
    if (length(A)!= length(B)){
        return false;
    }
    Node aNode = A->front;
    Node bNode = B->front;  
    while(bNode != NULL && aNode != NULL){
        if(bNode->data != aNode->data){
            return false;
        }
        aNode = aNode->next;
        bNode = bNode->next;
    }
    return true;
}

// Manipulation procedures ----------------------------------------------------

// Resets L to its original empty state.
void clear(List L){


    while(length(L) >0){
        deleteFront(L);
    }
    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->index = -1;
    L->length = 0;

}
 // Overwrites the cursor elementâ€™s data with x.
 // Pre: length()>0, index()>=0
void set(List L, int x){
    if (L == NULL){
        printf("List Error: calling set() on NULL list\n");
        exit(EXIT_FAILURE);
    }
    if(length(L) <= 0 ){
        printf("List Error: calling get() on empty list\n");
        exit(EXIT_FAILURE);
    }
    if(index(L)<0){
        printf("List Error: calling get() on non existent index\n");
        exit(EXIT_FAILURE);
    }
    L->cursor->data = x;
}
// If L is non-empty, sets cursor under the front element,
 // otherwise does nothing.
void moveFront(List L){
    if(L == NULL){
        printf("List Error: calling moveFront() on NULL list\n");
        exit(EXIT_FAILURE);
    }
    if(length(L) <=0){
        printf("List Error: calling moveFront() on empty list\n");
        exit(EXIT_FAILURE);

    }
    L->cursor = L->front;
    L->index = 0;

} 
// If L is non-empty, sets cursor under the back element,
 // otherwise does nothing.
void moveBack(List L){
    if(L==NULL){
        printf("List Error: calling moveBack() on NULL list\n");
        exit(EXIT_FAILURE);
    }
    if(length(L) <=0){
        printf("List Error: calling moveBack() on empty list\n");
        exit(EXIT_FAILURE);
    }
    L->cursor = L->back;
    L->index = L->length -1; 

} 
// If cursor is defined and not at front, move cursor one
 // step toward the front of L; if cursor is defined and at
 // front, cursor becomes undefined; if cursor is undefined
 // do nothing
void movePrev(List L){
    if(L==NULL){
        fprintf(stderr, "Calling movePrev() on a NULL List");
        exit(EXIT_FAILURE);
    }
    if (L->cursor == L->front) {
        L->cursor = NULL;
        L->index = -1;
    }else{
        L->cursor = L->cursor->prev;
        L->index--;
    }

}


// If cursor is defined and not at back, move cursor one
 // step toward the back of L; if cursor is defined and at
 // back, cursor becomes undefined; if cursor is undefined
 // do nothing
void moveNext(List L){
    if(L!= NULL && L->cursor != NULL){
        if(L->cursor == L->back){
            L->cursor = NULL;
            L->index = -1;
        }else{


            L->cursor = L->cursor->next;
            L->index++;
        }
    }
}
// Insert new element into L. If L is non-empty,
 // insertion takes place before front element.
void prepend(List L, int x){
    if(L==NULL){
        printf("List Error: calling prepend() on NULL list\n");
        exit(EXIT_FAILURE);
    }
    Node xNode = malloc(sizeof(NodeObj));
    assert(xNode != NULL);
    xNode->data = x;
    xNode->next = L->front;
    xNode->prev = NULL;
    if(length(L)==0){
        L->front =L->back = xNode;
        L->length++;
    }else{


        xNode->next = L->front;
        L->front->prev = xNode;
        L->front = xNode;
        if(L->cursor != NULL){
            L->index++;
        }
        L->length++;
    }
   
    
   


  
} 
// Insert new element into L. If L is non-empty,
// insertion takes place after back element.
void append(List L, int x){
    if(L== NULL){
        printf("List Error: calling append() on NULL list\n");
        exit(EXIT_FAILURE);
    }
    Node xNode = malloc(sizeof(NodeObj));
    xNode->data = x;
    xNode->next = NULL;
    assert(xNode != NULL);
    if(length(L)<=0){
        L->front = xNode;
        L->back = xNode;
    }else{
        L->back->next = xNode;
        xNode->prev = L->back;
    }
    L->back = xNode;
    L->length++;
}
// Insert new element before cursor.
 // Pre: length()>0, index()>=0
void insertBefore(List L, int x){
    if(L == NULL){
        printf("List Error: calling insertBefore() on NULL list\n");
        exit(EXIT_FAILURE);
    }
    if(length(L) <= 0 || index(L) < 0){
        printf("List Error: calling insertBefore() on empty list\n");
        exit(EXIT_FAILURE);
    }
    Node xNode = malloc(sizeof(NodeObj));
    assert(xNode!= NULL);
    xNode->data = x;
    if(L->cursor == L->front){
        xNode->next = L->front;
        xNode->prev = NULL;
        L->front->prev = xNode;
        L->front = xNode;
    }else{
        xNode->prev = L->cursor->prev;
        xNode->next = L->cursor;

        if(L->cursor->prev != NULL){
            L->cursor->prev->next = xNode;
        }
        L->cursor->prev = xNode;
    }
    L->index++;
    L->length++;
}
// Insert new element after cursor.
 // Pre: length()>0, index()>=0
void insertAfter(List L, int x){
    if(L == NULL){
        printf("List Error: calling insertAfter() on NULL list\n");
        exit(EXIT_FAILURE);
    }
    if(length(L) <= 0 || index(L) < 0){
        printf("List Error: calling insertAfter() on empty list\n");
        exit(EXIT_FAILURE);
    }
    Node xNode = malloc(sizeof(NodeObj));
    assert(xNode!= NULL);
    xNode->data = x;
    xNode->next = L->cursor->next;
    xNode->prev = L->cursor;
    
    if(L->cursor->next !=NULL){
        L->cursor->next->prev = xNode;
    }
    L->cursor->next = xNode;
    if(L->cursor == L->back){
        L->back = xNode;
    }
    L->length++;

}
// Delete the front element. Pre: length()>0
void deleteFront(List L){
if (L == NULL) {
        printf("List Error: calling deleteFront() on NULL list\n");
        exit(EXIT_FAILURE);
    }
    if (L->length <= 0) {
        printf("List Error: calling deleteFront() on empty list\n");
        exit(EXIT_FAILURE);
    }

    Node node = L->front; 
    L->front = L->front->next; 

    if (L->front != NULL) {
        L->front->prev = NULL;
    } else {
        L->back = NULL;
    }
    free(node); 
    L->length--;   
    if (L->cursor == node) {
        L->cursor = NULL; 
        L->index = -1;    
    } else {
        L->index--; 
    }
} 
// Delete the back element. Pre: length()>0
void deleteBack(List L){
    if(L == NULL){
        printf("List Error: calling deleteBack() on NULL list\n");
        exit(EXIT_FAILURE);
    }
    if(L->length<= 0 ){
        printf("List Error: calling deleteBack() on empty list\n");
        exit(EXIT_FAILURE);
    }  
    Node tempNode = L->back;
    if(L->length == 1){
        L->front = NULL;
        L->back = NULL;
    }else{
        Node curr = L->front;
        while(L->back != curr->next){
            curr = curr->next;
        }
        curr->next = NULL;
        L->back = curr;
    }


    free(tempNode);
    L->length--;
    if(L->cursor== tempNode){
        L->cursor= NULL;
        L->index =-1;
    }
}
// Delete cursor element, making cursor undefined.
 // Pre: length()>0, index()>=0
void delete(List L){
    if(L == NULL){
        printf("List Error: calling delete() on NULL list\n");
        exit(EXIT_FAILURE);
    }
    if(L->length <= 0 || L->index < 0 ){


        printf("List Error: calling delete() on empty list\n");
        exit(EXIT_FAILURE);
    }     
    Node tempNode = L->cursor;
    if(L->cursor == L->front){
        deleteFront(L);
    }else if(L->cursor == L->back){
        deleteBack(L);
    }else{
        Node curr = L->front;
        while(curr->next != L->cursor && curr != NULL){
            curr = curr->next;
        }
        if(curr!=NULL){
            curr->next = L->cursor->next;

        }
        free(tempNode);
        L->cursor = NULL;
        L->length--;
        L->index = -1;
    }
}
// Other operations -----------------------------------------------------------
 // Prints to the file pointed to by out, a
 // string representation of L consisting
// of a space separated sequence of integers,
// with front on left.
void printList(FILE* out, List L){
    if(L== NULL){
        printf("List Error: calling printList() on NULL list\n");
        return;
    }
    Node curr = L->front;
    while(curr!=NULL){
        fprintf(out,"%d ",curr->data);
        curr = curr->next;
    }
}
// Returns a new List representing the same integer
 // sequence as L. The cursor in the new list is undefined,
// regardless of the state of the cursor in L. The state
// of L is unchanged.
List copyList(List L){
    List copiedList = newList();
    Node curr = L->front;
    while(curr!=NULL){
        append(copiedList,curr->data);
        curr = curr->next;
    }
    return copiedList;

}
