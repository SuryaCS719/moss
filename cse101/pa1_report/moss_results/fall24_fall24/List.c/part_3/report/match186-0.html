#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>
#include "List.h"

// NodeObj
typedef struct NodeObj{
	int data;
	struct NodeObj* next;
	struct NodeObj* prev;
} NodeObj;

// Node
typedef NodeObj* Node;

// List Struct
typedef struct ListObj {
	Node front;
	Node back;
	Node cursor;
	int length;
	int index;
} ListObj;

typedef ListObj* List;

// Constructors-Destructors ---------------------------------------------------
List newList(void){	// Creates and returns a new empty List.
	List L = malloc(sizeof(ListObj));
	L -> front = NULL;
	L -> back = NULL;
	L -> cursor = NULL;
	L -> length = 0;
	L -> index = -1;
	return L;
}

void freeList(List* pL) {
	if (pL != NULL && *pL != NULL) {
        	while (length(*pL) != 0) {
            		deleteFront(*pL);
        	}
        	free(*pL);
        	*pL = NULL;
    }
}

// Access functions -----------------------------------------------------------
int length(List L){ // Returns the number of elements in L.

	if(L == NULL){
		printf("List Error: calling length() on NULL List reference\n");
		exit(1);
	}
	return L->length;
}
int index(List L){ // Returns index of cursor element if defined, -1 otherwise.

	if(L == NULL){
		printf("List Error: calling index() on NULL List reference\n");
        	exit(1);
	}
	return  L ->index;
}

int front(List L){ // Returns front element of L. Pre: length()>0

	if(L == NULL){
		printf("List Error: calling front() on NULL list reference\n");
		exit(1);
	}
	if(L->length == 0){
		printf("List Error: calling front() on an empty list\n");
		exit(1);
	}
	return L->front->data;

}
int back(List L){ // Returns back element of L. Pre: length()>0
	
	if(L == NULL){
		printf("List Error: calling back() on NULL list reference\n");
		exit(1);
	}
	if(L->length == 0){
		printf("List ErrorL calling back() on an empty list\n");
		exit(1);
	}
	return L->back->data;
}


int get(List L){ // Returns cursor element of L. Pre: length()>0, index()>=0

	if(L == NULL){
		printf("List Error: calling get() on NULL list reference\n");
		exit(1);
	}
	if(L->length == 0){
		printf("List Error: calling get() on an empty list\n");
		exit(1);
	}
	if(L->index < 0){
		printf("List Error: calling get() when cursor is undefined\n");
		exit(1);
	}
	return L->cursor->data;

}
bool equals(List A, List B){ // Returns true iff Lists A and B contain the same
	if(A == NULL || B == NULL){		// sequence of elements, returns false otherwise.
		printf("List Error: calling equals() on NULL list referenxe\n");
		exit(1);
	}

	if(A -> length != B->length){
		return false;
	}

	Node aNode = A->front; 
	Node bNode = B->front;

	while (aNode != NULL && bNode != NULL){
		if (aNode->data != bNode -> data){
			return false;
		}

		aNode = aNode->next;
		bNode = bNode->next;
	}

	return true;

}


// Manipulation procedures ----------------------------------------------------
void clear(List L){ // Resets L to its original empty state.
	if(L == NULL){
		printf("List Error: calling clear() on NULL list reference\n");
		exit(1);
	}	

	while(L->length>0){
		deleteFront(L);
	}
	L->cursor = NULL;
	L->index = -1;
}

void set(List L, int x){ // Overwrites the cursor elementâ€™s data with x.
 			// Pre: length()>0, index()>=0

	if(L == NULL){
		printf("List Error: calling set() on NULL list reference\n");
		exit(1);
	}
	if(L->length == 0){
		printf("List Error: calling set() on empty List\n");
		exit(1);
	}
	if(L->index<0){


		printf("List Error: calling set() on undefined cursor\n");
		exit(1);
	}

	L->cursor->data = x;
}

void moveFront(List L){ // If L is non-empty, sets cursor under the front element,
 			// otherwise does nothing.
	if(L!=NULL && L->length>0){
		L->cursor = L->front;
		L->index = 0;
	}		
			
}

void moveBack(List L){ // If L is non-empty, sets cursor under the back element,
 		       // otherwise does nothing.		       
	if(L!=NULL && L->length>0){
		L->cursor = L->back;
		L->index = L->length-1;
	}
}
		       

void movePrev(List L){  // If cursor is defined and not at front, move cursor one
  			// step toward the front of L; if cursor is defined and at
 			// front, cursor becomes undefined; if cursor is undefined
	     		// do nothing
	if(L!=NULL && L->cursor!=NULL){
		if(L->cursor == L->front){
			L->cursor = NULL;
			L->index = -1;
		}
		else{
			L->cursor = L->cursor->prev;
			L->index--;
		}
	}		
}


void moveNext(List L){ // If cursor is defined and not at back, move cursor one
        		// step toward the back of L; if cursor is defined and at
			// back, cursor becomes undefined; if cursor is undefined
 			// do nothing

	if(L!=NULL && L->cursor!=NULL){
		if(L->cursor == L->back){
			L->cursor = NULL;
			L->index = -1;
		}
		else{
			L->cursor = L->cursor->next;
			L->index++;
		}
	}
}
void prepend(List L, int x){ // Insert new element into L. If L is non-empty,
          		     // insertion takes place before front element.
	if(L == NULL){
		printf("List Error: calling prepend() on NULL List reference\n");
		exit(1);
	}

	Node newNode = malloc(sizeof(NodeObj));
	newNode->data = x;
	newNode->prev = NULL;
	newNode->next = L->front;

	if(L->length == 0){
		L->back = newNode;
	}
	else{
		L->front->prev = newNode;
	}

	L->front = newNode;
	L->length++;

	if(L->index != -1){
		L->index++;
	}
}

void append(List L, int x){ // Insert new element into L. If L is non-empty,
            		// insertion takes place after back element.
	if (L == NULL) {
        	printf("List Error: calling append() on NULL List reference\n");
        	exit(1);
    	}	
    	
   	Node newNode = malloc(sizeof(NodeObj));
    	newNode->data = x;
    	newNode->next = NULL;
    	newNode->prev = L->back;
    
    	if (L->length == 0) {
        	L->front = newNode;
    	} else {
        	L->back->next = newNode;
    	}	
    
    	L->back = newNode;
    	L->length++;
}

void insertBefore(List L, int x){ // Insert new element before cursor.
			          // Pre: length()>0, index()>=0

	if (L == NULL || L->length == 0 || L->index < 0) {
        	printf("List Error: calling insertBefore() on invalid state\n");
        	exit(1);
    	}	
    
    	if (L->cursor == L->front) {
        	prepend(L, x);
    	} 
	else {
        	Node newNode = malloc(sizeof(NodeObj));
        	newNode->data = x;
       		newNode->next = L->cursor;
        	newNode->prev = L->cursor->prev;
        
        	L->cursor->prev->next = newNode;
        	L->cursor->prev = newNode;
        
        	L->length++;
        	L->index++;
    	}	
}

void insertAfter(List L, int x){  // Insert new element after cursor.
 				  //Pre: length()>0, index()>=0

	if (L == NULL || L->length == 0 || L->index < 0) {
        	printf("List Error: calling insertAfter() on invalid state\n");
        	exit(1);
    	}	

    	if (L->cursor == L->back) {
        	append(L, x);
    	} 
	else {
        	Node newNode = malloc(sizeof(NodeObj));
        	newNode->data = x;
        	newNode->prev = L->cursor;
        	newNode->next = L->cursor->next;

        	L->cursor->next->prev = newNode;
        	L->cursor->next = newNode;

        	L->length++;
    	}	
}

void deleteFront(List L){ // Delete the front element. Pre: length()>0
	if (L == NULL || L->length == 0) {
		printf("List Error: calling deleteFront() on invalid state\n");
        	exit(1);
    	}
    
    	Node temp = L->front;
    
   	if (L->length == 1) {
        	L->front = L->back = NULL;
   	} 
	else {
        	L->front = L->front->next;
        	L->front->prev = NULL;
    	}
    
    	free(temp);
    	L->length--;
    
    	if (L->index != -1) {
        	L->index--; 
    	}
}

void deleteBack(List L){  // Delete the back element. Pre: length()>0
	if (L == NULL || L->length == 0) {
        	printf("List Error: calling deleteBack() on invalid state\n");
        	exit(1);
    	}

	if (L->cursor == L->back) {
        	L->cursor = NULL;
        	L->index = -1;
    	}	
    
    	Node temp = L->back;
    
    	if (L->length == 1) {
        	L->front = L->back = NULL;
    	} 
	else {
        	L->back = L->back->prev;
        	L->back->next = NULL;
    	}
    
    	free(temp);
    	L->length--;
}

void delete(List L){ 	  // Delete cursor element, making cursor undefined.
			  // Pre: length()>0, index()>=0
    	if (L == NULL || L->length == 0 || L->index < 0) {
        	printf("List Error: calling delete() on invalid state\n");
        	exit(1);
    	}
    
    	if (L->cursor == L->front) {
        	deleteFront(L);
    	} 
	else if (L->cursor == L->back) {
        	deleteBack(L);
    	}	 
	else {
        	Node temp = L->cursor;
        	L->cursor->prev->next = L->cursor->next;
        	L->cursor->next->prev = L->cursor->prev;
        	free(temp);
        	L->length--;
    	}	
    
    	L->cursor = NULL;
    	L->index = -1;
}



void printList(FILE* out, List L){ // Prints to the file pointed to by out, a
				   //string representation of L consisting
 				   // of a space separated sequence of integers,
 				   // with front on left.
	if (L == NULL) {
        	fprintf(stderr, "List Error: calling printList() on NULL List reference\n");
        	exit(1);
    	}

    	Node current = L->front;
    	
	while (current != NULL) {
        	fprintf(out, "%d ", current->data);
        	current = current->next;
    	}
}
				   
List copyList(List L){ // Returns a new List representing the same integer
  		       // sequence as L. The cursor in the new list is undefined,
  		       // regardless of the state of the cursor in L. The state
		       // of L is unchanged.


	if (L == NULL) {
        	fprintf(stderr, "List Error: calling copyList() on NULL List reference\n");
        	exit(1);
    	}

    	List newL = newList();  // Create a new empty list
    	Node current = L->front;
    	while (current != NULL) {
        	append(newL, current->data);  // Copy each element to the new list
        	current = current->next;
    	}

    	return newL; 		       
}
