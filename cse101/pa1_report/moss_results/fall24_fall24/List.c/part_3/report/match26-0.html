#include "List.h"

struct NodeObj {
    int data;
    Node next;
    Node prev;
};

struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
    int index;
};



List newList(void){
    List L = malloc(sizeof(struct ListObj));
    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->length = 0;
    L->index = -1;
    return L;
}

void freeList(List* pL) {


    if (pL != NULL && *pL != NULL) {
        clear(*pL);
        free(*pL);
        *pL = NULL;
    }
}

int length(List L) {
    if (L == NULL) {
        printf("Error: list is null\n");
        exit(1);
    }
    return L->length;
}

int index(List L) {
    if (L == NULL) {
        printf("Error: list is null\n");
        exit(1);
    }
    return L->index;
}

int front(List L) {
    if (L == NULL) {
        printf("Error: list is null\n");
        exit(1);
    }
    if (length(L) <= 0) {
        printf("Error: list is empty\n");
        exit(1);
    }
    return L->front->data;
}

int back(List L) {
    if (L == NULL) {
        printf("Error: list is null\n");
        exit(1);
    }
    if (length(L) <= 0) {
        printf("Error: list is empty\n");
        exit(1);
    }
    return L->back->data;
}

int get(List L){
    if(L == NULL){
        printf("Error: list is null\n");
        exit(1);
    }
    if(length(L) <= 0){
        printf("Error: list is empty\n");
        exit(1);
    }
    if(index(L) < 0){
        printf("Error: cursor is undefined\n");
        exit(1);
    }
    return L->cursor->data;
}

bool equals(List A, List B){
    if(A == NULL || B == NULL){
        printf("Error: list(s) is null\n");
        exit(1);
    }
    if(length(A) != length(B)){
        return false;
    }
    Node node1 = A->front;
    Node node2 = B->front;

    while (node1 != NULL) {
        if (node1->data != node2->data) {
            return false; 
        }
        node1 = node1->next;
        node2 = node2->next;
    }
    return true;
}

void clear(List L){
    if(L == NULL){
        printf("Error: list is null\n");
        exit(1);
    }
    while(length(L) > 0){
        deleteFront(L);
    }
    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->length = 0;
    L->index = -1;
}

        exit(1);
    }
    if(index(L) < 0){
        printf("Error: cursor is undefined\n");

    }
    L->cursor->data = x;
}



void moveFront(List L){
    if(L == NULL){
        printf("Error: list is null\n");
        exit(1);
    }
    if(length(L) > 0){
        L->cursor = L->front;
        L->index = 0;
    }
}

void moveBack(List L){
    if(L == NULL){
        printf("Error: list is null\n");
        exit(1);
    }
    if(length(L) > 0){
        L->cursor = L->back;
        L->index = L->length - 1;
    }
}

void movePrev(List L){
    if(L == NULL){
        printf("Error: list is null\n");
        exit(1);
    }
    if (L->cursor != NULL && L->cursor != L->front){
        L->cursor = L->cursor->prev;
        L->index--;
    } 
    else{


        L->cursor = NULL;
        L->index = -1;
    }
}

void moveNext(List L){
    if(L == NULL){
        printf("Error: list is null\n");
        exit(1);
    }
    if(L->cursor != NULL && L->index != length(L) - 1){
        L->cursor = L->cursor->next;
        L->index++;
    }
    else if(L->cursor != NULL && L->index == length(L) - 1){
        L->cursor = NULL;
        L->index = -1;
    }
}
void prepend(List L, int x){
    if(L == NULL){
        printf("Error: list is null\n");
        exit(1);
    }
    Node temp = malloc(sizeof(struct NodeObj));
    temp->data = x;
    temp->next = NULL;
    temp->prev = NULL;
    if(length(L) <= 0){
        L->front = temp;
        L->back = temp;
    }
    else{


        L->front->prev = temp;
        temp->next = L->front;
        L->front = temp;
    }
    L->length++;
    if(L->cursor != NULL){
        L->index++;
    }
}

    Node temp = malloc(sizeof(struct NodeObj));

    temp->prev = NULL;
    if(length(L) <= 0){
        L->front = temp;
        L->back = temp;
    }
    else{
        L->back->next = temp;
        temp->prev = L->back;
        L->back = temp;
    }
    L->length++;
}

void deleteFront(List L){
    if(L == NULL){
        printf("Error: list is null\n");
        exit(1);
    }
    if(length(L) <= 0){
        printf("Error: list is empty\n");
        exit(1);
    }
    if(L->cursor == L->front){
        L->cursor = NULL;
        L->index = -1;
    } 
    else if (L->index != -1) {
        L->index--;
    }
    Node temp = L->front;
    if(length(L) > 1){
        L->front = L->front->next;
        L->front->prev = NULL;
    }
    else{
        L->front = NULL;
        L->back = NULL;
    }
    L->length--;
    
    freeNode(&temp);
}


void deleteBack(List L){
    if(L == NULL){
        printf("Error: list is null\n");
        exit(1);
    }
    if(length(L) <= 0){
        printf("Error: list is empty\n");
        exit(1);
    }
    if(L->cursor == L->back){
        L->cursor = NULL;
        L->index = -1;
    }
    Node temp = L->back;
    if(length(L) > 1){
        L->back = L->back->prev;
        L->back->next = NULL;
    }
    else{
        L->front = NULL;
        L->back = NULL;
    }
    L->length--;
    freeNode(&temp);
}

void delete(List L){
    if(L == NULL){
        printf("Error: list is null\n");
        exit(1);
    }
    if(length(L) <= 0){
        printf("Error: list is empty\n");
        exit(1);
    }
    if(index(L) < 0){
        printf("Error: cursor is undefined\n");
        exit(1);
    }
    if(L->cursor == L->front){
        deleteFront(L);
    }
    else if(L->cursor == L->back){
        deleteBack(L);
    }
    else{
        Node temp = L->cursor;
        L->cursor->prev->next = L->cursor->next;
        L->cursor->next->prev = L->cursor->prev;
        L->cursor = NULL;
        L->index = -1;
        L->length--;
        freeNode(&temp);
    }
}




void printList(FILE* out, List L){
    if(L == NULL){
        printf("Error: list is null\n");
        exit(1);
    }
    Node temp = L->front;
    while(temp != NULL){
        fprintf(out, "%d ", temp->data);
        temp = temp->next;
    }
}

List copyList(List L){
    if(L == NULL){
        printf("Error: list is null\n");
        exit(1);
    }
    List temp = newList();
    Node current = L->front;
    while(current != NULL){
        append(temp, current->data);
        current = current->next;
    }
    return temp;
}


void freeNode(Node* pN) {
    if (pN != NULL && *pN != NULL) {
        free(*pN);
        *pN = NULL;
    }
}
