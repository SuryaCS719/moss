#include "List.h"
#include <stdlib.h> 
#include <assert.h>

// private Node type


typedef struct NodeObj* Node;

// private NodeObj type
typedef struct NodeObj{
   int data;
   Node next;
   Node prev;
} NodeObj;

//------------------------------------------------------------------------------------------------

Node newNode(int data){
   Node N = malloc(sizeof(NodeObj));
   assert( N!=NULL );
   N->data = data;
   N->next = NULL;
   N->prev = NULL;

   return(N);
}

void freeNode(Node* pN){
   if (pN != NULL || pN != NULL) {
      free(*pN);
      *pN = NULL;
   }
}

//------------------------------------------------------------------------------------------------

// private ListObj type
typedef struct ListObj {
   Node front;
   Node back;
   Node cursor;
   int length;
   int index;
} ListObj;

List newList(void) {
   //List L;
   List L = malloc(sizeof(ListObj));
   assert( L!=NULL );
   L->front = NULL;
   L->back = NULL;
   L->length = 0;
   L->index = -1;
   return(L);
}

void freeList(List* pL) {
   if(pL!=NULL && *pL!=NULL) { 
      clear(*pL);
      free(*pL);
      *pL = NULL;
   }
}

//------------------------------------------------------------------------------------------------



int length(List L) {
   return(L->length);
}

int index(List L) {
   return(L->index);
}

int front(List L) {
   if( L->length <= 0 ){
      fprintf(stderr, "List Error: calling getFront() on empty List\n");
      exit(EXIT_FAILURE);
   }
   return(L->front->data);
}

int back(List L) {
   if( L->length <= 0 ){
      printf("List Error: calling length() on empty list\n");
      exit(EXIT_FAILURE);
   }
   return(L->back->data);
}

int get(List L) {
   if( L->length <= 0 || L->index < 0 ){
      printf("List Error: calling length() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   return(L->cursor->data);
}

bool equals(List A, List B) {
   if( A==NULL || B==NULL ){
      printf("List Error: calling equals() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   bool eq;
   eq = ( A->length == B->length );
   moveFront(A);
   moveFront(B);
   
   while( index(A) >= 0 && index(B) >= 0){
      
      int nData = get(A);
      int mData = get(B);
      eq = ( nData==mData && eq);
      moveNext(A);
      moveNext(B);
   }
   return eq;
}

//------------------------------------------------------------------------------------------------

void clear(List L) {
   while( length(L) > 0 ) { 
         deleteFront(L); 
   }
   L->front = NULL;
   L->back = NULL;
   L->length = 0;
   L->index = -1;
   L->cursor = NULL;
}

void set(List L, int x) {
   if (L->length <= 0 || L->index < 0) {
      printf("List is empty, can't use set()");
      exit(EXIT_FAILURE);
   }
   L->cursor->data = x;
}

void moveFront(List L) {
   if (L->length != 0) {
      L->cursor = L->front;
      L->index = 0;
   }
}

void moveBack(List L) {
   if (L->length != 0) {
      L->cursor = L->back;
      L->index = L->length - 1;
   }
}

void movePrev(List L) {
   if (L->cursor != L->front && L->cursor != NULL) {
      L->cursor = L->cursor->prev;
      L->index = L->index - 1;
   }
   else if (L->cursor == L-> front && L->cursor != NULL) {
      L->cursor = NULL;
      L->index = -1;
   }
}

void moveNext(List L) {
   if (L->cursor != L->back && L->cursor != NULL) {
      L->cursor = L->cursor->next;
      L->index = L->index + 1;
   }
   else if (L->cursor == L-> back && L->cursor != NULL) {
      L->cursor = NULL;
      L->index = -1;
   }
}

void prepend(List L, int x) {
   Node N = newNode(x);
   if (L->length != 0) {
      L->front->prev = N;
      N->next = L->front;
      L->front = N;
   }
   else {
      L->back = L->front = N;
   }
   L->length++;
   if (L->index >= 0) {
      L->index++;
   }
}

void append(List L, int x) {
   Node N = newNode(x);
   if (L->length != 0) {
      N->prev = L->back;
      L->back->next = N;
      L->back = N;
   }
   else {
      L->back = L->front = N;
   }
   L->length++;
}

void insertBefore(List L, int x) {
   if (L->length <= 0 || L->index < 0) {
      printf("List is empty, can't use insertBefore()");
      exit(EXIT_FAILURE);
   }
   if (L->cursor == L->front) {
      prepend(L, x);
      return;
   }
   Node N = newNode(x);
   L->cursor->prev->next = N;
   N->prev = L->cursor->prev;
   L->cursor->prev = N;
   N->next = L->cursor;
   if (L->index >= 0) {
      L->index++;
   }
   L->length++;
   
}

void insertAfter(List L, int x) {
   if (L->length <= 0 || L->index < 0) {
      printf("List is empty, can't use insertAfter()");
      exit(EXIT_FAILURE);
   }
   if (L->cursor == L->back) {
      append(L, x);
      return;
   }
   Node N = newNode(x);
   L->cursor->next->prev = N;
   N->next = L->cursor->next;
   L->cursor->next = N;
   N->prev = L->cursor;
   L->length++;
}

void deleteFront(List L) {
   Node N = NULL;
   if (length <= 0) {
      printf("List is empty, can't use deleteFront()");
      exit(EXIT_FAILURE);
   }
   else {
      N = L->front;
      if( L->length > 1 ) { 
         L->front = L->front->next;
         
      } else { 
         L->front = L->back = NULL; 
      }
      if (L->index >= 1) {
         L->index--;
      }
      else {
         L->cursor = NULL;
         L->index = -1;
      }
      L->length--;
      freeNode(&N);
   }
}

void deleteBack(List L) {
   Node N = NULL;
   if (length <= 0) {
      printf("List is empty, can't use deleteFront()");
      exit(EXIT_FAILURE);
   }
   else {
      N = L->back;
      if( L->length > 1 ) { 
         L->back = L->back->prev; 
      }
      else { 
         L->back = L->front = NULL; 
      }
      if (L->index == L->length - 1) {
         L->cursor = NULL;
         L->index = -1;
      }
      L->length--;
      freeNode(&N);
   }
}

void delete(List L) {
   if (L->length <= 0 || L->index < 0) {
      printf("List is empty, can't use delete()");
      exit(EXIT_FAILURE);
   }
   if (L->cursor == L->front) {
      deleteFront(L);
   }
   else if (L->cursor == L->back) {
      deleteBack(L);
   }
   else {


      L->cursor->prev->next = L->cursor->next;
      L->cursor->next->prev = L->cursor->prev;
      freeNode(&L->cursor);
      L->cursor = NULL;
      L->index = -1;
      L->length--;
   }
}

void printList(FILE* out, List L) {
   Node current = L->front; 

   while (current != NULL) {
      fprintf(out, "%d ", current->data);  
      current = current->next;             
   }
}

List copyList(List L) {
   List newL = newList();  
   Node current = L->front;
   while (current != NULL) {
      append(newL, current->data); 
      current = current->next;    
   }

   return newL;
}
