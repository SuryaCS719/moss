// Structs

// private Node type
typedef struct NodeObj* Node;

// private NodeObj type, every node has data and a pointer to next value (end node points to NULL)
typedef struct NodeObj{
  int data;
  Node next;
  Node prev;
} NodeObj;

// private ListObj type
typedef struct ListObj{
  Node front;
  Node back;
  Node cursor;
  int length;
  int cursor_index;
} ListObj;

// Constructors/Destructors

// newNode(), returns reference to new Node object; initializes next and data fields
Node newNode(int data) {
  Node N = malloc(sizeof(NodeObj));
  assert(N!=NULL);
  N->data = data;
  N->next = NULL;
  N->prev = NULL;
  return(N);
}

// freeNode(), frees the heap momory of node, sets pointer to NULL
void freeNode(Node* pN) {
  //printf("freeNode");
  if(pN!=NULL && *pN!=NULL) {
    free(*pN);
    *pN = NULL;
  }
}

// newList(), returns reference to new (empty) List object
List newList() {
  //printf("newList");
  List L; 
  L = malloc(sizeof(ListObj));
  assert(L!=NULL);
  L->front = L->back = L->cursor = NULL;
  L->length = 0;
  L->cursor_index = -1;
  return(L);
}

// freeList(), free all heap memory with list, set pointer of list to NULL
void freeList(List* pL) {
  //printf("freeList");
  if(pL!=NULL && *pL!=NULL) {
    while(length(*pL)>0) {
      deleteFront(*pL);
    }
    free(*pL);


    *pL = NULL;
  }
}

// Access Functions

// length(), return the length of the list
int length(List L) {
  //printf("length");
  if(L==NULL) {
    printf("List Error: calling length() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }
  return(L->length);
}

// index(), returns index of cursor element
int index(List L) {
  //printf("index");
  return L->cursor_index;
}

// front(), returns front element of the list
int front(List L) {
  //printf("front");
  if(L==NULL) {
    printf("List Error: calling front() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }
  if(length(L) == 0) {
    printf("List Error: calling front() on an empty List\n");
    exit(EXIT_FAILURE);
  }
  return(L->front->data);
}

// back(), return back element of the list
int back(List L) {
  //printf("back");
  if(L==NULL) {
    printf("List Error: calling back() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }
  if(length(L) == 0) {
    printf("List Error: calling back() on an empty List\n");
    exit(EXIT_FAILURE);
  }
  return(L->back->data);
}

// get(), returns value of element in the list at cursor
int get(List L) {
  //printf("get");
  if (L->cursor_index < 0) {
    printf("Cursor Error: calling get() on undefined cursor\n");
    exit(EXIT_FAILURE);
  }
  if(L == NULL) {
    printf("List Error: calling get() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }
  if(length(L) <= 0) {
    printf("List Error: calling get() on an empty List\n");
    exit(EXIT_FAILURE);
  }
  return L->cursor->data;
}

// equals(), returns if two lists have same sequence of elements
bool equals(List A, List B) {
  //printf("equals");
  if(A == NULL || B == NULL) {
    printf("List Error: calling equals() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }

  bool eq;
  Node N, M;

  eq = (A->length == B->length);
  N = A->front;
  M = B->front;

  while(eq && N!=NULL) {
    eq = (N->data == M->data);


    N = N->next;
    M = M->next;
  }
  return eq;
}

// Manipulation procedures

// clear(), resets list to empty list
void clear(List L) {
  //printf("clear");
  if (L == NULL) {
    printf("List Error: calling clear() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }
  while (length(L) != 0) {
    deleteBack(L);
  }
}

// moveFront(), if the list is not empty, the cursor is set to the first element, otherwise nothing
void moveFront(List L) {
  //printf("moveFront");
  if(L == NULL) {
    printf("List Error: calling moveFront() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }
  if(length(L) > 0) {
    L->cursor_index = 0;
    L->cursor = L->front;
    L->cursor->next = L->front->next;
    L->cursor->prev = NULL;
  }
}

// moveBack(), if the list is not empty, the cursor is set to the back element, otherwise nothing
void moveBack(List L) {
  //printf("moveBack");
  if(L == NULL) {
    printf("List Error: calling moveBack() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }
  if(length(L) > 0) {
    L->cursor_index = L->length - 1;
    L->cursor = L->back;
    L->cursor->prev = L->back->prev;
    L->cursor->next = NULL;
  }
}

// movePrev(), if cursor is defined and not at front, move cursor one step towards front of list; if cursor is defined and at front, cursor is undefined; if cursor is undefined, do nothing
void movePrev(List L) {
  //printf("movePrev");
  if(L == NULL) {
    printf("List Error: calling movePrev() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }
  if(L->cursor_index < 0) {
    return;
  }
  if(L->cursor_index == 0) {
    L->cursor_index = -1;
    L->cursor = NULL;
  } else {
    L->cursor_index = L->cursor_index - 1;
    L->cursor = L->cursor->prev;  
  }
}

// moveNext(), if cursor is defined and not at back, move cursor one step towards the back of L; if cursor is defined and at back, cursor becomes undefined; if cursor is undefined do nothing
void moveNext(List L) {
  //printf("moveNext");
  if(L == NULL) {
    printf("List Error: calling moveNext() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }
  if(L->cursor_index < 0) {
    return;
  }
  if(L->cursor_index == (L->length - 1)) {
    L->cursor_index = -1;
    L->cursor = NULL;
  } else {
    L->cursor_index = L->cursor_index + 1;
    L->cursor = L->cursor->next;
  }
}

// prepend(), insert element into list, if list is not empty, insertion is before front element
void prepend(List L, int data) {
  //printf("prepend");
  if(L == NULL) {
    printf("List Error: calling prepend() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }

  Node N = newNode(data);

  if(length(L) == 0) {
    L->front = L->back = N;
    L->front->next = NULL;
    L->back->next = NULL;
  } else {
    L->front->prev = N;
    N->next = L->front;
    L->front = N;
    N->prev = NULL;
  }
  
  L->length = L->length + 1;
  if(L->cursor_index == 0) {
    L->cursor_index = L->cursor_index + 1;
    L->cursor->prev = L->front;
  } else if (L->cursor_index > 0) {
    L->cursor_index = L->cursor_index + 1;
  }
}

// append(), insert element into list, if list is not empty, insertion is after back element
void append(List L, int data) {
  //printf("append");
  if(L == NULL) {
    printf("List Error: calling append() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }

  Node N = newNode(data);
  
  if(length(L) == 0) {
    L->front = L->back = N;
    L->front->next = NULL;
    L->back->next = NULL;
  } else {
    L->back->next = N;
    N->prev = L->back;
    L->back = N;
    N->next = NULL;
  }
  L->length = L->length + 1;
  if(((L->cursor_index + 2) == L->length) && (L->cursor_index > -1)) {
    L->cursor->next = N;
  }
}

// insertBefore(), insert element before cursor; Pre: length()>0, index()>=0
void insertBefore(List L, int x) {
  //printf("insertBefore");
  if(L == NULL) {
    printf("List Error: calling insertBefore() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }

  if (length(L) <= 0) {
    printf("List Error: calling insertBefore() on empty List reference\n");
    exit(EXIT_FAILURE);
  }

  if(L->cursor_index < 0) {
    printf("Cursor Error: calling insertBefore() on undefined cursor\n");
    exit(EXIT_FAILURE);
  }


  if(L->cursor_index == 0) {
    prepend(L, x);
  } else {
    Node new = newNode(x);
    Node N = L->cursor->prev;
    new->prev = N;
    new->next = L->cursor;
    N->next = new;
    L->cursor->prev = new;
    L->cursor_index = L->cursor_index + 1;
    L->length = L->length + 1;
  }
}

// insertAfter(), insert element after cursor; Pre: length()>0, index()>=0
void insertAfter(List L, int x) {
  //printf("insertAfter");
  if(L == NULL) {
    printf("List Error: calling insertAfter() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }

  if (length(L) <= 0) {
    printf("List Error: calling insertAfter() on empty List reference\n");
    exit(EXIT_FAILURE);
  }

  if(L->cursor_index < 0 || L->cursor == NULL) {
    printf("Cursor Error: calling insertAfter() on undefined cursor\n");
    exit(EXIT_FAILURE);
  }
  

  if((L->cursor_index + 1) == L->length) {
    append(L, x);
  } else {
    Node new = newNode(x);
    Node N = L->cursor->next;
    new->prev = L->cursor;
    new->next = N;
    L->cursor->next = new;
    N->prev = new;
    L->length = L->length + 1;
  }
}

// deleteFront(), delete front element
void deleteFront(List L) {
  //printf("deleteFront");
  if(L == NULL) {
    printf("List Error: calling deleteFront() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }
  if(length(L) <= 0) {
    printf("List Error: calling deleteFront() on an empty List\n");
    exit(EXIT_FAILURE);
  }
  Node N = L->front;
  if (L->length == 1) {
    L->front = L->back = NULL;
  } else {
    L->front = L->front->next;
    L->front->prev = NULL; 
  }
  
  if (L->cursor_index == 0) {
    L->cursor_index = -1;
    L->cursor = NULL;
    Node A = L->cursor;
    freeNode(&A);
  } else if (L->cursor_index == 1) {
    L->cursor->prev = NULL;
    L->cursor = L->front;
    if (L->front != NULL) {
      L->cursor->next = L->front->next;
    }
    L->cursor_index = L->cursor_index-1;
  } else if (L->cursor_index > 1) {
    L->cursor_index = L->cursor_index-1;
  }

  L->length--;
  freeNode(&N);
}

// deleteBack(), delete back element
void deleteBack(List L) {
  //printf("deleteBack");
  if(L == NULL) {
    printf("List Error: calling deleteFront() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }
  if(length(L) <= 0) {
    printf("List Error: calling deleteFront() on an empty List\n");
    exit(EXIT_FAILURE);
  }
  Node N = L->back;
  if (L->length == 1) {
    L->front = L->back = NULL;
  } else {
    L->back = L->back->prev;
    L->back->next = NULL;
  }

  if (L->cursor_index == (L->length - 1)) {
    L->cursor_index = -1;
    L->cursor = NULL;
  } else if ((L->cursor_index > -1) && (L->cursor_index == (L->length-2))) {
    L->cursor->next = NULL;
  }

  L->length--;
  freeNode(&N);
}

// delete(), delete cursor element, make cursor undefined; Pre: length()>0, index()>=0
void delete(List L) {
  //printf("delete");
  if(L == NULL) {
    printf("List Error: calling delete() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }

  if (length(L) <= 0) {
    printf("List Error: calling delete() on empty List reference\n");
    exit(EXIT_FAILURE);
  }

  if(L->cursor_index < 0) {
    printf("Cursor Error: calling delete() on undefined cursor\n");
    exit(EXIT_FAILURE);
  }
  
  if(L->cursor_index == (L->length-1)) {
    deleteBack(L);		  
  } else if (L->cursor_index == 0) {
    deleteFront(L);
  } else {
    Node N = L->cursor->next;
    Node P = L->cursor->prev;
    P->next = N;
    N->prev = P;
    L->cursor_index = -1;
    L->cursor = NULL;
    L->length--;
  }
}

// copyList(), Returns a new List representing the same integer sequence as L. The cursor in the new list is undefined, regardless of the state of the cursor in L. The state of L is unchanged.
List copyList(List L) {
  //printf("copyList");
  List new = newList();
  
  Node N = L->front;

  if (L == NULL) {
    printf("List Error: calling copyList() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }

  if(L==NULL) {
    return new;
  }
  while(N != NULL) {
    append(new, N->data);
    N = N->next;
  }
  new->length = length(L);
  return new;
}

// concatList(), Returns a new List which is the concatenation of A and B. The cursor in the new List is undefined, regardless of the states of the cursors in A and B. The states of A and B are unchanged.
List concatList(List A, List B) {
  //printf("concatList");
  if (A == NULL || B == NULL) {
    printf("List Error: calling concatList() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }

  if (A->length == 0) {
    List new = copyList(B);
    return new;    
  }

  if (B->length == 0) {
    List new = copyList(A);
    return new;
  }

  List new = copyList(A);
  List b_list = copyList(B);
  Node N = new->front;
  while(N->next != NULL) {
    N = N->next;
  }
  N->next = b_list->front;

  return new;
}
