#include <stdio.h>
#include <stdlib.h>
#include <assert.h>
#include <stdbool.h>
#include "List.h"
#include <stddef.h>
typedef struct NodeObj *Node;

typedef struct NodeObj
{
    int data;
    Node previous;
    Node next;
} NodeObj;

typedef struct ListObj
{
    Node front;
    Node back;
    Node cursor;
    int length;
    int index;
} ListObj;

List newList(void)
{
    List L = malloc(sizeof(ListObj));
    assert(L != NULL);
    L->front = L->back = L->cursor = NULL;
    L->length = 0;
    L->index = -1;
    return L;
}

void freeList(List *pL)
{
    if (pL != NULL && *pL != NULL)
    {
        clear(*pL);
        free(*pL);
        *pL = NULL;
    }
}

int length(List L)
{
    if (L == NULL)
    {
        fprintf(stderr, "List is Empty\n");
        exit(EXIT_FAILURE);
    }
    return L->length;
}

int index(List L)
{
    return L->index;
}



int front(List L)
{
    if (length(L) > 0)
    {
        return L->front->data;
    }
    else
    {
        fprintf(stderr, "List is Empty\n");
        exit(EXIT_FAILURE);
    }
}

int back(List L)
{
    if (length(L) > 0)
    {
        return L->back->data;
    }
    else
    {
        fprintf(stderr, "List is Empty\n");
        exit(EXIT_FAILURE);
    }
}

int get(List L)
{
    if (length(L) > 0 && index(L) >= 0)
    {
        return L->cursor->data;
    }
    else
    {
        fprintf(stderr, "List is Empty\n");
        exit(EXIT_FAILURE);
    }
}

bool equals(List A, List B)
{
    if (A == NULL || B == NULL)
    {
        fprintf(stderr, "Lists are Empty\n");
        exit(EXIT_FAILURE);
    }

    if (length(A) != length(B))
    {
        return false;
    }

    Node N = A->front;
    Node M = B->front;

    while (N != NULL && M != NULL)
    {
        if (N->data != M->data)
        {
            return false;
        }


        N = N->next;
        M = M->next;
    }

    return true;
}

void clear(List L)
{
    while (length(L) > 0)
    {
        deleteFront(L);
    }
}

void moveFront(List L)
{
    if (length(L) > 0)
    {
        L->cursor = L->front;


        L->index = 0;
    }
}

void moveBack(List L)
{
    if (length(L) > 0)
    {
        L->cursor = L->back;
        L->index = length(L) - 1;
    }
}

void movePrev(List L)
{
    if (L->cursor != NULL)
    {
        L->cursor = L->cursor->previous;
        if (L->cursor == NULL)
        {
            L->index = -1;
        }
        else
        {
            L->index--;
        }
    }
    else
    {
        fprintf(stderr, "Cursor is undefined\n");
        exit(EXIT_FAILURE);
    }
}

void moveNext(List L)
{
    if (L->cursor != NULL)
    {
        L->cursor = L->cursor->next;
        if (L->cursor == NULL)
        {
            L->index = -1;
        }
        else
        {
            L->index++;
        }
    }
    else
    {
        fprintf(stderr, "Cursor is Undefined\n");
        exit(EXIT_FAILURE);
    }
}

void prepend(List L, int x)
{
    Node N = malloc(sizeof(NodeObj));
    assert(N != NULL);
    N->data = x;
    N->next = L->front;
    N->previous = NULL;

    if (length(L) == 0)
    {
        L->back = N;
    }
    else
    {
        L->front->previous = N;
    }

    if (L->cursor == NULL){
       L->index = -1;
    }else{
          L->index++;
    }
    L->front = N;
    L->length++;
}

void append(List L, int x)
{
    Node N = malloc(sizeof(NodeObj));
    assert(N != NULL);
    N->data = x;
    N->previous = L->back;


    N->next = NULL;

    if (length(L) == 0)
    {
        L->front = N;
    }
    else
    {
        L->back->next = N;
    }

    L->back = N;
    L->length++;
}

void insertBefore(List L, int x)
{
    if (length(L) > 0 && index(L) >= 0)
    {
        Node N = malloc(sizeof(NodeObj));
        assert(N != NULL);
        N->data = x;

        if (L->cursor->previous != NULL)
        {
            N->previous = L->cursor->previous;
            L->cursor->previous->next = N;
        }
        else
        {
            N->previous = NULL;
            L->front = N;
        }

        N->next = L->cursor;
        L->cursor->previous = N;
        L->length++;
        L->index++;
    }
    else
    {


        fprintf(stderr, "Empty List or Undefined Cursor\n");
        exit(EXIT_FAILURE);
    }
}

void insertAfter(List L, int x)
{
    if (length(L) > 0 && index(L) >= 0)
    {
        Node N = malloc(sizeof(NodeObj));
        assert(N != NULL);
        N->data = x;

        if (L->cursor->next != NULL)
        {
            N->next = L->cursor->next;
            L->cursor->next->previous = N;
        }
        else
        {
            N->next = NULL;
            L->back = N;
        }

        N->previous = L->cursor;
        L->cursor->next = N;
        L->length++;
    }
    else
    {


        fprintf(stderr, "Empty List or Undefined Cursor\n");
        exit(EXIT_FAILURE);
    }
}

void deleteFront(List L)
{
    if (length(L) > 0)
    {
        Node N = L->front;

        if (length(L) == 1)
        {
            L->front = L->back = NULL;
        }
        else
        {
            L->front = N->next;
            L->front->previous = NULL;
        }

        free(N);
        L->length--;

        if (L->index == 0)
        {
            L->cursor = NULL;
            L->index = -1;
        }
        else if (L->index > 0)
        {


            L->index--;
        }
    }
    else
    {
        fprintf(stderr, "List is Empty\n");
        exit(EXIT_FAILURE);
    }
}

void deleteBack(List L)
{
    if (length(L) > 0)
    {
        Node N = L->back;

        if (length(L) == 1)
        {
            L->front = L->back = NULL;
        }
        else
        {
            L->back = N->previous;
            L->back->next = NULL;
        }

        free(N);
        L->length--;

        if (L->index == length(L))
        {
            L->cursor = NULL;
            L->index = -1;
        }
    }
    else
    {
        fprintf(stderr, "List is Empty\n");
        exit(EXIT_FAILURE);
    }
}

void delete (List L)
{
    if (length(L) > 0 && index(L) >= 0)
    {
        Node N = L->cursor;

        if (length(L) == 1)
        {
            L->front = L->back = NULL;
        }
        else if (index(L) == 0)
        {
            L->front = N->next;
            L->front->previous = NULL;
        }
        else if (index(L) == length(L) - 1)
        {
            L->back = N->previous;
            L->back->next = NULL;
        }
        else
        {
            N->previous->next = N->next;
            N->next->previous = N->previous;
        }

        free(N);
        L->length--;
        L->index = -1;
    }
    else
    {
        fprintf(stderr, "List is Empty\n");
        exit(EXIT_FAILURE);
    }
}

void printList(FILE *out, List L)
{
    if (out == NULL)
    {
        fprintf(stderr, "Output File is NULL\n");
        exit(EXIT_FAILURE);
    }

    Node N = L->front;

    while (N != NULL)
    {
        fprintf(out, "%d ", N->data);
        N = N->next;
    }

    //fprintf(out, "\n");
}

List copyList(List L)
{
    List copy = newList();
    Node N = L->front;

    while (N != NULL)
    {
        append(copy, N->data);
        N = N->next;
    }

    return copy;
}

List concatList(List A, List B)
{
    List C = copyList(A);
    Node N = B->front;

    while (N != NULL)
    {
        append(C, N->data);
        N = N->next;
    }

    return C;
}
