#include <stdio.h>
#include <stdlib.h>
#include <stdlib.h>
#include "List.h"

typedef struct NodeObj {
    int data;
    struct NodeObj* prev;
    struct NodeObj* next;
} NodeObj;

typedef struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
    int cursorIndex;
} ListObj;

List newList(void) {
    List L = (List)malloc(sizeof(ListObj));
    if (L == NULL) {
        exit(1);
    }
    L->length = 0;
    L->cursor = NULL;
    L->cursorIndex = -1;
    L->front = NULL;
    L->back = NULL;
    return L;
}

Node newNode(int data) {
    Node N = (Node)malloc(sizeof(NodeObj));
    if (N == NULL) {
        exit(1);
    }
    N->data = data;
    N->prev = NULL;
    N->next = NULL;
    return N;
}

void freeNodes(List L) {
    Node current = L->front;
    while (current != NULL) {
        Node nextNode = current->next;
        free(current);
        current = nextNode;
    }
}

void freeList(List* pL) {


    if (pL != NULL && *pL != NULL) {
        freeNodes(*pL);
        
        free(*pL);
        
        *pL = NULL;
    }
}

int length(List L) {
    if (L == NULL) {
        exit(1);
    }
    return L->length;
}

int index(List L) {
    if (L == NULL) {
        exit(1);
    }
    
    if (L->cursor == NULL) {
        return -1;
    }
    
    Node N = L->front;
    int i = 0;
    
    while (N != NULL) {
        if (N == L->cursor) {
            return i;
        }
        N = N->next;
        i++;
    }

    return -1;
}


int front(List L) {
    if (L == NULL) {
        exit(1);
    }
    if (L->length == 0) {
        exit(1);
    }
    return L->front->data;
}

int back(List L) {
    if (L == NULL) {
        exit(1);
    }
    if (L->length == 0) {
        exit(1);
    }
    return L->back->data;
}

int get(List L) {
    if (L == NULL) {
        exit(1);
    }
    if (L->length == 0) {
        exit(1);
    }
    if (index(L) < 0) {


        exit(1);

}

    if (A->length != B->length) {
        return false;
    }

    Node aNode = A->front;
    Node bNode = B->front;

    while (aNode != NULL && bNode != NULL) {
        if (aNode->data != bNode->data) {
        }
        aNode = aNode->next;
        bNode = bNode->next;
    }

}

void clear(List L) {
    if (L == NULL) {
        exit(1);
    }

    while (L->length > 0) {
        deleteFront(L);
    }

    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->length = 0;
}

void set(List L, int x) {
    if (L == NULL) {
        exit(1);
    }
    if (L->length == 0) {
        exit(1);
    }
    if (index(L) < 0) {


        exit(1);
    }

    L->cursor->data = x;
}

void moveFront(List L) {
    if (L == NULL) {
        exit(1);
    }
    if (L->length > 0) {
        L->cursor = L->front;
    }
}

void moveBack(List L) {
    if (L == NULL) {
        exit(1);
    }
    if (L->length > 0) {
        L->cursor = L->back;
    }
}

void movePrev(List L) {
    if (L == NULL) {
        exit(1);
    }
    if (L->cursor != NULL) {
        if (L->cursor == L->front) {
            L->cursor = NULL;
        } else {
            L->cursor = L->cursor->prev;
        }
    }
}

void moveNext(List L) {
    if (L == NULL) {
        exit(1);
    }
    if (L->cursor != NULL) {
        if (L->cursor == L->back) {
            L->cursor = NULL;
        } else {
            L->cursor = L->cursor->next;
        }
    }
}


void prepend(List L, int x) {
    if (L == NULL) {
        exit(1);
    }
    
    Node newNode = (Node)malloc(sizeof(NodeObj));
    if (newNode == NULL) {
        exit(1);
    }
    
    newNode->data = x;
    newNode->next = L->front;
    newNode->prev = NULL;
    
    if (L->front != NULL) {
        L->front->prev = newNode;
    } else {
        L->back = newNode;
    }
    
    L->front = newNode;


    L->length++;

    if (L->cursorIndex != -1) {
        L->cursorIndex++;
    }
}

void append(List L, int x) {
    if (L == NULL) {
        exit(1);
    }

    Node newNode = (Node)malloc(sizeof(NodeObj));
    if (newNode == NULL) {
        exit(1);
    }
    
    newNode->data = x;
    newNode->prev = L->back;
    newNode->next = NULL;

    if (L->back != NULL) {
        L->back->next = newNode;
    } else {


        L->front = newNode;
    }

    L->back = newNode;
    L->length++;
}


void insertBefore(List L, int x) {
    if (L == NULL) {
        exit(1);
    }
    if (L->length == 0 || index(L) < 0) {
        exit(1);
    }

    Node N = newNode(x);

    if (L->cursor == L->front) {
        prepend(L, x);
    } else {
        N->next = L->cursor;
        N->prev = L->cursor->prev;
        L->cursor->prev->next = N;
        L->cursor->prev = N;
        L->length++;
    }
}

void insertAfter(List L, int x) {
    if (L == NULL) {
        exit(1);
    }
    if (L->length == 0 || index(L) < 0) {
        exit(1);
    }

    Node N = newNode(x);

    if (L->cursor == L->back) {
        append(L, x);
    } else {
        N->prev = L->cursor;
        N->next = L->cursor->next;


        L->cursor->next->prev = N;
        L->cursor->next = N;
        L->length++;
    }
}

void deleteFront(List L) {
    if (L == NULL) {
        exit(1);
    }
    if (L->length == 0) {
        exit(1);
    }

    Node temp = L->front;

    if (L->length == 1) {
        L->front = L->back = NULL;
    } else {
        L->front = L->front->next;
        L->front->prev = NULL;
    }

    if (L->cursor == temp) {
        L->cursor = NULL;
    }

    free(temp);
    L->length--;
}

void deleteBack(List L) {
    if (L == NULL) {
        exit(1);
    }
    if (L->length == 0) {
        exit(1);
    }

    Node temp = L->back;

    if (L->length == 1) {
        L->front = L->back = NULL;
    } else {
        L->back = L->back->prev;
        L->back->next = NULL;
    }

    if (L->cursor == temp) {
        L->cursor = NULL;
    }

    free(temp);
    L->length--;
}

void delete(List L) {
    if (L == NULL) {
        exit(1);
    }
    if (L->length == 0) {
        exit(1);
    }
    if (index(L) < 0) {
        exit(1);
    }

    Node temp = L->cursor;

    if (L->cursor == L->front) {
        deleteFront(L);
    } else if (L->cursor == L->back) {
        deleteBack(L);
    } else {
        L->cursor->prev->next = L->cursor->next;
        L->cursor->next->prev = L->cursor->prev;
        free(temp);
        L->length--;
    }

    L->cursor = NULL;
}



void printList(FILE* out, List L) {
    if (L == NULL) {
        exit(1);
    }
    
    Node N = L->front;
    
    while (N != NULL) {
        fprintf(out, "%d ", N->data);
        N = N->next;
    }
    fprintf(out, "\n");
}



List copyList(List L) {
    if (L == NULL) {
        exit(1);
    }

    List newL = newList();

    Node N = L->front;
    
    while (N != NULL) {
        append(newL, N->data);
        N = N->next;
    }

    return newL;
}
