#include<stdio.h>
#include<stdlib.h>
#include<stdbool.h>
#include<string.h>
#include<assert.h>
#include "List.h"

//---------------------------------------------------
// structure 
//---------------------------------------------------

// private NodeObj type


typedef struct NodeObj *Node;

// private NodeObj type
typedef struct NodeObj {
    ListElement data;
    Node next;
    Node prev;
} NodeObj;

// private ListObj type
typedef struct ListObj{
    int length;
    int index;
    Node front;
    Node back;
    Node cursor;
}ListObj;

//---------------------------------------------------
// Constructors-Destructors 
//---------------------------------------------------

static Node newNode(ListElement data) {
    Node N = malloc(sizeof(NodeObj));
    assert(N != NULL);
    N->data = data;
    N->next = N->prev = NULL;
    return N;
}

static bool checkNLLList(List L, const char *function) {
    if (L == NULL) {
        fprintf(stderr, "List Error: %s() NULL List\n", function);
        return false;
    }
    return true;
}

static bool checkEmptyList(List L, const char *function) {
    if (!checkNLLList(L, __func__)) return false;

    if (L->length == 0) {
        fprintf(stderr, "List Error: %s() empty List\n", function);
        return false;
    }
    return true;
}

static bool checkListCursor(List L, const char *function) {
    if (L->index == -1) {
        assert(L->cursor == NULL);
        fprintf(stderr, "List Error: %s() undefined cursor\n", function);
        return false;
    }
    return true;
}

List newList(void) {
    List list;
    list = malloc(sizeof(ListObj));
    list->front = list->back = list->cursor = NULL;
    list->length = 0;
    list->index = -1;
    return list;
}

void freeList(List *pL) {
    if (pL == NULL || *pL == NULL) {
        return;
    }
    clear(*pL);
    free(*pL);
    *pL = NULL;
}


int length(List L) {
    if (!checkNLLList(L, __func__)) {
        return -1;
    }
    return L->length;
}

int index(List L) {
    if (!checkNLLList(L, __func__)) {
        return -1;
    }
    return L->index;
}

ListElement front(List L) {
    if (!checkEmptyList(L, __func__)) {
        return -1;
    }
    return L->front->data;
}

ListElement back(List L) {
    if (!checkEmptyList(L, __func__)) {
        return -1;
    }
    return L->back->data;
}

ListElement get(List L) {
    if (!checkEmptyList(L, __func__)) {
        return -1;
    }

    if (!checkListCursor(L, __func__)) {
        return -1;
    }

    return (L->cursor->data);
}

bool equals(List A, List B) {
    if (A == NULL && B == NULL) return true;
    if (A != NULL && B == NULL) return false;
    if (A == NULL && B != NULL) return false;

    if (length(A) != length(B)) return false;
    Node N = A->front;
    Node M = B->front;
    while (N && M) {
        if (N->data != M->data) return false;
        N = N->next;
        M = M->next;
    }
    return true;
}

void clear(List L) {
    moveFront(L);
    while (L->length > 0) {
        deleteFront(L);
    }
}

void set(List L, ListElement x) {
    if (!checkEmptyList(L, __func__))
        return;
    if (!checkListCursor(L, __func__))
        return;

    L->cursor->data = x;
}

void moveFront(List L) {
    if (!checkNLLList(L, __func__))
        return;

    if (L->length == 0) { return; }
    L->cursor = L->front;
    L->index = 0;
}

void moveBack(List L) {
    if (!checkNLLList(L, __func__)) {
        return;
    }

    if (L->length == 0) { return; }
    L->cursor = L->back;
    L->index = (L->length - 1);
}

void movePrev(List L) {
    if (!checkNLLList(L, __func__)) {
        return;
    }
    if (L->cursor == NULL) { return; }
    L->cursor = L->cursor->prev;
    L->index -= 1;
}

void moveNext(List L) {
    if (!checkNLLList(L, __func__)) {
        return;
    }
    if (L->cursor == NULL) { return; }
    L->cursor = L->cursor->next;
    L->index += 1;
    if (L->cursor == NULL) L->index = -1;
}

void prepend(List L, ListElement x) {
    if (!checkNLLList(L, __func__)) {
        return;
    }

    Node node = newNode(x);
    if (L->front == NULL) {
        L->front = L->back = node;
    } else {
        L->front->prev = node;
        node->next = L->front;
        L->front = node;
    }
    if (L->index >= 0) {
        L->index += 1;
    }
    L->length += 1;
}

void append(List L, ListElement x) {
    if (!checkNLLList(L, __func__)) {
        return;
    }
    Node node = newNode(x);

    if (L->front == NULL) {
        L->front = L->back = node;
    } else {
        L->back->next = node;
        node->prev = L->back;
        L->back = node;
    }
    L->length += 1;
}

void insertBefore(List L, int x) {
    if (!checkEmptyList(L, __func__)) {
        return;
    }

    if (!checkListCursor(L, __func__)) {
        return;
    }

    if (L->index == 0) {
        prepend(L, x);
        return;
    }

    Node node = newNode(x);
    L->cursor->prev->next = node;
    node->prev = L->cursor->prev;
    L->cursor->prev = node;
    node->next = L->cursor;
    L->length += 1;
    L->index += 1;
}

void insertAfter(List L, int x) {
    if (!checkEmptyList(L, __func__)) {
        return;
    }
    if (!checkListCursor(L, __func__)) {
        return;
    }

    if (L->cursor == L->back) {
        append(L, x);
    } else {
        Node node = newNode(x);
        L->cursor->next->prev = node;
        node->next = L->cursor->next;
        L->cursor->next = node;
        node->prev = L->cursor;
        L->length += 1;
    }
}

void deleteFront(List L) {
    if (!checkEmptyList(L, __func__)) {
        return;
    }
    if (L->cursor == L->front) {
        L->cursor = NULL;
    }
    if (L->index != -1) {
        L->index -= 1;
    }
    Node newFront = L->front->next;
    free(L->front);
    L->front = newFront;

    if (L->front != NULL) {
        L->front->prev = NULL;
    } else {
        L->front = L->back = NULL;
    }
    L->length -= 1;
}

void deleteBack(List L) {
    if (!checkEmptyList(L, __func__)) {
        return;
    }

    if (L->index == (L->length - 1)) {
        L->index = -1;
        L->cursor = NULL;
    }
    Node newBack = L->back->prev;
    free(L->back);
    L->back = newBack;
    if (L->back != NULL) {
        L->back->next = NULL;
    } else {
        L->front = newBack;
    }
    L->length -= 1;
}

void delete(List L) {
    if (!checkEmptyList(L, __func__)) {
        return;
    }
    if (!checkListCursor(L, __func__)) {
        return;
    }

    if (L->index == 0) {
        deleteFront(L);
    }
    else if (L->index == (L->length - 1)) {
        deleteBack(L);
    } else {
        L->cursor->prev->next = L->cursor->next;
        L->cursor->next->prev = L->cursor->prev;
        free(L->cursor);
        L->length -= 1;
        L->index = -1;
    }
}

void printList(FILE *out, List L) {
    if (!checkEmptyList(L, __func__)) {
        return;
    }
    for (Node node = L->front; node != NULL; node = node->next) {
        fprintf(out, "%i ", node->data);
    }
}

List copyList(List L) {
    if (!checkNLLList(L, __func__)) {
        return NULL;
    }

    List out = newList();
    for (Node node = L->front; node != NULL; node = node->next) {
        append(out, node->data);
    }
    out->cursor = L->cursor;
    out->index = L->index;
    return out;
}

List concatList(List A, List B) {
    if(A == NULL && B == NULL) return NULL;
    List out = newList();

    for (Node node = (A == NULL ? NULL : A->front); node != NULL; node = node->next) {
        append(out, node->data);
    }

    for (Node node = (B == NULL ? NULL : B->front); node != NULL; node = node->next) {
        append(out, node->data);
    }

    return out;
}
