#include <stdio.h>
#include <stdlib.h>
#include "List.h"

// NodeObj type
typedef struct NodeObj {
    int data;
    struct NodeObj* next;
    struct NodeObj* prev;
} NodeObj;

// Node type
typedef NodeObj* Node;

// ListObj type
typedef struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
    int index;
} ListObj;

// Constructors-Destructors ---------------------------------------------------
List newList(void) {
    List L = malloc(sizeof(ListObj));
    L->front = L->back = L->cursor = NULL;
    L->length = 0;
    L->index = -1;
    return L;
}

void freeList(List* pL) {
    if (pL != NULL && *pL != NULL) {
        while ((*pL)->length > 0) {


            deleteFront(*pL);
        }
        free(*pL);
        *pL = NULL;
    }
}

// Access Functions -----------------------------------------------------------
int length(List L) {
    return L->length;
}

int index(List L) {
    return L->cursor == NULL ? -1 : L->index;
}

int front(List L) {
    if (L->length == 0) {
        fprintf(stderr, "List Error: calling front() on empty List\n");
        exit(EXIT_FAILURE);
    }
    return L->front->data;
}

int back(List L) {
    if (L->length == 0) {
        fprintf(stderr, "List Error: calling back() on empty List\n");
        exit(EXIT_FAILURE);
    }
    return L->back->data;
}

int get(List L) {
    if (L->length == 0 || L->index < 0) {
        fprintf(stderr, "List Error: calling get() on undefined cursor\n");
        exit(EXIT_FAILURE);
    }
    return L->cursor->data;
}

bool equals(List A, List B) {
    if (A->length != B->length) return false;
    Node N = A->front;
    Node M = B->front;
    while (N != NULL && M != NULL) {
        if (N->data != M->data) return false;
        N = N->next;
        M = M->next;
    }
    return true;
}

// Manipulation Procedures ----------------------------------------------------
void clear(List L) {
    while (L->length > 0) {


        deleteFront(L);
    }
    L->cursor = NULL;
    L->index = -1;
}

void set(List L, int x) {
    if (L->length == 0 || L->index < 0) {
        fprintf(stderr, "List Error: calling set() on undefined cursor\n");
        exit(EXIT_FAILURE);
    }
    L->cursor->data = x;
}

void moveFront(List L) {
    if (L->length > 0) {
        L->cursor = L->front;
        L->index = 0;
    }
}

void moveBack(List L) {
    if (L->length > 0) {
        L->cursor = L->back;
        L->index = L->length - 1;
    }
}

void movePrev(List L) {
    if (L->cursor != NULL) {
        if (L->cursor == L->front) {
            L->cursor = NULL;
            L->index = -1;
        } else {
            L->cursor = L->cursor->prev;
            L->index--;
        }
    }
}

void moveNext(List L) {
    if (L->cursor != NULL) {
        if (L->cursor == L->back) {
            L->cursor = NULL;
            L->index = -1;
        } else {
            L->cursor = L->cursor->next;
            L->index++;
        }
    }
}

void prepend(List L, int x) {
    Node N = malloc(sizeof(NodeObj));
    N->data = x;
    N->next = L->front;
    N->prev = NULL;

    if (L->front != NULL) {
        L->front->prev = N;
    } else {
        L->back = N; // If the list was empty, back also points to the new node
    }
    L->front = N;
    L->length++;

    if (L->cursor != NULL) {
        L->index++;  // Increment the index if cursor is defined
    }
}

void append(List L, int x) {
    Node N = malloc(sizeof(NodeObj));
    N->data = x;
    N->prev = L->back;
    N->next = NULL;
    if (L->length == 0) {
        L->front = N;
    } else {


        L->back->next = N;
    }
    L->back = N;
    L->length++;
}

void insertBefore(List L, int x) {
    if (L->length == 0 || L->index < 0) {
        fprintf(stderr, "List Error: calling insertBefore() on undefined cursor\n");
        exit(EXIT_FAILURE);
    }
    if (L->cursor == L->front) {
        prepend(L, x);  // Special case: If cursor is at the front, just prepend
    } else {
        Node N = malloc(sizeof(NodeObj));
        N->data = x;
        N->prev = L->cursor->prev;
        N->next = L->cursor;
        L->cursor->prev->next = N;


        L->cursor->prev = N;
        L->length++;
        L->index++;  // Update the index since we're adding an element before the cursor
    }
}

void insertAfter(List L, int x) {
    if (L->length == 0 || L->index < 0) {
        fprintf(stderr, "List Error: calling insertAfter() on undefined cursor\n");
        exit(EXIT_FAILURE);
    }
    if (L->cursor == L->back) {
        append(L, x);
    } else {
        Node N = malloc(sizeof(NodeObj));
        N->data = x;
        N->next = L->cursor->next;
        N->prev = L->cursor;


        L->cursor->next->prev = N;
        L->cursor->next = N;
        L->length++;
    }
}

void deleteFront(List L) {
    if (L->length == 0) {
        fprintf(stderr, "List Error: calling deleteFront() on empty list\n");
        exit(EXIT_FAILURE);
    }

    Node N = L->front;
    if (L->length > 1) {
        L->front = L->front->next;
        L->front->prev = NULL;
    } else {
        L->front = L->back = NULL;
    }

    free(N);
    L->length--;

    if (L->cursor != NULL) {
        if (L->cursor == N) {
            L->cursor = NULL;
            L->index = -1;  // Cursor becomes undefined if it was pointing to the front
        } else {


            L->index--;  // Decrement the cursor's index
        }
    }
}

void deleteBack(List L) {
    if (L->length == 0) {
        fprintf(stderr, "List Error: calling deleteBack() on empty list\n");
        exit(EXIT_FAILURE);
    }

    Node N = L->back;
    if (L->length > 1) {
        L->back = L->back->prev;
        L->back->next = NULL;
    } else {
        L->front = L->back = NULL;
    }

    free(N);
    L->length--;

    if (L->cursor != NULL && L->cursor == N) {
        L->cursor = NULL;
        L->index = -1;  // Cursor becomes undefined if it was pointing to the back
    }
}

void delete(List L) {
    if (L->length == 0 || L->index < 0) {
        fprintf(stderr, "List Error: calling delete() on undefined cursor\n");
        exit(EXIT_FAILURE);
    }
    Node N = L->cursor;
    if (N == L->front) {
        deleteFront(L);
    } else if (N == L->back) {
        deleteBack(L);
    } else {
        N->prev->next = N->next;
        N->next->prev = N->prev;
        free(N);
        L->length--;
    }
    L->cursor = NULL;
    L->index = -1;
}

// Other Operations -----------------------------------------------------------
void printList(FILE* out, List L) {
    Node N = L->front;
    while (N != NULL) {
        fprintf(out, "%d ", N->data);
        N = N->next;
    }
}

List copyList(List L) {
    List C = newList();
    Node N = L->front;
    while (N != NULL) {
        append(C, N->data);
        N = N->next;
    }
    return C;
}

List concatList(List A, List B) {
    List C = newList();
    Node N = A->front;
    while (N != NULL) {
        append(C, N->data);
        N = N->next;
    }
    N = B->front;
    while (N != NULL) {
        append(C, N->data);
        N = N->next;
    }
    return C;
}
