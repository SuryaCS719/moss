#include <stdio.h>

// structs --------------------------------------------------------------------

// private Node type
typedef struct NodeObj* Node;

// private NodeObj type

typedef struct NodeObj {
    Node previous;
    Node next;
    ListElement data;
} NodeObj;

// private ListObj type

typedef struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
    int cursorIndex;
} ListObj;


// Constructors-Destructors ---------------------------------------------------

// newNode()
// Returns reference to new Node object. Initializes next, previous, and data fields.

Node newNode(ListElement data) {
    Node N = malloc(sizeof(NodeObj));
    assert(N != NULL);
    N->data = data;
    N->next = NULL;
    N->previous = NULL;
    return (N);
}

// freeNode()
// Frees heap memory pointed to by *pN, sets *pN to NULL.

void freeNode(Node* pN) {
    if (pN != NULL && *pN != NULL) {
        free(*pN);
        *pN = NULL;
    }
}

// newList()
// Returns reference to new empty List object.

List newList() {
    List L;
    L = malloc(sizeof (ListObj));
    assert(L != NULL);
    L->front = L->back = L->cursor = NULL;
    L->cursorIndex = -1;


    L->length = 0;
    return (L);
}

// freeList()
// Frees all heap memory associated with List *pL, and sets *pL to NULL.

void freeList(List* pL) {
    if (pL != NULL && *pL != NULL) {
        /*
        while (!isEmpty(*pL)) {
            deleteBack(*pL);
        }*/
        free(*pL);
        *pL = NULL;
    }
}


// Access functions -----------------------------------------------------------

// length()
// Returns the length of A.

int length(List A) {
    if (A == NULL) {
        printf("List Error: calling length() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    return (A->length);
}

// index()
// Returns index of cursor element if defined, -1 otherwise.

int index(List A) {
    if (A->cursor == NULL) {
        return -1;
    }
    return (A->cursorIndex);
}

// front()
// Returns front element of A.
// Pre: !isEmpty(A)

int front(List A) {
    if (A == NULL) {
        printf("List Error: calling front() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (isEmpty(A)) {
        printf("List Error: calling front() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    return (A->front->data);
}

// back()
// Returns back element of A. 
// Pre: length()>0

int back(List A) {
    if (A == NULL) {
        printf("List Error: calling back() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (isEmpty(A)) {
        printf("List Error: calling back() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    return (A->back->data);
}

// get()
// Returns cursor element of L. 
// Pre: length()>0, index()>=0

int get(List A) {
    if (isEmpty(A)) {
        printf("List Error: calling get() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    if (index(A) < 0) {
        printf("List Error: calling get() on an undefined cursor.\n");
        exit(EXIT_FAILURE);
    }
    return (A->cursor->data);
}


// isEmpty()
// Returns true if A is empty, otherwise returns false.
bool isEmpty(List A) {
   if( A == NULL ){
      printf("List Error: calling isEmpty() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   return(A->length == 0);
}
 

// Manipulation procedures ----------------------------------------------------

// clear()
// Resets L to its original empty state.
void clear(List L) {
    assert(L != NULL);
    L->front = L->back = L->cursor = NULL;
    L->cursorIndex = -1;
    L->length = 0;
}                        

// set()
// Overwrites the cursor element's data with x.
// Pre: length()>0, index()>=0
void set(List L, int x) {
    if (isEmpty(L)) {
        printf("List Error: calling set() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    if (index(L) < 0) {
        printf("List Error: calling set() on an undefined cursor.\n");
        exit(EXIT_FAILURE);
    }
    
    L->cursor->data = x;
}

// moveFront()
// If L is non-empty, sets cursor under the front element,
// otherwise does nothing.
void moveFront(List L) {
    if (!isEmpty(L)) {
        L->cursor = L->front;
        L->cursorIndex = 0;
    }
} 

// moveBack()
// If L is non-empty, sets cursor under the back element,
// otherwise does nothing.
void moveBack(List L) {
    if (!isEmpty(L)) {
        L->cursor = L->back;
        L->cursorIndex = L->length-1;
    }
}

// movePrev()
// If cursor is defined and not at front, move cursor one
// step toward the front of L; if cursor is defined and at
// front, cursor becomes undefined; if cursor is undefined
// do nothing
void movePrev(List L) {
    if (L->cursor == L->front) {
        L->cursor = NULL;
    }
    if (L->cursor != NULL) {
        L->cursor = L->cursor->previous;
        L->cursorIndex--;
    }
}

// moveNext();
// If cursor is defined and not at back, move cursor one
// step toward the back of L; if cursor is defined and at
// back, cursor becomes undefined; if cursor is undefined
// do nothing
void moveNext(List L) {
    if (L->cursor == L->back) {
        L->cursor = NULL;
    }
    if (L->cursor != NULL) {
        L->cursor = L->cursor->next;
        L->cursorIndex++;
    }
}

// prepend()
// Insert new element into L. If L is non-empty,
// insertion takes place before front element.
void prepend(List L, int x) {
    
    Node N = newNode(x);
    if (L->length == 0) {
        L->front = L->back = L->cursor = N;
        L->cursorIndex = 0;
        L->length = 1;
    }
    else {
        N->previous = NULL;
        N->next = L->front;
        L->front->previous = N;
        L->front = N;
        L->cursorIndex++;
        L->length++;
    }
}

// append()
// Insert new element into L. If L is non-empty,
// insertion takes place after back element.
void append(List L, int x) {
    
    Node N = newNode(x);
    if (L->length == 0) {
        L->front = L->back = L->cursor = N;
        L->cursorIndex = 0;
        L->length = 1;
    }
    else {
        N->next = NULL;
        N->previous = L->back;
        L->back->next = N;
        L->back = N;
        // cursorIndex does not increment as appending does not affect prior 
        // ordering.
        L->length++;
    }
}

// insertBefore()
// Insert new element before cursor.
// Pre: length()>0, index()>=0
void insertBefore(List L, int x) {
    if (isEmpty(L)) {
        printf("List Error: calling insertBefore() on an empty List\n");
        //exit(EXIT_FAILURE);
        return;
    }
    if (index(L) < 0) {
        printf("List Error: calling insertBefore() on an undefined cursor.\n");
        //exit(EXIT_FAILURE);
        
    }
    
    
                                  // previous List:     A<->B<->C<->D
                                  // cursor pos:            ^
    Node N = newNode(x);                        //          v
    N->next = L->cursor;                        //  A   N ->B<->C<->D
    N->previous = L->cursor->previous;          //  A<- N ->B<->C<->D
    
    if (index(L)>0) {
        L->cursor->previous->next = N;              //  A<->N ->B<->C<->D
    }
    if (index(L) == 0) {
        L->front = N;
    }
    L->cursor->previous = N;                    //  A<->N<->B<->C<->D
    
    
    L->cursorIndex++;
    L->length++;
}

// insertAfter()
// Insert new element after cursor.
// Pre: length()>0, index()>=0
void insertAfter(List L, int x) {
    if (isEmpty(L)) {
        printf("List Error: calling insertAfter() on an empty List\n");
        //exit(EXIT_FAILURE);
        return;
    }
    if (index(L) < 0) {
        printf("List Error: calling insertAfter() on an undefined cursor.\n");
        //exit(EXIT_FAILURE);
        return;
    }
    
                        // previous List:   A<->B<->C<->D
                        // cursor pos:          ^
    Node N = newNode(x);                //      v
    N->previous = L->cursor;            //  A<->B<- N   C<->D
    N->next = L->cursor->next;          //  A<->B<- N ->C<->D
    
    if (index(L) == length(L)-1) {
        L->back = N;
    }
    else {
        L->cursor->next->previous = N;      //  A<->B<- N<->C<->D
    }
    L->cursor->next = N;                //  A<->B<->N<->C<->D
    
    L->length++;
}

// deleteFront()
// Delete the front element. 
// Pre: length()>0
void deleteFront(List L) {
    if (isEmpty(L)) {
        printf("List Error: calling delete() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    
    if (index(L) == 0) {
        L->cursor = NULL;
        L->cursorIndex = -1;
    }
    if (length(L)>1) {
        L->front = L->front->next;
        L->front->previous = NULL;
        L->length--;
        L->cursorIndex--;
    }
    else {
        L->back = NULL;
        L->front = NULL;
        L->length = 0;
        L->cursorIndex = -1;
    }
} 

// deleteBack()
// Delete the back element. Pre: length()>0
void deleteBack(List L) {
    if (isEmpty(L)) {
        printf("List Error: calling delete() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    if (index(L) == length(L)-1) {
        L->cursor = NULL;
        L->cursorIndex = -1;
    }
    if (length(L)>1) {
        L->back = L->back->previous;
        L->back->next = NULL;
        L->length--;
    }
    else {
        L->back = NULL;
        L->front = NULL;
        L->length = 0;
        L->cursorIndex = -1;
    }
}

// delete()
// Delete cursor element, making cursor undefined.
 // Pre: length()>0, index()>=0
void delete(List L) {
    if (isEmpty(L)) {
        printf("List Error: calling delete() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    if (index(L) < 0) {
        printf("List Error: calling delete() on an undefined cursor.\n");
        exit(EXIT_FAILURE);
    }
    if (length(L) > 1) {
        if (L->cursor == L->front) {
            L->front = L->cursor->next;
        }
        else {
            L->cursor->previous->next = L->cursor->next;
        }
        if (L->cursor == L->back) {
            L->back = L->cursor->previous;
        }
        else {
            L->cursor->next->previous = L->cursor->previous;
        }
    }

    L->cursor->previous = NULL;
    L->cursor->next = NULL;
    L->cursor = NULL;
    L->cursorIndex = -1;
    L->length--;
}

// equals()
// Returns true if A is same int sequence as B, false otherwise.
bool equals(List A, List B) {
    if (A == NULL || B == NULL) {
        printf("List Error: calling equals() on NULL List reference\n");
        //exit(EXIT_FAILURE);
        return 0;
    }

    bool eq;
    Node N, M;

    eq = (A->length == B->length);
    N = A->front;
    M = B->front;
    while (eq && N != NULL) {
        eq = (N->data == M->data);
        N = N->next;
        M = M->next;
    }
    return eq;
}

List copyList(List A) {
    List B = newList();
    B->back = A->back;
    B->cursor = A->cursor;
    B->cursorIndex = A->cursorIndex;
    B->front = A->front;
    B->length = A->length;
    return (B);
}

// Custom Functions -----------------------------------------------------------

// moveCursor()
// positions the Node cursor to a Node matching the argument index
// pre: length()>0, index()>=0, cursorPosition <= length()
void moveCursor(List L, int cursorPosition) {
    if (isEmpty(L)) {
        printf("List Error: calling moveCursor() on an empty List\n");
        //exit(EXIT_FAILURE);
        return;
    }
    if (index(L) < 0) {
        printf("List Error: calling moveCursor() on an undefined cursor.\n");
        //exit(EXIT_FAILURE);
        return;
    }
    if (cursorPosition > length(L)) {
        printf("List Error: calling moveCursor() with a position longer than List.\n");
        //exit(EXIT_FAILURE);
        return;
    }
    
    if (index(L) < cursorPosition) {
        //move next
        while (index(L) != cursorPosition) {
            moveNext(L);
        }
    }
    if (index(L) > cursorPosition) {
        // move prev
        while (index(L) != cursorPosition) {
            movePrev(L);
        }
    }
    
}

int getFromIndex(List L, int cursorPosition) {
    int savedCursor = index(L);
    int output;
    moveCursor(L, cursorPosition);
    output = get(L);
    moveCursor(L, savedCursor);
    return output;
}
