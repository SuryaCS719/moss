#include <stdio.h>
#include "List.h"


//structues---------------------------------------------------------------------------


typedef struct NodeObj {
    int data;
    Node next;
    Node previous;
} NodeObj;

typedef struct ListObj{
    int length;
    int index;
    Node front;
    Node back;
    Node cursor;
} ListObj;


//constructor/decontructor functions--------------------------------------------------

List newList(void){
    List newList= malloc(sizeof(ListObj));
    if(newList){
        newList->front = NULL;
        newList->back = NULL;
        newList->cursor = NULL;

        newList->index = -1;
        newList->length = 0;
    }
    return newList;
}

void freeList(List* pL){
    if(pL && *pL){
        clear(*pL);
        free(*pL);
        *pL = NULL;
    }


    return;
}

Node createNode(int data) {
    Node NodeComponent = malloc(sizeof(NodeObj));
    if(NodeComponent){
        NodeComponent->data = data;
        NodeComponent->previous = NULL;
        NodeComponent->next = NULL;
    }

    return NodeComponent;
}

void deleteNode(Node* target){
    if((*target)){
        free(*target);
        (*target) = NULL;
    }

    return;
}

// Access functions ----------------------------------------------------------------

int length(List L){
    return L->length;
}

int index(List L){
    if(L->index < 0){
        return -1;
    }
    return L->index;
}

int front(List L){

    if(length(L) > 0){
        return L->front->data;
    }
    //return -2 to indicate an empty list
    return -2;
}

int back(List L){
    if(length(L) > 0){
        return L->back->data;
    }
    //return -2 to indicate an empty list
    return -2;
}

int get(List L){
    if(length(L) > 0 && index(L) >= 0){
        return L->cursor->data;
    }
    //return -2 to indicate an empty list
    return -2;
}

bool equals(List A, List B){
    Node A_node = A->front;
    Node B_node = B->front;

    if(length(A) != length(B)){
        return false;
    }
    //loops through the node chains and returns false if any difference is detected between A and B
    for(int i = 0; i < length(A); i++){
        if(A_node->data != B_node->data){
            return false;
        }

        A_node = A_node->next;
        B_node = B_node->next;
    }

    return true;
}

//Manipulation functions------------------------------------------------------------

void clear(List L){
    if(length(L) > 0){
        //delete's the front node until nothing is left
        while(length(L) > 0){
            deleteFront(L);
        }
        L->length = 0;
        L->index = -1;
        //printf("index modified under clear!\n");
    }

    return;
}

void set(List L, int x){
    //length and index check
    if(length(L) > 0 && index(L) >= 0){
        //modify data of node that's under cursor
        L->cursor->data = x;
    }
    return;
}

void moveFront(List L){
    //length check
    if(length(L) > 0){
        //set cursor to front and index to zero
        L->cursor = L->front;
        L->index = 0;
        //printf("index modified under movefront, is now %d!\n", L->index);
    }
    //printf("index return %d!\n", L->index);
    return;
}

void moveBack(List L){
    //length check
    if(length(L)>0){
        //set cursor to back and index to -1
        L->cursor = L->back;
        L->index = length(L) - 1;
        //printf("index modified under moveBack!\n");
    }
    return;
}

void movePrev(List L){
    //checks if cursor is at the front and if it isn't, then is moves to the previous node
    if(L->cursor && L->cursor != L->front){
        L->cursor = L->cursor->previous;
        L->index -= 1;
        //printf("index modified under moveprev1!\n");
    //otherwise, cursor becomes undefined
    }else if(L->cursor && L->cursor == L->front){
        L->cursor = NULL;
        L->index = -1;
        //printf("index modified under MovePrev2!\n");
    }else{
        return;
    }
}

void moveNext(List L){
    //checks if cursor is at the back and if it isn't, then is moves to the next node
    if(L->cursor && L->cursor != L->back){
        L->cursor = L->cursor->next;
        L->index += 1;
        //printf("index modified under move next1!\n");
    //otherwise, cursor becomes undefined
    }else if(L->cursor && L->cursor == L->back){
        L->cursor = NULL;
        L->index = -1;
        //printf("index modified under moenext2!\n");
    }else{
        return;
    }
}

void prepend(List L, int x){
    Node preNode = createNode(x);
    //check if empty
    if(length(L) > 0){
        //links the current front and new node together before making "preNode" the new front
        preNode->next = L->front;
        L->front->previous = preNode;
        L->front = preNode;
        L->length += 1;
        if(L->cursor){
            L->index += 1;
            //printf("index modified under prepend1!\n");
        }
    }else{
        //otherwise makes preNode the back and front
        L->front = preNode;
        L->back = preNode;
        L->length += 1;
        L->index += 1;
        //printf("index modified under prepend2!\n");
    }
}

void append(List L, int x){
    Node postNode = createNode(x);
    if(length(L) > 0){
        //links the current back and new node together before making "postNode" the new back
        postNode->previous = L->back;
        L->back->next = postNode;
        L->back = postNode;
        L->length += 1;
        
    }else{
        //otherwise makes postNode the back and front
        L->front = postNode;
        L->back = postNode;
        L->length += 1;
        //printf("index modified under append2!\n");
    }
}

void insertBefore(List L, int x){
    //prereq check
    if(length(L) <= 0 || index(L) < 0){
        return;
    }
    
    if(L->cursor != L->front){
        Node beforeNode = createNode(x);
        beforeNode->next = L->cursor;
        beforeNode->previous = L->cursor->previous;
        L->cursor->previous->next= beforeNode;
        L->cursor->previous = beforeNode;
        L->index += 1;
        L->length += 1;
        //printf("index modified under insertB!\n");
    }else{
    //if at front, prepend
        prepend(L, x);
    }
    return;
}

void insertAfter(List L, int x){
    //prereq check
    if(length(L) <= 0 || index(L) < 0){
        //checks if cursor is at the back
        return;
    }
    if(L->cursor != L->back){
        Node afterNode = createNode(x);
        afterNode->previous = L->cursor;
        afterNode->next = L->cursor->next;
        L->cursor->next->previous = afterNode;
        L->cursor->next = afterNode;
        L->length += 1;
        //printf("index modified under insertA!\n");

    }else{
    //if at back, append
        append(L, x);
    }

    return;
}

void deleteFront(List L){
    if(length(L) > 0){
        //get the front node
        Node delete = L->front;
        //check if cursor exists
        if(L->cursor){
            L->index -= 1;
            //printf("index modified under delete1!\n");

        }
        //if front is the only node in the list
        if(L->cursor == L->front){
            L->cursor = NULL;
            L->index -= 1;
            //printf("index modified under delete2!\n");
        }
        //if non-empty create a new front, otherwise clear the front and back
        if(length(L) > 1){
            L->front = L->front->next;
            L->front->previous = NULL;
        }else{
            L->front = NULL;
            L->back = NULL;
        }

        L->length -= 1;
        //printf("index modified under delete3!\n");
        //delete the old front node
        deleteNode(&delete);
    }
    return;
}

void deleteBack(List L){
    if(length(L) > 0){
        //get the front node
        Node delete = L->back;

        //if back is the only node in the list
        if(L->cursor == L->back){
            L->cursor = NULL;
            L->index = -1;
        }
        //if non-empty create a new front, otherwise clear the front and back
        if(length(L) > 1){
            L->back = L->back->previous;
            L->back->next = NULL;
        }else{
            L->front = NULL;
            L->back = NULL;
        }

        L->length -= 1;
        //delete the old front node
        deleteNode(&delete);
    }
    return;
}

void delete(List L){
    if(length(L) > 0 && index(L) >= 0){
        Node curse = L->cursor;
        if(L->cursor == L->front){
            deleteFront(L);
        }else if(L->cursor == L->back){
            deleteBack(L);
        }else{
            //links the nodes before and after cursor before deleting cursor
            curse->next->previous = curse->previous;
            curse->previous->next = curse->next;
            deleteNode(&curse);
            L->length -= 1;
            L->index = -1;
            L->cursor = NULL;
        }
    }

    return;
}

void printList(FILE* output, List L){
    if(L == NULL){
        fprintf(stderr, "Warning, can't copy Null list");
        exit(EXIT_FAILURE);
    }

    for(Node N = L->front; N != NULL; N = N->next){
        fprintf(output, "%d", N->data);
    }
}

List copyList(List L){
    if(L == NULL){
        fprintf(stderr, "Warning, can't copy Null list");
        exit(EXIT_FAILURE);
    }
    List listCopy = newList();

    for(Node N = L->front; N != NULL; N = N->next){
        append(listCopy, N->data);
    }

    return listCopy;
}
