#include<stdio.h>
#include<stdlib.h>
#include<assert.h>
#include<stdbool.h>
#include "List.h"

// Struct Definitions


typedef struct NodeObj* Node; 

typedef struct NodeObj {
    ListElement data;
    int index;
    Node next;
    Node prev;
} NodeObj;

typedef struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
} ListObj;

// Create a new node
Node newNode(int data) {
    Node N = malloc(sizeof(NodeObj));
    assert(N != NULL);
    N->data = data;
    N->next = NULL;
    return(N);
}

// Create a new list
List newList(void) {
    List L = malloc(sizeof(ListObj));
    assert(L != NULL);
    L->front = L->back = L->cursor = NULL;
    L->length = 0;
    return(L);
}

// Free the memory for the list
void freeList(List* pL) {
    if (pL != NULL && *pL != NULL) {
        clear(*pL);
        free(*pL);
        *pL = NULL; // Mark the pointer as null after freeing
    }
}

// Access Functions
int length(List L) {
    if (L == NULL) {
        printf("List Error: calling length() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    return(L->length);
} 

int index(List L) {
    if (L->cursor == NULL) {
        return -1;
    }
    return(L->cursor->index);
}

int front(List L) {
    if (L == NULL) {
        printf("Error: calling front on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->length == 0) {
        printf("List Error: calling front on an empty List\n");
        exit(EXIT_FAILURE);
    }
    return(L->front->data);
}

int back(List L) {
    if (L == NULL) {
        printf("Error: calling back on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->length == 0) {
        printf("List Error: calling back on an empty List\n");
        exit(EXIT_FAILURE);
    }
    return(L->back->data);
}



int get(List L) {
    if (L == NULL) {
        printf("Error: calling get on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->length == 0) {
        printf("List Error: calling get on an empty List\n");
        exit(EXIT_FAILURE);
    }
    return(L->cursor->data);
}

// Compare two lists for equality
bool equals(List A, List B) {
    Node nodeA = A->front;
    Node nodeB = B->front;
    while (nodeA != NULL && nodeB != NULL) {
        if (nodeA->data != nodeB->data) {
            return false;
        }
        nodeA = nodeA->next;
        nodeB = nodeB->next;
    }
    return (nodeA == NULL && nodeB == NULL);
}

// Manipulation Procedures
void clear(List L) {
    L->cursor = NULL;
    while (L->front != NULL) {
        Node temp = L->front;
        L->front = L->front->next;
        free(temp);
    }
    L->length = 0;
}

void set(List L, int x) {
    if (L == NULL) {
        printf("Error: calling set on NULL List reference\n");
        return;
    }
    if (L->length == 0) {
        printf("Error: calling set on an empty List\n");
        return;
    }
    L->cursor->data = x;
}

void moveFront(List L) {
    if (L == NULL) {
        printf("Error: calling moveFront on NULL List reference\n");
        return;
    }
    if (L->length == 0) {
        printf("Error: calling moveFront on an empty List\n");
        return;
    }
    L->cursor = L->front;
    L->cursor->index = 0;
}

void moveBack(List L) {
    if (L == NULL) {
        printf("Error: calling moveBack on NULL List reference\n");
        return;
    }
    if (L->length == 0) {
        printf("Error: calling moveBack on an empty List\n");
        return;
    }
    L->cursor = L->back;
    L->cursor->index = L->length - 1;
}

void movePrev(List L) {
    if (L->cursor != NULL && L->cursor != L->front) {
        L->cursor = L->cursor->prev;
    } else if (L->cursor == L->front) {
        L->cursor = NULL;
    }
}

void moveNext(List L) {
    if (L->cursor != NULL && L->cursor != L->back) {
        L->cursor = L->cursor->next;
        L->cursor->index = L->cursor->prev->index + 1;
    } else if (L->cursor == L->back) {
        L->cursor = NULL;
    }
}

// Insert an element at the front
void prepend(List L, int x) {
    Node N = newNode(x);
    if (L->length > 0) {
        Node temp = L->front;
        L->front->prev = N;
        L->front = N;
        N->next = temp;
        Node current = L->front;
        while (current != NULL) {
            current->index++;
            current = current->next;
        }
    } else {
        N->index = 0;
        L->front = L->back = N;
    }
    L->length++;
}

// Insert an element at the back
void append(List L, int x) {
    Node N = newNode(x);
    if (L->length > 0) {
        Node temp = L->back;
        L->back->next = N;
        L->back = N;
        N->prev = temp;
        N->index = L->length;
    } else {
        N->index = 0;
        L->front = L->back = N;
    }
    L->length++;
}

// Insert a new element before the cursor
void insertBefore(List L, int x) {
    if (L->length > 0 && L->cursor->index >= 0) {
        if (L->cursor == L->front) {
            prepend(L, x);
        } else {
            Node N = newNode(x);
            Node temp = L->cursor;
            Node prevNode = L->cursor->prev;
            L->cursor->prev = N;
            prevNode->next = N;
            N->prev = prevNode;
            N->next = temp;
            N->index = L->cursor->index;
            L->cursor->index++;
            L->length++;
        }
    }
}

// Insert a new element after the cursor
void insertAfter(List L, int x) {
    if (L->length > 0 && L->cursor->index >= 0) {
        if (L->cursor == L->back) {
            append(L, x);
        } else {
            Node N = newNode(x);
            Node temp = L->cursor;
            Node nextNode = L->cursor->next;
            L->cursor->next = N;
            nextNode->prev = N;
            N->prev = temp;
            N->next = nextNode;
            N->index = L->cursor->index + 1;
            L->length++;
        }
    }
}

// Delete the front element
void deleteFront(List L) {
    if (L->length > 0) {
        Node temp = L->front;
        if (L->length == 1) {
            L->front = L->back = L->cursor = NULL;
        } else {
            if (L->front == L->cursor) {
                L->cursor = NULL;
            }
            L->front = L->front->next;
            L->front->prev = NULL;
        }
        free(temp);
        if (L->cursor != NULL) {
            L->cursor->index--;
        }
        L->length--;
    }
}

// Delete the back element
void deleteBack(List L) {
    if (L->length > 0) {
        Node temp = L->back;
        if (L->length == 1) {
            L->front = L->back = L->cursor = NULL;
        } else {
            if (L->back == L->cursor) {
                L->cursor = NULL;
            }
            L->back = L->back->prev;
            L->back->next = NULL;
        }
        free(temp);
        L->length--;
    }
}

// Delete the cursor element
void delete(List L) {
    if (L->length > 0 && L->cursor->index >= 0) {
        Node temp = L->cursor;
        Node before = temp->prev;
        Node after = temp->next;
        if (L->cursor == L->front) {
            deleteFront(L);
        } else if (L->cursor == L->back) {
            deleteBack(L);
        } else {
            before->next = after;
            after->prev = before;
            free(temp);
            L->cursor = NULL;
            L->length--;
        }
    }
}

// Print the list to a file
void printList(FILE* out, List L) {
    if (out == NULL) {
        printf("Error: output file is not defined\n");
        return;
    }
    if (L == NULL) {
        printf("Error: printList called with a NULL list reference\n");
        return;
    }
    Node current = L->front;
    while (current != NULL) {
        fprintf(out, "%d", current->data);
        current = current->next;
        if (current != NULL) {
            fprintf(out, " ");
        }
    }
    fprintf(out, " ");
}

// Create a copy of the list
List copyList(List L) {
    if (L == NULL) {
        fprintf(stderr, "Error: copyList called with NULL list input\n");
        return NULL;
    }
    List copy = newList();
    Node current = L->front;
    while (current != NULL) {
        append(copy, current->data);
        current = current->next;
    }
    return copy;
}
