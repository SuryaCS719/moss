#include<stdio.h>
#include<stdlib.h>
#include<assert.h>
#include<stdbool.h>
#include "List.h"

// NodeObj definition


typedef struct NodeObj* Node;
typedef struct NodeObj {
    int data;
    Node next;
    Node prev;
} NodeObj;

// ListObj definition
typedef struct ListObj* List;
typedef struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
    int index;
} ListObj;

// Node constructor
Node newNode(int data) {
    Node N = malloc(sizeof(NodeObj));
    assert(N != NULL);
    N->data = data;
    N->next = NULL;
    N->prev = NULL;
    return N;
}

// Free node
void freeNode(Node *node) {
    if (node != NULL && *node != NULL) {


        free(*node);
        *node = NULL;
    }
}

// List constructor
List newList(void) {
    List L = malloc(sizeof(ListObj));
    assert(L != NULL);
    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->length = 0;
    L->index = -1;
    return L;
}

// Free list
void freeList(List* pL) {
    if (pL != NULL && *pL != NULL) {
        while ((*pL)->length > 0) {
            deleteFront(*pL);
        }
        free(*pL);


        *pL = NULL;
    }
}

// Access functions
int length(List L) {
    return L->length;
}

int index(List L) {
    return L->index;
}

int front(List L) {
    if (L != NULL && L->length > 0) {
        return L->front->data;
    }
    exit(EXIT_FAILURE);  // Invalid state
}

int back(List L) {
    if (L != NULL && L->length > 0) {
        return L->back->data;
    }
    exit(EXIT_FAILURE);  // Invalid state
}

int get(List L) {
    if (L != NULL && L->length > 0 && L->index >= 0 && L->cursor != NULL) {
        return L->cursor->data;
    }
    exit(EXIT_FAILURE);  // Invalid state
}

bool equals(List A, List B) {
    if (A->length != B->length) {
        return false;
    }
    Node nodeA = A->front;
    Node nodeB = B->front;
    while (nodeA != NULL && nodeB != NULL) {


        if (nodeA->data != nodeB->data) {
            return false;
        }
        nodeA = nodeA->next;
        nodeB = nodeB->next;
    }
    return true;
}

// Clear list
void clear(List L) {
    if (L == NULL) return;
    while (L->length > 0) {
        deleteFront(L);
    }
    L->cursor = NULL;
    L->index = -1;
}

// Manipulation procedures
void moveFront(List L) {
    if (L != NULL && L->length > 0) {
        L->cursor = L->front;
        L->index = 0;
    }
}

void moveBack(List L) {


    if (L != NULL && L->length > 0) {
        L->cursor = L->back;
        L->index = L->length - 1;
    }
}

void movePrev(List L) {
    if (L != NULL && L->cursor != NULL) {
        if (L->cursor != L->front) {
            L->cursor = L->cursor->prev;
            L->index--;
        } else {
            L->cursor = NULL;
            L->index = -1;
        }
    }
}

void moveNext(List L) {
    if (L != NULL && L->cursor != NULL) {
        if (L->cursor != L->back) {
            L->cursor = L->cursor->next;
            L->index++;
        } else {
            L->cursor = NULL;
            L->index = -1;
        }
    }
}

void prepend(List L, int x) {
    Node newFront = newNode(x);
    if (L->length == 0) {
        L->front = L->back = newFront;
    } else {
        newFront->next = L->front;
        L->front->prev = newFront;
        L->front = newFront;
    }
    L->length++;
    if (L->index != -1) L->index++;
}

void append(List L, int x) {
    Node newBack = newNode(x);
    if (L->length == 0) {
        L->front = L->back = newBack;
    } else {
        newBack->prev = L->back;


        L->back->next = newBack;
        L->back = newBack;
    }
    L->length++;
}

void insertBefore(List L, int x) {
    if (L != NULL && L->length > 0 && L->index >= 0) {
        if (L->cursor == L->front) {
            prepend(L, x);
        } else {
            Node newNode = malloc(sizeof(NodeObj));
            newNode->data = x;
            newNode->prev = L->cursor->prev;
            newNode->next = L->cursor;
            L->cursor->prev->next = newNode;


            L->cursor->prev = newNode;
            L->length++;
            L->index++;
        }
    }
}

void insertAfter(List L, int x) {
    if (L != NULL && L->length > 0 && L->index >= 0) {
        if (L->cursor == L->back) {
            append(L, x);
        } else {
            Node newNode = malloc(sizeof(NodeObj));
            newNode->data = x;
            newNode->prev = L->cursor;
            newNode->next = L->cursor->next;
            L->cursor->next->prev = newNode;
            L->cursor->next = newNode;
            L->length++;
        }
    }
}

void deleteFront(List L) {
    if (L != NULL && L->length > 0) {
        Node oldFront = L->front;
        if (L->cursor == L->front) {
            L->cursor = NULL;
            L->index = -1;
        } else if (L->index > 0) {
            L->index--;
        }
        L->front = oldFront->next;
        if (L->front != NULL) L->front->prev = NULL;
        freeNode(&oldFront);
        L->length--;
    }
}

void deleteBack(List L) {
    if (L != NULL && L->length > 0) {
        Node oldBack = L->back;
        if (L->cursor == L->back) {
            L->cursor = NULL;
            L->index = -1;
        }
        L->back = oldBack->prev;
        if (L->back != NULL) L->back->next = NULL;
        freeNode(&oldBack);
        L->length--;
    }
}

void delete(List L) {
    if (L != NULL && L->length > 0 && L->index >= 0) {
        Node toDelete = L->cursor;
        if (toDelete == L->front) {
            deleteFront(L);
        } else if (toDelete == L->back) {
            deleteBack(L);
        } else {
            toDelete->prev->next = toDelete->next;
            toDelete->next->prev = toDelete->prev;
            freeNode(&toDelete);
            L->cursor = NULL;
            L->index = -1;
            L->length--;
        }
    }
}

// Output function
void printList(FILE* out, List L) {
    Node currentNode = L->front;
    while (currentNode != NULL) {
        fprintf(out, "%d ", currentNode->data);
        currentNode = currentNode->next;
    }
}

// Copy List
List copyList(List L) {
    List copy = newList();
    Node currentNode = L->front;
    while (currentNode != NULL) {
        append(copy, currentNode->data);
        currentNode = currentNode->next;
    }
    return copy;
}
