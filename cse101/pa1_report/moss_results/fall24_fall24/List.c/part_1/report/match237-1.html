#include "List.h"

#include <assert.h>
#include <stdbool.h>
#include <stdio.h>
#include <stdlib.h>

// structs --------------------------------------------------------------------

// private Node type
typedef struct NodeObj* Node;

// private NodeObj type
typedef struct NodeObj {
  int data;
  Node previous;
  Node next;
} NodeObj;

// private ListObj type
typedef struct ListObj {
  Node front;
  Node back;
  Node cursor;
  int length;
  int index;
} ListObj;

// Constructors-Destructors ---------------------------------------------------

/**
 * @brief creates a new node
 * @param data
 * @return node
 */
Node newNode(int data) {
  Node N = malloc(sizeof(NodeObj));
  assert(N != NULL);
  N->data = data;
  N->next = N->previous = NULL;
  return (N);
}

/**
 * @brief frees the heap memory of a node
 * @param pN pointer to node
 */
void freeNode(Node* pN) {
  if (pN != NULL && *pN != NULL) {
    free(*pN);
    *pN = NULL;
  }
}

/**
 * @brief creates a new empty list
 * @return reference to list
 */
List newList() {


  List L = malloc(sizeof(ListObj));
  assert(L != NULL);
  L->front = L->back = NULL;
  L->cursor = NULL;
  L->index = -1;
  L->length = 0;
  return (L);
}

/**
 * @brief frees the heap memory of a list
 * @param pL pointer to list
 */
void freeList(List* pL) {
  if (pL != NULL && *pL != NULL) {
    while (length(*pL) > 0) {
      deleteBack(*pL);
    }
    free(*pL);
    *pL = NULL;
  }
}

// Access functions -----------------------------------------------------------

/**
 * @brief gets the length of list L
 * @param L List
 * @return length of L
 */
int length(List L) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling length() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }
  return (L->length);
}

/**
 * @brief gets the index of the cursor
 * @param L List
 * @return index of cursor element if defined, -1 otherwise
 */
int index(List L) {
  if (L->cursor != NULL) {
    return L->index;
  } else {
    return -1;
  }
}

/**
 * @brief gets the data in the element at the front of the list
 * @param L List
 * @return value at front of L
 * @pre length(L) > 0
 */
int front(List L) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling front() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }
  if (length(L) == 0) {
    fprintf(stderr, "List Error: calling front() on an empty List\n");
    exit(EXIT_FAILURE);
  }
  return (L->front->data);
}

/**
 * @brief gets the data in the element at the back of the list
 * @param L List
 * @return value at the back of L
 * @pre length(L) > 0
 */
int back(List L) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling back() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }
  if (length(L) == 0) {
    fprintf(stderr, "List Error: calling back() on an empty List\n");
    exit(EXIT_FAILURE);
  }
  return (L->back->data);
}

/**
 * @brief gets the data at the current cursor location
 * @param L List
 * @return data at cursor
 * @pre length()>0, index()>=0
 */


int get(List L) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling get() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }
  if (length(L) == 0) {
    fprintf(stderr, "List Error: calling get() on an empty List\n");
    exit(EXIT_FAILURE);
  }
  if (index(L) == -1) {


    fprintf(stderr, "List Error: calling get() on an undefined cursor\n");
    exit(EXIT_FAILURE);
  }
  return (L->cursor->data);
}

/**
 * @brief compares two lists for equality
 * @param A List
 * @param B List
 * @retval true iff Lists A and B contain the same sequence of elements
 * @retval false otherwise
 */
bool equals(List A, List B) {
  if (A == NULL || B == NULL) {
    fprintf(stderr, "List Error: calling equals() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }

  Node prevA = A->cursor;
  Node prevB = B->cursor;
  int prevIndexA = A->index;
  int prevIndexB = B->index;

  if (A->length == 0 && B->length == 0) return true;
  if (A->length != B->length) return false;

  moveFront(A);
  moveFront(B);

  while (A->index >= 0 && B->index >= 0) {
    if (get(A) != get(B)) {
      A->cursor = prevA;
      B->cursor = prevB;
      return false;
    }
    moveNext(A);
    moveNext(B);
  }
  A->cursor = prevA;
  B->cursor = prevB;
  A->index = prevIndexA;
  B->index = prevIndexB;
  return true;
}

// Manipulation procedures ----------------------------------------------------

/**
 * @brief resets list to empty state
 * @param L list
 */
void clear(List L) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling clear() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }
  while (L->length != 0) {
    deleteBack(L);
  }
}

/**
 * @brief overwrites the cursor element's data with x
 * @param L list
 * @param x data
 * @pre length()>0, index()>=0
 */
void set(List L, int x) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling set() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }
  if (length(L) == 0) {
    fprintf(stderr, "List Error: calling set() on empty List\n");
    exit(EXIT_FAILURE);
  }



  if (index(L) == -1) {
    fprintf(stderr, "List Error: calling set() with undefined cursor\n");
    exit(EXIT_FAILURE);
  }
  L->cursor->data = x;
}

/**
 * @brief If L is non-empty, sets cursor under the front element, otherwise does
 * nothing
 * @param L list
 */
void moveFront(List L) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling moveFront() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }
  if (length(L) == 0) {
    fprintf(stderr, "List Error: calling moveFront() on empty List\n");
    exit(EXIT_FAILURE);
  }
  L->cursor = L->front;
  L->index = 0;
}

/**
 * @brief If L is non-empty, sets cursor under the back element, otherwise does
 * nothing
 * @param L list
 */
void moveBack(List L) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling moveBack() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }
  if (length(L) == 0) {
    fprintf(stderr, "List Error: calling moveBack() on empty List\n");
    exit(EXIT_FAILURE);
  }
  L->cursor = L->back;
  L->index = length(L) - 1;
}

/**
 * @brief If cursor is defined and not at front, move cursor one step toward the
 * front of L; if cursor is defined and at front, cursor becomes undefined; if
 * cursor is undefined do nothing
 * @param L list
 */
void movePrev(List L) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling movePrev() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }
  if (length(L) == 0) {
    fprintf(stderr, "List Error: calling movePrev() on empty List\n");
    exit(EXIT_FAILURE);
  }
  if (L->cursor == L->front && index(L) != -1) {
    L->cursor = NULL;
    L->index = -1;
  }
  if (L->cursor != L->front && index(L) != -1) {
    L->cursor = L->cursor->previous;
    L->index--;
  }
}

/**
 * @brief If cursor is defined and not at back, move cursor one step toward the
 * back of L; if cursor is defined and at back, cursor becomes undefined; if
 * cursor is undefined do nothing
 * @param L list
 */
void moveNext(List L) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling moveNext() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }
  if (length(L) == 0) {
    fprintf(stderr, "List Error: calling moveNext() on empty List\n");
    exit(EXIT_FAILURE);
  }
  if (L->cursor == L->back && index(L) != -1) {
    L->cursor = NULL;
    L->index = -1;
  }
  if (L->cursor != L->back && index(L) != -1) {
    L->cursor = L->cursor->next;


    L->index++;
  }
}

/**
 * @brief Insert new element into L. If L is non-empty, insertion takes place
 * before front element
 * @param L list
 * @param x data
 */
void prepend(List L, int x) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling prepend() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }
  if (length(L) == 0) {
    Node N = newNode(x);
    L->front = N;
    L->back = N;
  }
  if (length(L) > 0) {
    Node N = newNode(x);
    Node prevFront = L->front;
    L->front->previous = N;
    L->front = N;
    L->front->next = prevFront;
  }

  L->length++;


  L->index++;
}

/**
 * @brief Insert new element into L. If L is non-empty, insertion takes place
 * after back element.
 * @param L list
 * @param x data
 */
void append(List L, int x) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling append() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }
  if (length(L) == 0) {
    Node N = newNode(x);
    L->front = N;
    L->back = N;
  }
  if (length(L) > 0) {
    Node N = newNode(x);
    N->next = NULL;
    N->previous = L->back;
    L->back->next = N;
    L->back = N;
  }

  L->length++;
}

/**
 * @brief Insert new element before cursor
 * @param L list
 * @param x data
 * @pre length()>0, index()>=0
 *  */
void insertBefore(List L, int x) {
  if (L == NULL) {
    fprintf(stderr,
            "List Error: calling insertBefore() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }
  if (length(L) == 0) {
    fprintf(stderr, "List Error: calling insertBefore() on empty List\n");
    exit(EXIT_FAILURE);
  }


  if (index(L) == -1) {
    fprintf(stderr,
            "List Error: calling insertBefore() with undefined cursor\n");
    exit(EXIT_FAILURE);
  }

  if (L->cursor == L->front) {
    prepend(L, x);
  } else {
    Node N = newNode(x);
    Node tempPrev = L->cursor->previous;

    L->cursor->previous = N;

    tempPrev->next = N;

    N->previous = tempPrev;
    N->next = L->cursor;

    L->length++;
    L->index++;
  }
}

/**
 * @brief Insert new element after cursor
 * @param L list
 * @param x data
 * @pre length()>0, index()>=0
 */
void insertAfter(List L, int x) {
  if (L == NULL) {
    fprintf(stderr,
            "List Error: calling insertAfter() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }
  if (length(L) == 0) {
    fprintf(stderr, "List Error: calling insertAfter() on empty List\n");
    exit(EXIT_FAILURE);
  }


  if (index(L) == -1) {
    fprintf(stderr,
            "List Error: calling insertAfter() with undefined cursor\n");
    exit(EXIT_FAILURE);
  }

  if (L->cursor == L->back) {
    append(L, x);
  } else {
    Node N = newNode(x);
    Node tempNext = L->cursor->next;

    L->cursor->next = N;

    tempNext->previous = N;

    N->previous = L->cursor;
    N->next = tempNext;
    L->length++;
  }
}

/**
 * @brief Delete the front element
 * @param L list
 * @pre length()>0
 */
void deleteFront(List L) {
  if (L == NULL) {
    fprintf(stderr,
            "List Error: calling deleteFront() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }
  if (length(L) == 0) {
    fprintf(stderr, "List Error: calling deleteFront() on empty List\n");
    exit(EXIT_FAILURE);
  }

  Node N = L->front;
  if (length(L) > 1) {
    L->front = L->front->next;
  } else {
    L->front = L->back = NULL;
  }
  L->length--;
  L->index--;
  freeNode(&N);
}

/**
 * @brief Delete the back element
 * @param L list
 * @pre length()>0
 */
void deleteBack(List L) {
  if (L == NULL) {
    fprintf(stderr,
            "List Error: calling deleteBack() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }
  if (length(L) == 0) {
    fprintf(stderr, "List Error: calling deleteBack() on empty List\n");
    exit(EXIT_FAILURE);
  }

  Node N = L->back;

  if (L->cursor == L->back) {
    L->index = -1;
  }
  if (length(L) > 1) {
    L->back = L->back->previous;
  } else {
    L->back = L->front = NULL;
  }
  L->length--;

  freeNode(&N);
}

/**
 * @brief Delete the cursor element, making the cursor undefined
 * @param L list
 * @pre length()>0, index()>=0
 */
void delete(List L) {
  if (L == NULL) {
    fprintf(stderr, "List Error: calling delete() on NULL List reference\n");
    exit(EXIT_FAILURE);
  }
  if (length(L) == 0) {
    fprintf(stderr, "List Error: calling delete() on empty List\n");
    exit(EXIT_FAILURE);
  }
  if (index(L) == -1) {
    fprintf(stderr, "List Error: calling delete() on undefined cursor\n");
    exit(EXIT_FAILURE);
  }

  if (L->cursor != L->front && L->cursor != L->back) {
    L->cursor->next->previous = L->cursor->previous;
    L->cursor->previous->next = L->cursor->next;
    freeNode(&(L->cursor));
    L->cursor = NULL;
    L->length--;
  } else {
    if (L->cursor == L->front) deleteFront(L);
    if (L->cursor == L->back) deleteBack(L);
  }
}

// Other Functions ------------------------------------------------------------

/**
 * @brief Prints to the file pointed to by out, a string representation of L
 * consisting of a space separated sequence of integers, with front on left
 * @param out file or stream
 * @param L list
 */
void printList(FILE* out, List L) {
  int prevIndex = L->index;
  Node prevCursor = L->cursor;
  if (L->length > 0) {
    moveFront(L);

    for (moveFront(L); index(L) >= 0; moveNext(L)) {
      fprintf(out, "%d ", get(L));
    }

    L->index = prevIndex;
    L->cursor = prevCursor;
  }
}

/**
 * @brief Returns a new List representing the same integer sequence as L. The
 * cursor in the new list is undefined, regardless of the state of the cursor in
 * L. The state of L is unchanged.
 * @param L list to copy
 * @return the new copied list
 */
List copyList(List L) {
  List temp = newList();

  Node prevCursor = L->cursor;
  int prevIndex = L->index;

  if (L->length == 0) {
    return temp;
  }
  for (moveFront(L); index(L) >= 0; moveNext(L)) {
    append(temp, get(L));
  }
  L->cursor = prevCursor;
  L->index = prevIndex;
  return temp;
}
