#include<stdio.h>

// structs --------------------------------------------------------------------

// private Node type


typedef struct NodeObj* Node;

// private NodeObj type
typedef struct NodeObj{
   ListElement data;
   Node next;
} NodeObj;

// private ListObj type
typedef struct ListObj{
   Node front;
   Node back;
   Node cursor;
   int pointer;
   int length;
} ListObj;


// Constructors-Destructors ---------------------------------------------------

// newNode()
// Returns reference to new Node object. Initializes next and data fields.
Node newNode(ListElement data){
   Node N = malloc(sizeof(NodeObj));
   assert( N!=NULL );
   N->data = data;
   N->next = NULL;
   return(N);
}

// freeNode()
// Frees heap memory pointed to by *pN, sets *pN to NULL.
void freeNode(Node* pN){
   if( pN!=NULL && *pN!=NULL ){
      free(*pN);
      *pN = NULL;
   }
}

// newList()
// Returns reference to new empty List object.
List newList(){
   List L;
   L = malloc(sizeof(ListObj));
   assert( L!=NULL );
   L-> front = L->back = L->cursor = NULL;
   L-> length = 0;
   L-> pointer = -1;
   return(L);
}

// freeList()
// Frees all heap memory associated with List *pL, and sets *pL to NULL.
void freeList(List* pL){
   if(pL!=NULL && *pL!=NULL) { 
      while( !isEmpty(*pL) ) { 
         deleteFront(*pL); 
      }
      free(*pL);
      *pL = NULL;
   }
}


// Access functions -----------------------------------------------------------

// front()/back()
// Returns the value at the front/back of L.
// Pre: !isEmpty(L)


ListElement front(List L){
   if( L==NULL ){
      printf("List Error: calling front() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if( isEmpty(L) ){
      printf("List Error: calling front() on an empty List\n");
      exit(EXIT_FAILURE);
   }
   return(L->front->data);
}
ListElement back(List L){
   if( L==NULL ){
      printf("List Error: calling back() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if( isEmpty(L) ){
      printf("List Error: calling back() on an empty List\n");
      exit(EXIT_FAILURE);
   }
   return(L->back->data);
}

// length()
// Returns the length of L.
int length(List L){
   if( L==NULL ){


      printf("List Error: calling length() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   return(L->length);
}

// isEmpty()
// Returns true if L is empty, otherwise returns false.
bool isEmpty(List L){
   if( L==NULL ){
      printf("List Error: calling isEmpty() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   return(L->length==0);
}


// Other Functions ------------------------------------------------------------

// equals()
// Returns true if A is same int seLuence as B, false otherwise.
bool equals(List A, List B){
   if( A==NULL || B==NULL ){
      printf("List Error: calling equals() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }

   bool eq;
   Node N, M;

   eq = ( A->length == B->length );
   N = A->front;
   M = B->front;
   while( eq && N!=NULL){


      eq = ( N->data==M->data );
      N = N->next;
      M = M->next;
   }
   return eq;
}



///NEW FUNCTIONS

//index()
//Returns index value, otherwise -1 if undefined
int index(List L){
   if (isEmpty(L) == true) {return -1;}
   return L->pointer;
}

//get()
//Returns cursor element of L. Pre: length()>0, index()>=0
int get(List L){
   if (isEmpty(L) == true) {
      printf("List Error: calling get() on an empty List reference\n");
      exit(EXIT_FAILURE);
   }
   if (index(L) == -1) {
      printf("Index Error: calling get() while cursor is not on List\n");
      exit(EXIT_FAILURE);
   }
   return L->cursor->data;

}

//clear()
// Resets L to its original empty state.
void clear(List L){
   while(isEmpty(L) == false) { 
         deleteFront(L); 
   }
      L->front = NULL;
      L->back = NULL;

   L->cursor = NULL;
   L->pointer = -1;
}

//set()
// Overwrites the cursor elementâ€™s data with x.
// Pre: length()>0, index()>=0
void set(List L, int x){
   if (isEmpty(L) == true) {
      printf("List Error: calling set() on an empty List reference\n");
      exit(EXIT_FAILURE);
   }
   if (index(L) == -1) {
      printf("Index Error: calling set() while cursor is not pointing to anything\n");
      exit(EXIT_FAILURE);
   }
   
   L->cursor->data = x;

}

//moveFront/moveBack()
// If L is non-empty, sets cursor under the front/back element,
// otherwise does nothing.
void moveFront(List L){
   if(index(L) != 0) {
      L->cursor = L->front;
      L->pointer = 0;
      } 
}; 
void moveBack(List L){
   if(index(L) != length(L)-1) {
      L->cursor = L->back;
      L->pointer = length(L)-1;
   } 
}; 

//movePrev/moveNext()
// If cursor is defined and not at front/back, move cursor one
// step toward the front/back of L; if cursor is defined and at
// front/back, cursor becomes undefined; if cursor is undefined
// do nothing
void moveNext(List L){
   if(index(L) != -1) {
      if (L->cursor->next == NULL){
         L->cursor = NULL;
         L->pointer = -1;
      } else {
         L->cursor = L->cursor->next;
         L->pointer++;   
      }
   } 
}; 
void movePrev(List L){
   if(index(L) != -1) {
      if (L->pointer == 0){
         L->cursor = NULL;
         L->pointer = -1;
      } else {
         const int prevValue = L->pointer;
         moveFront(L);
         while (L->pointer != (prevValue-1)){
            moveNext(L);
         }
      }
   } 
}; 

//prepend()
// Insert new element into L. If L is non-empty,
// insertion takes place before front element.
void prepend(List L, int x){
   Node N = newNode(x);

   if( L==NULL ){
      printf("List Error: calling prepend() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   
   if( isEmpty(L) ) { 
      L->front = L->back = N; 
   }else{ 
      N->next = L->front;
      L->front = N; 
   }
   L->length++;
};

//append()
// Insert new element into L. If L is non-empty,
// insertion takes place after back element.
void append(List L, int x){
   Node N = newNode(x);

   if( L==NULL ){
      printf("List Error: calling append() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   
   if( isEmpty(L) ) { 
      L->front = L->back = N; 
   }else{ 


      L->back->next = N; 
      L->back = N; 
   }
   L->length++;
}; 

//insertBefore()/insertAfter()
// Insert new element before/after cursor.
// Pre: length()>0, index()>=0
void insertAfter(List L, int x){
   if (isEmpty(L) == true) {
      printf("List Error: calling inesrtAfter() on an empty List reference\n");
      exit(EXIT_FAILURE);
   }
   if (index(L) == -1) {
      printf("Index Error: calling insertAfter() while cursor is not on List\n");
      exit(EXIT_FAILURE);
   }
   Node N = newNode(x);

   N->next = L->cursor->next;
   L->cursor->next = N;
   

   L->length++;
}; 

void insertBefore(List L, int x){
   if (isEmpty(L) == true) {
      printf("List Error: calling inesrtBefore() on an empty List reference\n");
      exit(EXIT_FAILURE);
   }
   if (index(L) == -1) {
      printf("Index Error: calling insertBefore() while cursor is not on List\n");
      exit(EXIT_FAILURE);
   }

   if (index(L) == 0) {
      prepend(L, x);
   } else {
      movePrev(L);
      insertAfter(L, x);

      moveNext(L);
      moveNext(L);
      L->pointer++;
   }
   
}; 

//delete()
// Delete cursor element, making cursor undefined.
// Pre: length()>0, index()>=0
void delete(List L){
   if (isEmpty(L) == true) {
      printf("List Error: calling delete() on an empty List reference\n");
      exit(EXIT_FAILURE);
   }
   if (index(L) == -1) {
      printf("Index Error: calling delete() while cursor is not on List\n");
      exit(EXIT_FAILURE);
   }
   movePrev(L);
   Node delNode = NULL;
   delNode = L->cursor->next;
   L->cursor->next = L->cursor->next->next;
   
   L->cursor = NULL;
   L->pointer = -1;
   freeNode(&delNode);

   L->length--;
}; 


//deleteFront()/deleteBack()
// Delete the front/back element. Pre: length()>0
void deleteFront(List L){
   Node N = NULL;

   if( L==NULL ){
      printf("List Error: calling deleteFront() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if( isEmpty(L) ){
      printf("List Error: calling deleteFront() on an empty List\n");
      exit(EXIT_FAILURE);
   }
   
   N = L->front;
   if( length(L)>1 ) { 


      L->front = L->front->next; 
   }else{ 
      L->front = L->back = NULL; 
   }
   L->length--;
   freeNode(&N);
}; 
void deleteBack(List L){
   if (isEmpty(L) == true) {
      printf("List Error: calling deleteBack() on an empty List reference\n");
      exit(EXIT_FAILURE);
   }
   Node Temp = L->cursor;
   Node dN = NULL;
   moveBack(L);
   movePrev(L);
   
   dN = L->back;
   L->back = L->cursor;

   freeNode(&dN);
   
   L->length--;
   if (index(L) == length(L)) {
      L->cursor = NULL;
      L->pointer=-1;
   } else {
      L->cursor = Temp;
   }
   
};
