List newList(void){
    List L = malloc(sizeof(ListObj));
    if (L == NULL){
        fprintf(stderr, "Error: Unable to allocate memory\n");
        exit(EXIT_FAILURE);
    }
    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->length = 0;
    L->index = -1;
    return L;
} // Creates and returns a new empty List.
void freeList(List *pL){
    if (pL == NULL){
        fprintf(stderr, "Error: Calling function on NULL reference\n");
        exit(EXIT_FAILURE);
    }
    if ((*pL) == NULL) return;

    List toDelete = (*pL);
    Node *curr = toDelete->front;
    while (curr != NULL){
        Node *temp = curr->next;
        free(curr);
        curr = temp;
    }
    free((*pL));
    *pL = NULL;
    return;
}
 // Frees all heap memory associated with *pL, and sets
 // *pL to NULL.

bool equals(List A, List B){
    if (A == NULL || B == NULL){
        fprintf(stderr, "Error: Calling function on NULL reference\n");
        exit(EXIT_FAILURE);
    }
    if (length(A) != length(B)){
        return false;
    }
    Node *curr1 = A->front;
    Node *curr2 = B->front;
    while (curr1 != NULL && curr2 != NULL){
        if (curr1->val != curr2->val){
            return false;
        }
        curr1 = curr1->next;
        curr2 = curr2->next;
    }
    return curr1 == NULL && curr2 == NULL;

} // Returns true iff Lists A and B contain the same
 // sequence of elements, returns false otherwise.

        exit(EXIT_FAILURE);
    }
    L->length = 0;
    L->index = -1;
    L->cursor = NULL;
    Node* curr = L->front;
    Node* temp;
    while (curr != NULL) {
        temp = curr->next;
        free(curr);
        curr = temp;
    }
    L->front = NULL;
    L->back = NULL;
    return;

} // Resets L to its original empty state.

void movePrev(List L){
    if (L == NULL){
        fprintf(stderr, "Error: Calling function on NULL reference\n");
        exit(EXIT_FAILURE);
    }
    if (index(L) < 0){
        return;
    }
    if (L->cursor == L->front){
        L->cursor = NULL;
        L->index = -1;
        return;
    }
    L->cursor = L->cursor->prev;
    L->index--;
    return;
    
}; // If cursor is defined and not at front, move cursor one
 // step toward the front of L; if cursor is defined and at
 // front, cursor becomes undefined; if cursor is undefined

void prepend(List L, int x){
    if (L == NULL){
        fprintf(stderr, "Error: Calling function on NULL reference\n");
        exit(EXIT_FAILURE);
    }
    Node* curr = malloc(sizeof(Node));
    if (curr == NULL){
        fprintf(stderr, "Error: Unable to allocate memory\n");
        exit(EXIT_FAILURE);
    }
    curr->val = x;
    if (length(L) == 0) {
        curr->next = NULL;
        L->front = curr;
        L->back = curr;
    } else {
        curr->next = L->front;
        L->front->prev = curr;
        L->front = curr;
        if (L->cursor != NULL) L->index++;
    }
    curr->prev = NULL;
    L->length++;
    return;
}; // Insert new element into L. If L is non-empty,
 // insertion takes place before front element.
void append(List L, int x){
    if (L == NULL){
        fprintf(stderr, "Error: Calling function on NULL reference\n");
        exit(EXIT_FAILURE);
    }
    Node* curr = malloc(sizeof(Node));
    if (curr == NULL){
        fprintf(stderr, "Error: Unable to allocate memory\n");
        exit(EXIT_FAILURE);
    }
    curr->val = x;
    if (length(L) == 0) {
        curr->prev = NULL;
        L->front = curr;
    } else {
        curr->prev = L->back;
        L->back->next = curr;
    }
    L->back = curr;
    curr->next = NULL;
    L->length++;
    return;

}; // Insert new element into L. If L is non-empty,
 // insertion takes place after back element.
void insertBefore(List L, int x){
    if (L == NULL){
        fprintf(stderr, "Error: Calling function on NULL reference\n");
        exit(EXIT_FAILURE);
    }
    if (length(L) <= 0){
        fprintf(stderr, "Error: Calling function on empty list\n");
        exit(EXIT_FAILURE);
    }
    if (index(L) < 0){
        fprintf(stderr, "Error: Calling function on undefined cursor\n");
        exit(EXIT_FAILURE);
    }
    else if (index(L) == 0) {
        prepend(L, x);
        return;
    }
    Node * curr = malloc(sizeof(Node));
    if (curr == NULL){
        fprintf(stderr, "Error: Unable to allocate memory\n");
        exit(EXIT_FAILURE);
    }
    curr->val = x;

    Node* temp = L->cursor->prev;
    if (temp != NULL) temp->next = curr;
    curr->prev = temp;
    curr->next = L->cursor;
    L->cursor->prev = curr;
    L->index++;
    L->length++;
    return;    
}; // Insert new element before cursor.
 // Pre: length()>0, index()>=0
void insertAfter(List L, int x){
    if (L == NULL){
        fprintf(stderr, "Error: Calling function on NULL reference\n");

    }
    if (length(L) <= 0){

    if (index(L) < 0){
        fprintf(stderr, "Error: Calling function on undefined cursor\n");
        exit(EXIT_FAILURE);
    }
    if (index(L) == length(L) - 1) {
        append(L, x);

    }
    Node * curr = malloc(sizeof(Node));
    if (curr == NULL){
        fprintf(stderr, "Error: Unable to allocate memory\n");
        exit(EXIT_FAILURE);

    Node* temp = L->cursor->next;
    if (temp != NULL) temp->prev = curr;
    curr->next = temp;
    curr->prev = L->cursor;
    L->cursor->next = curr;
    L->length++;
    return;

}; // Insert new element after cursor.
 // Pre: length()>0, index()>=0
void deleteFront(List L){
    if (L == NULL){
        fprintf(stderr, "Error: Calling function on NULL reference\n");
        exit(EXIT_FAILURE);
    }
    if (length(L) <= 0){
        fprintf(stderr, "Error: Calling function on empty list\n");
        exit(EXIT_FAILURE);
    }
    Node * temp = L->front;
    if (length(L) == 1){
        L->front = NULL;
        L->back = NULL;
    }
    else{
        L->front = L->front->next;
        L->front->prev = NULL;
    }
    L->length--;
    L->index--;
    free(temp);
    return;

}; // Delete the front element. Pre: length()>0
void deleteBack(List L){
    if (L == NULL){
        fprintf(stderr, "Error: Calling function on NULL reference\n");
        exit(EXIT_FAILURE);
    }
    if (length(L) <= 0){
        fprintf(stderr, "Error: Calling function on empty list\n");
        exit(EXIT_FAILURE);
    }
    Node * temp = L->back;
    if (temp == L->cursor){
        L->index = -1;
    }
    if (length(L) == 1){
        L->front = NULL;
        L->back = NULL;
    }
    else{
        L->back = L->back->prev;
        L->back->next = NULL;
    }
    free(temp);
    L->length--;
    return;
}; // Delete the back element. Pre: length()>0
void delete(List L){
    if (L == NULL){
        fprintf(stderr, "Error: Calling function on NULL reference\n");
        exit(EXIT_FAILURE);
    }
    if (length(L) <= 0){
        fprintf(stderr, "Error: Calling function on empty list\n");
        exit(EXIT_FAILURE);
    }
    if (index(L) < 0){
        fprintf(stderr, "Error: Calling function on undefined cursor\n");
        exit(EXIT_FAILURE);
    }
    if (index(L) == 0){
        deleteFront(L);
        L->index = -1;
        L->cursor = NULL;
        return;
    }
    if (index(L) == length(L) - 1){
        deleteBack(L);
        L->index = -1;
        L->cursor = NULL;
        return;
    }
    Node * temp = L->cursor;
    temp->prev->next = temp->next;
    temp->next->prev = temp->prev;
    free(temp);
    L->cursor = NULL;
    L->index = -1;
    L->length--;
    return;
}; // Delete cursor element, making cursor undefined.
 // Pre: length()>0, index()>=0

 void printList(FILE* out, List L){
    if (L == NULL){
        fprintf(stderr, "Error: Calling function on NULL reference\n");
        exit(EXIT_FAILURE);
    }
    Node * curr = L->front;
    while (curr != NULL){
        fprintf(out, "%d ", curr->val);
        curr = curr->next;
    }
    return;
 }; // Prints to the file pointed to by out, a
 // string representation of L consisting
// of a space separated sequence of integers,
// with front on left.
List copyList(List L){
    if (L == NULL){
        fprintf(stderr, "Error: Calling function on NULL reference\n");
        exit(EXIT_FAILURE);
    }
    List new_list = newList();
    Node * curr = L->front;
    if (curr == NULL){
        return new_list;
    }
    while (curr != NULL){
        append(new_list, curr->val);
        curr = curr->next;
    }
    return new_list;
} // Returns a new List representing the same integer
 // sequence as L. The cursor in the new list is undefined,
// regardless of the state of the cursor in L. The state
// of L is unchanged.
