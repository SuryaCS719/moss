/*
 * private struct and functions for underlying data structure of double linked list --------
 */



typedef struct NodeObj* Node;

typedef struct NodeObj{
    int value;
    Node previous;
    Node next;
} NodeObj;

Node newNode(int value, Node previous) {
    Node N = malloc(sizeof(NodeObj));
    assert(N!=NULL);
    N->previous = previous;
    N->value = value;
    N->next = NULL;
    return N;
}

void freeNode(Node* pN) {


    if(pN!=NULL && *pN!=NULL) {
        free(*pN);
        *pN = NULL;
    }
}

typedef struct ListObj {
    Node front;
    Node cursor;
    Node back;
    int index;
    int length;

} ListObj;

//Public Functions to use for List ADT -----------------------------------------------------

/*
 * Constructor/Destructor
 */

List newList(void) {
    List L;
    L = malloc(sizeof(ListObj));
    assert(L!=NULL);
    L->front = NULL;
    L->back = NULL;
    L->length = 0;
    L->cursor = NULL;
    L->index = -1;
    return L;
}
void freeList(List* pL) {
    if(pL!=NULL && *pL!=NULL) {
        while(((*pL)->length) > 0) {
            deleteBack(*pL);
        }
        free(*pL);
        *pL = NULL;
    }
}

    return L->index;

    return (L->front)->value;

    return (L->back)->value;

int get(List L) {
    return (L->cursor)->value;
}

bool equals(List A, List B) {
    if(A->length != B->length) {
        return false;
    }

    Node currentA = (A->front);
    Node currentB = (B->front);
    for(int i = 0; i < (A->length); i++) {
        if (currentA->value != currentB->value)
            return false;
        
        currentA = currentA->next;
        currentB = currentB->next;
    }

    return true;
}

/*
 * Manipulation functions ------------------------------------------------------------------
 */

void clear(List L) {
    while((L->length) > 0) {
        deleteBack(L);
    }
    L->index = -1;
    L->length = 0;
}

void set(List L, int x) {
    (L->cursor)->value = x;
}

void moveFront(List L) {
    (L->cursor) = (L->front);
    L->index = 0;
}

void moveBack(List L) {
    (L->cursor) = (L->back);
    L->index = (L->length)-1;
}

void movePrev(List L) {
    (L->cursor) = (L->cursor)->previous;
    (L->index)--;
}

void moveNext(List L) {
    (L->cursor) = (L->cursor)->next;
    (L->index)++;
    if(L->index >= L->length)
        L->index = -1;
}

        (L->front) = N;

void append(List L, int x) {
    Node N = newNode(x, NULL);
    
    if (L->back == NULL) {
        //printf("empty list adding val %d\n", x);
        L->front = N;
        L->back = N;
        //printf("empty list added val\n");
    } else {
        //printf("non-empty list adding val %d\n", x);
        (L->back)->next = N;
        N->previous = L->back;
        (L->back) = N;
        //printf("non-empty list added val\n");
    }
    (L->length)++;
}

void insertAfter(List L, int x) {
    Node N = newNode(x, (L->cursor));

    if (L->cursor == L->back) {
        N->previous = L->cursor;
        (L->cursor)->next = N;
        L->back = N;
    } else {
        N->next = (L->cursor)->next;
        N->previous = L->cursor;


        ((L->cursor)->next)->previous = N;
        (L->cursor)->next = N;
    }

    (L->length)++;
}

void deleteFront(List L) {
    // Node temp = ((L->front)->next);
    // temp->previous = NULL;
    // freeNode(&(L->front));
    // (L->front) = temp;
    // (L->length)--;
    // (L->index)--;

    Node temp = L->front;
    if (L->front == L->back) {
        L->front = NULL;
        L->back = NULL;
    } else {
        L->front = (L->front)->next;
        (L->front)->previous = NULL;
    }
    freeNode(&temp);
    (L->length)--;
    (L->index)--;
}

void deleteBack(List L) {
    Node temp = L->back;
    if (L->back->previous != NULL) {
        L->back = L->back->previous;
        L->back->next = NULL;
    } else {
        L->front = NULL;  // List is now empty
        L->back = NULL;
    }

    freeNode(&temp);
    (L->length)--;
    if(L->index >= L->length)
        L->index = -1;
}

void delete(List L) {
    if (L->cursor == L->front) {
        deleteFront(L);
    } else if (L->cursor == L->back) {
        deleteBack(L);
    } else {
        Node temp = ((L->cursor)->previous);
        (temp->next) = (L->cursor)->next;
        ((L->cursor)->next)->previous = temp;
        freeNode(&(L->cursor));
        L->cursor = NULL;
        (L->length)--;
        (L->index) = -1;
    }
}

    }
        
    for(N = L->front; N != L->back; N = N->next) {
        fprintf(out, "%d ", N->value);

}

List copyList(List L) {
    List copyL = newList();

    Node N = NULL;
    for(N = L->front; N != NULL; N = N->next) {
        append(copyL, N->value);
    }

    return copyL;
}

List concatList(List A, List B) {
    List L = newList();

    Node N = NULL;
    for(N = A->front; N != NULL; N = N->next) {
        append(L, N->value);
    }

    N = NULL;
    for(N = B->front; N != NULL; N = N->next) {
        append(L, N->value);
    }

    return L;
}
