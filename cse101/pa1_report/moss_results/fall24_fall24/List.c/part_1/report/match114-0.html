#include<stdio.h>
#include<stdlib.h>
#include "List.h"
#undef index

// Structs

typedef struct NodeObj {
	int data;
	struct NodeObj* next;
	struct NodeObj* prev;
}NodeObj;

typedef NodeObj* Node;

typedef struct ListObj {
	Node front;
	Node back;
	Node cursor;
	int length;
	int index;
}ListObj;


// Constructors / Destructos

List newList(void) { //creates and returns new empty list
	List L = malloc(sizeof(ListObj));
	L->front = L->back = L->cursor = NULL;
	L->length = 0;
	L->index = -1;
	return L;
}

void freeList(List* pL){
	if(pL != NULL && *pL != NULL) {
		while(length(*pL) > 0){
			deleteFront(*pL);
		}
		free(*pL);
		*pL = NULL;

	}
}

// Access Functions



int length(List L){
	return L->length;
}

int index(List L){
	return L->index;
}

int front(List L){
	if(L->length <= 0) {
		fprintf(stderr, "Error: front() called on an empty list\n");
		exit(EXIT_FAILURE);
	}
	return L->front->data;
}

int back(List L){
	if(L->length <= 0) {
		fprintf(stderr, "Error: back() called on an empty list\n");
		exit(EXIT_FAILURE);
	}
	return L->back->data;
}

int get(List L){
	if(L->length <= 0){
		fprintf(stderr, "Error: get() called on empty list");
		exit(EXIT_FAILURE);
	}
	if(L->index < 0){
		fprintf(stderr, "Error: get() called on undefined cursor");
		exit(EXIT_FAILURE);
	}

	return L->cursor->data;
}
bool equals(List A, List B){
	if(A->length != B->length){
		return false;
	}
	Node nodeA = A->front;


	Node nodeB = B->front;
	//loops through each data point to check if equal. If not then returns false
	while(nodeA != NULL && nodeB != NULL){
		if(nodeA->data != nodeB->data){
			return false;
		}else{
			nodeA = nodeA->next;
			nodeB = nodeB->next;
		}
	}
	return true;
}

// Manipulation Procedures

void clear(List L){
	while(length(L) > 0){ //loops through and deletes all elements
		deleteFront(L);
	}
	L->cursor = NULL;
	L->index = -1;
}

void set(List L, int x){
	if(L->length <= 0){
		fprintf(stderr, "Error: length <= 0");
		exit(EXIT_FAILURE);
	}
	if(L->index < 0){
		fprintf(stderr, "Error: index <0");
		exit(EXIT_FAILURE);
	}

	L->cursor->data = x;
}

void moveFront(List L){
	if(L->length > 0){
		L->cursor = L->front;
		L->index = 0;
	}
}

void moveBack(List L){
	if(L->length > 0){
		L->cursor = L->back;
		L->index = L->length -1;
	}
}

void movePrev(List L){
	if(L->cursor != NULL){
		if(L->cursor == L->front){//if at the front make the cursor and index undefined
			L->cursor = NULL;
			L->index = -1;
		}
		else {
			L->cursor = L->cursor->prev;
			L->index--;
		}
	}
}

void moveNext(List L) {
	if(L->cursor != NULL){
		if(L->cursor == L->back){//if at the back make the cursor and index undefined
			L->cursor = NULL;
			L->index = -1;
		}
		else {
			L->cursor = L->cursor->next;
			L->index++;
		}
	}
}

void prepend(List L, int x){
	Node n = malloc(sizeof(NodeObj));
	n->data = x;
	n->prev = NULL;
	n->next = L->front;
	if(L->front != NULL){ //update the previous front node


		L->front->prev = n;
	}

	L->front = n;

	if(L->back == NULL){//if empty update the back node
		L->back = n;
	}

	L->length++;
	L->index++; //move the cursor so it stays in original position
}

void append(List L, int x){
	Node N = malloc(sizeof(NodeObj));
	N->data = x;
	N->next = NULL;
	N->prev = L->back;
	if(L->back != NULL){
		L->back->next = N;
	}

	L->back = N;



	if(L->front == NULL){
		L->front = N;
	}

	L->length++;
}


void insertBefore(List L, int x){
	if(L->length <= 0){
		fprintf(stderr, "Error length <= 0");
		exit(EXIT_FAILURE);
	}
	if(L->index < 0){
		fprintf(stderr, "Error index < 0");
		exit(EXIT_FAILURE);
	}

	Node N = malloc(sizeof(NodeObj));
	N->data = x;
	N->prev = L->cursor->prev;
	N->next = L->cursor;

	if(L->cursor->prev != NULL){ //if cursor is not on the front node
		L->cursor->prev->next = N;
	}
	else{
		L->front = N;
	}


	L->cursor->prev = N;
	L->length++;
	L->index++;
}

void insertAfter(List L, int x){
	if(L->length <= 0){
		fprintf(stderr, "Error length <= 0");
		exit(EXIT_FAILURE);
	}
	if(L->index < 0){
		fprintf(stderr, "Error index < 0");
		exit(EXIT_FAILURE);
	}

	Node N = malloc(sizeof(NodeObj));
	N->data = x;
	N->next = L->cursor->next;
	N->prev = L->cursor;

	if(L->cursor->next != NULL){ //if cursor is not on the back node
		L->cursor->next->prev = N;
	}


	else{
		L->back = N;
	}

	L->cursor->next = N;
	L->length++;
}

void deleteFront(List L){
	if(L->length <= 0){
		fprintf(stderr, "Error length <= 0");
		exit(EXIT_FAILURE);
	}

	Node N = L->front;
	if(L->front == L->back){ //if only one element, delete it
		L->front = NULL;
		L->back = NULL;
	}
	else{
		L->front = L->front->next;
		L->front->prev = NULL;
	}

	free(N);
	L->length--;
	L->index--;
}

void deleteBack(List L){
	if(L->length <= 0){
		fprintf(stderr, "Error length <= 0");
		exit(EXIT_FAILURE);
	}


	Node N = L->back;
	if(L->front == L->back){ //if only one element, delete it
		L->front = NULL;
		L->back = NULL;
		L->cursor = NULL;
		L->index = -1;
	}
	else{
		if(L->back == L->cursor){
			L->back = L->back->prev;
			L->back->next = NULL;
			L->cursor = NULL;
			L->index = -1;
		}
		else{
			L->back = L->back->prev;
			L->back->next = NULL;
		}

	}

	free(N);
	L->length--;

}

void delete(List L){
	if(L->length <= 0){
		fprintf(stderr, "Error length <= 0");
		exit(EXIT_FAILURE);
	}
	if(L->index < 0){
		fprintf(stderr, "Error index < 0");
		exit(EXIT_FAILURE);
	}

	Node N = L->cursor;
	if(N == L->front){
		deleteFront(L); //using the already written function
	}
	else if(N == L->back){
		deleteBack(L);
	}
	else{
		N->prev->next = N->next;
		N->next->prev = N->prev;
		free(N);
		L->length--;
	}
	L->cursor = NULL;
	L->index = -1;

}


// Other Functions

void printList(FILE* out, List L){
	Node N = L->front;
	while(N != NULL){ 
		fprintf(out, "%d ", N->data);
		N = N->next;
	}
}

List copyList(List L){
	List NEWlist = newList();
	Node N = L->front;
	while(N != NULL){
		append(NEWlist, N->data);
		N = N->next;
	}
	return NEWlist;
}
