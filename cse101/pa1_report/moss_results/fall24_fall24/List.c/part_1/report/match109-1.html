// Constructor for creating a new list


List newList(void) {
    List L = malloc(sizeof(ListObj));
    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->length = 0;
    L->cursorIndex = -1;  // Initialize cursorIndex to -1 
    return L;
}

// Destructor for freeing all memory 


void freeList(List* pL) {
    if (pL != NULL && *pL != NULL) {
        clear(*pL);
        free(*pL);
        *pL = NULL;
    }
}



// Resets the list to its empty state
void clear(List L) {
    while (L->length > 0) {
        deleteFront(L);
    }
    L->cursor = NULL;

    L->cursorIndex = -1;
}

// Returns the number of elements in the list
int length(List L) {
    return L->length;
}

// Returns the element at the front of the list
int front(List L) {
    if (L->length > 0) {

        return L->front->data;
    }
    return -1;  // Handle underflow case
}

// Returns the element at the back of the list
int back(List L) {
    if (L->length > 0) {
        return L->back->data;

    }
    return -1;  // Handle underflow case
}

// Returns the element at the cursor
int get(List L) {
    if (L->cursor != NULL) {
        return L->cursor->data;
    }
    return -1;  // Handle null cursor case
}

// Move the cursor to the front element
void moveFront(List L) {
    if (L->length > 0) {
        L->cursor = L->front;

        L->cursorIndex = 0;
    }
}

// Move the cursor to the back element
void moveBack(List L) {
    if (L->length > 0) {
        L->cursor = L->back;
        L->cursorIndex = L->length - 1;
    }
}

// Move the cursor to the next element
void moveNext(List L) {


    if (L->cursor != NULL && L->cursor != L->back) {
        L->cursor = L->cursor->next;
        L->cursorIndex++;
    } else {
        L->cursor = NULL;

        L->cursorIndex = -1;
    }
}

// Move the cursor to the previous element
void movePrev(List L) {
    if (L->cursor != NULL && L->cursor != L->front) {
        L->cursor = L->cursor->prev;
        L->cursorIndex--;


    } else {
        L->cursor = NULL;
        L->cursorIndex = -1;
    }
}

// Returns the cursor index
int index(List L) {
    return (L->cursor != NULL) ? L->cursorIndex : -1;
}

// Insert a new element after the cursor
void insertAfter(List L, int data) {
    if (L->cursor == NULL) return;
    
    Node N = malloc(sizeof(NodeObj));


    N->data = data;
    N->next = L->cursor->next;
    N->prev = L->cursor;
    
    if (L->cursor->next != NULL) {


        L->cursor->next->prev = N;
    } else {
        L->back = N;
    }
    
    L->cursor->next = N;
    L->length++;
}

// Insert a new element before the cursor
void insertBefore(List L, int data) {
    if (L->cursor == NULL) {
        printf("List Error: insertBefore() called on NULL cursor.\n");
        exit(EXIT_FAILURE);
    }
    Node N = malloc(sizeof(NodeObj));
    N->data = data;
    N->next = L->cursor;


    N->prev = L->cursor->prev;
    if (L->cursor->prev != NULL) {
        L->cursor->prev->next = N;
    } else {
        L->front = N;
    }


    L->cursor->prev = N;
    L->length++;


    // Increment cursor index to reflect the new element
    L->cursorIndex++;
}


// Insert a new element at the end of the list
void append(List L, int data) {
    Node N = malloc(sizeof(NodeObj));
    N->data = data;


    N->next = NULL;
    N->prev = L->back;
    
    if (L->back != NULL) {
        L->back->next = N;


    } else {
        L->front = N;
    }
    
    L->back = N;
    L->length++;
}

// Insert a new element at the beginning of the list
void prepend(List L, int data) {
    Node N = malloc(sizeof(NodeObj));
    N->data = data;
    N->next = L->front;


    N->prev = NULL;
    if (L->front != NULL) {


        L->front->prev = N;
    } else {
        L->back = N;
    }
    L->front = N;
    L->length++;

    // If the cursor is defined, increment its index to reflect the new element
    if (L->cursor != NULL) {
        L->cursorIndex++;
    }
}


// Delete the front element

void deleteFront(List L) {
    if (L->length == 0) {
        printf("List Error: deleteFront() called on an empty list.\n");
        exit(EXIT_FAILURE);


    }
    Node N = L->front;
    if (L->cursor == L->front) {
        // Reset the cursor if it's at the front
        L->cursor = NULL;
        L->cursorIndex = -1;
    }
    if (L->length > 1) {
        L->front = L->front->next;


        L->front->prev = NULL;
    } else {
        L->front = L->back = NULL;
    }
    free(N);
    L->length--;
    // If the cursor is defined, decrement its index
    if (L->cursor != NULL) {
        L->cursorIndex--;
    }
}


// Delete the back element
void deleteBack(List L) {
    if (L->length > 0) {
        Node N = L->back;
        
        if (L->length == 1) {


            L->front = L->back = NULL;
        } else {
            L->back = L->back->prev;
            L->back->next = NULL;
        }
        


        if (L->cursor == N) {
            L->cursor = NULL;
            L->cursorIndex = -1;
        }
        
        free(N);
        L->length--;
    }
}

// Delete the cursor element
void delete(List L) {
    if (L->cursor != NULL) {
        Node N = L->cursor;


        
        if (L->cursor == L->front) {
            deleteFront(L);
        } else if (L->cursor == L->back) {
            deleteBack(L);
        } else {
            L->cursor->prev->next = L->cursor->next;


            L->cursor->next->prev = L->cursor->prev;
            free(N);
            L->length--;
        }
        
        L->cursor = NULL;


        L->cursorIndex = -1;
    }
}

// Check if two lists are equal
int equals(List A, List B) {
    if (A->length != B->length) return 0;
    
    Node N = A->front;
    Node M = B->front;
    
    while (N != NULL && M != NULL) {
        if (N->data != M->data) return 0;
        N = N->next;


        M = M->next;
    }
    
    return 1;
}

// Create a copy of the list
List copyList(List L) {
    List copy = newList();
    Node N = L->front;
    
    while (N != NULL) {
        append(copy, N->data);
        N = N->next;


    }
    
    return copy;
}
