// List.c
#include <stdio.h>
#include <stdlib.h>
#include "List.h"

// Node structure


struct NodeObj {
    int data;
    Node prev;
    Node next;
};

// List structure
struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
    int index; 
};


// Creating a new list
List newList(void) {
    List L = malloc(sizeof(struct ListObj));
    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->length = 0;
    L->index = -1;
    return L;
}

// Freeing the list
void freeList(List* pL) {
    if (pL != NULL && *pL != NULL) {
        // Freeing all nodes 
        Node current = (*pL)->front;
        while (current != NULL) {
            Node temp = current;
            current = current->next;
            free(temp);
        }
        free(*pL);
        *pL = NULL;
    }
}

// Returning number of elememts in L 
int length(List L) {
    return L->length;
}

// Returns index of cursor element if defined, otherwise -1 
int getIndex(List L) {
    return L->index;
}

// Return cursor element of L
int get(List L) {
    if (L->cursor != NULL){
        return L->cursor->data;
    }
    // If cursor is undefined...
    return -1;
}

// Moving cursor to the front
void moveFront(List L) {
    if (L->length > 0){
        L->cursor = L->front;
        L->index = 0;
    }
}

// Moving cursor to the back
void moveBack(List L) {
    if (L->length > 0) {


        L->cursor = L->back;
        L->index = L->length -1;
    }
}

// Move cursor previous
void movePrev(List L) {
    if (L->cursor == NULL || L->cursor == L->front) {  // Handle out-of-bounds cases first
        L->cursor = NULL;
        L->index = -1;
    } else {
        L->cursor = L->cursor->prev;
        L->index--;
    }
}


// Move cursor next
void moveNext(List L) {
    if (L->cursor == NULL || L->cursor == L->back) {  // Handle end of list
        L->cursor = NULL;
        L->index = -1;
    } else {
        L->cursor = L->cursor->next;
        L->index++;
    }
}


// Prepend element
void prepend(List L, int data) {
    Node newNode = malloc(sizeof(struct NodeObj));
    newNode->data = data;
    newNode->prev = NULL;
    newNode->next = L->front;

    if (L->front != NULL) {
        L->front->prev = newNode;
    } else {
        L->back = newNode;
    }
    L->front = newNode; 
    L->length++;
}

// Append element 
void append(List L, int data) {
    Node newNode = malloc(sizeof(struct NodeObj));
    newNode->data = data;
    newNode->next = NULL;
    newNode->prev = L->back;
	
    if (L->back != NULL) {
        L->back->next = newNode;
    } else {


        L->front = newNode; // In case the list was empty
    }
    L->back = newNode; // Update the back pointer
    L->length++; // Increase the length of the list


}


// Insert before cursor
void insertBefore(List L, int data) {
    if (L->cursor != NULL) {
        Node newNode = malloc(sizeof(struct NodeObj));
        newNode->data = data;

        newNode->prev = L->cursor->prev;
        newNode->next = L->cursor;

        if (L->cursor->prev != NULL) {


            L->cursor->prev->next = newNode;
        } else {
            L->front = newNode;
        }
        L->cursor->prev = newNode;
        L->length++;
        L->index++;
    } else {
	    fprintf(stderr, "Cannot insert before cursor; cursor is undefined.\n");
	}
}    

// Insert After Cursor
void insertAfter(List L, int data) {
    if (L->cursor != NULL) {
        Node newNode = malloc(sizeof(struct NodeObj));
        newNode->data = data;

        newNode->next = L->cursor->next;
        newNode->prev = L->cursor;

        if (L->cursor->next != NULL) {


            L->cursor->next->prev = newNode;
        } else {
            L->back = newNode;
        }
        L->cursor->next = newNode;
        L->length++;
    } else {
	    fprintf(stderr, "Cannot insert after cursor; cursor is undefined.\n");
    }
}

// Delete Cursor Element
void deleteCursor(List L) {
    if (L->cursor != NULL) {
        Node temp = L->cursor;
        if (temp->prev != NULL) {
            temp->prev->next = temp->next;
        } else {
            L->front = temp->next; 
        }
        if (temp->next != NULL) {
            temp->next->prev = temp->prev;
        } else {
            L->back = temp->prev; 
        }
        L->cursor = NULL; 
        free(temp);
        L->length--;
        L->index = -1;
    }
}

// Concatenating two lists
List concatList(List A, List B) {
    List C = newList();
    // Add elements from A
    Node current = A->front;
    while (current != NULL) {
        append(C, current->data);
        current = current->next;
    }
    // Add elements from B
    current = B->front;
    while (current != NULL) {
        append(C, current->data);
        current = current->next;
    }
    return C;
}

// Printing the list
void printList(FILE *out, List L) {
    Node current = L->front;
    while (current != NULL) {
        fprintf(out, "%d", current->data);
        current = current->next;
    }
    fprintf(out, "\n");
}
