#include "List.h"
#include <stdlib.h>
#include <stdbool.h>



typedef struct Node{
	int data;
	struct Node* next;
	struct Node* prev;
} Node;

typedef struct ListObj{
	Node* front;
	Node* back;
	Node* cursor;
	int length;
	int index;
} ListObj;

typedef struct ListObj* List;

// Constructors-Destructors ---------------------------------------------------
// Creates and returns a new node.
Node* newNode(int data){
	Node* N = malloc(sizeof(Node));
	N->data = data;
	N->next = NULL;
	N->prev = NULL;
	return N;
}

//frees node
void freeNode(Node** pN){
        if(pN != NULL && *pN != NULL){
                free(*pN);
                *pN = NULL;
        }
}

// Creates and returns a new empty List.
List newList(void){
	List L = malloc(sizeof(ListObj));
	L->front = NULL;
	L->back = NULL;
	L->cursor = NULL;
	L->length = 0;
	L->index = -1;
	return L;
}

// Frees all heap memory associated with *pL, and sets *pL to NULL.
void freeList(List* pL){
	if(pL != NULL && *pL != NULL){
		clear(*pL);
      		free(*pL);
      		*pL = NULL;
      		pL = NULL;
	}
}

// Access functions -----------------------------------------------------------
int length(List L){
	return L-> length; // Returns the number of elements in L.
}


// Returns index of cursor element if defined, -1 otherwise.
int index(List L){
	return (L->cursor == NULL) ? -1 : L->index; //return current as null if =-1, else make L -> index
}

// Returns front element of L. Pre: length()>0
int front(List L){
	if (length(L) > 0){ //Pre: length()>0
		return L->front->data;
	}

	else{
		printf("Error: List is empty.");
                exit(1);
	}
}

int back(List L){
	if (length(L) > 0){ //Pre: length()>0
                return L->back->data; // Returns back element of L.
        }

        else{
                printf("Error: List is empty.");
                exit(1);
        }
}

int get(List L){
	if (length(L)>0 && index(L)>=0){ //Pre: length()>0, index()>=0
		return L->cursor->data; // Return cursor element of L.
        }
        printf("Error: Cursor is undefined.");
        exit(1);
}

// Returns true if Lists A and B contain the same sequence of elements, returns false otherwise.


bool equals(List A, List B){
	if (A->length != B->length){ ////if lengths of A and B aren't equal
		return false;
	}

	Node* nodeA = A->front; ////Start at front of List A
	Node* nodeB = B->front; ////Start at front of List B

	while(nodeA != NULL && nodeB != NULL){ ////while node is defined
		if(nodeA->data != nodeB->data){ ////if node in A != node in B
			return false;
		}
		nodeA = nodeA->next; ////move to next node in List A
		nodeB = nodeB->next; ////move to next node in List B
    	}

	return true;
}


// Manipulation procedures ----------------------------------------------------
// Resets L to its original empty state.
void clear(List L){
	while(L->length > 0){ //while list not empty
		deleteFront(L); //free nodes
	}
	
	L->front = NULL; //reset L
	L->back = NULL; //reset L
	L->cursor = NULL; //reset L
	L->index = -1; //reset L
}


void set(List L, int x){
	if(L->length > 0 && L->index >= 0){ // Pre: length()>0, index()>=0
		L->cursor->data = x; // Overwrites the cursor elementâ€™s data with x.
	}

	else{
		printf("Error: Cursor not defined.\n");
		exit(1);
	}
}

void moveFront(List L){
	if(L->length > 0){ // If L is non-empty,
		L->cursor = L->front; // sets cursor under the front element
		L->index = 0; ////set the index to 0
	}
	// otherwise does nothing.
}

void moveBack(List L){ // If L is non-empty,
	if(L->length > 0){
		L->cursor = L->back; // sets cursor to back element
		L->index = L->length - 1; ////set index to last element
	}
	// otherwise does nothing.
}

void moveNext(List L){
	if(L->cursor != NULL && L->index < L->length - 1){ // If cursor is defined and not at front,
		L->cursor = L->cursor->next; // move cursor one step toward the front of L
		L->index++;
	}

	else if(L->cursor != NULL && L->index == L->length - 1){ //if cursor is defined and at front,
		L->cursor = NULL;  // Cursor becomes undefined
		L->index = -1;
	}
	// if cursor is undefined do nothing
}

void movePrev(List L){
	if(L->cursor != NULL && L->index > 0){ // If cursor is defined and not at front,
		L->cursor = L->cursor->prev; // move cursor one step toward the back of L;
		L->index--;
	}

	else if(L->cursor != NULL && L->index == 0){ //if cursor is defined and at back,
		L->cursor = NULL; // cursor becomes undefined
		L->index = -1;
	}
	// if cursor is undefined do nothing
}

// Insert new element into L. If L is non-empty,
// insertion takes place before front element.
void prepend(List L, int x){
	Node* new_node = newNode(x);

	if (L->length == 0){ ////if list is empty
		L->front = L->back = new_node; ////insert new element into L (which will be both front and back since there's only 1 element)
	}

	else{
		new_node->next = L->front;
		L->front->prev = new_node;
		L->front = new_node;
	}
	L->length++;

	if(L->index != -1){
		L->index++; ////Adjust index if cursor is defined
	}

}

// Insert new element into L. If L is non-empty,
// insertion takes place after back element.
void append(List L, int x){
	Node* new_node = newNode(x);
	if(L->length == 0){  // List is empty
		L->front = L->back = new_node;
	}

	else{
		new_node->prev = L->back;
		L->back->next = new_node;


		L->back = new_node;
	}
	L->length++;
}

// Insert new element before cursor.
void insertBefore(List L, int x){
	if(L->length > 0 && L->index >= 0){
		if(L->cursor == L->front){
			prepend(L, x); ////Use prepend if cursor is at the front
		}

		else{
			Node* new_node = newNode(x);
			new_node->next = L->cursor;
			new_node->prev = L->cursor->prev;
			L->cursor->prev->next = new_node;
			L->cursor->prev = new_node;
			L->length++;
			L->index++;
		}
	}

	else{
		printf("Error: Cursor is undefined or list is empty.\n");
		exit(1);
	}
}

// Insert new element after cursor.
void insertAfter(List L, int x){


	if(L->length > 0 && L->index >= 0){ // Pre: length()>0, index()>=0
		if (L->cursor == L->back){
			append(L, x); ////Use append if cursor is at the back
		}

		else{
			Node* new_node = newNode(x);
			new_node->prev = L->cursor;


			new_node->next = L->cursor->next;
			L->cursor->next->prev = new_node;
			L->cursor->next = new_node;
			L->length++;
		}
	}

	else{
		printf("Error: Cursor is undefined or list is empty.\n");
		exit(1);
    }
}

// Delete the front element.


void deleteFront(List L){
	if(L->length > 0){ //Pre: length()>0
		Node* temp = L->front;

		if (L->length == 1){
			L->front = L->back = NULL;
		}

		else{
			L->front = L->front->next;
			L->front->prev = NULL;
		}

		if (L->index == 0) {  ////Cursor was at the front
			L->cursor = NULL;
			L->index = -1;
		}

		free(temp);
		L->length--;

		if (L->index != -1){
			L->index--; ////Adjust cursor index if necessary
		}
	}

	else{
		printf("Error: List is empty.\n");
		exit(1);
	}
}

// Delete the back element.


void deleteBack(List L){
	if(L->length > 0){ //Pre: length()>0
		Node* temp = L->back;

		if(L->length == 1){
			L->front = L->back = NULL;
		}

		else{
			L->back = L->back->prev;
			L->back->next = NULL;
		}

		if(L->index == L->length - 1){ ////Cursor was at the back
			L->cursor = NULL;
			L->index = -1;
		}
		free(temp);
		L->length--;
	}

	else{
		printf("Error: List is empty.\n");
		exit(1);
	}
}

// Delete cursor element, making cursor undefined.


void delete(List L){
	if(L->length > 0 && L->index >= 0){ // Pre: length()>0, index()>=0
		if(L->cursor == L->front){ ////if cursor is at front
			deleteFront(L); ////delete front
		}

		else if(L->cursor == L->back){ ////if cursor is at back
		    deleteBack(L); ////delete back
		}

		else{
			Node *temp = L->cursor;
			L->cursor->prev->next = L->cursor->next;
			L->cursor->next->prev = L->cursor->prev;
			freeNode(&temp);
            		L->cursor = NULL;
            		L->index = -1;
            		L->length--;
		}
	}

	else{
		printf("Error: Cursor is undefined or list is empty.\n");
		exit(1);
	}
}

// Other operations -----------------------------------------------------------

// Prints to the file pointed to by out, a
// string representation of L consisting
// of a space separated sequence of integers, with front on left.
void printList(FILE* out, List L){
	if(L == NULL){
		printf("Error: List is NULL.\n");
		exit(1);
	}
	Node* current = L->front; ////start from front of list

	while(current != NULL){
		fprintf(out, "%d ", current->data); ////print current node's data
		current = current->next; ////move to next node
	}
	fprintf(out, "\n"); ////new line after printing list
}


// Returns a new List representing the same integer
// sequence as L. The cursor in the new list is undefined,
// regardless of the state of the cursor in L. The state of L is unchanged.
List copyList(List L){
	if(L == NULL){
		printf("Error: List is NULL.\n");
		exit(1);
	}
	List copy = newList(); ////create new empty list
	Node* current = L->front; ////start from front of original list

	while (current != NULL){
		append(copy, current->data); ////append each element to new list
		current = current->next; ////move to the next node
	}
	return copy; ////return copied list
}
