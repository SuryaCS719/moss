#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>
#include "List.h"

// Node structure
typedef struct NodeObj {
    int data;
    struct NodeObj* next;
    struct NodeObj* prev;
} NodeObj;

typedef NodeObj* Node;

// List structure
typedef struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
    int index; // -1 when undefined
} ListObj;

// Creates and returns a new empty List
List newList(void) {
    List L = malloc(sizeof(ListObj));
    L->front = L->back = L->cursor = NULL;
    L->length = 0;
    L->index = -1;
    return L;
}

// Frees all heap memory associated with *pL, and sets *pL to NULL
void freeList(List* pL) {
    if (pL != NULL && *pL != NULL) {
        while ((*pL)->length > 0) {
            deleteFront(*pL);
        }
        free(*pL);
        *pL = NULL;
    }
}


// Returns the number of elements in L


int length(List L) {
    return L->length;
}

// Returns the index of the cursor element if defined, -1 otherwise
int index(List L) {
    return L->cursor != NULL ? L->index : -1;
}

// Returns front element of L. Pre: length() > 0
int front(List L) {
    if (L->length > 0) {
        return L->front->data;
    }
    printf("List Error: calling front() on an empty List\n");
    exit(EXIT_FAILURE);
}

// Returns back element of L. Pre: length() > 0
int back(List L) {
    if (L->length > 0) {
        return L->back->data;
    }
    printf("List Error: calling back() on an empty List\n");
    exit(EXIT_FAILURE);
}

// Returns cursor element of L. Pre: length() > 0, index() >= 0
int get(List L) {
    if (L->length > 0 && L->index >= 0) {
        return L->cursor->data;
    }
    printf("List Error: calling get() with undefined cursor\n");
    exit(EXIT_FAILURE);
}

// Returns true iff Lists A and B contain the same sequence of elements
bool equals(List A, List B) {
    if (A->length != B->length) return false;
    Node nodeA = A->front;
    Node nodeB = B->front;
    while (nodeA != NULL && nodeB != NULL) {
        if (nodeA->data != nodeB->data) return false;
        nodeA = nodeA->next;
        nodeB = nodeB->next;
    }
    return true;
}


// Resets L to its original empty state
void clear(List L) {
    while (L->length > 0) {
        deleteFront(L);
    }
    L->cursor = NULL;
    L->index = -1;
}

// Overwrites the cursor elementâ€™s data with x
void set(List L, int x) {
    if (L->length > 0 && L->index >= 0) {


        L->cursor->data = x;
    } else {
        printf("List Error: calling set() with undefined cursor\n");
        exit(EXIT_FAILURE);
    }
}

// If L is non-empty, sets cursor under the front element
void moveFront(List L) {
    if (L->length > 0) {
        L->cursor = L->front;
        L->index = 0;
    }
}

// If L is non-empty, sets cursor under the back element
void moveBack(List L) {
    if (L->length > 0) {
        L->cursor = L->back;
        L->index = L->length - 1;
    }
}

// Move cursor one step toward the front of L
void movePrev(List L) {
    if (L->cursor != NULL && L->cursor != L->front) {
        L->cursor = L->cursor->prev;
        L->index--;
    } else if (L->cursor == L->front) {
        L->cursor = NULL;
        L->index = -1;
    }
}

// Move cursor one step toward the back of L
void moveNext(List L) {
    if (L->cursor != NULL && L->cursor != L->back) {
        L->cursor = L->cursor->next;
        L->index++;
    } else if (L->cursor == L->back) {
        L->cursor = NULL;
        L->index = -1;
    }
}

// Insert new element into L. Insertion takes place before front element
void prepend(List L, int x) {
    Node newNode = malloc(sizeof(NodeObj));
    newNode->data = x;
    newNode->next = L->front;
    newNode->prev = NULL;
    if (L->front != NULL) {
        L->front->prev = newNode;
    } else {
        L->back = newNode; // If list was empty
    }
    L->front = newNode;
    L->length++;
    if (L->index != -1) {
        L->index++; // Shift cursor position
    }
}

// Insert new element into L. Insertion takes place after back element
void append(List L, int x) {
    Node newNode = malloc(sizeof(NodeObj));
    newNode->data = x;
    newNode->prev = L->back;
    newNode->next = NULL;
    if (L->back != NULL) {
        L->back->next = newNode;
    } else {
        L->front = newNode; // If list was empty
    }
    L->back = newNode;
    L->length++;
}

// Insert new element before cursor. Pre: length() > 0, index() >= 0
void insertBefore(List L, int x) {
    if (L->length > 0 && L->index >= 0) {
        Node newNode = malloc(sizeof(NodeObj));
        newNode->data = x;
        newNode->prev = L->cursor->prev;
        newNode->next = L->cursor;
        if (L->cursor->prev != NULL) {
            L->cursor->prev->next = newNode;
        } else {
            L->front = newNode; // If cursor was at front
        }


        L->cursor->prev = newNode;
        L->length++;
        L->index++;
    } else {
        printf("List Error: calling insertBefore() with undefined cursor\n");
        exit(EXIT_FAILURE);
    }
}

// Insert new element after cursor. Pre: length() > 0, index() >= 0
void insertAfter(List L, int x) {
    if (L->length > 0 && L->index >= 0) {
        Node newNode = malloc(sizeof(NodeObj));
        newNode->data = x;
        newNode->next = L->cursor->next;
        newNode->prev = L->cursor;
        if (L->cursor->next != NULL) {
            L->cursor->next->prev = newNode;
        } else {
            L->back = newNode; // If cursor was at back
        }


        L->cursor->next = newNode;
        L->length++;
    } else {
        printf("List Error: calling insertAfter() with undefined cursor\n");
        exit(EXIT_FAILURE);
    }
}

// Delete the front element. Pre: length() > 0
void deleteFront(List L) {
    if (L->length > 0) {
        Node temp = L->front;
        if (L->front == L->back) {
            L->front = L->back = NULL;
        } else {
            L->front = L->front->next;


            L->front->prev = NULL;
        }
        free(temp);
        L->length--;
        if (L->index != -1) {
            L->index--;
        }
    } else {
        printf("List Error: calling deleteFront() on empty list\n");
        exit(EXIT_FAILURE);
    }
}

// Delete the back element. Pre: length() > 0
void deleteBack(List L) {
    if (L->length > 0) {
        Node temp = L->back;
        if (L->front == L->back) {
            L->front = L->back = NULL;
        } else {
            L->back = L->back->prev;
            L->back->next = NULL;
        }
        free(temp);
        L->length--;
        if (L->index == L->length) {
            L->index = -1;
        }
    } else {
        printf("List Error: calling deleteBack() on empty list\n");
        exit(EXIT_FAILURE);
    }
}

// Delete cursor element. Pre: length() > 0, index() >= 0
void delete(List L) {
    if (L->length > 0 && L->index >= 0) {
        Node temp = L->cursor;
        if (L->cursor == L->front) {
            deleteFront(L);
        } else if (L->cursor == L->back) {
            deleteBack(L);
        } else {
            L->cursor->prev->next = L->cursor->next;


            L->cursor->next->prev = L->cursor->prev;
            free(temp);
            L->length--;
        }
        L->cursor = NULL;
        L->index = -1;
    } else {
        printf("List Error: calling delete() with undefined cursor\n");
        exit(EXIT_FAILURE);
    }
}

// Prints a string representation of L to out
void printList(FILE* out, List L) {
    Node node = L->front;
    while (node != NULL) {
        fprintf(out, "%d ", node->data);
        node = node->next;
    }
}

// Returns a copy of L
List copyList(List L) {
    List copy = newList();
    Node node = L->front;
    while (node != NULL) {
        append(copy, node->data);
        node = node->next;
    }
    return copy;
}

// Concatenates two lists
List concatList(List A, List B) {
    List concat = copyList(A);
    Node node = B->front;
    while (node != NULL) {
        append(concat, node->data);
        node = node->next;
    }
    return concat;
}
