#include <stdio.h>
#include <stdlib.h>
#include "List.h"

typedef struct NodeObj {
    int data;
    struct NodeObj* next;
    struct NodeObj* prev;
} NodeObj;

typedef NodeObj* Node;

typedef struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
    int index;
} ListObj;

List newList(void) {
    List L = malloc(sizeof(ListObj));
    L->front = L->back = L->cursor = NULL;
    L->length = 0;
    L->index = -1;
    return L;
}

void freeList(List* pL) {
    if (pL != NULL && *pL != NULL) {
        clear(*pL);
        free(*pL);
        *pL = NULL;
    }
}


int length(List L) {
    return L->length;
}



int index(List L) {
    return L->index;
}

int front(List L) {
    if (L->length > 0) return L->front->data;
    exit(EXIT_FAILURE);
}

int back(List L) {
    if (L->length > 0) return L->back->data;
    exit(EXIT_FAILURE);
}

int get(List L) {
    if (L->length > 0 && L->index >= 0) return L->cursor->data;
    exit(EXIT_FAILURE);
}

bool equals(List A, List B) {
    if (A->length != B->length) return false;
    Node N = A->front;
    Node M = B->front;
    while (N != NULL) {
        if (N->data != M->data) return false;
        N = N->next;
        M = M->next;
    }
    return true;
}


void clear(List L) {
    while (L->length > 0) deleteFront(L);
    L->cursor = NULL;
    L->index = -1;
}



void set(List L, int x) {
    if (L->length > 0 && L->index >= 0) L->cursor->data = x;
    else exit(EXIT_FAILURE);
}

void moveFront(List L) {
    if (L->length > 0) {
        L->cursor = L->front;
        L->index = 0;
    }
}

void moveBack(List L) {
    if (L->length > 0) {
        L->cursor = L->back;
        L->index = L->length - 1;
    }
}

void movePrev(List L) {
    if (L->cursor != NULL && L->cursor != L->front) {
        L->cursor = L->cursor->prev;
        L->index--;
    } else if (L->cursor == L->front) {
        L->cursor = NULL;
        L->index = -1;
    }
}

void moveNext(List L) {
    if (L->cursor != NULL && L->cursor != L->back) {
        L->cursor = L->cursor->next;
        L->index++;


    } else if (L->cursor == L->back) {
        L->cursor = NULL;
        L->index = -1;
    }
}

void prepend(List L, int x) {
    Node N = malloc(sizeof(NodeObj));
    N->data = x;
    N->prev = NULL;
    N->next = L->front;


    if (L->front != NULL) L->front->prev = N;
    L->front = N;
    if (L->length == 0) L->back = N;
    L->length++;
    if (L->index != -1) L->index++;
}

void append(List L, int x) {
    Node N = malloc(sizeof(NodeObj));
    N->data = x;
    N->next = NULL;
    N->prev = L->back;
    if (L->back != NULL) L->back->next = N;


    L->back = N;
    if (L->length == 0) L->front = N;
    L->length++;
}

void insertBefore(List L, int x) {
    if (L->length > 0 && L->index >= 0) {
        if (L->cursor == L->front) prepend(L, x);
        else {
            Node N = malloc(sizeof(NodeObj));
            N->data = x;
            N->prev = L->cursor->prev;
            N->next = L->cursor;
            L->cursor->prev->next = N;
            L->cursor->prev = N;
            L->length++;
            L->index++;
        }
    }
}

void insertAfter(List L, int x) {


    if (L->length > 0 && L->index >= 0) {
        if (L->cursor == L->back) append(L, x);
        else {
            Node N = malloc(sizeof(NodeObj));
            N->data = x;
            N->next = L->cursor->next;
            N->prev = L->cursor;
            L->cursor->next->prev = N;
            L->cursor->next = N;


            L->length++;
        }
    }
}

void deleteFront(List L) {
    if (L->length > 0) {
        Node N = L->front;
        if (L->front == L->back) {
            L->front = L->back = NULL;
        } else {


            L->front = L->front->next;
            L->front->prev = NULL;
        }
        free(N);
        L->length--;
        if (L->index != -1) L->index--;
    }
}

void deleteBack(List L) {
    if (L->length > 0) {
        Node N = L->back;
        if (L->front == L->back) {
            L->front = L->back = NULL;
        } else {


            L->back = L->back->prev;
            L->back->next = NULL;
        }
        free(N);
        L->length--;
        if (L->index >= L->length) L->index = -1;
    }
}

void delete(List L) {
    if (L->length > 0 && L->index >= 0) {
        if (L->cursor == L->front) deleteFront(L);
        else if (L->cursor == L->back) deleteBack(L);
        else {
            Node N = L->cursor;
            N->prev->next = N->next;
            N->next->prev = N->prev;
            free(N);
            L->length--;
        }
        L->cursor = NULL;
        L->index = -1;
    }
}


void printList(FILE* out, List L) {
    Node N = L->front;
    while (N != NULL) {
        fprintf(out, "%d ", N->data);
        N = N->next;
    }
}

List copyList(List L) {
    List copy = newList();
    Node N = L->front;
    while (N != NULL) {
        append(copy, N->data);
        N = N->next;
    }
    return copy;
}
