#include<stdio.h>

// structs --------------------------------------------------------------------

typedef struct NodeObj* Node; // private Node type

typedef struct NodeObj{ // private NodeObj type
   int data;
   Node next;
   Node prev;
} NodeObj;

typedef struct ListObj{ // private ListObj type
   Node front;
   Node back;
   Node cursor;
   int length;
   int index;
} ListObj;


// Constructors-Destructors ---------------------------------------------------

// Returns reference to new Node object. Initializes next and data fields.
Node newNode(int data) {


   Node N = malloc(sizeof(NodeObj));
   assert( N!=NULL );
   N->data = data;
   N->next = NULL;
   N->prev = NULL;
   return(N);
}

List newList(void) { // Creates and returns a new empty List
   List L;
   L= malloc(sizeof(ListObj));
   assert(L != NULL);
   L->front = L->back = L->cursor = NULL;
   L->length = 0;


   L->index = -1;
   return(L);
}

void freeNode(Node* pL) { // frees heap memory pointed to by *pL then sets *pL to NULL
   if( pL!=NULL && *pL!=NULL ){ // check to dereference pointer
      free(*pL);
      *pL = NULL;
   }
}

void freeList(List* pL) { // frees all heap memory associated with *pL then sets *pL to NULL
   if( pL!=NULL && *pL!=NULL ){ // check to dereference pointer
      Node freel = (*pL)->front; // sets pointer to front of pL
      while (freel != NULL) {
         Node temp = freel; //temporary point to curent node
         freel = freel->next; 
         free(temp);
      }
      
      (*pL)->front = NULL; // nullify front pointer
      (*pL)->back = NULL; // nullify back pointer
      (*pL)->cursor = NULL; // nullify cursor pointer
      (*pL)->length = 0;
      
      free(*pL); // free list structure 
      *pL = NULL; // set pointer to NULL


   }
   
}

// Access functions -----------------------------------------------------------

int front(List L){ // Returns the element at the front of L.
                   // Pre: !isEmpty(Q)
   if( L==NULL ){
      printf("List Error: attempting to call front() on a NULL Queue reference\n");
      exit(EXIT_FAILURE);
   }
   
   if (L->length > 0){
      return L->front->data;
   
   }else {
      exit(EXIT_FAILURE);
   }

}

int index(List L) { // Returns index of cursor element if defined, -1 otherwise.
   if (L == NULL) {
      printf("Index of cursor element isn't defined\n");
      exit(EXIT_FAILURE);
   }

   if (L->cursor == NULL) {
         return -1;
   }
  
   if (L->index < 0) {
	return -1;
}
   return L->index;

}

int length(List L) { // Returns the number of length in L.
   if( L==NULL ){
      printf("List Error: attempting to call int Length() on a NULL Queue reference\n");
      exit(EXIT_FAILURE);
   }
   return(L->length);
}

int back(List L) { // Returns back element of L. Pre: length()>0
   if( L==NULL ){
      printf("List Error: attempting to call back() on a NULL Queue reference\n");
      exit(EXIT_FAILURE);
   }
   
   if (L->length > 0) {
      return L->back->data;

   }else{
      printf("List Error: attempting to call int back() on a NULL Queue reference\n");
      exit(EXIT_FAILURE);
   }
}

int get(List L) { //Returns cursor element of L. Pre: length()>0, index()>=0
   if( L==NULL ){
         printf("List Error: attempting to call get() on a NULL Queue reference\n");
         exit(EXIT_FAILURE);
      }

   if (length(L) < 1 && index(L) < 0) {
      printf("List Error: attempting to call get() on a NULL Queue reference\n");
      exit(EXIT_FAILURE);
   }
      
      
   return L->cursor->data;
}

bool equals(List A, List B){ // Returns true if A is same int sequence as B, false otherwise.
   if(A==NULL || B==NULL ){
      printf("Queue Error: attempting to call equals() on a NULL Queue reference\n");
      exit(EXIT_FAILURE);
   }	

   bool eq;
   Node N, M;
   eq = (A->length == B->length );
   N = A->front;
   M = B->front;
   while( eq && N!=NULL){
      eq = ( N->data==M->data );


      N = N->next;
      M = M->next;
   }

   return eq;

}



// Manipulation procedures ----------------------------------------------------

void clear(List L) { // Resets L to its original empty state.

   if( L==NULL ){
      printf("List Error: attempting to call clear() on a NULL Queue reference\n");
      exit(EXIT_FAILURE);
   }

   while (L->front != NULL) { // traverse list and free nodes
      Node temp = L->front;
      L->front = L->front->next;
      freeNode(&temp);
   }

   L->front = NULL;
   L->back = NULL;
   L->cursor = NULL;
   L->length = 0;

}

void set(List L, int x) { // Overwrites the cursor elementâ€™s data with x.
                          // Pre: length()>0, index()>=0
   if( L==NULL ){
      printf("List Error: attempting to call set() on a NULL Queue reference\n");
      exit(EXIT_FAILURE);
   }
   
   
   if(length(L) < 1 && index(L) < 0) {
      printf("List Error: attempting to call set() on a NULL Queue reference\n");
      exit(EXIT_FAILURE);
   }

   L->cursor->data = x;

   
}

void moveFront(List L) { // If L is non-empty, sets cursor under the front element,
                         // otherwise does nothing.
   if( L==NULL ){
      printf("List Error: attempting to call movefront() on a NULL Queue reference\n");
      exit(EXIT_FAILURE);
   }
   
  if (L->length > 0) {
	L->cursor = L->front;
	L->index = 0;
   }
}

void moveBack(List L) { // If L is non-empty, sets cursor under the back element,
                        // otherwise does nothing.
    if( L==NULL ){
      printf("List Error: attempting to call moveBack() on a NULL Queue reference\n");
      exit(EXIT_FAILURE);
   }
   
   if (L->length > 0) {
	L->cursor = L->back;


	L->index = L->length - 1;
}    

}

 
   





void movePrev(List L) { // If cursor is defined and not at front, move cursor one
                        // step toward the front of L; if cursor is defined and at
                        // front, cursor becomes undefined; if cursor is undefined
                        // do nothing
   if( L==NULL ){
      printf("List Error: attemtping to call movePrev() on a NULL Queue reference\n");
      exit(EXIT_FAILURE);
   }

   if (L->cursor != NULL) { // check
      if (L->cursor != L->front) {
         L->cursor = L->cursor->prev;
         L->index--;
      }else {
         L->cursor = NULL;


         L->index = -1;
      }
   }
}

void moveNext(List L) { // If cursor is defined and not at back, move cursor one
                        // step toward the back of L; if cursor is defined and at
                        // back, cursor becomes undefined; if cursor is undefined
                        // do nothing
   if( L==NULL ){
      printf("List Error: attempting to call moveNext() on a NULL Queue reference\n");
      exit(EXIT_FAILURE);
   }
   
   if (L->cursor != NULL) { // check
      if (L->cursor != L->back) {
         L->cursor = L->cursor->next;
         L->index++;
      }else {
         L->cursor = NULL;
         L->index = -1;
      }
   }
}
   
void prepend(List L, int x) { // Insert new element into L. If L is non-empty,
                              // insertion takes place before front element.
   if( L==NULL ){
      printf("List Error: attempting to call prepend() on a NULL Queue reference\n");
      exit(EXIT_FAILURE);
   }

   Node inserte = newNode(x);

   if (L->length == 0) { // check if list is empty
      L->front = inserte;
      L->back = inserte;
   } else {
      L->front->prev = inserte;
      inserte->next = L->front;
      L->front = inserte;
   }
   
   if (L->cursor != NULL) {
	if (L->index == 0) {
	 L->cursor->prev = inserte;
   }
   L->index++;
}
   L->front = inserte;
   L->length++;

   

} 

void append(List L, int x) { // Insert new element into L. If L is non-empty,
                             // insertion takes place after back element.
   if( L==NULL ){
      printf("List Error: attempting to call append() on a NULL Queue reference\n");
      exit(EXIT_FAILURE);
   }

   Node inserte = newNode(x); // create new node
   
   if (L->length == 0) { // checks when list is empty
      L->front = inserte;
      L->back = inserte;
   } else {
      inserte->prev = L->back; // sets current node after back element
      L->back->next = inserte;


      L->back = inserte;

   }
   
   L->length++; // increment len
}

void insertBefore(List L, int x) { // Insert new element before cursor.
                                   // Pre: length()>0, index()>=0
   if( L==NULL ){
      printf("List Error: calling insertBefore() on NULL Queue reference\n");
      exit(EXIT_FAILURE);
   }
   
   
   if(length(L) < 1 && index(L) < 0) {
      printf("List Error: attempting to call insertBefore() on a NULL Queue reference\n");
      exit(EXIT_FAILURE);
   }

   Node inserte = newNode(x);
   if (L->cursor != NULL) {
      inserte->next = L->cursor; // link new node next to cursor

      if (L->cursor->prev != NULL) { // link new node prev to node before cursor
         inserte->prev = L->cursor->prev;
         L->cursor->prev->next = inserte;
      }

      L->cursor->prev = inserte; // update cursor prev to point to new node

      if (L->cursor == L->front) { // update front pointer if cursor was at front
         L->front = inserte;
      }

      L->index++; // increment len
      L->length++;
   }
}
    
void insertAfter(List L, int x) { // Insert new element after cursor.
                                  // Pre: length()>0, index()>=0
   if( L==NULL ){
      printf("List Error: attempting to call insertAfter() on a NULL Queue reference\n");
      exit(EXIT_FAILURE);
   }
   
   
   if(length(L) < 1 && index(L) < 0) {
      printf("List Error: attempting to call insertAfter() on a NULL Queue reference\n");
      exit(EXIT_FAILURE);
   }

   Node inserte = newNode(x);
   if (L->cursor != NULL) {
      inserte->next = L->cursor->next; // link new node next to cursor next

      inserte->prev = L->cursor; // link new node prev to cursor

      L->cursor->next = inserte; // update cursor next to point to new node

      if (L->cursor == L->back) { // update back pointer if cursor was at back
         L->back = inserte;
      }

      L->length++; // increment len

   }

}

void deleteFront(List L) { // Delete the front element. Pre: length()>0
   if( L==NULL ){
      printf("List Error: attempting to call deleteFront() on a NULL Queue reference\n");
      exit(EXIT_FAILURE);
   }
   
   
   if(L->length < 1)  {
      printf("List Error: attempting to call deleteFront() on a NULL Queue reference\n");
      exit(EXIT_FAILURE);
   }

   Node frontn = L->front;
   if (L->front == L->cursor) {
	L->cursor = NULL;
	L->index = -1;
}

   L->front = frontn->next;

  if (L->length == 0) { // check if list becomes empty after deletion
      L->back = NULL;
   }

   L->length--;
   L->index--;
   freeNode(&frontn);
}
void deleteBack(List L) { // Delete the back element. Pre: length()>0
   if( L==NULL ){
      printf("List Error: attempting to call deleteBack() on a NULL Queue reference\n");
      exit(EXIT_FAILURE);
   }
   
   
   if(length(L) < 1)  {
      printf("List Error: attempting to call deleteBack() on a NULL Queue reference\n");
      exit(EXIT_FAILURE);
   }
   
   if (L->back == L->cursor) {
	L->cursor = NULL;
	L->index = -1;
}   

   Node backn = L->back;
   
   if (L->length == 1) {
	L->front = L->back = NULL;
   } else {
	L->back = backn->prev;
	L->back->next = NULL;
   }
	
   freeNode(&backn);
   L->length--;	 	
   
}

void delete(List L) { // Delete cursor element, making cursor undefined.
                      // Pre: length()>0, index()>=0
    if( L==NULL ){
      printf("List Error: attempting to call isEmpty() on a NULL Queue reference\n");
      exit(EXIT_FAILURE);
   }
   
   
   if(length(L) < 1 && index(L) < 0) {
      printf("List Error: attempting to call isEmpty() on a NULL Queue reference\n");
      exit(EXIT_FAILURE);
   }

      Node cursornode = L->cursor;

      if (cursornode == L->front) {
         L->front = cursornode->next;
      } else if (cursornode == L->back) {
         L->back = cursornode->prev;
         L->back->next = NULL;
      } else {
         cursornode->prev->next = cursornode->next;
         cursornode->next->prev = cursornode->prev;
      }


      freeNode(&cursornode);
      L->cursor = NULL;
      L->length--;
   
}



// Other Functions ------------------------------------------------------------

void printList(FILE* out, List L) { // Prints to the file pointed to by out, a
                                    // string representation of L consisting
                                    // of a space separated sequence of integers,
                                    // with front on left.
   if (out == NULL && L == NULL){
      printf("Queue Error: invalid file or NULL List reference encountered in printList()\n");
      exit(EXIT_FAILURE);
   }
   
   Node currentNode = L->front;

   while (currentNode != NULL) {
      fprintf(out, "%d", currentNode->data);
      if (currentNode->next != NULL) { // print space after each integer except last
         fprintf(out, " ");
      } else{
         fprintf(out, " ");
      }
      currentNode = currentNode->next;
   }

}

List copyList(List L) { // Returns a new List representing the same integer
                        // sequence as L. The cursor in the new list is undefined,
                        // regardless of the state of the cursor in L. The state
                        // of L is unchanged.
   if (L == NULL) {
      printf("List Error: attempting to call copyList() on a NULL List reference\n");
      exit(EXIT_FAILURE);
   }

   List new_List = newList(); // create new empty list to hold copied elements

   Node currentNode = L->front;

   while (currentNode != NULL) {
      append(new_List, currentNode->data);
      currentNode = currentNode->next;
   }
   
   new_List->cursor = NULL;
   new_List->index = -1; 
   return new_List;
   
}
