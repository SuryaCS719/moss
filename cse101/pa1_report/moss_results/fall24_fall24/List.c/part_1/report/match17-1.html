#include <stdio.h>
#include <stdlib.h>
#include <assert.h>
#include <stdbool.h>
#include <time.h>
#include "List.h"

// Node
typedef struct NodeObj* Node;

// NodeObj
typedef struct NodeObj {
    int data; // contained data
    Node beforeNode; // pointer to element before
    Node afterNode; // pointer to element after
} NodeObj;

// ListObj
typedef struct ListObj {
    Node frontNode; // pointer to front
    Node cursorNode; // pointer to element selected by the cursor
    Node backNode; // pointer to back
    int cursor; // index of cursor
    int length; // length of the list
} ListObj;


Node newNode(int data) {
    Node N = malloc(sizeof(NodeObj));
    assert(N != NULL);
    N->data = data;
    N->afterNode = NULL;
    N->beforeNode = NULL;
    return N;
}

void freeNode(Node* pN) {
    if (pN != NULL && *pN != NULL) {


        free(*pN);
        *pN = NULL;
    }
}


List newList() {
    List L = malloc(sizeof(ListObj));
    assert(L != NULL);
    L->frontNode = NULL;
    L->cursorNode = NULL;
    L->backNode = NULL;
    L->cursor = -1;
    L->length = 0;
    return L;
}

void freeList(List *pL) {
    if (pL != NULL && *pL != NULL) {
        clear(*pL);
        free(*pL);


        *pL = NULL;
    }
}

int length(List L) {
    return L->length;
}

int index(List L) {
    return L->cursor;
}

int front(List L) {
    if (L == NULL) {
        fprintf(stderr, "Integer List Error: calling front() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->length < 1) {
        fprintf(stderr, "Integer List Error: calling front() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    return L->frontNode->data;
}

int back(List L) {
    if (L == NULL) {
        fprintf(stderr, "Integer List Error: calling back() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->length < 1) {
        fprintf(stderr, "Integer List Error: calling back() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    return L->backNode->data;
}

int get(List L) {
    if (L == NULL) {
        fprintf(stderr, "Integer List Error: calling get() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }


    if (L->length < 1) {
        fprintf(stderr, "Integer List Error: calling get() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    return L->cursorNode->data;
}

bool equals(List A, List B) {
    if (A == NULL) {
        fprintf(stderr, "Integer List Error: calling equals() on NULL A List reference\n");
        exit(EXIT_FAILURE);
    }
    if (B == NULL) {
        fprintf(stderr, "Integer List Error: calling equals() on NULL B List reference(s)\n");
        exit(EXIT_FAILURE);
    }

    if (A->length != B->length) {
        return false;
    }
    Node aItem = A->frontNode;
    Node bItem = B->frontNode;
    for (int i = 0; i < A->length; i++) {
        if (aItem->data != bItem->data) {
            return false;
        }
        if (aItem->afterNode != NULL) {
            aItem = aItem->afterNode;
            bItem = bItem->afterNode;
        }
    }
    return true;
}

void clear(List L) {
    if (L == NULL) {
        fprintf(stderr, "Integer List Error: calling clear() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    L->cursor = -1;
    L->cursorNode = NULL;
    while (L->length > 0) {
        deleteBack(L);
    }
}

void set(List L, int x) {
    if (L == NULL) {
        fprintf(stderr, "Integer List Error: calling set() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->length < 1) {
        fprintf(stderr, "Integer List Error: calling set() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    if (L->cursorNode < 0) {
        fprintf(stderr, "Integer List Error: calling set() with no cursor selected\n");
        exit(EXIT_FAILURE);
    }


    L->cursorNode->data = x;
}

void moveFront(List L) {
    if (L == NULL) {
        fprintf(stderr, "Integer List Error: calling moveFront() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->length > 0) {
        L->cursor = 0;
        L->cursorNode = L->frontNode;
    }
}

void moveBack(List L) {
    if (L == NULL) {
        fprintf(stderr, "Integer List Error: calling moveBack() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->length > 0) {
        L->cursor = L->length-1;
        L->cursorNode = L->backNode;
    }
}

void movePrev(List L) {
    if (L == NULL) {
        fprintf(stderr, "Integer List Error: calling movePrev() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->length > 0 && L->cursor > -1) {
        if (L->cursor == 0) {
            L->cursor = -1;
            L->cursorNode = NULL;
        } else {
            L->cursor--;
            L->cursorNode = L->cursorNode->beforeNode;
        }
    }
}

void moveNext(List L) {
    if (L == NULL) {
        fprintf(stderr, "Integer List Error: calling moveNext() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->length > 0 && L->cursor > -1) {
        if (L->cursor == L->length - 1) {
            L->cursor = -1;
            L->cursorNode = NULL;
        } else {
            L->cursor++;
            L->cursorNode = L->cursorNode->afterNode;
        }
    }
}

void prepend(List L, int x) {
    if (L == NULL) {
        fprintf(stderr, "Integer List Error: calling prepend() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }

    Node N = newNode(x);

    if (L->length == 0) {
        L->frontNode = N;
        L->backNode = N;
    } else {
        Node prevFirst = L->frontNode;
        L->frontNode = N;
        N->afterNode = prevFirst;
        prevFirst->beforeNode = N;
    }
    L->cursor++;
    L->length++;
}

void append(List L, int x) {
    if (L == NULL) {
        fprintf(stderr, "Integer List Error: calling append() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }



    Node N = newNode(x);

    if (L->length == 0) {
        L->frontNode = N;
        L->backNode = N;
    } else {
        Node prevLast = L->backNode;
        L->backNode = N;
        N->beforeNode = prevLast;
        prevLast->afterNode = N;
    }

    L->length++;
}

void insertBefore(List L, int x) {
    if (L == NULL) {
        fprintf(stderr, "Integer List Error: calling insertBefore() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }


    if (L->length < 1) {
        fprintf(stderr, "Integer List Error: calling insertBefore() on empty list\n");
        exit(EXIT_FAILURE);
    }
    if (L->cursor < 0) {
        fprintf(stderr, "Integer List Error: calling insertBefore() with undefined cursor\n");
        exit(EXIT_FAILURE);
    }

    Node N = newNode(x);

    if (L->cursor == 0) {
        Node prevFirst = L->frontNode;
        L->frontNode = N;
        N->afterNode = prevFirst;
        prevFirst->beforeNode = N;
    } else {
        Node prevNode = L->cursorNode->beforeNode;
        prevNode->afterNode = N;
        N->beforeNode = prevNode;
        L->cursorNode->beforeNode = N;
        N->afterNode = L->cursorNode;
    }
    L->cursor++;
    L->length++;
}

void insertAfter(List L, int x) {
    if (L == NULL) {
        fprintf(stderr, "Integer List Error: calling insertAfter() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->length < 1) {
        fprintf(stderr, "Integer List Error: calling insertAfter() on empty list\n");
        exit(EXIT_FAILURE);
    }
    if (L->cursor < 0) {
        fprintf(stderr, "Integer List Error: calling insertAfter() with undefined cursor\n");
        exit(EXIT_FAILURE);
    }

    Node N = newNode(x);

    if (L->cursor == L->length-1) {
        Node prevLast = L->backNode;
        L->backNode = N;
        N->beforeNode = prevLast;
        prevLast->afterNode = N;
    } else {
        Node nextNode = L->cursorNode->afterNode;
        nextNode->beforeNode = N;
        N->afterNode = nextNode;
        L->cursorNode->afterNode = N;
        N->beforeNode = L->cursorNode;
    }

    L->length++;
}

void deleteFront(List L) {
    if (L == NULL) {
        fprintf(stderr, "Integer List Error: calling deleteFront() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->length < 1) {
        fprintf(stderr, "Integer List Error: calling deleteFront() on empty list\n");
        exit(EXIT_FAILURE);
    }

    Node frontNode = L->frontNode;
    if (L->length > 1) {
        frontNode->afterNode->beforeNode = NULL;
        L->frontNode = frontNode->afterNode;
    }
    if (L->cursor == 0) {
        L->cursorNode = NULL;
        L->cursor = -1;
    } else {
        L->cursor--;
    }
    freeNode(&frontNode);
    L->length--;
}

void deleteBack(List L) {
    if (L == NULL) {
        fprintf(stderr, "Integer List Error: calling deleteBack() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->length < 1) {
        fprintf(stderr, "Integer List Error: calling deleteBack() on empty list\n");
        exit(EXIT_FAILURE);
    }

    Node backNode = L->backNode;
    if (L->length > 1) {
        backNode->beforeNode->afterNode = NULL;
        L->backNode = backNode->beforeNode;
    }
    if (L->cursor == L->length-1) {
        L->cursorNode = NULL;
        L->cursor = -1;
    }
    freeNode(&backNode);
    L->length--;
}

void delete(List L) {
    if (L == NULL) {
        fprintf(stderr, "Integer List Error: calling delete() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (L->length < 1) {
        fprintf(stderr, "Integer List Error: calling delete() on empty list\n");
        exit(EXIT_FAILURE);
    }

    Node cursorNode = L->cursorNode;
    if (cursorNode->beforeNode != NULL) {
        cursorNode->beforeNode->afterNode = cursorNode->afterNode;
    } else {
        if (cursorNode->afterNode !=NULL) {
            L->frontNode = cursorNode->afterNode;
        }
    }
    if (cursorNode->afterNode != NULL) {
        cursorNode->afterNode->beforeNode = cursorNode->beforeNode;
    } else {
        if (cursorNode->beforeNode != NULL) {
            L->backNode = cursorNode->beforeNode;
        }
    }

    freeNode(&cursorNode);
    L->cursor = -1;
    L->length--;
}

void printList(FILE *out, List L) {
    if (out == NULL) {
        fprintf(stderr, "Integer List Error: calling printList() on NULL file reference\n");
        exit(EXIT_FAILURE);
    }
    if (L == NULL) {
        fprintf(stderr, "Integer List Error: calling printList() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    Node activeNode = L->frontNode;
    for (int i = 0; i < L->length; i++) {
        fprintf(out, "%d", activeNode->data);
        if (activeNode->afterNode != NULL) {
            fprintf(out, " ");
            activeNode = activeNode->afterNode;
        }
    }
}

List copyList(List L) {
    if (L == NULL) {
        fprintf(stderr, "Integer List Error: calling copyList() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }

    List copy = newList();
    Node activeNode = L->frontNode;
    for (int i = 0; i < L->length; i++) {
        append(copy, activeNode->data);
        if (activeNode->afterNode != NULL) {
            activeNode = activeNode->afterNode;
        }
    }
    return copy;
}

List concatList(List A, List B) {
    if (A == NULL) {
        fprintf(stderr, "Integer List Error: calling concatList() on NULL A List reference\n");
        exit(EXIT_FAILURE);
    }
    if (B == NULL) {
        fprintf(stderr, "Integer List Error: calling concatList() on NULL B List reference\n");
        exit(EXIT_FAILURE);
    }

    List copy = newList();
    Node activeNode = A->frontNode;

    for (int i = 0; i < A->length; i++) {
        append(copy, activeNode->data);
        if (activeNode->afterNode != NULL) {
            activeNode = activeNode->afterNode;
        }
    }
    activeNode = B->frontNode;
    for (int i = 0; i < B->length; i++) {
        append(copy, activeNode->data);
        if (activeNode->afterNode != NULL) {
            activeNode = activeNode->afterNode;
        }
    }

    return copy;
}
