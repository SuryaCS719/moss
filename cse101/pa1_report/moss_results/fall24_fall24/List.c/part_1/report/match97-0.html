#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>
#include <assert.h>
#include "List.h"

typedef struct NodeObj* Node;

typedef struct NodeObj {
    int data;
    Node next;
    Node prev;
} NodeObj;


typedef struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
    int index;
} ListObj;


Node newNode(int data) {
    Node N = malloc(sizeof(NodeObj));
    assert( N!=NULL );
    N->data = data;
    N->next = NULL;
    N->prev = NULL;
    return(N);
}

void freeNode(Node* pN) {
    if(pN!=NULL && *pN!=NULL ){


      free(*pN);
      *pN = NULL;
   }
}

List newList(void) {
    List L = malloc(sizeof(ListObj));
    assert(L != NULL);

    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->length = 0;
    L->index = -1;

    return L;
}

void freeList(List* pL) {
    if (pL != NULL && *pL != NULL) {
        while((*pL)->length > 0) {
            deleteFront(*pL);
        }
        free(*pL);
        *pL = NULL;
    }
}

int length(List L) {
    return L->length;
}

int index(List L) {
    return L->index;
}



int front(List L) {
    return L->front->data;
}

int back(List L) {
    return L->back->data;
}


int get(List L) {
    if (L == NULL) {
        fprintf(stderr, "Get called on NULL\n");
        exit(EXIT_FAILURE);
    } else {
        return L->cursor->data;
    }
}

bool equals(List A, List B) {
    if (A == NULL || B == NULL) {
        fprintf(stderr, "Equals is called on NULL");
        exit(EXIT_FAILURE);
    }
    if (length(A) != length(B)) {
        return false;
    }
    moveFront(A);
    moveFront(B);

    while (index(A) >= 0 && index(B) >= 0) {
        if (get(A) != get(B)) {
            return false; 
        }
        moveNext(A);
        moveNext(B);
    }
    return true;
}


void clear(List L) {
    if (L == NULL) {
        fprintf(stderr, "clear called on NULL\n");
        exit(EXIT_FAILURE);
    } else {
        while (L->length != 0) {
            deleteFront(L);
        }
    }
}

void set(List L, int x) {
    if (L == NULL) {
        fprintf(stderr, "Set is called on NULL\n");
        exit(EXIT_FAILURE);
    } 
    else {
        L->cursor->data = x;
    }
}

void moveFront(List L) {
    if (L == NULL) {
        fprintf(stderr, "moveFront called on NULL\n");
        exit(EXIT_FAILURE);
    } else if (L->length > 0) {
        L->cursor = L->front;
        L->index = 0;
    }
}

void moveBack(List L) {
    if (L == NULL) {
        fprintf(stderr, "moveBack called on NULL\n");
        exit(EXIT_FAILURE);
    } 
    if (L->length > 0) {
        L->cursor = L->back;


        L->index = L->length - 1;
    }
}

void movePrev(List L) {
    if (L == NULL) {
        fprintf(stderr, "movePrev called on NULL\n");
        exit(EXIT_FAILURE);
    } 
    if (L->cursor != NULL) {
        L->cursor = L->cursor->prev;
        L->index--;
    }
}

void moveNext(List L) {
    if (L == NULL) {
        fprintf(stderr, "moveNext called on NULL\n");
        exit(EXIT_FAILURE);
    } else if (L->cursor != NULL) {
        L->cursor = L->cursor->next;
        if (L->cursor == NULL) {
            L->index = -1;  //reset to -1 if it is at the end of the list 
        }
        else {
        L->index++;
        }
    }
}

void prepend(List L, int x) {
    if (L == NULL) {
        fprintf(stderr, "prepend called on NULL\n");
        exit(EXIT_FAILURE);
    } 
    
    Node N = newNode(x);

    if (L->front == NULL) {
        L->front = N;
        L->back = N;
    } else {
        N->next = L->front;  
        L->front->prev = N; 
        L->front = N; 
    }
    L->length++;
    if (L->index != -1) {


        L->index++; 
    }
}

void append(List L, int x) {
    if (L == NULL) {
        fprintf(stderr, "append called on NULL\n");
        exit(EXIT_FAILURE);
    } else {
        Node N = newNode(x);
        if (L->length == 0) {
            L->front = L->back = N;
        } else {
            L->back->next = N;
            N->prev = L->back;
            L->back = N;
        }


        L->length++;
    }
}

void insertBefore(List L, int x) {
    if (L == NULL) {
        fprintf(stderr, "insertBefore called on NULL\n");
        exit(EXIT_FAILURE);
    } 
    //Node N = newNode(x);

    if (L->cursor == L->front) {
        prepend(L, x);
    } else {
        Node N = newNode(x);
        N->next = L->cursor;          
        N->prev = L->cursor->prev;     
        L->cursor->prev->next = N;   
        L->cursor->prev = N;     
        L->length++;


 	L->index++;	
    }
}

void insertAfter(List L, int x) {
    if (L == NULL) {
        fprintf(stderr, "insertAfter called on NULL\n");
        exit(EXIT_FAILURE);
    } 
    if (L->cursor == NULL) {
        fprintf(stderr, "insertAfter called on NULL\n");
        exit(EXIT_FAILURE);
    } 
    Node N = newNode(x);

    if (L->cursor == L->back) {
        L->back->next = N;


        N->prev = L->back;
        L->back = N;
    } else {
        N->next = L->cursor->next;
        N->prev = L->cursor;
        L->cursor->next->prev = N;
        L->cursor->next = N;
    }


    L->length++;
}



void deleteFront(List L) {
    if (L == NULL || L->length == 0) {
        fprintf(stderr, "Standard Error: deleteFront is called on NULL reference\n");
        exit(EXIT_FAILURE);
    }
    Node N = L->front;

    if (L->length == 1) {
        L->front = NULL;
        L->back = NULL;
    } else {
        L->front = L->front->next;
        L->front->prev = NULL;
    }

    if (L->cursor == N) {
        L->cursor = NULL;
        L->index--;
    } else if (L->cursor != NULL) {
        L->index--;
        //-=1
    }

    freeNode(&N);
    L->length--;
}


void deleteBack(List L) {
    if (L == NULL) {
        fprintf(stderr, "deleteBack called on NULL\n");
        exit(EXIT_FAILURE);
    } else if (L->length == 0) {
        fprintf(stderr, "deleteBack called on empty list\n");
        exit(EXIT_FAILURE);
    } 
    Node N = L->back;

    if (L->length == 1) {
        L->front = NULL;
        L->back = NULL;
    } else {
        L->back = L->back->prev;  
        L->back->next = NULL;   
    }

    if (L->cursor == N) {


        L->cursor = NULL;     
        L->index = -1;         
    }
    freeNode(&N);
    L->length--;
}

void delete(List L) {
    
    if (L->cursor == NULL) {
        fprintf(stderr, "delete is called on NULL\n");
        exit(EXIT_FAILURE);
    } 
    Node N = L->cursor;
    if (L->cursor == L->front) {
        deleteFront(L);
    }
    else if (L->cursor == L->back) {
        deleteBack(L);
    }
    else { 
        L->cursor->prev->next = L->cursor->next;


        L->cursor->next->prev = L->cursor->prev;
        freeNode(&N);  
        L->length--;
    }
    L->cursor = NULL;
    L->index = -1;
}

void printList(FILE* out, List L, char** lines) {
    Node N = L->front;
    while (N != NULL) {
        fprintf(out, "%s\n", lines[N->data]);
        N = N->next;    
    }
}

List copyList(List L) {
    if (L == NULL) {
        fprintf(stderr, "copyList called on NULL\n");
        exit(EXIT_FAILURE);
    } else {
        List C = newList();
        Node N = L->front;
        while (N != NULL) {
            append(C, N->data);
            N = N->next;
        }
        return C;
    }
}

List concatList(List A, List B) {
    if (A == NULL || B == NULL) {
        fprintf(stderr, "concatList called on NULL\n");
        exit(EXIT_FAILURE);
    } else {
        List C = copyList(A);
        Node N = B->front;
        while (N != NULL) {
            append(C, N->data);
            N = N->next;
        }
        return C;
    }
}
