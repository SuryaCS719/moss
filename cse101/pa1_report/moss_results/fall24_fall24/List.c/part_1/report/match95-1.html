#include <stdio.h>
#include <stdlib.h>
#include "List.h"

typedef struct NodeObj {
    int data;
    struct NodeObj* next;
    struct NodeObj* prev;
} NodeObj;

typedef NodeObj* Node;

typedef struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
    int index;
} ListObj;

Node newNode(int data) {
    Node N = malloc(sizeof(NodeObj));
    N->data = data;
    N->prev = NULL;
    N->next = NULL;
    return N;
}



void freeNode(Node* pN) {
    if (pN != NULL && *pN != NULL) {
        free(*pN);
        *pN = NULL;
    }
}
List newList(void) {
    List L = calloc(1, sizeof(ListObj));
    return L;
}
void freeList(List* pL) {
    if (pL != NULL && *pL != NULL) {
        while (length(*pL) > 0) {
            deleteFront(*pL); 
        }
        free(*pL); 
        *pL = NULL; 
    }
}
int length(List L) {
    return L->length;
}

int index(List L) {
    return L->cursor == NULL ? -1 : L->index;
}

int front(List L) {
    return L->front->data;
}

int back(List L) {
    return L->back->data;
}

int prev (List L){
    return L->cursor->prev->data;
}

int next (List L){
    return L->cursor->next->data;
}

int get(List L) {
    return L->cursor->data;
}

bool equals(List A, List B) {
    if (A == NULL || B == NULL) return false; // Check if either list is null
    
    if (A->length != B->length) return false;
    Node C = A->front;
    Node D = B->front;
    while (D != NULL){
	    if (C->data != D->data) return false;
	    C = C->next;
	    D = D->next;
    }
    return true;
}

void clear (List L) {
    while (L->length > 0){
	    deleteFront(L);}
    L->cursor = NULL;
    L->index = -1;
}
void set (List L, int x) {


    if (L->length > 0 && L->index > 0) L->cursor->data = x;
}

void moveFront (List L) {
    if (L->length > 0){
	    L->cursor = L->front;
	    L->index = 0;
	    return;
    }

}
void moveBack (List L) {
    if (L->length > 0){
        L->cursor = L->back;
	L->index = L->length - 1;
    }
}
void movePrev (List L) {
    if (L->cursor != L->front && L->cursor != NULL){
	    L->cursor = L->cursor->prev;
	    L->index--;
	    return;
   } if (L->cursor == L->front && L->cursor != NULL){
	    L->cursor = NULL;
	    L->index = -1;
	    return;
    }
}

void moveNext (List L) {
    if (L->cursor != L->back && L->cursor != NULL){
            L->cursor = L->cursor->next;
            L->index = L->index + 1;
	    return;
    }
    if (L->cursor == L->back && L->cursor != NULL){
            L->cursor = NULL;
	    L->index = -1;
            return;
	    }
}
void prepend (List L, int x) {
    Node newN = newNode(x); 
    if (L->length == 0){
	    L->front = newN;
	    L->back = newN;
	    newN->prev = NULL;
	    newN->next = NULL;
	    
    }
    else{
	    newN->next = L->front;
	    newN->prev = NULL;
	    L->front->prev = newN;
	    L->front = newN;
	    if (L->index != -1 && L->cursor != NULL){
		    L->index++;
		    
	    }
	    }
    L->length++;
}

void append (List L, int x) {
    Node newN = newNode(x);
    if (L->length == 0){
            L->front = newN;
            L->back = newN;
            newN->prev = NULL;
            newN->next = NULL;
    }
    else{
            newN->next = NULL;
            newN->prev = L->back;
            L->back->next = newN;
            L->back = newN;
    }
    L->length++;
}

    if (L == NULL || L->cursor == NULL || L->index < 0){

    if (L->length > 0 && L->index >= 0) {

            N->prev = L->cursor->prev;
            L->cursor->prev->next = N;

void deleteFront (List L){
    if (L->length > 0){
            Node dN = L->front;
            L->front = L->front->next;
            if (L->front != NULL){
                L->front->prev = NULL;
           }else {
                L->back = NULL;
           }
           if (L->cursor == dN) {           
                L->cursor = NULL;
                L->index = -1;
           }else{
                if (L->index > 0) L->index--;
           }
           free(dN);
           L->length--;

             }
}
void deleteBack(List L) {
    if (L->length > 0){
            Node ddN = L->back;
            L->back = L->back->prev;
            if (L->back != NULL){
                L->back->next = NULL;
           }else {
                L->front = NULL;
           }
           if (L->cursor == ddN) {
                L->cursor = NULL;
                L->index = -1;
           }else{
                if (L->index > 0) L->index--;
           }
           free(ddN);
           L->length--;

             }
}

void delete(List L){
    if (L == NULL || L->cursor == NULL || L->index < 0){
	    exit(EXIT_FAILURE);}
    else{
        Node cN = L->cursor;
	if (L->cursor == L->back) deleteBack(L);
	else if(L->cursor == L->front) deleteFront(L);
	else{
		cN->prev->next = cN->next;
		cN->next->prev = cN->prev;
		free(cN);
		L->length--;
	}
     
     L->cursor = NULL;
     L->index = -1;
}
}

void printList(FILE* out, List L){
    Node current = L->front;  
    while (current != NULL) {
        fprintf(out, "%d ", current->data);  
        current = current->next; 
    }

    fprintf(out, "\n");  
}
