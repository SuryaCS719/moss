#include <stdio.h>
#include <stdlib.h>
#include "List.h"

// Node structure
typedef struct NodeObj {
    int data;
    struct NodeObj* next;
    struct NodeObj* prev;
} NodeObj;

// Node type
typedef NodeObj* Node;

// List structure
typedef struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
    int index;   // -1 when cursor is undefined
} ListObj;

// Constructors-Destructors

// newNode() creates a new node
Node newNode(int data) {
    Node N = malloc(sizeof(NodeObj));
    N->data = data;
    N->next = NULL;
    N->prev = NULL;
    return N;
}

// freeNode() frees a node
void freeNode(Node* pN) {
    if (pN != NULL && *pN != NULL) {
        free(*pN);
        *pN = NULL;
    }
}

// newList() creates and returns a new empty List.
List newList(void) {
    List L = malloc(sizeof(ListObj));
    L->front = L->back = L->cursor = NULL;
    L->length = 0;
    L->index = -1;
    return L;
}

// deleteFront() deletes front element. Pre: length() > 0
void deleteFront(List L) {
    if (L->length > 0) {
        Node N = L->front;
        printf("Freeing node with data: %d\n", N->data);
        if (L->length == 1) {
            freeNode(&N);


            L->front = L->back = NULL;
        } else {
            L->front = L->front->next;
            L->front->prev = NULL;
            freeNode(&N);
        }
        L->length--;
        if (L->index != -1) L->index--;
    } else {
        exit(EXIT_FAILURE);  // Error: deleteFront() called on empty list
    }
}

// freeList() frees all heap memory associated with *pL, and sets *pL to NULL.
void freeList(List* pL) {
    if (pL != NULL && *pL != NULL) {
        while (length(*pL) > 0) {
            deleteFront(*pL);
        }
        free(*pL);
        *pL = NULL;
    }
}

// Access functions

// length() returns the number of elements in L.
int length(List L) {
    return L->length;
}

// index() returns the index of the cursor element if defined, -1 otherwise.
int index(List L) {
    return L->index;
}

// front() returns front element of L. Pre: length() > 0
int front(List L) {
    if (L->length > 0) {
        return L->front->data;
    }
    exit(EXIT_FAILURE);  // Error: calling front() on empty list
}

// back() returns back element of L. Pre: length() > 0
int back(List L) {
    if (L->length > 0) {
        return L->back->data;
    }
    exit(EXIT_FAILURE);  // Error: calling back() on empty list
}

// get() returns cursor element of L. Pre: length() > 0, index() >= 0
int get(List L) {
    if (L->length > 0 && L->index >= 0) {
        return L->cursor->data;
    }
    exit(EXIT_FAILURE);  // Error: calling get() on invalid cursor
}

// equals() returns true iff Lists A and B contain the same sequence of elements.
bool equals(List A, List B) {
    if (A->length != B->length) return false;
    Node nA = A->front;
    Node nB = B->front;
    while (nA != NULL && nB != NULL) {
        if (nA->data != nB->data) return false;
        nA = nA->next;
        nB = nB->next;
    }
    return true;
}

// Manipulation procedures

// clear() resets L to its original empty state.
void clear(List L) {
    while (L->length > 0) {
        deleteFront(L);
    }
    L->cursor = NULL;
    L->index = -1;
}

// set() overwrites the cursor element's data with x. Pre: length() > 0, index() >= 0
void set(List L, int x) {
    if (L->length > 0 && L->index >= 0) {
        L->cursor->data = x;
    } else {
        exit(EXIT_FAILURE);  // Error: calling set() on invalid cursor
    }
}

// moveFront() moves cursor to the front element if L is non-empty.
void moveFront(List L) {
    if (L->length > 0) {
        L->cursor = L->front;
        L->index = 0;
    }
}

// moveBack() moves cursor to the back element if L is non-empty.
void moveBack(List L) {
    if (L->length > 0) {
        L->cursor = L->back;
        L->index = L->length - 1;
    }
}

// movePrev() moves cursor one step towards the front if it's not at front.
void movePrev(List L) {
    if (L->cursor != NULL && L->index > 0) {
        L->cursor = L->cursor->prev;
        L->index--;
    } else if (L->cursor != NULL && L->index == 0) {
        L->cursor = NULL;
        L->index = -1;
    }
}

// moveNext() moves cursor one step towards the back if it's not at back.
void moveNext(List L) {
    if (L->cursor != NULL && L->index < L->length - 1) {
        L->cursor = L->cursor->next;
        L->index++;
    } else if (L->cursor != NULL && L->index == L->length - 1) {
        L->cursor = NULL;
        L->index = -1;
    }
}

// prepend() inserts new element before front element.
void prepend(List L, int x) {
    Node N = newNode(x);
    if (L->length == 0) {
        L->front = L->back = N;
    } else {
        N->next = L->front;
        L->front->prev = N;
        L->front = N;
    }
    L->length++;
    if (L->index != -1) L->index++;
}

// append() inserts new element after back element.
void append(List L, int x) {
    Node N = newNode(x);
    if (L->length == 0) {
        L->front = L->back = N;
    } else {
        N->prev = L->back;
        L->back->next = N;
        L->back = N;
    }
    L->length++;
}

// insertBefore() inserts new element before cursor. Pre: length() > 0, index() >= 0


void insertBefore(List L, int x) {
    if (L->length > 0 && L->index >= 0) {
        Node N = newNode(x);
        if (L->cursor == L->front) {
            prepend(L, x);
        } else {
            N->next = L->cursor;
            N->prev = L->cursor->prev;
            L->cursor->prev->next = N;
            L->cursor->prev = N;
            L->length++;
            L->index++;
        }
    } else {
        exit(EXIT_FAILURE);  // Error: insertBefore() called on invalid cursor
    }
}

// insertAfter() inserts new element after cursor. Pre: length() > 0, index() >= 0


void insertAfter(List L, int x) {
    if (L->length > 0 && L->index >= 0) {
        Node N = newNode(x);
        if (L->cursor == L->back) {
            append(L, x);
        } else {
            N->prev = L->cursor;
            N->next = L->cursor->next;
            L->cursor->next->prev = N;
            L->cursor->next = N;
            L->length++;
        }
    } else {


        exit(EXIT_FAILURE);  // Error: insertAfter() called on invalid cursor
    }
}

// deleteBack() deletes back element. Pre: length() > 0
void deleteBack(List L) {
    if (L->length > 0) {
        Node N = L->back;
        if (L->length == 1) {
            L->front = L->back = NULL;
        } else {
            L->back = L->back->prev;
            L->back->next = NULL;
        }
        freeNode(&N);
        L->length--;
        if (L->index == L->length) {
            L->cursor = NULL;
            L->index = -1;
        }
    } else {
        exit(EXIT_FAILURE);  // Error: deleteBack() called on empty list
    }
}

// delete() deletes cursor element. Pre: length() > 0, index() >= 0
void delete(List L) {
    if (L->length > 0 && L->index >= 0) {
        if (L->cursor == L->front) {
            deleteFront(L);
        } else if (L->cursor == L->back) {
            deleteBack(L);
        } else {
            Node N = L->cursor;
            L->cursor->prev->next = L->cursor->next;
            L->cursor->next->prev = L->cursor->prev;
            freeNode(&N);
            L->length--;
            L->cursor = NULL;
            L->index = -1;
        }
    } else {
        exit(EXIT_FAILURE);  // Error: delete() called on invalid cursor
    }
}

// Other operations

// printList() prints space-separated sequence of elements to out.
void printList(FILE* out, List L) {
    Node N = L->front;
    while (N != NULL) {
        fprintf(out, "%d ", N->data);
        N = N->next;
    }
    fprintf(out, "\n");
}

// copyList() returns a copy of L.
List copyList(List L) {
    List copy = newList();
    Node N = L->front;
    while (N != NULL) {
        append(copy, N->data);
        N = N->next;
    }
    return copy;
}
