#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <assert.h>
#include <stdbool.h>
#include "List.h"


//underlying data structure is a doubly linked list



//Constructor, Deconstructor for Node/List
typedef struct NodeObj* Node;

typedef struct NodeObj{
    int data;
    struct NodeObj* prev;
    struct NodeObj* next;
} NodeObj;


Node newNode(int data){
    Node N = malloc(sizeof(NodeObj));
    assert(N != NULL);
    N->data = data;
    N->prev = NULL;
    N->next = NULL;
    return(N);
}

void freeNode(Node* pN){
    if(pN != NULL && *pN != NULL){
        free(*pN);
        *pN = NULL;
    }
}

typedef struct ListObj{
    Node front;
    Node back;
    Node cursor;
    int length;
    int index;
} ListObj;

List newList(){
    List L = malloc(sizeof(ListObj));
    assert(L != NULL);
    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->length = 0;
    L->index = -1;
    return(L);
}

void freeList(List* pL){
    if(pL != NULL && *pL != NULL){
        clear(*pL);
        free(*pL);
    }
}


//Access functions

int length(List L){
    if(L == NULL){
        fprintf(stderr, "Error: calling length() on NULL List\n");
        exit(EXIT_FAILURE);
    }


    return(L->length);
}


int index(List L){
    if(L == NULL){
        fprintf(stderr, "Error: calling index() on NULL List\n");
        exit(EXIT_FAILURE);
    }
    if(L->cursor == NULL){
        return -1;
    }
    return(L->index);
}

int front(List L){
    if(L == NULL){
        fprintf(stderr, "Error: calling front() on NULL List\n");
        exit(EXIT_FAILURE);
    }
    return(L->front->data);
}


int back(List L){
    if(L == NULL){
        fprintf(stderr, "Error: calling back() on NULL List\n");
        exit(EXIT_FAILURE);
    }
    return(L->back->data);
}


int get(List L){
    if(L == NULL){
        fprintf(stderr, "Error: calling get() on NULL List\n");
        exit(EXIT_FAILURE);
    }
    return(L->cursor->data);
}


////Manipulation procedures

bool equals(List A, List B){
    if (A == NULL || B == NULL) {
        fprintf(stderr, "Error: calling equals() on NULL List\n");
        exit(EXIT_FAILURE);
    }
    if (A->length != B->length) {
        return false;
    }
    Node nodeA = A->front;
    Node nodeB = B->front;
    while (nodeA != NULL && nodeB != NULL) {
        if (nodeA->data != nodeB->data) {
            return false;
        }
        nodeA = nodeA->next;
        nodeB = nodeB->next;
    }
    return true;
}


void clear(List L){


    if(L == NULL){
        return;
    }
    while(L->front != NULL){
        deleteFront(L);
    }
}

void set(List L, int x){
    if(L == NULL){
        fprintf(stderr, "Error: calling set() on NULL List\n");
        exit(EXIT_FAILURE);
    }
    L->cursor->data = x;
}

void moveFront(List L){
    if(L == NULL){
        fprintf(stderr, "Error: calling moveFront() on NULL List\n");
        exit(EXIT_FAILURE);
    }


    if(length(L) > 0){
        L->index = 0;
        L->cursor = L->front;
    }
}

void moveBack(List L){
    if(L == NULL){
        fprintf(stderr, "Error: calling moveBack() on NULL List\n");
        exit(EXIT_FAILURE);
    }

    if(length(L) > 0){
        L->index = length(L) - 1;
        L->cursor = L->back;
    }
}




void movePrev(List L){
    if(L->cursor != L->front){
        L->cursor = L->cursor->prev;
        L->index--;
    } else if ((L->cursor != NULL) && (L->cursor == L->front)) {
        L->cursor = NULL;
        L->index = -1;
    }
}

void moveNext(List L){
    if(L->cursor != L->back){
        L->cursor = L->cursor->next;
        L->index++;
    } else if ((L->cursor != NULL) && (L->cursor == L->back)) {
        L->cursor = NULL;


        L->index = -1;
    }
}

void prepend(List L, int x){
    if(L == NULL){
        fprintf(stderr, "Error: calling prepend() on NULL List\n");
        exit(EXIT_FAILURE);
    }
    Node N = newNode(x);
    N->prev = NULL;
    if(L->length == 0){
        L->front = N;
        L->back = N;
    } else {
        L->front->prev = N;
        N->next = L->front;


        L->front = N;
        if (L->index != -1) {
            L->index++;
        }
    }
    L->length++;
}


void append(List L, int x){
    if (L == NULL) {
        fprintf(stderr, "Error: calling append() on NULL List\n");
        exit(EXIT_FAILURE);
    }
    Node N = newNode(x);
    //empty list case
    if(L->length == 0){
        L->front = N;
        L->back = N;
    } else {
        L->back->next = N;
        N->prev = L->back;
        L->back = N;
    }
    L->length++;
}

void insertBefore(List L, int x){
    if(L == NULL){
        fprintf(stderr, "Error: calling insertBefore() on NULL List\n");
        exit(EXIT_FAILURE);
    }
    Node N = newNode(x);
    if(L->cursor == L->front){
        N->next = L->front;
        L->front->prev = N;
        L->front = N;
    } else {
        N->next = L->cursor;
        N->prev = L->cursor->prev;
        L->cursor->prev->next = N;
        L->cursor->prev = N;
    }
    L->length++;
    L->index++;
}


void insertAfter(List L, int x){
    if(L == NULL) {
        fprintf(stderr, "Error: calling insertAfter() on NULL List\n");
        exit(EXIT_FAILURE);
    }
    Node N = newNode(x);
    if(L->cursor == L->back){
        N->prev = L->back;
        L->back->next = N;
        L->back = N;
    } else {
        N->prev = L->cursor;
        N->next = L->cursor->next;
        L->cursor->next->prev = N;
        L->cursor->next = N;
    }
    L->length++;
    //L->index++;
}


void deleteFront(List L){
    if(L == NULL){
        fprintf(stderr, "Error: calling deleteFront() on NULL List\n");
        exit(EXIT_FAILURE);
    }
    if(length(L) > 0){
        Node N = L->front;
        if(length(L) > 1){
            L->front = L->front->next;
            L->front->prev = NULL;
        } else {
            L->front = NULL;
            L->back = NULL;
        }
        if(L->index == 0){
            L->cursor = NULL;
            L->index = -1;
        } else if (L->index != -1) {


            L->index--;
        }
        freeNode(&N);
        L->length--;
    }
}

void deleteBack(List L){
    if(L == NULL){
        fprintf(stderr, "Error: calling deleteBack() on NULL List\n");
        exit(EXIT_FAILURE);
    }
    if(length(L) > 0){
        Node N = L->back;
        if(length(L) > 1){
            L->back = L->back->prev;


            L->back->next = NULL;
        } else {
            L->front = NULL;
            L->back = NULL;
        }

        if(L->cursor == N){
            L->cursor = NULL;
            L->index = -1;
        }
        freeNode(&N);
        L->length--;
    }
}

void delete(List L){
    if(L == NULL || L->cursor == NULL){
        fprintf(stderr, "Error: calling delete() on NULL List\n");
        exit(EXIT_FAILURE);
    }
    if(L->cursor == L->front){
        deleteFront(L);
    } else if(L->cursor == L->back){
        deleteBack(L);
    } else {
        Node N = L->cursor;
        L->cursor->prev->next = L->cursor->next;
        L->cursor->next->prev = L->cursor->prev;
        L->cursor = NULL;
        L->index = -1;
        freeNode(&N);


        L->length--;
    }
}

//Other operations

void printList(FILE* out, List L){
    if(L == NULL){
        fprintf(stderr, "Error: calling printList() on NULL List\n");
        exit(EXIT_FAILURE);
    }
    Node N = L->front;
    while(N != NULL){
        fprintf(out, "%d ", N->data);
        N = N->next;
    }
}

List copyList(List L){
    if(L == NULL){
        fprintf(stderr, "Error: calling copyList() on NULL List\n");
        exit(EXIT_FAILURE);
    }
    List copy = newList();
    Node N = L->front;
    while(N != NULL){
        append(copy, N->data);
        N = N->next;
    }
    copy->length = L->length;
    copy->index = -1; 
    copy->cursor = NULL;
    return(copy);
}
