#include "List.h"
#include <stdbool.h>
#include <stdio.h>
#include <stdlib.h>



typedef struct NodeObj* Node;



typedef struct NodeObj{
    int data;
    Node next;
    Node prev;
} NodeObj;

typedef struct ListObj{
	Node head;
	Node tail;
    Node cursor;
    int length;
    int index;
} ListObj;

void deleteNode(Node *n){
    free(*n);
    *n = NULL;
}

List newList(void) {
    List l = (List)malloc(sizeof(ListObj));
    if (l == NULL) {
        fprintf(stderr, "List Error: list initialization List() failing due to NULL malloc\n");
        return NULL;
    }
    l->head = NULL;
    l->tail = NULL;
    l->cursor = NULL;
    l->length = 0;
    l->index = -1;
    return l;
}



void freeList(List *pL) {
    if (pL != NULL && *pL != NULL) {

        clear(*pL);
        free(*pL);
        *pL = NULL;
    }
    return;
}

int length(List L){
    if( L == NULL ){
      fprintf(stderr, "List Error: calling length() on NULL List reference\n");
      exit(EXIT_FAILURE);
    }
    return L->length;
}

int index(List L){
    if( L == NULL ){
      fprintf(stderr, "List Error: calling index() on NULL List reference\n");
      exit(EXIT_FAILURE);
    }
    
    return L->index;
}

int front(List L){
    if( L == NULL ){
      fprintf(stderr, "List Error: calling front() on NULL List reference\n");
      exit(EXIT_FAILURE);
    }
    if(length(L) == 0){
      fprintf(stderr, "List Error: calling front() on an empty List\n");
      exit(EXIT_FAILURE);
    }
    return (L->head)->data;
}

int back(List L){
    if( L == NULL ){
      fprintf(stderr, "List Error: calling back() on NULL List reference\n");
      exit(EXIT_FAILURE);
    }
    if(length(L) <= 0){
        fprintf(stderr, "List Error: calling back() on an empty List\n");
        exit(1);
    }
    return (L->tail)->data;
}

int get(List L){
    if( L == NULL ){
      fprintf(stderr, "List Error: calling get() on NULL List reference\n");
      exit(EXIT_FAILURE);
    }
    return (L->cursor)->data;
}

bool equals(List A, List B){
    if( A == NULL ){
      fprintf(stderr, "List Error: calling equals() on NULL List reference \"A\"\n");
      exit(EXIT_FAILURE);
    }
    if( B == NULL ){
      fprintf(stderr, "List Error: calling equals() on NULL List reference \"B\"\n");
      exit(EXIT_FAILURE);
    }


    if (A->length != B->length){
        return false;
    }
    if(A->length == 0 && B->length == 0){
        return true;
    }
    
    moveFront(A);
    moveFront(B);
    while(1){
        if(A->cursor == NULL && B->cursor == NULL){
            return true;
        }
        if((A->cursor)->data != (B->cursor)->data){
            return false;
        }
        moveNext(A);
        moveNext(B);
    }
}

void clear(List L){
    if( L == NULL ){
      fprintf(stderr, "List Error: calling set() on NULL List reference\n");
      exit(EXIT_FAILURE);
    }
    while(length(L) > 0){
        deleteBack(L);
    }

    return;
}

void set(List L, int x){
    if( L == NULL ){
      fprintf(stderr, "List Error: calling set() on NULL List reference\n");
      exit(EXIT_FAILURE);
    }
    if(length(L) <= 0){
        fprintf(stderr, "List Error: calling set() on an empty List\n");
        exit(1);
    }
    
    (L->cursor)->data = x;

}

void moveFront(List L){
    if( L == NULL ){
      fprintf(stderr, "List Error: calling moveFront() on NULL List reference\n");
      exit(EXIT_FAILURE);
    }
    if(length(L) <= 0){
        fprintf(stderr, "List Error: calling moveFront() on an empty List\n");
        exit(1);
    }

    L->cursor = L->head; 
    L->index = 0;
}

void moveBack(List L){
    if( L == NULL ){
      fprintf(stderr, "List Error: calling moveBack() on NULL List reference\n");
      exit(EXIT_FAILURE);
    }
    if(length(L) <= 0){
        fprintf(stderr, "List Error: calling moveBack() on an empty List\n");
        exit(1);
    }

    L->cursor = L->tail;
    L->index = L->length-1;
}

void movePrev(List L){
    if( L == NULL ){
      fprintf(stderr, "List Error: calling movePrev() on NULL List reference\n");
      exit(EXIT_FAILURE);
    }
    
    if(L->cursor == NULL){
        return;
    }

    if((L->cursor)->prev == NULL){
        L->cursor = NULL;
        L->index = -1;
        return;
    }
    
    L->cursor = (L->cursor)->prev;
    L->index -= 1;
}

void moveNext(List L){
    if( L == NULL ){
      fprintf(stderr, "List Error: calling movePrev() on NULL List reference\n");
      exit(EXIT_FAILURE);
    }
    
    if(L->cursor == NULL){
        return;
    }

    if((L->cursor)->next == NULL){
        L->cursor = NULL;
        L->index = -1;
        return;
    }
    
    L->cursor = (L->cursor)->next;
    L->index += 1;
}

void prepend(List L, int x){
    if( L == NULL ){
      fprintf(stderr, "List Error: calling prepend() on NULL List reference\n");
      exit(EXIT_FAILURE);
    }

    Node rep = (Node) malloc(sizeof(NodeObj));

    if (rep == NULL) {
        fprintf(stderr, "List Error: malloc failed in prepend()\n");
        exit(EXIT_FAILURE);
    }

    rep->prev = NULL;
    rep->data = x;

    if(L->head == NULL){
        rep->next = NULL;
        L->head = rep;
        L->tail = rep;
        L->length +=1;
        L->index +=1;
        return;
    }
    else {
        rep->next = L->head;
        (L->head)->prev = rep;
        L->head = rep;
        L->length +=1;
        L->index +=1;
    }

    return;
}

void append(List L, int x){
    if( L == NULL ){
      fprintf(stderr, "List Error: calling append() on NULL List reference\n");
      exit(EXIT_FAILURE);
    }

    Node rep = (Node) malloc(sizeof(NodeObj));

    if (rep == NULL) {
        fprintf(stderr, "List Error: malloc failed in append()\n");
        exit(EXIT_FAILURE);
    }

    rep->next = NULL;
    rep->data = x;

    if(L->tail == NULL){
        rep->prev = NULL;
        L->head = rep;
        L->tail = rep;
        L->length +=1;
        return;
    }
    else {
        rep->prev = L->tail;
        (L->tail)->next = rep;
        L->tail = rep;
        L->length +=1;
    }
    return;
}

void insertBefore(List L, int x){
    if( L == NULL ){
      fprintf(stderr, "List Error: calling insertBefore() on NULL List reference\n");
      exit(EXIT_FAILURE);
    }
    if(length(L) <= 0){
        fprintf(stderr, "List Error: calling insertBefore() on an empty List\n");
        exit(1);
    }
    if(L->cursor == NULL){
        fprintf(stderr, "List Error: calling insertBefore() with a NULL cursor Node\n");
        exit(1);
    }

    //if at start, prepend
    if(L->index == 0){
        prepend(L, x);
        return;
    }

    Node rep = (Node) malloc(sizeof(NodeObj));

    if (rep == NULL) {
        fprintf(stderr, "List Error: malloc failed in insertBefore()\n");
        exit(EXIT_FAILURE);
    }

    rep->data = x;
    rep->next = L->cursor;
    rep->prev = (L->cursor)->prev;

    ((L->cursor)->prev)->next = rep;
    (L->cursor)->prev = rep;



    L->index +=1;
    L->length +=1;
}

void insertAfter(List L, int x){

    if( L == NULL ){
      fprintf(stderr, "List Error: calling insertAfter() on NULL List reference\n");
      exit(EXIT_FAILURE);
    }
    if(length(L) <= 0){
        fprintf(stderr, "List Error: calling insertAfter() on an empty List\n");
        exit(1);
    }
    if(L->cursor == NULL){
        fprintf(stderr, "List Error: calling insertAfter() with a NULL cursor Node\n");
        exit(1);
    }

    //if at start, append
    if(L->index == length(L)-1){
        append(L, x);
        return;
    }

    Node rep = (Node) malloc(sizeof(NodeObj));

    if (rep == NULL) {
        fprintf(stderr, "List Error: malloc failed in insertAfter()\n");
        exit(EXIT_FAILURE);
    }

    rep->data = x;
    rep->prev = L->cursor;
    rep->next = (L->cursor)->next;

    ((L->cursor)->next)->prev = rep;
    (L->cursor)->next = rep;

    L->length +=1;
   
}

void deleteFront(List L){
if( L == NULL ){
      fprintf(stderr, "List Error: calling deleteBack() on NULL List reference\n");
      exit(EXIT_FAILURE);
    }
    if(length(L) <= 0){
        fprintf(stderr, "List Error: calling deleteBAck() on an empty List\n");
        exit(1);
    }
    
    Node delete = L->head;
    
    // if cursor is in the front
    if(L->cursor == L->head){
        L->cursor = NULL;
        L->index = -1;
    }
    else{
        L->index -=1;
    }


    if(length(L) > 1){
        L->head = (L->head)->next;
        (L->head)->prev = NULL;
    } else{
        L->head = NULL;
        L->tail = NULL;
    }
    // remove old back node
    deleteNode(&delete);
    L->length -= 1;

    
}

void deleteBack(List L){
    if( L == NULL ){
      fprintf(stderr, "List Error: calling deleteBack() on NULL List reference\n");
      exit(EXIT_FAILURE);
    }
    if(length(L) <= 0){
        fprintf(stderr, "List Error: calling deleteBAck() on an empty List\n");
        exit(1);
    }
    
    Node delete = L->tail;
    
    // if cursor is in the back
    if(L->cursor == L->tail){
        L->cursor = NULL;
        L->index = -1;
    }

    if(length(L) > 1){
        L->tail = (L->tail)->prev;
        (L->tail)->next = NULL;
    } else{
        L->head = NULL;
        L->tail = NULL;
    }

    // remove old back node
    deleteNode(&delete);
    L->length -= 1;
}

void delete(List L){
    if( L == NULL ){
      fprintf(stderr, "List Error: calling delete() on NULL List reference\n");
      exit(EXIT_FAILURE);
    }

    if (length(L) <= 0 || index(L) < 0) {
        fprintf(stderr, "List Error: calling delete() on an empty List or with undefined cursor\n");
        exit(EXIT_FAILURE);
    }

    Node delete = L->cursor;

    
    if (L->cursor == L->head) {
        deleteFront(L);
    }
   
    else if (L->cursor == L->tail) {
        deleteBack(L);
    }
    
    else {


        ((L->cursor)->prev)->next = (L->cursor)->next;  
        ((L->cursor)->next)->prev = (L->cursor)->prev;  
        free(delete);  
        L->length -= 1; 
    }

    // Reset the cursor and index
    L->cursor = NULL;
    L->index = -1;


}

// Other Operations
void printList(FILE* out, List L){
    if(L == NULL){
        fprintf(stderr, "List Error: calling printList() on a NULL List\n");
        exit(EXIT_FAILURE);
    }

    if(length(L) == 0){
        fprintf(out, "\n");
        return;
    }
    
    //save spot of cursor
    int save = index(L);

    //start printing
    moveFront(L);
    while (index(L)>=0){
        int x = get(L);
        if(index(L) == length(L)-1){
            fprintf(out, "%d\n", x);
        }
        else{        
            fprintf(out, "%d ", x);
        }
        moveNext(L);
    }

    //go back to saved cursor spot
    if(save != -1){
        moveFront(L);
        while(index(L) != save){
            moveNext(L);
        }
    }
}

List copyList(List L){

    List two = newList();

    //save spot of cursor
    int save = index(L);

    if(length(L)== 0){
        return two;
    }

    moveFront(L);
    while(index(L)>=0){
        append(two, get(L));
        moveNext(L);
    }

    if(save != -1){
        moveFront(L);
        while(index(L) != save){
            moveNext(L);
        }
    }

    return two;
}
