#include <stdio.h>
#include <stdlib.h>
#include "List.h"

typedef struct NodeObj{
	int data;
	struct NodeObj* next;
	struct NodeObj* prev;
} NodeObj;

typedef NodeObj* Node;

typedef struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
    int index;
} ListObj;

List newList(void) {
    List L = malloc(sizeof(ListObj));
    L->front = NULL;
    L->back = NULL;
    L->cursor = NULL;
    L->length = 0;
    L->index = -1;
    return L;
}

void freeList(List* pL) {
    if (pL != NULL && *pL != NULL) {
        while (length(*pL) > 0) {
            deleteFront(*pL);
        }
        free(*pL);
        *pL = NULL;
    }
}

int length(List L) {
    return L->length;
}



int index(List L) {
    return L->index;
}

int front(List L) {
    if (L->length > 0){
        return L->front->data;
    }
    else{
        printf("List Error: calling front() on an empty list\n");
        exit(1);
    }
}

int back(List L) {
    if (L->length > 0){
        return L->back->data;
    }
    else{
        printf("List Error: calling back() on an empty list\n");
        exit(1);
    }
}

int get(List L) {
    if (L->length > 0 && L->index >= 0) {
        return L->cursor->data;
    }
    else{
        printf("List Error: calling get() on undefined cursor\n");
        exit(1);
    }
}

void clear(List L) {
    while (L->length > 0) {
        deleteFront(L);    
    }
    L->cursor = NULL;
    L->index = -1;
}

void moveFront(List L) {
    if (L->length > 0) {
        L->cursor = L->front;
        L->index = 0;
    }
}

void moveBack(List L) {
    if (L->length > 0) {
        L->cursor = L->back;
        L->index = L->length - 1;
    }
}

void append(List L, int data) {
    Node N = malloc(sizeof(NodeObj));
    N->data = data;
    N->next = NULL;
    N->prev = L->back;

    if (L->length == 0) {
        L->front = L->back = N;
    } else {
        L->back->next = N;
        L->back = N;
    }
    L->length++;
}

void prepend(List L, int data) {
    Node N = malloc(sizeof(NodeObj));
    if (N == NULL) {
        printf("List Error: unable to allocate memory for new node\n");
        exit(1);
    }
    N->data = data;
    N->prev = NULL;
    N->next = L->front;

    if (L->length == 0) {
        L->front = L->back = N;
    } else {
        L->front->prev = N;
        L->front = N;
    }
    L->length++;

    if (L->index != -1) { // Check if cursor is defined
        L->index++;  // Increment cursor index since list head is now different
    }
}

void deleteFront(List L) {
    if (L->length > 0) {
        Node N = L->front;
        if (L->length == 1) {
            L->front = L->back = NULL;
            L->cursor = NULL;
            L->index = -1;
        } else {
            L->front = L->front->next;


            L->front->prev = NULL;
            if (L->index == 0) { // Cursor was at the front
                L->cursor = NULL;
                L->index = -1;
            } else if (L->index > 0) {
                L->index--; // Adjust index as the front has been removed
            }
        }
        free(N);
        L->length--;
    } else {
        printf("List Error: calling deleteFront() on an empty list\n");
        exit(1);
    }
}


void deleteBack(List L) {
    if (L->length > 0) {
        Node N = L->back;
        if (L->length == 1) {
            L->front = L->back = NULL;
            L->cursor = NULL;
            L->index = -1;
        } else {
            L->back = L->back->prev;


            L->back->next = NULL;
            if (L->index == L->length - 1) { // Cursor was at the back
                L->cursor = NULL;
                L->index = -1;
            }
        }
        free(N);
        L->length--;
    } else {
        printf("List Error: calling deleteBack() on an empty list\n");
        exit(1);
    }
}


void delete(List L) {
    if (L->length > 0 && L->index >= 0) {
        Node N = L->cursor;
        if (L->cursor == L->front) {
            deleteFront(L);
        } else if (L->cursor == L->back) {
            deleteBack(L);
        } else {
            L->cursor->prev->next = L->cursor->next;


            L->cursor->next->prev = L->cursor->prev;
            free(N);
            L->length--;
        }
        L->cursor = NULL;
        L->index = -1;
    } else {
        printf("List Error: calling delete() on undefined cursor or empty list\n");
        exit(1);
    }
}

void moveNext(List L) {
    if (L->cursor != NULL && L->cursor != L->back) {
        L->cursor = L->cursor->next;
        L->index++;
    } else if (L->cursor == L->back) {
        L->cursor = NULL;
        L->index = -1;
    }
}

void movePrev(List L) {
    if (L->cursor != NULL && L->cursor != L->front) {
        L->cursor = L->cursor->prev;
        L->index--;


    } else if (L->cursor == L->front) {
        L->cursor = NULL;
        L->index = -1;
    }
}

void insertBefore(List L, int data) {
    if (L->length > 0 && L->index >= 0) {
        if (L->cursor == L->front) {
            prepend(L, data);
        } else {
            Node N = malloc(sizeof(NodeObj));
            if (N == NULL) {
                printf("List Error: unable to allocate memory for new node\n");
                exit(1);
            }
            N->data = data;
            N->next = L->cursor;
            N->prev = L->cursor->prev;
            L->cursor->prev->next = N;
            L->cursor->prev = N;
            L->length++;
            L->index++;
        }
    } else {
        printf("List Error: calling insertBefore() on undefined cursor or empty list\n");
        exit(1);
    }
}



void insertAfter(List L, int data) {
    if (L->length > 0 && L->index >= 0) {
        if (L->cursor == L->back) {
            append(L, data);
        } else {
            Node N = malloc(sizeof(NodeObj));
            if (N == NULL) {
                printf("List Error: unable to allocate memory for new node\n");
                exit(1);
            }
            N->data = data;
            N->prev = L->cursor;
            N->next = L->cursor->next;
            L->cursor->next->prev = N;
            L->cursor->next = N;
            L->length++;
        }
    } else {
        printf("List Error: calling insertAfter() on undefined cursor or empty list\n");
        exit(1);
    }
}

        printf("List Error: calling equals() on NULL List reference\n");

    if (A->length != B->length) {
        return false;

        if (nodeA->data != nodeB->data) {
            return false;
        }
        nodeA = nodeA->next;
        nodeB = nodeB->next;
    }

    return true;
}
