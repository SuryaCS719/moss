// structs --------------------------------------------------------------------

// private Node type


typedef struct NodeObj* Node;

// private NodeObj type
typedef struct NodeObj{
   int data;
   Node next;
   Node prev;
} NodeObj;

// private ListObj type
typedef struct ListObj{
   Node front;
   Node back;
   int list_length;
   int cursor_index;
   Node cursor;
} ListObj;

// newList()
// Returns reference to new empty List object.
List newList(){
   List L;
   L = malloc(sizeof(ListObj));
   assert( L!=NULL );
   L->front = L->back = NULL;
   L->list_length = 0;
   L->cursor_index = -1;
   L->cursor = NULL;
   return(L);
}

// freeList()
// Frees all heap memory associated with List *pL, and sets *pL to NULL.
void freeList(List* pL){
   if(pL == NULL || *pL == NULL) {
        printf("ListError: calling freeList on NULL List reference");
        exit(EXIT_FAILURE);
   }
      clear(*pL);
      free(*pL);
      *pL = NULL;
}
//pre-condition function
void preConditions(List L, char* functionName, bool checkLength, bool checkCursor) {
    if(L ==NULL){
        printf("ListError: calling %s on NULL List reference", functionName);
        exit(EXIT_FAILURE);
    }
    if(checkLength) {
        if(!(length(L) > 0)) {
            printf("ListError: calling %s on empty List", functionName);
            exit(EXIT_FAILURE);
        }
    }
    if(checkCursor) {
        if(!(index(L) >= 0) || L->cursor == NULL) {
            printf("ListError: calling %s on List with NULL cursor", functionName);
            exit(EXIT_FAILURE);
        }
    }
}
// Access functions -----------------------------------------------------------

/// Returns the number of elements in L.
int length(List L){
    preConditions(L, "length", false, false);
    return(L->list_length);
}

// Returns index of cursor element if defined, -1 otherwise.
int index(List L) {
    preConditions(L, "index", false, false);
    return(L->cursor_index);
}

// Returns front element of L. Pre: length()>0
int front(List L){
    preConditions(L, "front", true, false);
    return(L->front->data);
}

// Returns back element of L. Pre: length()>0
int back(List L){
    preConditions(L, "back", true, false);
    return(L->back->data);
}

// Returns cursor element of L. Pre: length()>0, index()>=0
int get(List L) {
    preConditions(L, "get", true, true);
    return(L->cursor->data);
}
// Returns true if Lists A and B contain the same sequence of elements, returns false otherwise.
bool equals(List A, List B){
    if(A == NULL || B == NULL){
        printf("ListError: calling equals on NULL List reference");
        exit(EXIT_FAILURE);
    }
    if(length(A) != length(B)){
        return(false);
    }
    int i;
    int len = length(A);
    Node nodeA = A->front;
    Node nodeB = B->front;
    for (i = 0; i < len; i++) {
        if(nodeA->data != nodeB->data) {
            return(false);
        }
        else {
            nodeA = nodeA->next;
            nodeB = nodeB->next;
        }
    }
    return(true);
}

// Manipulation procedures ----------------------------------------------------

// Resets L to its original empty state.
void clear(List L) {
    preConditions(L, "clear", false, false);
    Node temp1 = L->front;
    while(temp1 != NULL) {
        Node temp2 = temp1->next;
        free(temp1);
        temp1 = temp2;
    }
    L->front = L->back = L->cursor = NULL;
    L->cursor_index = -1;
    L->list_length = 0;
}
 // Overwrites the cursor elementâ€™s data with x.
 // Pre: length()>0, index()>=0
void set(List L, int x) {
    preConditions(L, "set", true, true);
    L->cursor->data = x;
}
// If L is non-empty, sets cursor under the front element,
// otherwise does nothing.
void moveFront(List L){
    preConditions(L, "moveFront", false, false);
    if(length(L) > 0) {
        L->cursor = L->front;
        L->cursor_index = 0;
    }
}

// If L is non-empty, sets cursor under the back element,
// otherwise does nothing.
void moveBack(List L) {
    preConditions(L, "moveBack", false, false);
    if(length(L) > 0) {
        L->cursor = L->back;
        L->cursor_index = length(L) - 1;
    }
}

// If cursor is defined and not at front, move cursor one
// step toward the front of L; if cursor is defined and at
// front, cursor becomes undefined; if cursor is undefined
// do nothing
void movePrev(List L) {
    preConditions(L, "movePrev", true, false);
    if(index(L) >= 0 && L->cursor != NULL) {
        if(L->cursor == L->front) {
            L->cursor = NULL;
            L->cursor_index = -1;
        }
        else {
            L->cursor = L->cursor->prev;
            L->cursor_index--;
        }
    }
}
// If cursor is defined and not at back, move cursor one
// step toward the back of L; if cursor is defined and at
// back, cursor becomes undefined; if cursor is undefined
// do nothing
void moveNext(List L) {
    preConditions(L, "moveNext", true, false);
    if(index(L) >= 0 && L->cursor != NULL) {
        if(L->cursor == L->back) {
            L->cursor = NULL;
            L->cursor_index = -1;
        }
        else {
            L->cursor = L->cursor->next;
            L->cursor_index++;
        }
    }
}
// Insert new element into L. If L is non-empty,
// insertion takes place before front element.
void prepend(List L, int x) {
    preConditions(L, "prepend", false, false);
    Node new_front = calloc(1, sizeof(NodeObj));
    new_front->data = x;
    new_front->next = L->front;
    new_front->prev = NULL;

    if (L->list_length == 0) {
        L->back = new_front;
    } else {
        L->front->prev = new_front;
    }
    L->front = new_front;
    if (L->cursor_index >= 0) {
        L->cursor_index++;
    }
    L->list_length++;
}
// Insert new element into L. If L is non-empty,
// insertion takes place after back element.
void append(List L, int x) {
    preConditions(L, "append", false, false);
    Node new_back = calloc(1, sizeof(NodeObj));
    new_back->data = x;
    new_back->prev = L->back;
    new_back->next = NULL;

    if (L->list_length == 0) {
        L->front = new_back;
    } else {
        L->back->next = new_back;
    }
    L->back = new_back;
    L->list_length++;
}
// Insert new element before cursor.
// Pre: length()>0, index()>=0
void insertBefore(List L, int x) {
    preConditions(L, "insertBefore", true, true);
    Node new_node = calloc(1, sizeof(NodeObj));
    new_node->data = x;
    new_node->next = L->cursor;

    if (L->cursor == L->front) {
        L->front = new_node;
    }
    else {
        new_node->prev = L->cursor->prev;
        L->cursor->prev->next = new_node;
    }

    L->cursor->prev = new_node;
    L->list_length++;
    L->cursor_index++;
}
// Insert new element after cursor.
// Pre: length()>0, index()>=0
void insertAfter(List L, int x) {
    preConditions(L, "insertAfter", true, true);
    Node new_node = calloc(1, sizeof(NodeObj));
    new_node->data = x;
    new_node->prev = L->cursor;

    if (L->cursor == L->back) {
        L->back = new_node;
    }
    else {
        new_node->next = L->cursor->next;
        L->cursor->next->prev = new_node;
    }
    L->cursor->next = new_node;
    L->list_length++;
}
// Delete the front element. Pre: length()>0
void deleteFront(List L) {
    preConditions(L, "deleteFront", true, false);
    if(L->cursor == L->front) {
        L->cursor = NULL;
        L->cursor_index = -1;
    }
    else if(L->cursor_index >= 1){
        L->cursor_index--;
    }
    Node temp = L->front->next;
    free(L->front);
    L->front = temp;
    if (L->front != NULL) {
        L->front->prev = NULL;
    } else {
        L->back = NULL; // Handle case when list becomes empty
    }
    L->list_length--;
}
// Delete the back element. Pre: length()>0
void deleteBack(List L) {
    preConditions(L, "deleteBack", true, false);
    if(L->cursor == L->back) {
        L->cursor = NULL;
        L->cursor_index = -1;
    }
    Node temp = L->back->prev;
    free(L->back);
    L->back = temp;
    if (L->back != NULL) {
        L->back->next = NULL;
    } else {
        L->front = NULL; // Handle case when list becomes empty
    }
    L->list_length--;
}
// Delete cursor element, making cursor undefined.
// Pre: length()>0, index()>=0
void delete(List L) {
    preConditions(L, "delete", true, true);
    Node temp = L->cursor;
    if (temp->prev != NULL) {temp->prev->next = temp->next;}
    if (temp->next != NULL) {temp->next->prev = temp->prev;}
    if (temp == L->front) {L->front = temp->next;}
    if (temp == L->back) {L->back = temp->prev;}

    free(temp);
    L->cursor = NULL;
    L->cursor_index = -1;
    L->list_length--;
}

// Other operations -----------------------------------------------------------

// Prints to the file pointed to by out, a
// string representation of L consisting
// of a space separated sequence of integers,
// with front on left.
void printList(FILE* out, List L) {
    if(out == NULL) {
        printf("ListError: calling printList on NULL file reference");
        exit(EXIT_FAILURE);
    }
    else if(L ==NULL){
        printf("ListError: calling printList on NULL List reference");
        exit(EXIT_FAILURE);
    }
    Node temp = L->front;
    while(temp != NULL) {
        fprintf(out, "%d ", temp->data);
        temp = temp->next;
    }
}

// Returns a new List representing the same integer
// sequence as L. The cursor in the new list is undefined,
// regardless of the state of the cursor in L. The state
// of L is unchanged.
List copyList(List L) {
    if(L == NULL){
        printf("ListError: calling copyList on NULL List reference");
        exit(EXIT_FAILURE);
    }
    List NL = newList();
    NL->list_length = L->list_length;
    Node current = L->front;
    Node prev_node = NULL;
    while(current != NULL) {
        Node new_node = calloc(1, sizeof(NodeObj));
        new_node->data = current->data;
        new_node->prev = prev_node;
        if(prev_node == NULL) {
            NL->front = new_node;
        }
        else {
            prev_node->next = new_node;
        }
        prev_node = new_node;
        current = current->next;
    }
    if(prev_node != NULL) {
       NL->back = prev_node;
    }
    return(NL);
}
