#include <stdio.h>
#include <stdlib.h>
#include <assert.h>
#include <stdbool.h>
#include "List.h"

//structs -----------

// private Node type


typedef struct NodeObj* Node;

//private NodeObj type
typedef struct NodeObj{
	int data;
	Node next;
	Node previous;
} NodeObj;

// private List type
typedef struct ListObj{
	Node front;
	Node back;
	int length;
	int cursor;
} ListObj;

//Constructors-Destructors-------
Node newNode(int data){
	Node N = malloc(sizeof(NodeObj));
	assert( N!=NULL );
	N->data = data;
	N->next = NULL;
	N->previous = NULL;
	return(N);
}

// freeNode()
// Frees heap memory pointed to by *pN, sets *pN to NULL.
void freeNode(Node* pN){
   if( pN!=NULL && *pN!=NULL ){
      free(*pN);
      *pN = NULL;
   }
}

List newList(){
	List L;


	L = malloc(sizeof(ListObj));
	assert( L!=NULL);
	L->front = L->back = NULL;
	L->length = 0;
	L->cursor = -1;
	return(L);
}

void freeList(List* pL){
	if(pL!=NULL && *pL!= NULL){
		clear(*pL);
		free(*pL);


		*pL = NULL;
	}
}

//access functions

int length(List L){
	return L->length;
}
	
int index(List L){
	return L->cursor;
}

int front(List L){
	assert ( length(L) > 0);
	return L->front->data;
}

int back(List L){
	assert ( length(L) > 0);
	return L->back->data;
}

int get(List L){ // Returns cursor element of L. Pre: length()>0, index()>=0
	assert(length(L) > 0 && index(L) >= 0);
	Node current = L->front;
	for(int i = 0; i < index(L); i++){
		current = current->next;
	}
	return current->data;
}

bool equals(List A, List B){ // Returns true iff Lists A and B contain the same
	if( length(A) != length(B)){
		return false;
	}
	int tempA = index(A);
	int tempB = index(B);
	moveFront(A);
	moveFront(B);
	for(int i = 0; i < length(A); i++){
		if(get(A) != get(B)){
			return false;
		}
		moveNext(A);
		moveNext(B);
	}
	set(A, tempA);
	set(B, tempB);
	return true;
}
void clear(List L){
	while(length(L) > 0){
		deleteFront(L);
	}
	L->cursor = -1;
}

void set(List L, int x){
	L->cursor = x;
}

void moveFront(List L){
	if(length(L) > 0){
		set(L, 0);
	}
}

void moveBack(List L){
	if(length(L) > 0){
		set(L, length(L) - 1);
	}
}

void movePrev(List L){
	if(index(L) != -1 && index(L) != 0){
		set(L, index(L)-1);
	}
	else if(index(L) != -1 && index(L) == 0){
		set(L, -1);
	}
}

void moveNext(List L){
	if(index(L) != -1 && index(L) != length(L)-1){
		set(L, index(L)+1);
	}
	else if(index(L) != -1 && index(L) == length(L)-1){
		set(L, -1);
	}

}
	
void prepend(List L, int x){
	Node N = newNode(x);
	if(length(L) == 0){
		L->front = N;
		L->back = L->front;
	}
	else{
		N->next = L->front;
		L->front->previous = N;
		L->front = N;
		L->cursor += 1; 
	}
	L->length = L->length + 1;
	
}

void append(List L, int x){
	if(length(L) == 0){
		L->back = newNode(x);
		L->front = L->back;
		L->length = L->length + 1;
	}
	else{
		Node temp = newNode(x);
		temp->previous = L->back;
		L->back->next = temp;
		L->back = temp;
		L->length = L->length + 1;
	}
}

void insertBefore(List L, int x){
	assert(length(L) > 0 && index(L) >= 0);
	if(index(L) == 0){
		prepend(L,x);
	}
	else{
		Node current = L->front;
		for(int i = 0; i < index(L); i++){
			current = current->next;
		}
		Node insert = newNode(x); //Make a new node with value x we are going to insert.
		insert->next = current; //Set the next pointer in the new node to the node the cursor points to.
		insert->previous = current->previous; //set the new nodes previous pointer to the node before the cursor.
		insert->previous->next = insert; //change the next field of the node before the one we just inserted in order to have it correctly point to the inserted node. 
		insert->next->previous = insert; //do the same as above but with the node in front of the three pointing back to the node we just inserted.
		L->length = L->length + 1; 
		L->cursor += 1;
	}
}
	
void insertAfter(List L, int x){
	assert(length(L) > 0 && index(L) >= 0);
	if(length(L) == 0){
		append(L,x);
	}
	else if(index(L) == (L->length - 1)){
		append(L,x);
	}
	else{
		Node current = L->front;
		for(int i = 0; i < index(L); i++){
			current = current->next;
		}
		Node insert = newNode(x); //Make a new node with value x we are going to insert.
		insert->previous = current; //Set the next pointer in the new node to the node the cursor points to.
		insert->next = current->next; //set the new nodes previous pointer to the node after the cursor.
		insert->next->previous = insert; //change the next field of the node before the one we just inserted in order to have it correctly point to the inserted node. 
		insert->previous->next = insert; //do the same as above but with the node in front of the three pointing back to the node we just inserted. 
		L->length = L->length + 1;
	}
}	

void deleteFront(List L){
	assert( length(L) > 0);
	Node temp=L->front;
	L->front = L->front->next;
	freeNode(&temp);
	L->length = L->length -1;
	L->cursor -= 1;
}

void deleteBack(List L){
	assert( length(L) > 0);
	Node temp = L->back;
	L->back = L->back->previous;
	freeNode(&temp);
	if(L->cursor == L->length-1){
		L->cursor = -1;
	}
	L->length = L->length -1;

}

void delete(List L){
	assert(length(L) > 0 && index(L)>=0);
	if(index(L) == 0){
		deleteFront(L);
	}
	else if(index(L) == (L->length-1)){
		deleteBack(L);
	}
	else{
		Node temp = L->front;
		for(int i = 0; i<index(L); i++){
			temp = temp->next;
		}
		temp->previous->next = temp->next;
		temp->next->previous = temp->previous;
		freeNode(&temp);
		L->length = L->length - 1;
		set(L,-1);
	}
}

//Other operations
void printList(FILE* out, List L){
	moveFront(L);
	for(int i = 0; i < length(L); i++){
		fprintf(out, "%d ", get(L));
		moveNext(L);
	}
}

List copyList(List L){
	List copy = newList();
	int temp = index(L);
	moveFront(L);
	for(int i = 0; i < length(L); i++){
		append(copy,get(L));
		moveNext(L);
	}
	set(L, temp);
	set(copy, temp);
	return copy;
}
