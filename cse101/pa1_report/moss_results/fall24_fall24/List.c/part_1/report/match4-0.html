// Creates and returns a new empty List.
List newList(){

    //assert( L->front!=NULL );

} 

// Frees all heap memory associated with *pL, and sets
 // *pL to NULL.
void freeList(List* pL){ 
    if (pL != NULL && *pL != NULL){
        clear(*pL);
        free(*pL);
        *pL = NULL;
    }
}

// Returns the number of elements in L.
int length(List L){
    if (L == NULL){
        fprintf(stderr, "List Error: Cannot find length, list is Null\n");

    }
    return(L->len);
}

// Returns index of cursor element if defined, -1 otherwise.
int index(List L){
    if (L->cursor == NULL){
        return -1;
    }
    return(L->index);
}

void set(List L, int x){
    if (length(L) <= 0){
        fprintf(stderr, "List Error: List is empty\n");

    }
    if (L->index < 0){

// If L is non-empty, sets cursor under the front element,
 // otherwise does nothing.
void moveFront(List L){
    if (L != NULL && length(L) != 0){
        L->cursor = L->front;
        L->index = 0;
    }
}

// If L is non-empty, sets cursor under the back element,
 // otherwise does nothing.
void moveBack(List L){
    if (L != NULL && length(L) != 0){
        L->cursor = L->back;
        L->index = length(L) - 1;
    }
}

// If cursor is defined and not at front, move cursor one
// step toward the front of L; if cursor is defined and at
// front, cursor becomes undefined; if cursor is undefined
// do nothing
void movePrev(List L){
    if (L->index > 0){
        L->index -= 1;
        L->cursor = L->cursor->prev;
    }
    else if(L->index == 0){
        L->index -= 1;
        L->cursor = NULL;
    }
    else if ( L->cursor == L->front ){
        L->index -= 1;
        L->cursor = NULL;
    }
    else{
        ;
    }
}

// If cursor is defined and not at back, move cursor one
// step toward the back of L; if cursor is defined and at
// back, cursor becomes undefined; if cursor is undefined
// do nothing
void moveNext(List L){
    if (L->index < length(L) - 1){
        L->index += 1;
        L->cursor = L->cursor->next;
    }
    else if(L->cursor == L->back){
        L->index = -1;
        L->cursor = NULL;
    }
    else{
        ;
    }
} 

// Insert new element into L. If L is non-empty,
// insertion takes place before front element.
void prepend(List L, int x){
    if (L == NULL){
        fprintf(stderr, "List Error: List is NULL\n");
        exit(EXIT_FAILURE);
    }
    Node node = malloc(sizeof(NodeObj));
    assert(node != NULL);
    node->data = x;
    if (L->len == 0){
        node->prev = NULL;
        L->front = node;
        L->back = node;
    }
    if (L->len > 0){
        L->front->prev = node;
        node->next = L->front;
        L->front = node;
        L->front->prev = NULL;
    }
    if ( index(L) != -1 ){
        L->index++;
    }
    L->len += 1;
}

// Insert new element before cursor.
// Pre: length()>0, index()>=0

    if (length(L) <= 0){

    Node node = malloc(sizeof(NodeObj));

    if ( L->cursor == L->front ){

        node->data = x;
        node->next = L->cursor;
        node->prev = L->cursor->prev;
        L->cursor->prev->next = node;
        L->cursor->prev = node;
        L->len += 1;
        if (index(L) != -1){
            L->index++;
        }
    }

}

        exit(EXIT_FAILURE);

        deleteFront(L);
    } else if ( L->cursor == L->back) {
        deleteBack(L);
    } else if (length(L) > 0 ){
        L->cursor->next->prev = L->cursor->prev;
        L->cursor->prev->next = L->cursor->next;
        L->cursor = NULL;
        L->index = -1;
        L->len--;
        free(a);
    } else {
        L->front = L->back = NULL;
        L->cursor = NULL;
        L->index = -1;
        L->len--;
        free(a);
    }

}

// Prints to the file pointed to by out, a
// string representation of L consisting
// of a space separated sequence of integers,
// with front on left.
void printList(FILE* out, List L){ 
    if (L == NULL){
        fprintf(stderr, "List Error: List is NULL\n");
        exit(EXIT_FAILURE);
    }
    if (out != stdout){
        for (Node i = L->front; i != NULL; i = i->next){
            fprintf(out, "%d ", i->data);
        }
        fprintf(out, "\n");
        fclose(out);
    } else {
        for (Node i = L->front; i != NULL; i = i->next){
            printf("%d ", i->data);
    }
    }
}

// Returns a new List representing the same integer
// sequence as L. The cursor in the new list is undefined,
// regardless of the state of the cursor in L. The state
// of L is unchanged.
List copyList(List L){
    List M = newList();
    if ( length(L) == 0){
        return M;
    }
    for (Node l = L->front; l != NULL; l = l->next){
        append(M, l->data);
    }
    return M;
} 

// (extra) Returns a new List which is the concatenation of
// A and B. The cursor in the new List is undefined,
// regardless of the states of the cursors in A and B.
// The states of A and B are unchanged.
List concatList(List A, List B){
    List C = newList();
    for (Node a = A->front; a != NULL; a = a->next){
        append(C, a->data);
    }
    for (Node b = B-> front; b != NULL; b = b->next){
        append(C, b->data);
    }
