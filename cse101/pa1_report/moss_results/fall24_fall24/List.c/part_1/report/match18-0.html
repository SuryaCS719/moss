#include "List.h"
#include<stdio.h>
#include<stdlib.h>
#include<assert.h>
#include<stdbool.h>
#include<string.h>


// private Node type


typedef struct NodeObj* Node;

// private NodeObj type
typedef struct NodeObj{
   int data;
   Node next;
   Node prev;
} NodeObj;

// private ListObj type
typedef struct ListObj{
   Node front; // head
   Node back;  // tail
   int length;
   int cursor_idx; // idx of the cursor element 
   Node cursor; // cursor element
} ListObj;


// Constructors-Destructors ---------------------------------------------------

// newList()
// Returns reference to new empty List object.
List newList(void){
    List L = malloc(sizeof(ListObj));
    assert( L != NULL );
    L->front = L->back = NULL;
    L->length = 0;
    L->cursor_idx = -1;
    L->cursor = NULL;
    return L;
}

// isEmpty()
// Returns true if L is empty, otherwise returns false.
bool isEmpty(List L){
   if( L==NULL ){
      printf("List Error: calling isEmpty() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   return L->length==0;
}


// freeList()
// Frees all heap memory associated with *pL, and sets *pL to NULL.
void freeList(List* pL){
    if (pL != NULL && *pL != NULL) {
        while( !isEmpty(*pL) ) { 
            deleteFront(*pL); // delete head node, and point the next node as new front node
        }
        free(*pL); // clear memory
        *pL = NULL; // point to NULL
    }
}

// Access functions -----------------------------------------------------------
int length(List L){
    if( L==NULL ){
        printf("List Error: calling length() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    return L->length;
}

int index(List L){


    if( L==NULL ){
        printf("List Error: calling index() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    return L->cursor_idx;
}

int front(List L){
    if( L==NULL ){
        printf("List Error: calling front() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if( isEmpty(L) == true ){
        printf("List Error: calling front() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    return L->front->data;
}

int back(List L){
    if( L==NULL ){
        printf("List Error: calling back() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if( isEmpty(L) == true ){
        printf("List Error: calling back() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    return L->back->data;
}

int get(List L){
    if( L==NULL ){
        printf("List Error: calling back() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if( isEmpty(L) == true ){
        printf("List Error: calling back() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    return L->cursor->data;
}

bool equals(List A, List B){
    if( A==NULL || B==NULL ){
        printf("List Error: calling equals() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }

    bool eq;
    Node N, M;

    eq = ( A->length == B->length );
    N = A->front;
    M = B->front;
    while( eq && N!=NULL){
        eq = ( N->data==M->data );


        N = N->next;
        M = M->next;
    }
    return eq;
}


// Manipulation procedures ----------------------------------------------------
void clear(List L){
    if (L == NULL) {
        printf("List Error: calling clear() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    while (!isEmpty(L)) {
        deleteFront(L);
    }
    L->cursor_idx = -1;
    L->cursor = NULL;
    L->length = 0;
    L->front = NULL;
    L->back = NULL;
}

void set(List L, int x){
    if( L==NULL ){
        printf("List Error: calling set() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if( isEmpty(L) == true ){
        printf("List Error: calling set() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    if( L->cursor_idx < 0 ){
        printf("List Error: calling set() on an empty cursor element\n");
        exit(EXIT_FAILURE);
    }
    L->cursor->data = x;
}

void moveFront(List L){
    if( L==NULL ){
        printf("List Error: calling moveFront() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if( isEmpty(L) == true ){
        printf("List Error: calling moveFront() on an empty List\n");
        exit(EXIT_FAILURE);
    }

    L->cursor = L->front;
    L->cursor_idx = 0;
}

void moveBack(List L){
    if( L==NULL ){
        printf("List Error: calling moveBack() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if( isEmpty(L) == true ){
        printf("List Error: calling moveBack() on an empty List\n");
        exit(EXIT_FAILURE);
    }

    L->cursor = L->back;
    L->cursor_idx = length(L)-1;
}

void movePrev(List L){
    if( L==NULL ){
        printf("List Error: calling movePrev() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if( isEmpty(L) == true ){
        printf("List Error: calling movePrev() on an empty List\n");
        exit(EXIT_FAILURE);
    }

    if (L->cursor == NULL) {
        return;
    }
    else if (L->cursor_idx == 0) {
        L->cursor_idx = -1;
        L->cursor = NULL;
        return;
    }
    else {
        L->cursor = L->cursor->prev;
        L->cursor_idx--;
    }
}

void moveNext(List L){
    if( L==NULL ){
        printf("List Error: calling moveNext() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if( isEmpty(L) == true ){
        printf("List Error: calling moveNext() on an empty List\n");
        exit(EXIT_FAILURE);
    }

    if (L->cursor == NULL) {
        return;
    }
    else if (L->cursor == L->back) {
        L->cursor_idx = -1;
        L->cursor = NULL;
        return;
    }
    else {
        L->cursor = L->cursor->next;
        L->cursor_idx++;
    }
}

// Insert new element into L. If L is non-empty, takes place before front element.
void prepend(List L, int x){
    if (L == NULL) {
        printf("List Error: calling prepend() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    // 1. create new node
    Node N = malloc(sizeof(NodeObj));
    assert( N != NULL );
    N->data = x;
    N->prev = NULL;
    N->next = L->front;

    // 2. if list is not empty, update update current front's prev
    if (L->front != NULL) {
        L->front->prev = N;
    } else {
        L->back = N;
    }

    // 3. set front to this new node
    L->front = N;
    L->length++;
    L->cursor_idx++;
}

// Insert new element into L. If L is non-empty, insertion takes place after back element.
void append(List L, int x){
    if (L == NULL) {
        printf("List Error: calling append() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }

    // 1. create new node
    Node N = malloc(sizeof(NodeObj));
    assert( N != NULL );
    N->data = x;
    N->prev = L->back;
    N->next = NULL;

    // 2. if list is not empty, update update current front's prev
    if (L->back != NULL) {
        L->back->next = N;
    } else {
        L->front = N;
    }

    // 3. set front to this new node


    L->back = N;
    L->length++;
}

// Insert new element before cursor. Pre: length()>0, index()>=0
void insertBefore(List L, int x){
    if (L == NULL) {
        printf("List Error: calling insertBefore() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (isEmpty(L)) {
        printf("List Error: calling insertBefore() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    if (L->cursor_idx < 0) {
        printf("List Error: calling insertBefore() with an undefined cursor\n");
        exit(EXIT_FAILURE);
    }

    // 1. create new node
    Node N = malloc(sizeof(NodeObj));
    assert( N != NULL );
    N->data = x;
    N->next = L->cursor;
    N->prev = L->cursor->prev;

    // 2. if cursor is at the front, update front
    if (L->cursor->prev != NULL) {
        L->cursor->prev->next = N;
    } else {
        L->front = N;
    }

    // 3. update cursor's previous node
    L->cursor->prev = N;


    L->length++;
    L->cursor_idx++;
}

// Insert new element after cursor. Pre: length()>0, index()>=0
void insertAfter(List L, int x){
    if (L == NULL) {
        printf("List Error: calling insertAfter() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (isEmpty(L)) {
        printf("List Error: calling insertAfter() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    if (L->cursor_idx < 0) {
        printf("List Error: calling insertAfter() with an undefined cursor\n");
        exit(EXIT_FAILURE);
    }

    // 1. create new node
    Node N = malloc(sizeof(NodeObj));
    assert( N != NULL );
    N->data = x;
    N->next = L->cursor->next;
    N->prev = L->cursor;

    // 2. if cursor is at the back, update back
    if (L->cursor->next != NULL) {
        L->cursor->next->prev = N;
    } else {
        L->back = N;
    }

    // 3. update cursor's previous node


    L->cursor->next = N;
    L->length++;
}

// deleteFront()
// Delete front element
// Pre: !isEmpty(Q)
void deleteFront(List L){
    if( L==NULL ){
        printf("List Error: calling deleteFront() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if( isEmpty(L) ){
        printf("List Error: calling deleteFront() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    Node N = NULL;
    
    N = L->front;
    if( length(L)>1 ) { 
        L->front = L->front->next; 
        L->front->prev = NULL;
    }else{ // when we're deleting the last element already.
        L->front = L->back = NULL; 
    }

    if (L->cursor == N) { // If cursor is at the front, reset it
        L->cursor = NULL;
        L->cursor_idx = -1;
    } else if (L->cursor != NULL) {
        L->cursor_idx--; // Adjust cursor index if it's not null
    }

    L->length--;
    free(N);
}

// Delete the back element. Pre: length()>0
void deleteBack(List L){
    if( L==NULL ){
        printf("List Error: calling deleteBack() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if( isEmpty(L) ){
        printf("List Error: calling deleteBack() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    Node N = L->back;
    if( length(L)>1 ) { 
        L->back = L->back->prev; 
        L->back->next = NULL;
    }else{ // when we're deleting the last element already.


        L->front = L->back = NULL; 
    }

    if (L->cursor == N) { // If cursor is at the back, reset it
        L->cursor = NULL;
        L->cursor_idx = -1;
    }

    L->length--;
    free(N);
}

// Delete cursor element, making cursor undefined.
void delete(List L){
    if( isEmpty(L) ){
        printf("List Error: calling delete() on an empty List\n");
        exit(EXIT_FAILURE);
    }
    if (L->cursor_idx < 0) {
        printf("List Error: calling insertAfter() with an undefined cursor\n");
        exit(EXIT_FAILURE);
    }
    
    /*
    1. cursor at front
    2. cursor at back
    3. cursor somewhere in between
    */
    Node N = NULL;
    N = L->cursor;

    if (L->front == L->cursor){
        deleteFront(L);
    }
    else if (L->back == L->cursor){
        deleteBack(L);
    }
    else {


        L->cursor->prev->next = L->cursor->next;
        L->cursor->next->prev = L->cursor->prev;
        L->length--;
    }
    L->cursor = NULL;
    L->cursor_idx = -1;
    free(N);
}

/*
Prints to the file pointed to by out, a
string representation of L consisting
of a space separated sequence of integers,
with front on left.
*/
void printList(FILE* out, List L){
    if (L == NULL) {
        fprintf(out, "List Error: calling printList() on NULL List reference\n");
        return;
    }
    Node N = L->front;
    while (N != NULL) {
        fprintf(out, "%d ", N->data);
        N = N->next;
    }
    fprintf(out, "\n");
}

void saveList(FILE* out, List L, char** array){
    if (L == NULL) {
        fprintf(out, "List Error: calling saveList() on NULL List reference\n");
        return;
    }
    Node N = L->front;
    while (N != NULL) {
        char* str_val = array[N->data];
        fprintf(out, "%s\n", str_val);
        N = N->next;
    }
}


/*
Returns a new List representing the same integer
sequence as L. The cursor in the new list is undefined,
regardless of the state of the cursor in L. The state
of L is unchanged.
*/
List copyList(List L){
    if (L == NULL) {
        printf("List Error: calling copyList() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }

    List newL = newList();
    Node N = L->front;

    while (N != NULL) {
        append(newL, N->data); // Append each element to the new list
        N = N->next;
    }

    return newL;
}

    printList(stdout, L); 
    if (L->cursor != NULL) {
        printf("Cursor is at index %d pointing at %d.\n", L->cursor_idx, L->cursor->data);
    } else {
        printf("Cursor is at index %d pointing at NULL.\n", L->cursor_idx);
    }
    if (length(L) > 0) {
        printf("Front (head): %d\n", front(L));
    } else {
        printf("Front (head): NULL\n");
    }
    if (length(L) > 0) {
        printf("Back (tail): %d\n", back(L));
    } else {
        printf("Back (tail): NULL\n");
    }
    printf("Length: %d\n", length(L));
    printf("\n");
}
