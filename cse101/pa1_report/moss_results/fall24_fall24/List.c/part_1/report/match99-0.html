#include<stdio.h>
#include<stdlib.h>
#include<assert.h>
#include<stdbool.h>
#include "List.h"

// structs


typedef struct NodeObj* Node;
typedef int ListElement;

typedef struct NodeObj {
    ListElement data;
    Node next;
    Node prev;
} NodeObj;

typedef struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int index;
    int length;
} ListObj;

// newList defined as front, back, cursor are NULL. Length = 0, index = -1
List newList(void) {
    List L = malloc(sizeof(ListObj));
    L->front = L->back = L->cursor = NULL;
    L->length = 0;
    L->index = -1;
    return L;
}

// free heap for *pL and set that pointer of the pointer to NULL
// Need to account for each Node in the list
void freeList(List *pL) {
    if (pL != NULL && *pL != NULL) {
        while ((*pL)->length > 0) {
            deleteFront(*pL); 
        }
        free(*pL);


        *pL = NULL;
    }
}

int length(List L) {
    return L->length;
}

int index(List L) {
    return L->index;
}

int front(List L) {
    if (length(L) > 0) {
        return L->front->data;
    }
    fprintf(stderr, "List Error: calling front() on empty List\n");
    exit(EXIT_FAILURE);
}

int back(List L) {
    if (length(L) > 0) {
        return L->back->data;
    }
    fprintf(stderr, "List Error: calling back() on empty List\n");
    exit(EXIT_FAILURE);
}

int get(List L) {
    if (length(L) > 0 && index(L) >= 0) {
        return L->cursor->data;
    }
    fprintf(stderr, "List Error: calling get() on undefined cursor\n");
    exit(EXIT_FAILURE);
}

bool equals(List A, List B) {
    if (A == NULL || B == NULL) {
        printf("List Error: calling equals() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    if (A->length != B->length) {
        return false;
    }
    Node N = A->front;
    Node M = B->front;
    while (N != NULL) {
        if (N->data != M->data) {
            return false;
        }
        N = N->next;
        M = M->next;
    }
    return true;
}

// Manipulation procedures ----------------------------------------------------

void clear(List L) {
    while (length(L) > 0) {


        deleteFront(L);
    }
    L->cursor = NULL;
    L->index = -1;
}

void set(List L, int x) {
    if (L->cursor != NULL) {
        L->cursor->data = x;
    } else {
        fprintf(stderr, "List Error: calling set() on undefined cursor\n");
        exit(EXIT_FAILURE);
    }
}

void moveFront(List L) {
    if (L->length > 0) {
        L->cursor = L->front;
        L->index = 0;
    }
}

void moveBack(List L) {
    if (L->length > 0) {
        L->cursor = L->back;
        L->index = L->length - 1;
    }
}

void movePrev(List L) {
    if (L->cursor != NULL) {
        if (L->cursor == L->front) {
            L->cursor = NULL;
            L->index = -1;
        } else {
            L->cursor = L->cursor->prev;
            L->index--;
        }
    }
}

void moveNext(List L) {
    if (L->cursor != NULL) {
        if (L->cursor == L->back) {
            L->cursor = NULL;
            L->index = -1;
        } else {


            L->cursor = L->cursor->next;
            L->index++;
        }
    }
}

void prepend(List L, int x) {
    Node N = (Node)malloc(sizeof(NodeObj));
    if (N == NULL) {
        fprintf(stderr, "Memory allocation error\n");
        exit(EXIT_FAILURE);
    }
    N->data = x;
    N->next = NULL;
    N->prev = NULL;

    if (L->length == 0) {
        L->front = L->back = N;
    } else {
        N->next = L->front;
        L->front->prev = N;
        L->front = N;
    }


    L->length++;
    if (L->index != -1) L->index++;
}

void append(List L, int x) {
    Node N = (Node)malloc(sizeof(NodeObj));
    if (N == NULL) {
        fprintf(stderr, "Memory allocation error\n");
        exit(EXIT_FAILURE);
    }
    N->data = x;
    N->next = NULL;
    N->prev = NULL;

    if (L->length == 0) {
        L->front = L->back = N;
    } else {
        N->prev = L->back;


        L->back->next = N;
        L->back = N;
    }
    L->length++;
}

void insertBefore(List L, int x) {
    if (L->index >= 0) {
        if (L->cursor == L->front) {
            prepend(L, x);
        } else {
            Node N = (Node)malloc(sizeof(NodeObj));
            if (N == NULL) {
                fprintf(stderr, "Memory allocation error\n");
                exit(EXIT_FAILURE);
            }
            N->data = x;
            N->prev = L->cursor->prev;
            N->next = L->cursor;
            L->cursor->prev->next = N;
            L->cursor->prev = N;
            L->index++;
            L->length++;
        }
    } else {
        fprintf(stderr, "List Error: calling insertBefore() on undefined cursor\n");
        exit(EXIT_FAILURE);
    }
}

void insertAfter(List L, int x) {
    if (L->index >= 0) {
        if (L->cursor == L->back) {
            append(L, x);
        } else {
            Node N = (Node)malloc(sizeof(NodeObj));
            if (N == NULL) {
                fprintf(stderr, "Memory allocation error\n");
                exit(EXIT_FAILURE);
            }
            N->data = x;
            N->next = L->cursor->next;
            N->prev = L->cursor;
            L->cursor->next->prev = N;
            L->cursor->next = N;
            L->length++;
        }
    } else {
        fprintf(stderr, "List Error: calling insertAfter() on undefined cursor\n");
        exit(EXIT_FAILURE);
    }
}

void deleteFront(List L) {
    if (L->length > 0) {
        Node delete = L->front;
        if (L->front == L->back) {
            L->front = L->back = NULL;
        } else {


            L->front = L->front->next;
            L->front->prev = NULL;
        }
        free(delete);
        L->length--;
        if (L->index >= 0) {
            L->index--;
        }
    } else {
        fprintf(stderr, "List Error: calling deleteFront() on empty List\n");
        exit(EXIT_FAILURE);
    }
}

void deleteBack(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling deleteBack() on NULL List\n");
        exit(EXIT_FAILURE);
    }
    
    if (L->length == 0) {
        fprintf(stderr, "List Error: calling deleteBack() on empty List\n");
        exit(EXIT_FAILURE);
    }
    Node deleteNode = L->back; 
    if (L->length == 1) {
        L->front = L->back = NULL;
        L->cursor = NULL; 
    } else { 
        L->back = L->back->prev; 
        if (L->back != NULL) {
            L->back->next = NULL; 
        }
        if (L->cursor == deleteNode) {
            L->cursor = NULL;
            L->index = -1; 
        }
    }
    free(deleteNode); 
    L->length--; 
}


void delete(List L) {
    if (L->length > 0 && L->index >= 0) {
        Node delete = L->cursor;
        if (L->cursor == L->front) {
            deleteFront(L);
        } else if (L->cursor == L->back) {
            deleteBack(L);
        } else {
            L->cursor->prev->next = L->cursor->next;


            L->cursor->next->prev = L->cursor->prev;
            free(delete);
            L->length--;
        }
        L->cursor = NULL;
        L->index = -1;
    } else {
        fprintf(stderr, "List Error: calling delete() on undefined cursor or empty List\n");
        exit(EXIT_FAILURE);
    }
}

// Other operations -----------------------------------------------------------
void printList(FILE* out, List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling printList() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    Node temp = L->front;
    while (temp != NULL) {
        fprintf(out, "%d ", temp->data);
        temp = temp->next;
    }
}

List copyList(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling copyList() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    List newL = newList();  // Create a new empty list
    Node temp = L->front;
    while (temp != NULL) {
        append(newL, temp->data);


        temp = temp->next;
    }
    return newL;
}

List concatList(List A, List B) {
    if (A == NULL || B == NULL) {
        fprintf(stderr, "List Error: calling concatList() on NULL List reference\n");
        exit(EXIT_FAILURE);
    }
    List newL = newList();
    Node Atemp = A->front;
    while (Atemp != NULL) {
        append(newL, Atemp->data);
        Atemp = Atemp->next;
    }
    Node Btemp = B->front;
    while (Btemp != NULL) {
        append(newL, Btemp->data);
        Btemp = Btemp->next;
    }
    return newL;
}
