#include "List.h"




Node* newNode(int data) {
	Node* N = malloc(sizeof(Node));
	N->data = data;
	N->next = NULL;
	N->prev = NULL;
	return N;
}

void freeNode(Node** ptp) { // ptp -> pointer to a pointer
        if (ptp != NULL && *ptp != NULL) {
                free(*ptp);
                *ptp = NULL;
        }
}

List newList(void){
	List L = malloc(sizeof(ListObj)); 
	L->front = L->back = L->cursor = NULL;
	L->length = 0;
	L->index = -1;
	return L;

}	
void freeList(List* pL){
	if (pL != NULL && *pL != NULL) {
		while(length(*pL) > 0) {


			deleteFront(*pL);
		}
		free(*pL);
		*pL = NULL;
	}
}
int length(List L){
return L->length;

}
int index(List L){
	return L->cursor == NULL ? -1 : L->index; //ternary operator (reminder)

}
int front(List L){
	if (L == NULL || length(L) == 0) {
		printf("Error, list is not found");
		exit(1);
	}
	return L->front->data;

}	
int back(List L){
        if (L == NULL || length(L) == 0) {
                printf("Error, list is not found");
                exit(1);
        }
        return L->back->data;
}
int get(List L){
	if (L == NULL || L->cursor == NULL) {
		printf("Error, cursor is NULL");
		exit(1);
	}
	return L->cursor->data;

}	
bool equals(List A, List B){
	if (A == NULL || B == NULL) {
		printf("Error, lists are not the same");
		exit(1);
	}

	if (A->length != B->length) {
		return false;
	}

	Node* nodeA = A->front;
	Node* nodeB = B->front;

	while (nodeA != NULL && nodeB != NULL) {
		if (nodeA->data != nodeB->data) {
			return false;
		}



		nodeA = nodeA->next;
		nodeB = nodeB->next;
	}

	return true;
}	
void clear(List L){
	while (length(L) > 0) {
		deleteFront(L);
	}
}

void set(List L, int x){
	if (L == NULL || L->cursor == NULL) {
		printf("Error, cursor is NULL");
		exit(1);
	}
	L->cursor->data = x;
}	
void moveFront(List L){
	if (L != NULL && length(L) > 0) {
		L->cursor = L->front;
		L->index = 0;
	}
}	
void moveBack(List L){
        if (L != NULL && length(L) > 0) {
                L->cursor = L->back;
                L->index = length(L) - 1;
        }
}	
void movePrev(List L){
        if (L != NULL && L->cursor != NULL) {
		if (L->cursor == L->front) {
			L->cursor = NULL;
			L->index = -1;
		} else {
			L->cursor = L->cursor->prev;
			L->index--;
		}
        }
}	
void moveNext(List L){
        if (L != NULL && L->cursor != NULL) {
                if (L->cursor == L->back) {
                        L->cursor = NULL;
                        L->index = -1;
                } else {
                        L->cursor = L->cursor->next;
                        L->index++;
                }
        }
}
 
void prepend(List L, int x) { 
	Node* N = newNode(x);
	if (length(L) == 0) {
		L->front = L->back = N;
	} else {
		N->next = L->front;
		L->front->prev = N;


		L->front = N;
	}
	L->length++;
	if (L->index != -1) {
		L->index++;
	}

}
void append(List L, int x) {
        Node* N = newNode(x);
        if (length(L) == 0) {
                L->front = L->back = N;
        } else {
                N->prev = L->back;
                L->back->next = N;
                L->back = N;
        }
        L->length++;

}
void insertBefore(List L, int x) {
    if (length(L) > 0 && index(L) >= 0) {
        Node* N = newNode(x);
        if (L->cursor == L->front) {
            prepend(L, x);
        } else {
            N->prev = L->cursor->prev;
            N->next = L->cursor;
            L->cursor->prev->next = N;
            L->cursor->prev = N;
            L->length++;
            L->index++;
        }
    } else {
        printf("Error, list is not found");
        exit(1);
    }
}

void insertAfter(List L, int x) {
    if (length(L) > 0 && index(L) >= 0) {
        Node* N = newNode(x);
        if (L->cursor == L->back) {
            append(L, x);
        } else {
            N->next = L->cursor->next;
            N->prev = L->cursor;
            L->cursor->next->prev = N;
            L->cursor->next = N;
            L->length++;
        }
    } else {
        printf("Error, list is not found");
        exit(1);
    }
}

void deleteFront(List L) {
    if (length(L) > 0) {
        Node* N = L->front;
        if (length(L) > 1) {
            L->front = L->front->next;
            L->front->prev = NULL;
        } else {
            L->front = L->back = NULL;
        }
        freeNode(&N);
        L->length--;
        if (L->index != -1) {
            L->index--;
        }
    } else {
        printf("Error, list is not found");
        exit(1);
    }
}

void deleteBack(List L) {
    if (length(L) > 0) {
        Node* N = L->back;
        if (length(L) > 1) {
            L->back = L->back->prev;
            L->back->next = NULL;
        } else {
            L->front = L->back = NULL;
        }
        freeNode(&N);
        L->length--;
        if (L->index == L->length) {
            L->index = -1;
            L->cursor = NULL;
        }
    } else {
        printf("Error, list is not found");
        exit(1);
    }
}

void delete(List L) {
    if (length(L) > 0 && index(L) >= 0) {
        Node* N = L->cursor;
        if (L->cursor == L->front) {
            deleteFront(L); 
        } else if (L->cursor == L->back) {
            deleteBack(L);
        } else {
            L->cursor->prev->next = L->cursor->next;


            L->cursor->next->prev = L->cursor->prev;
            freeNode(&N);
            L->length--;
        }
        L->cursor = NULL;
        L->index = -1;
    } else {
        printf("Error, list is not found");
        exit(1);
    }
}
void printList(FILE* out, List L){
	if (L == NULL) {
		printf("Error, list is not found");
		exit(1);
	}
	Node* N = L->front;
	while (N != NULL) {
		fprintf(out, "%d ", N->data);
		N = N->next;
	}
	fprintf(out, "\n");
}
