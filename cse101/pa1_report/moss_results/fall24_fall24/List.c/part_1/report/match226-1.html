#include<stdio.h>
#include<stdlib.h>
#include<assert.h>
#include<stdbool.h>
#include "List.h"

// isEmpty()
// Returns true if list is empty, otherwise returns false.
bool isEmpty(List list);

// structs --------------------------------------------------------------------

// private Node type
typedef struct NodeObj* Node;

// private NodeObj type
typedef struct NodeObj{
   int data;
	Node prev;
   Node next;
} NodeObj;

// private ListObj type
typedef struct ListObj{
   Node front;
   Node back;
	Node cursor;
   int length;
	int cursorIndex;
} ListObj;


// Constructors-Destructors ---------------------------------------------------

// Returns reference to new Node object. Initializes next and data fields.
Node newNode(int data){
   Node node = (Node)malloc(sizeof(NodeObj));
   assert( node!=NULL );
   node->data = data;
   node->next = NULL;
	node->prev = NULL;
   return(node);
}

// Frees heap memory pointed to by *pN, sets *pN to NULL.
void freeNode(Node* pN){
   if( pN!=NULL && *pN!=NULL ){
      free(*pN);
      *pN = NULL;
   }
}

// Returns reference to new empty List object.
List newList(){
   List list;
   list = (List)malloc(sizeof(ListObj));
   assert( list!=NULL );
	list->front = list->back = list->cursor = NULL;
   list->length = 0;
	list->cursorIndex = -1;
   return(list);
}

// Frees all heap memory associated with List *pList, and sets *pList to NULL.
void freeList(List* pList){
   if(pList!=NULL && *pList!=NULL) { 
      while( !isEmpty(*pList) ) { 
         deleteFront(*pList); 
      }
      free(*pList);
      *pList = NULL;
   }
}


// Access functions -----------------------------------------------------------

// Returns the length of list.
int length(List list){
   if( list==NULL ){
      printf("List Error: calling length() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   return(list->length);
}

// Returns index of cursor element if defined, -1 otherwise.
int index(List list)
{


   if( list==NULL ){
      printf("List Error: calling index() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
	return(list->cursorIndex);
}

// Returns the value at the front of list.
// Pre: !isEmpty(list)
int front(List list){
   if( list==NULL ){
      printf("List Error: calling front() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if( isEmpty(list) ){
      printf("List Error: calling front() on an empty List\n");
      exit(EXIT_FAILURE);
   }
   return(list->front->data);
}

// Returns back element of list. Pre: length()>0
int back(List list)
{
   if( list==NULL ){
      printf("List Error: calling back() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if( isEmpty(list) ){
      printf("List Error: calling back() on an empty List\n");
      exit(EXIT_FAILURE);
   }


	return(list->back->data);
}

// Returns cursor element of list. Pre: length()>0, index()>=0
int get(List list)
{
   if( list==NULL ){
      printf("List Error: calling get() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if( isEmpty(list) ){
      printf("List Error: calling get() on an empty List\n");
      exit(EXIT_FAILURE);
   }
	if( list->cursorIndex < 0 ){
      printf("List Error: calling get() with undefined cursor\n");
      exit(EXIT_FAILURE);
   }
	return(list->cursor->data);
}

// Returns true if A is same int sequence as B, false otherwise.
bool equals(List A, List B){
   if( A==NULL || B==NULL ){
      printf("List Error: calling equals() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }

   bool eq;
   Node node, otherNode;

   eq = ( A->length == B->length );
   node = A->front;
   otherNode = B->front;
   while( eq && node!=NULL){
      eq = ( node->data==otherNode->data );


      node = node->next;
      otherNode = otherNode->next;
   }
   return eq;
}

// Returns true if list is empty, otherwise returns false.
bool isEmpty(List list){
   if( list==NULL ){
      printf("List Error: calling isEmpty() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   return(list->length==0);
}


// Manipulation procedures ----------------------------------------------------

// Resets list to its original empty state.
void clear(List list)
{
   if( list==NULL ){
      printf("List Error: calling clear() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   while( !isEmpty(list) ) { 


      deleteFront(list); 
   }
}

// Overwrites the cursor element’s data with x.
// Pre: length()>0, index()>=0
void set(List list, int x)
{
   if( list==NULL ){
      printf("List Error: calling set() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if( isEmpty(list) ){
      printf("List Error: calling set() on an empty List\n");
      exit(EXIT_FAILURE);
   }
	if( list->cursorIndex < 0 ){


      printf("List Error: calling set() with invalid cursor\n");
      exit(EXIT_FAILURE);
   }
	list->cursor->data = x;
}

// If list is non-empty, sets cursor under the front element,
// otherwise does nothing.
void moveFront(List list)
{
   if( list==NULL ){
      printf("List Error: calling moveFront() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if( !isEmpty(list) ){
		list->cursor = list->front;
		list->cursorIndex = 0;
   }
}

// If list is non-empty, sets cursor under the back element,
// otherwise does nothing.
void moveBack(List list)
{
   if( list==NULL ){
      printf("List Error: calling moveBack() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if( !isEmpty(list) ){
		list->cursor = list->back;
		list->cursorIndex = list->length - 1;
   }
}

// If cursor is defined and not at front, move cursor one
// step toward the front of list; if cursor is defined and at
// front, cursor becomes undefined; if cursor is undefined
// do nothing
void movePrev(List list)
{
   if( list==NULL ){
      printf("List Error: calling movePrev() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
	if(list->cursorIndex >= 0) {
		list->cursor = list->cursor->prev;
		list->cursorIndex--;
	}
}

// If cursor is defined and not at back, move cursor one
// step toward the back of list; if cursor is defined and at
// back, cursor becomes undefined; if cursor is undefined
// do nothing
void moveNext(List list)
{
   if( list==NULL ){
      printf("List Error: calling moveNext() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
	if(list->cursorIndex >= 0) {
		list->cursor = list->cursor->next;
		list->cursorIndex++;
		if (list->cursor == NULL)
			list->cursorIndex = -1;
	}
}

// Insert new element into list. If list is non-empty,
// insertion takes place before front element.
void prepend(List list, int x)
{
   Node node = newNode(x);

   if( list==NULL ){
      printf("List Error: calling prepend() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   
   if( isEmpty(list) ) { 
      list->front = list->back = node; 
   }else{ 
      list->front->prev = node;
		node->next = list->front;
      list->front = node; 
   }
	if (list->cursorIndex >= 0)
		list->cursorIndex++;
   list->length++;
}

// Places new data at the back of list.
void append(List list, int data)
{
   Node node = newNode(data);

   if( list==NULL ){
      printf("List Error: calling append() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   
   if( isEmpty(list) ) { 
      list->front = list->back = node; 
   }else{ 
      list->back->next = node;
		node->prev = list->back;
      list->back = node; 
   }
   list->length++;
}


// Insert new element before cursor.
// Pre: length()>0, index()>=0
void insertBefore(List list, int x)
{
   Node node = newNode(x);



   if( list==NULL ){
      printf("List Error: calling insertBefore() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if( isEmpty(list) ){
      printf("List Error: calling insertBefore() on an empty List\n");
      exit(EXIT_FAILURE);
   }
	if( list->cursorIndex < 0 ){
      printf("List Error: calling insertBefore() with invalid cursor\n");
      exit(EXIT_FAILURE);
   }
	if (list->cursor->prev == NULL){
		list->front = node;
	}
	else {
		list->cursor->prev->next = node;
	}

	node->prev = list->cursor->prev;
	list->cursor->prev = node;
	node->next = list->cursor;
	list->cursorIndex++;
	list->length++;
}

// Insert new element after cursor.
// Pre: length()>0, index()>=0
void insertAfter(List list, int x)
{
   Node node = newNode(x);

   if( list==NULL ){
      printf("List Error: calling insertAfter() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if( isEmpty(list) ){
      printf("List Error: calling insertAfter() on an empty List\n");
      exit(EXIT_FAILURE);
   }
	if( list->cursorIndex < 0 ){
      printf("List Error: calling insertAfter() with invalid cursor\n");
      exit(EXIT_FAILURE);
   }
	if (list->cursor->next == NULL){
		list->back = node;
	}
	else {
		list->cursor->next->prev = node;
	}

	node->next = list->cursor->next;
	list->cursor->next = node;
	node->prev = list->cursor;
	list->length++;
}

// Deletes data at front of list.
// Pre: !isEmpty(list)
void deleteFront(List list){
   Node node = NULL;

   if( list==NULL ){
      printf("List Error: calling deleteFront() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if( isEmpty(list) ){
      printf("List Error: calling deleteFront() on an empty List\n");
      exit(EXIT_FAILURE);
   }
   
   node = list->front;

	if(list->cursorIndex == 0)
		list->cursor = NULL;

	if(list->cursorIndex >= 0)
		list->cursorIndex--;

   if( length(list)>1 ) { 
		if (list->front->next != NULL)
			list->front->next->prev = NULL;
      list->front = list->front->next; 
   }else{ 
      list->front = list->back = NULL; 
   }
   list->length--;
   freeNode(&node);
}

// Delete the back element. Pre: length()>0
void deleteBack(List list)
{
   Node node = NULL;

   if( list==NULL ){
      printf("List Error: calling deleteBack() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if( isEmpty(list) ){
      printf("List Error: calling deleteBack() on an empty List\n");
      exit(EXIT_FAILURE);
   }
   
   node = list->back;

	if(list->back == list->cursor)
	{
		list->cursor = NULL;
		list->cursorIndex = -1;
	}

   if( length(list)>1 ) { 
		if (list->back->prev != NULL)
			list->back->prev->next = NULL;
      list->back = list->back->prev; 
   }else{ 
      list->back = list->front = NULL; 
   }
   list->length--;
   freeNode(&node);
}

// Delete cursor element, making cursor undefined.
// Pre: length()>0, index()>=0
void delete(List list)
{
   Node node = NULL;

   if( list==NULL ){
      printf("List Error: calling delete() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }
   if( isEmpty(list) ){
      printf("List Error: calling delete() on an empty List\n");
      exit(EXIT_FAILURE);
   }
	if( list->cursorIndex < 0 ){
      printf("List Error: calling delete() with invalid cursor\n");
      exit(EXIT_FAILURE);
   }

   node = list->cursor;

   if( length(list)>1 ) { 
		if (list->cursor->prev != NULL)
			list->cursor->prev->next = list->cursor->next;
		if (list->cursor->next != NULL)
			list->cursor->next->prev = list->cursor->prev;
   }else{ 
      list->back = list->front = NULL; 
   }

	list->cursor = NULL;
	list->cursorIndex = -1;
   list->length--;
   freeNode(&node);
}

// Other Functions ------------------------------------------------------------

// Prints to the file pointed to by out, a
// string representation of list consisting
// of a space separated sequence of integers,
// with front on left.
void printList(FILE* out, List list){
   Node node = NULL;

   if( list==NULL ){
      fprintf(out, "List Error: calling printList() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }

   for(node = list->front; node != NULL; node = node->next){
      fprintf(out, FORMAT" ", node->data);
   }
}

// Returns a new List representing the same integer
// sequence as list. The cursor in the new list is undefined,
// regardless of the state of the cursor in list. The state
// of list is unchanged.
List copyList(List list)
{
   if( list==NULL ){
      printf("List Error: calling copyList() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }

	List otherList = newList();

   Node node = NULL;

   for(node = list->front; node != NULL; node = node->next){
		append(otherList, node->data);
	}

	return otherList;
}

// Returns a new List which is the concatenation of
// A and B. The cursor in the new List is undefined,
// regardless of the states of the cursors in A and B.
// The states of A and B are unchanged.
List concatList(List A, List B)
{
   if( A==NULL || B==NULL ){
      printf("List Error: calling concatList() on NULL List reference\n");
      exit(EXIT_FAILURE);
   }

	List otherList = newList();

   Node node = NULL;

   for(node = A->front; node != NULL; node = node->next){
		append(otherList, node->data);
	}

   for(node = B->front; node != NULL; node = node->next){
		append(otherList, node->data);
	}

	return otherList;
}
