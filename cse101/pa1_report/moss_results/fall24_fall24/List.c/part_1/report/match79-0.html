	Node N = malloc(sizeof(NodeObj));

bool equals(List A, List B) {
	if (A->length != B->length) {
		printf("auto");
		return false;
	}
	Node X = A->front;
	Node Y = B->front;
	while (X || Y) {
		//printf("It is %X");
		if (X->data != Y->data) {
			return false;
		}
		X = X->next;
		Y = Y->next;
	}

	return true;
}
