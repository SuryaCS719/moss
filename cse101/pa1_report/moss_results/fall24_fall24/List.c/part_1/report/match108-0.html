#include <stdio.h>
#include <stdlib.h>
#include "List.h"

// Node structure
typedef struct NodeObj {
    int data;
    struct NodeObj* prev;
    struct NodeObj* next;
} NodeObj;

typedef NodeObj* Node;

// List structure
typedef struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
    int cursorIndex;
} ListObj;

// Constructors-Destructors

List newList(void) {
    List L = malloc(sizeof(ListObj));
    L->front = L->back = L->cursor = NULL;
    L->length = 0;
    L->cursorIndex = -1;  // Initially, the cursor is undefined
    return L;
}

void freeList(List* pL) {
    if (pL != NULL && *pL != NULL) {
        while (length(*pL) > 0) {
            deleteFront(*pL);
        }
        free(*pL);


        *pL = NULL;
    }
}

// Access Functions

int length(List L) {
    return L->length;
}

int index(List L) {
    return L->cursorIndex;
}

int front(List L) {
    if (L->length == 0) {
        fprintf(stderr, "List Error: calling front() on an empty List\n");
        exit(1);
    }
    return L->front->data;
}

int back(List L) {
    if (L->length == 0) {
        fprintf(stderr, "List Error: calling back() on an empty List\n");
        exit(1);
    }
    return L->back->data;
}

int get(List L) {
    if (L->length == 0 || L->cursor == NULL) {
        fprintf(stderr, "List Error: calling get() with undefined cursor\n");
        exit(1);
    }
    return L->cursor->data;
}

bool equals(List A, List B) {
    if (A->length != B->length) return false;

    Node aNode = A->front;
    Node bNode = B->front;
    while (aNode != NULL && bNode != NULL) {
        if (aNode->data != bNode->data) return false;
        aNode = aNode->next;
        bNode = bNode->next;
    }
    return true;
}

// Manipulation Procedures

void clear(List L) {
    while (L->length > 0) {


        deleteFront(L);
    }
    L->cursor = NULL;
    L->cursorIndex = -1;
}

void set(List L, int x) {
    if (L->cursor == NULL) {
        fprintf(stderr, "List Error: calling set() with undefined cursor\n");
        exit(1);
    }
    L->cursor->data = x;
}

void moveFront(List L) {
    if (L->length > 0) {
        L->cursor = L->front;
        L->cursorIndex = 0;
    }
}

void moveBack(List L) {
    if (L->length > 0) {
        L->cursor = L->back;
        L->cursorIndex = L->length - 1;
    }
}

void movePrev(List L) {


    if (L->cursor != NULL && L->cursorIndex > 0) {
        L->cursor = L->cursor->prev;
        L->cursorIndex--;
    } else if (L->cursorIndex == 0) {
        L->cursor = NULL;
        L->cursorIndex = -1;
    }
}

void moveNext(List L) {


    if (L->cursor != NULL && L->cursorIndex < L->length - 1) {
        L->cursor = L->cursor->next;
        L->cursorIndex++;
    } else if (L->cursorIndex == L->length - 1) {
        L->cursor = NULL;
        L->cursorIndex = -1;
    }
}

void prepend(List L, int x) {
    Node newNode = malloc(sizeof(NodeObj));
    newNode->data = x;
    newNode->prev = NULL;
    newNode->next = L->front;

    if (L->front != NULL) {
        L->front->prev = newNode;


    } else {
        L->back = newNode;
    }

    L->front = newNode;
    L->length++;
}

void append(List L, int x) {
    Node newNode = malloc(sizeof(NodeObj));
    newNode->data = x;
    newNode->next = NULL;
    newNode->prev = L->back;

    if (L->back != NULL) {
        L->back->next = newNode;
    } else {


        L->front = newNode;
    }

    L->back = newNode;
    L->length++;
}

void insertBefore(List L, int x) {
    if (L->cursor == NULL) {
        fprintf(stderr, "List Error: calling insertBefore() with undefined cursor\n");
        exit(1);
    }

    Node newNode = malloc(sizeof(NodeObj));
    newNode->data = x;
    newNode->prev = L->cursor->prev;
    newNode->next = L->cursor;

    if (L->cursor->prev != NULL) {


        L->cursor->prev->next = newNode;
    } else {
        L->front = newNode;
    }

    L->cursor->prev = newNode;
    L->length++;
}

void insertAfter(List L, int x) {
    if (L->cursor == NULL) {
        fprintf(stderr, "List Error: calling insertAfter() with undefined cursor\n");
        exit(1);
    }

    Node newNode = malloc(sizeof(NodeObj));
    newNode->data = x;
    newNode->next = L->cursor->next;
    newNode->prev = L->cursor;

    if (L->cursor->next != NULL) {
        L->cursor->next->prev = newNode;


    } else {
        L->back = newNode;
    }

    L->cursor->next = newNode;
    L->length++;
}

void deleteFront(List L) {
    if (L->length == 0) {
        fprintf(stderr, "List Error: calling deleteFront() on empty List\n");
        exit(1);
    }

    Node temp = L->front;
    if (L->front->next != NULL) {
        L->front = L->front->next;
        L->front->prev = NULL;
    } else {
        L->front = L->back = NULL;
    }

    free(temp);
    L->length--;
}

// Implementation of deleteBack()
void deleteBack(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling deleteBack() on NULL List reference\n");
        exit(1);
    }
    if (L->length == 0) {
        fprintf(stderr, "List Error: calling deleteBack() on empty List\n");
        exit(1);
    }

    Node temp = L->back;
    if (L->back == L->front) {  // List has only one element
        L->front = L->back = NULL;
    } else {
        L->back = L->back->prev;
        L->back->next = NULL;
    }

    free(temp);
    L->length--;
    if (L->cursor == temp) {  // If cursor was at the back, it becomes undefined
        L->cursor = NULL;
        L->cursorIndex = -1;
    }
}

void delete(List L) {
    if (L->cursor == NULL) {
        fprintf(stderr, "List Error: calling delete() with undefined cursor\n");
        exit(1);
    }

    Node temp = L->cursor;
    if (L->cursor == L->front) {
        deleteFront(L);
    } else if (L->cursor == L->back) {
        deleteBack(L);
    } else {
        L->cursor->prev->next = L->cursor->next;
        L->cursor->next->prev = L->cursor->prev;
        free(temp);
        L->length--;
    }

    L->cursor = NULL;


    L->cursorIndex = -1;
}

// copyList() and printList() implementations

void printList(FILE* out, List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling printList() on NULL List reference\n");
        exit(1);
    }
    Node n = L->front;
    while (n != NULL) {
        fprintf(out, "%d ", n->data);
        n = n->next;
    }
}

List copyList(List L) {
    if (L == NULL) {
        fprintf(stderr, "List Error: calling copyList() on NULL List reference\n");
        exit(1);
    }
    List copy = newList();
    Node n = L->front;
    while (n != NULL) {
        append(copy, n->data);
        n = n->next;
    }
    return copy;
}
