#include <stdio.h>
#include <stdlib.h>
#include "List.h"

typedef struct NodeObj {
    int data;
    struct NodeObj* prev;
    struct NodeObj* next;
} NodeObj;

typedef NodeObj* Node;

typedef struct ListObj {
    Node front;
    Node back;
    Node cursor;
    int length;
    int index;
} ListObj;


List newList(void) {
    List L = malloc(sizeof(ListObj));
    L->front = L->back = L->cursor = NULL;
    L->length = 0;
    L->index = -1;
    return L;
}

void freeList(List* pL) {
    if (pL != NULL && *pL != NULL) {
        while (length(*pL) > 0) {


            deleteFront(*pL);
        }
        free(*pL);
        *pL = NULL;
    }
}


int length(List L) {
    if (L == NULL) {
        printf("List Error: calling length() on NULL List reference\n");
        exit(1);
    }
    return L->length;
}

int index(List L) {
    if (L == NULL) {
        printf("List Error: calling index() on NULL List reference\n");
        exit(1);
    }
    return L->index;
}

int front(List L) {
    if (L == NULL || L->length <= 0) {
        printf("List Error: calling front() on an empty List\n");
        exit(1);
    }
    return L->front->data;
}

int back(List L) {
    if (L == NULL || L->length <= 0) {
        printf("List Error: calling back() on an empty List\n");
        exit(1);
    }
    return L->back->data;
}

int get(List L) {


    if (L == NULL || L->length <= 0 || L->index < 0) {
        printf("List Error: calling get() on an undefined cursor\n");
        exit(1);
    }
    return L->cursor->data;
}

bool equals(List A, List B) {
    if (A == NULL || B == NULL) {
        printf("List Error: calling equals() on NULL List reference\n");
        exit(1);
    }
    if (A->length != B->length) return false;
    Node N = A->front;
    Node M = B->front;
    while (N != NULL && M != NULL) {
        if (N->data != M->data) return false;


        N = N->next;
        M = M->next;
    }
    return true;
}

void clear(List L) {
    if (L == NULL) {
        printf("List Error: calling clear() on NULL List reference\n");
        exit(1);
    }
    while (L->length > 0) {
        deleteFront(L);
    }
    L->front = L->back = L->cursor = NULL;
    L->length = 0;
    L->index = -1;
}

void set(List L, int x) {
    if (L == NULL || L->length <= 0 || L->index < 0) {
        printf("List Error: calling set() on undefined cursor\n");
        exit(1);
    }
    L->cursor->data = x;
}

void moveFront(List L) {
    if (L == NULL || L->length <= 0) return;
    L->cursor = L->front;
    L->index = 0;
}

void moveBack(List L) {
    if (L == NULL || L->length <= 0) return;
    L->cursor = L->back;
    L->index = L->length - 1;
}

void movePrev(List L) {
    if (L == NULL || L->cursor == NULL || L->index == -1) return;
    if (L->cursor == L->front) {
        L->cursor = NULL;
        L->index = -1;
    } else {
        L->cursor = L->cursor->prev;
        L->index--;
    }
}

void moveNext(List L) {
    if (L == NULL || L->cursor == NULL || L->index == -1) return;
    if (L->cursor == L->back) {
        L->cursor = NULL;
        L->index = -1;
    } else {


        L->cursor = L->cursor->next;
        L->index++;
    }
}

void prepend(List L, int x) {
    if (L == NULL) {
        printf("List Error: calling prepend() on NULL List reference\n");
        exit(1);
    }
    Node N = malloc(sizeof(NodeObj));
    N->data = x;
    N->prev = NULL;
    N->next = L->front;
    if (L->front != NULL) L->front->prev = N;
    if (L->length == 0) L->back = N;
    L->front = N;


    L->length++;
    if (L->index != -1) L->index++;
}

void append(List L, int x) {
    if (L == NULL) {
        printf("List Error: calling append() on NULL List reference\n");
        exit(1);
    }
    Node N = malloc(sizeof(NodeObj));
    N->data = x;
    N->prev = L->back;
    N->next = NULL;
    if (L->back != NULL) L->back->next = N;
    if (L->length == 0) L->front = N;
    L->back = N;
    L->length++;
}

void insertBefore(List L, int x) {
    if (L == NULL || L->length <= 0 || L->index < 0) {
        printf("List Error: calling insertBefore() on undefined cursor\n");
        exit(1);
    }
    Node N = malloc(sizeof(NodeObj));
    N->data = x;
    N->prev = L->cursor->prev;
    N->next = L->cursor;
    if (L->cursor->prev != NULL) L->cursor->prev->next = N;
    else L->front = N;
    L->cursor->prev = N;
    L->length++;
    L->index++;
}

void insertAfter(List L, int x) {
    if (L == NULL || L->length <= 0 || L->index < 0) {
        printf("List Error: calling insertAfter() on undefined cursor\n");
        exit(1);
    }
    Node N = malloc(sizeof(NodeObj));
    N->data = x;
    N->next = L->cursor->next;
    N->prev = L->cursor;
    if (L->cursor->next != NULL) L->cursor->next->prev = N;
    else L->back = N;
    L->cursor->next = N;
    L->length++;
}

void deleteFront(List L) {
    if (L == NULL || L->length <= 0) {
        printf("List Error: calling deleteFront() on empty List\n");
        exit(1);
    }
    Node N = L->front;
    if (L->length > 1) {
        L->front = L->front->next;
        L->front->prev = NULL;
    } else {


        L->front = L->back = NULL;
    }
    free(N);
    L->length--;
    if (L->index != -1) L->index--;
}

void deleteBack(List L) {
    if (L == NULL || L->length <= 0) {
        printf("List Error: calling deleteBack() on empty List\n");
        exit(1);
    }
    Node N = L->back;
    if (L->length > 1) {
        L->back = L->back->prev;
        L->back->next = NULL;
    } else {
        L->front = L->back = NULL;
    }
    free(N);
    L->length--;
}

void delete(List L) {
    if (L == NULL || L->length <= 0 || L->index < 0) {
        printf("List Error: calling delete() on undefined cursor\n");
        exit(1);
    }
    Node N = L->cursor;
    if (N->prev != NULL) N->prev->next = N->next;
    else L->front = N->next;
    if (N->next != NULL) N->next->prev = N->prev;
    else L->back = N->prev;
    free(N);
    L->cursor = NULL;
    L->index = -1;
    L->length--;
}

void printList(FILE* out, List L) {
    if (L == NULL) {
        printf("List Error: calling printList() on NULL List reference\n");
        exit(1);
    }
    Node N = L->front;
    while (N != NULL) {
        fprintf(out, "%d ", N->data);
        N = N->next;
    }
}

List copyList(List L) {
    if (L == NULL) {
        printf("List Error: calling copyList() on NULL List reference\n");
        exit(1);
    }
    List copy = newList();
    Node N = L->front;
    while (N != NULL) {
        append(copy, N->data);
        N = N->next;
    }
    return copy;
}
