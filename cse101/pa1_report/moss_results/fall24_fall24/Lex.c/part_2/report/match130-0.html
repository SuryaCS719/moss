#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LEN 160

int main(int argc, char* argv[]) {
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);


        return EXIT_FAILURE;
    }

    FILE* in = fopen(argv[1], "r");
    FILE* out = fopen(argv[2], "w");

    if (!in || !out) {
        fprintf(stderr, "Error: unable to open files.\n");
        return EXIT_FAILURE;
    }

    // Count lines in input file
    int lineCount = 0;
    char line[MAX_LEN];
    while (fgets(line, MAX_LEN, in)) lineCount++;

    // Reset file pointer to read again
    rewind(in);

    // Allocate memory for string array
    char** lines = calloc(lineCount, sizeof(char*));
    if (lines == NULL) {
        fprintf(stderr, "Error: memory allocation failed.\n");
        fclose(in);
        fclose(out);
        return EXIT_FAILURE;
    }

    // Read lines and allocate memory for each string
    for (int i = 0; i < lineCount; i++) {
        if (fgets(line, MAX_LEN, in)) {
            lines[i] = strdup(line);  // strdup allocates memory and copies the line
            if (lines[i] == NULL) {
                fprintf(stderr, "Error: memory allocation for line failed.\n");
                fclose(in);
                fclose(out);
                return EXIT_FAILURE;
            }
        }
    }

    // Create a List of indices
    List indices = newList();
    append(indices, 0); // First index is trivially sorted

    // Insertion sort to build the List of indices
    for (int i = 1; i < lineCount; i++) {
        moveFront(indices);
        while (index(indices) >= 0 && strcmp(lines[get(indices)], lines[i]) < 0) {
            moveNext(indices);
        }
        if (index(indices) == -1) {
            append(indices, i);
        } else {
            insertBefore(indices, i);
        }
    }

    // Write the sorted lines to output
    moveFront(indices);
    while (index(indices) >= 0) {
        fprintf(out, "%s", lines[get(indices)]);
        moveNext(indices);
    }

    // Free memory
    for (int i = 0; i < lineCount; i++) {
        free(lines[i]);
    }
    free(lines);
    freeList(&indices);

    fclose(in);
    fclose(out);
    return EXIT_SUCCESS;
}
