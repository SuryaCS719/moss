#include "List.h"
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <assert.h>
#include <stdbool.h>

#define MAX_LEN 100

int main(int argc, char* argv[]) {


    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        return 1;
    }

    FILE* inFile = fopen(argv[1], "r");
    FILE* outFile = fopen(argv[2], "w");

    if (inFile == NULL || outFile == NULL) {
        fprintf(stderr, "Error: unable to open file %s or %s for reading or writing\n", argv[1], argv[2]);
        return 1;
    }

    // Count the number of lines in the input file
    int lineCount = 0;
    char line[MAX_LEN];
    while (fgets(line, MAX_LEN, inFile) != NULL) {
        lineCount++;
    }

    // Reset file pointer to the beginning of the file
    rewind(inFile);

    // Allocate memory for storing lines of input
    char** lines = calloc(lineCount, sizeof(char*));
    for (int i = 0; i < lineCount; i++) {
        fgets(line, MAX_LEN, inFile);
        lines[i] = calloc(strlen(line) + 1, sizeof(char));  // Allocate memory for each line
        strcpy(lines[i], line);  // Copy the line from the input file
    }

    // Create a List to hold the indices of lines in sorted order
    List indices = newList();

    // Insertion Sort: Insert indices into the List in lexicographic order
    for (int i = 0; i < lineCount; i++) {
        moveFront(indices);
        while (index(indices) >= 0 && strcmp(lines[i], lines[get(indices)]) > 0) {
            moveNext(indices);
        }
        if (index(indices) >= 0) {
            insertBefore(indices, i);
        } else {
            append(indices, i);  // If we're at the end or the list is empty, append
        }
    }

    // Output the lines in lexicographic order
    moveFront(indices);
    while (index(indices) >= 0) {
        fprintf(outFile, "%s", lines[get(indices)]);
        moveNext(indices);
    }

    // Free allocated memory
    for (int i = 0; i < lineCount; i++) {
        free(lines[i]);
    }
    free(lines);
    freeList(&indices);

    // Close files
    fclose(inFile);
    fclose(outFile);

    return 0;
}
