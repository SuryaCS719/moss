#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<stdbool.h>
#include "List.h"

#define MAX_LEN 300

int main(int argc, char* argv[]){
    // Step 1: Check for correct number of arguments
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        return EXIT_FAILURE;
    }

    FILE *in, *out;
    in = fopen(argv[1], "r");
    out = fopen(argv[2], "w");

    if (in == NULL) {


        fprintf(stderr, "Unable to open input file %s\n", argv[1]);
        return EXIT_FAILURE;
    }

    if (out == NULL) {
        fprintf(stderr, "Unable to open output file %s\n", argv[2]);
        fclose(in); // close input file before exiting
        return EXIT_FAILURE;
    }

    // Step 2: Count the number of lines in the input file
    char line[MAX_LEN];
    int lineCount = 0;

    // First pass: count the number of lines
    while (fgets(line, MAX_LEN, in) != NULL) {
        lineCount++;
    }

    /* test print statement
    printf("Line count =%d", lineCount);
    */

    // Allocate memory for string array
    char** lines = calloc(lineCount, sizeof(char*));
    if (lines == NULL) {
        fprintf(stderr, "Memory allocation failed.\n");
        fclose(in);
        fclose(out);
        return EXIT_FAILURE;
    }

    // Rewind the file to the beginning to read the lines into the array
    rewind(in);
    int i = 0;
    while (fgets(line, MAX_LEN, in) != NULL) {
        // Just in case the last line of the file isn't terminated by a newline character
        if (i == lineCount - 1 && line[strlen(line) - 1] != '\n') {
            size_t len = strlen(line);  // Get the current length before modification
            // Check if there's room to add a newline
            if (len < MAX_LEN - 1) { // Ensure space for a newline
                line[len] = '\n';  // Add a newline character at the current end
                line[len + 1] = '\0'; // Ensure null-termination right after the newline
            }

            // Test print statement to debug
            /*
            fprintf(out, "here: %i\n", i);
            */

        }

        lines[i] = calloc(strlen(line) + 1, sizeof(char));
        if (lines[i] == NULL) {
            fprintf(stderr, "Memory allocation failed.\n");
            fclose(in);
            fclose(out);
            return EXIT_FAILURE;
        }



        strcpy(lines[i], line); // Copy line into array
        i++;
    }

    /* test representation of the arrary, printed
    for (int i = 0; i < lineCount; i++) {
        printf("%s", lines[i]);
    }
    */


    // Step 3: Create a List and sort indices based on lexicographic order
    List L = newList();
    for (int i = 0; i < lineCount; i++) {
        if (length(L) == 0) {
            append(L, i); // Append first index (zero) directly
        } else {
            moveFront(L);
            while (index(L) >= 0 && strcmp(lines[i], lines[get(L)]) > 0) {
                moveNext(L);
            }
            if (index(L) == -1) {
                append(L, i); // If cursor falls off, append at the end
            } else {
                insertBefore(L, i); // Insert in correct sorted position
            }
        }
    }

    // Step 4: Use List to print lines in sorted order
    for (moveFront(L); index(L) >= 0; moveNext(L)) {
        fprintf(out, "%s", lines[get(L)]);
    }

    // Cleanup
    for (int i = 0; i < lineCount; i++) {
        free(lines[i]);
    }
    free(lines);
    freeList(&L);
    fclose(in);
    fclose(out);

    return EXIT_SUCCESS;
}
