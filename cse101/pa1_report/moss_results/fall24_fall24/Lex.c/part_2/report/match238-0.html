#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LINE_LENGTH 1000



int main(int argc, char* argv[]) {
    if (argc != 3) {
        fprintf(stderr, "Usage: Lex <input file> <output file>\n");
        return EXIT_FAILURE;
    }

    // handle input file
    FILE* in = fopen(argv[1], "r");
    if (in == NULL) {
        fprintf(stderr, "Error: Cannot open input file %s\n", argv[1]);
        return EXIT_FAILURE;
    }

   // Count the number of lines in the input file
    int n = 0;
    char buffer[MAX_LINE_LENGTH];
    while (fgets(buffer, sizeof(buffer), in) != NULL) {
        n++;
    }

    // Read lines from input file
    rewind(in);
    char** lines = malloc(n * sizeof(char*));
    if (lines == NULL) {
        fprintf(stderr, "Error: malloc failed\n");
        fclose(in);
        return EXIT_FAILURE;
    }
    for (int i = 0; i < n; i++) {
        if (fgets(buffer, sizeof(buffer), in) == NULL) {
            fprintf(stderr, "Error: Unexpected end of file\n");
            for (int j = 0; j < i; j++) {
                free(lines[j]);
            }
            free(lines);
            fclose(in);
            return EXIT_FAILURE;
        }

        buffer[strcspn(buffer, "\n")] = '\0'; // remove newline
        lines[i] = strdup(buffer);
        if (lines[i] == NULL) {
            fprintf(stderr, "Error: strdup failed\n");
            for (int j = 0; j < i; j++) {
                free(lines[j]);
            }
            free(lines);
            fclose(in);
            return EXIT_FAILURE;
        }
    }
    fclose(in);

    // Create and build list of index - SORTED
    List sortedIndices = newList();
    for (int i = 0; i < n; i++) {
        moveFront(sortedIndices);
        bool inserted = false;
        while (index(sortedIndices) >= 0) {
            int currentIdx = get(sortedIndices);
            if (strcmp(lines[i], lines[currentIdx]) < 0) {
                insertBefore(sortedIndices, i);
                inserted = true;
                break;
            }
            moveNext(sortedIndices);
        }
        if (!inserted) {
            append(sortedIndices, i);
        }
    }

    // Open output file
    FILE* out = fopen(argv[2], "w");
    if (out == NULL) {
        fprintf(stderr, "Error: Cannot open output file %s\n", argv[2]);
        freeList(&sortedIndices);
        for (int i = 0; i < n; i++) {
            free(lines[i]);
        }
        free(lines);
        return EXIT_FAILURE;
    }

    // Write to output file
    moveFront(sortedIndices);
    while (index(sortedIndices) >= 0) {
        int idx = get(sortedIndices);
        fprintf(out, "%s\n", lines[idx]);
        moveNext(sortedIndices);
    }

    fclose(out);
    freeList(&sortedIndices);
    for (int i = 0; i < n; i++) {
        free(lines[i]);
    }
    free(lines);

    return EXIT_SUCCESS;
}
