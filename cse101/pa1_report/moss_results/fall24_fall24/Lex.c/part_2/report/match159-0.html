#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <stdbool.h>
#include "List.h"

int main(int argc, char **argv) {
  
  FILE *in, *out;
  char **lines = NULL;
  char line[1000];
  int numLines = 0;

  char* input = argv[1];
  char* output = argv[2];

  if (argc != 3) {
    fprintf(stderr, "More or less than 2 command line arguments were given\n");
    exit(EXIT_FAILURE);
  } 

  in = fopen(input, "r");
  if (in == NULL) {
    fprintf(stderr, "File does not exist\n");
    exit(EXIT_FAILURE);
  }

  while (fgets(line, sizeof(line), in)) {
    numLines++;
    lines = realloc(lines, numLines * sizeof(char*));
    if (!lines) {
      fprintf(stderr, "Memory could not be reallocated\n");
      fclose(in);
      exit(EXIT_FAILURE);
    }
    lines[numLines - 1] = malloc((strlen(line) + 1));
    if (!lines[numLines - 1]) {


      fprintf(stderr, "Memory allocation failed\n");
      fclose(in);
      exit(EXIT_FAILURE);
    }
    strcpy(lines[numLines - 1], line);
    lines[numLines - 1][strcspn(lines[numLines - 1], "\n")] = '\0';
  }
  fclose(in);

  List indices = newList();
  if (numLines > 1) {


    append(indices, 0); 

    for (int i = 1; i < numLines; i++) {
      int ind1 = i;
      char* parseLine = lines[ind1];

      moveBack(indices);
      while (index(indices) >= 0) {
        int ind2 = get(indices);
        if (strcmp(lines[ind2], parseLine) > 0) {
          movePrev(indices);
        } else {
          break;
        }
      }
      if (index(indices) == -1) {
        prepend(indices, ind1);
      } else {
        insertAfter(indices, ind1);
      }
    }
  }


  out = fopen(output, "w");
  if (out == NULL) {
    freeList(&indices);
    fprintf(stderr, "File does not exist\n");
    exit(EXIT_FAILURE);
  }
  moveFront(indices);
  while (indices != NULL && index(indices) >= 0) {
    int x = get(indices);
    fprintf(out, "%s\n", lines[x]);
    moveNext(indices);
  }

  fclose(out);
  for (int i = 0; i < numLines; i++) {
    free(lines[i]);
  }
  free(lines);
  freeList(&indices);

  return 0;
}
