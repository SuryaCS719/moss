#include<stdio.h>
#include<stdlib.h>
#include<stdbool.h>
#include<string.h>
#include<assert.h>
#include "List.h"

#define MAX_LEN 300

int main(int argc, char * argv[]){
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        exit(EXIT_FAILURE);
    }

    FILE *in = fopen(argv[1], "r");
    if (in == NULL) {
        fprintf(stderr, "Can't open file %s for reading\n", argv[1]);
        exit(EXIT_FAILURE);
    }



    FILE *out = fopen(argv[2], "w");
    if (out == NULL) {
        fprintf(stderr, "Can't open file %s for writing\n", argv[2]);
        fclose(in);
        exit(EXIT_FAILURE);
    }

    int line_tot = 0;
    char buffer[MAX_LEN];

    while (fgets(buffer, MAX_LEN, in) != NULL) {
        line_tot++;
    }

    rewind(in);

    char **lines = calloc(line_tot, sizeof(char *));
    assert(lines != NULL);

    if (lines == NULL) {
        fprintf(stderr, "Memory allocation failed\n");
        fclose(in);
        fclose(out);
        exit(EXIT_FAILURE);
    }

    for (int i = 0; i < line_tot; i++) {
        lines[i] = calloc(MAX_LEN, sizeof(char)); 
        assert(lines[i] != NULL);

        fgets(lines[i], MAX_LEN, in); 
    }
    
    List L = newList();
    //append(L, 0);

    for (int i = 0; i < line_tot; i++) {
        moveFront(L);

        while (index(L) >= 0 && strcmp(lines[i], lines[get(L)]) > 0) {
            moveNext(L); 
        }

        if (index(L) == -1) {
            append(L, i); 
        } else {
           insertBefore(L, i); 
        }
    }

    for (moveFront(L); index(L) >= 0; moveNext(L)) {
        fprintf(out, "%s", lines[get(L)]);
    }


    for (int i = 0; i < line_tot; i++) {
        free(lines[i]);
    }
    
    free(lines);
    freeList(&L);
    fclose(in);
    fclose(out);

    return 0;
}
