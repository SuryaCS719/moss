#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

// custom strdup function to duplicate a string


char* my_strdup(const char* s) {
    size_t len = strlen(s) + 1;
    char* dup = malloc(len);
    if (dup) {
        memcpy(dup, s, len);
    }
    return dup;
}

int main(int argc, char* argv[]) {
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        return EXIT_FAILURE;
    }

    FILE* in = fopen(argv[1], "r");
    FILE* out = fopen(argv[2], "w");
    if (in == NULL || out == NULL) {
        fprintf(stderr, "Error: cannot open file\n");
        if (in) fclose(in);
        if (out) fclose(out);
        return EXIT_FAILURE;
    }

    // count the number of lines in the input file
    char line[256];
    int lineCount = 0;
    while (fgets(line, sizeof(line), in) != NULL) {
        lineCount++;
    }
    rewind(in);

    // Allocate memory for lines and read them into the array


    char** lines = calloc(lineCount, sizeof(char*));
    for (int i = 0; i < lineCount; i++) {
        fgets(line, sizeof(line), in);
        lines[i] = my_strdup(line);  // Use my_strdup instead of strdup
    }

    // Create a List to store indices in sorted order
    List L = newList();
    append(L, 0); // Start with the first element

    for (int i = 1; i < lineCount; i++) {
        moveFront(L);
        while (index(L) >= 0 && strcmp(lines[i], lines[get(L)]) > 0) {
            moveNext(L);
        }
        if (index(L) >= 0) {
            insertBefore(L, i);
        } else {
            append(L, i);
        }
    }

    // Write the sorted lines to the output file
    for (moveFront(L); index(L) >= 0; moveNext(L)) {
        fprintf(out, "%s", lines[get(L)]);
    }

    // Clean up memory
    for (int i = 0; i < lineCount; i++) {
        free(lines[i]);
    }
    free(lines);
    freeList(&L);
    fclose(in);
    fclose(out);

    return EXIT_SUCCESS;
}
