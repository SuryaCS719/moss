#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h" 
#define MAX_LEN 256

int main(int argc, char* argv[]){
    
    FILE *input, *output; // Input, output file
    int lineCounter = 0; // Line counter
    char ch;
    // Check if 3 command line arguments are given
    if(argc != 3){
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        exit(EXIT_FAILURE);
            }

    // Open the input file for reading
    input = fopen(argv[1], "r");
    if(input == NULL){
        fprintf(stderr, "Fail to open file %s\n", argv[1]);
        exit(EXIT_FAILURE);
    }

    
    // Count lines in the input file
    while ((ch = getc(input)) != EOF){
        if(ch == '\n' || ch == '\r'){
            lineCounter++;
        }
    }

    rewind(input); // back to the beginning of the file

    // Allocate memory for array
    char** lines = (char**)calloc(lineCounter, sizeof(char*));
    if(lines == NULL){
        fprintf(stderr, "Fail to allocate memory for array\n");
        exit(EXIT_FAILURE);
    }

    // Allocate memory for each array element
    for(int i = 0; i < lineCounter; i++){
        lines[i] = (char*)calloc(MAX_LEN, sizeof(char));
        if (lines[i] == NULL){
            fprintf(stderr,"Memory allocation failed for string %d\n", i);
            exit(EXIT_FAILURE);
        }
    }

    // Read lines and store it in the array
    for (int i = 0; i < lineCounter; i++){
        if (fgets(lines[i], MAX_LEN, input) == NULL){
            fprintf(stderr, "Fail to read line  %d from file %s\n", i, argv[1]);
            exit(EXIT_FAILURE);
        }
    }
    // Close the input file
    fclose(input);

    List list = newList();
    if (list == NULL) {
        fprintf(stderr, "Fail to create a new list\n");
        exit(EXIT_FAILURE);
    }

    append(list, 0);  // Add the first line

    // Use insertion sort to add lines to the list
    for(int i = 1; i < lineCounter; i++){
        moveBack(list); // Move the cursor to the back of the list for traversal
        while(index(list)>=0){
            // Check if lines[i] is smaller than current line

            
           if (strcmp(lines[i], lines[get(list)]) >= 0) {
            insertAfter(list, i);
            break;
        }
        
        // Move the cursor one step back
        movePrev(list);
        }
        

        // Lines[i] is the smallest line, insert at the front
        if(index(list) == -1){
            prepend(list, i);
        }
        
    }


    // Open output file


    output = fopen(argv[2], "w");
    if(output == NULL){
        fprintf(stderr, "Fail to open file %s\n", argv[2]);
        freeList(&list);
        for (int i = 0; i < lineCounter; i++) {
            free(lines[i]);
        }
        free(lines);
        exit(EXIT_FAILURE);
    }

    // Write sorted list to output file
    moveFront(list);
    while (index(list) >= 0) {
        fprintf(output, "%s", lines[get(list)]);
        moveNext(list);
    }

    // Cleanup
    fclose(output);
    freeList(&list);
    // Free array elements
    for (int i = 0; i < lineCounter; i++) {
        free(lines[i]);
    }
    // Free array
    free(lines);

    return EXIT_SUCCESS;    
}
