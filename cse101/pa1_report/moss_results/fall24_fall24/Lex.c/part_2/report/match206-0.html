#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LEN 160

int main(int argc, char* argv[]) {
    if (argc != 3) {
	fprintf(stderr, "Error either too many or too few arguments given\n");
        exit(1);
    }

    FILE* in = fopen(argv[1], "r");
    if (in == NULL) {
        exit(1);
    }

    FILE* out = fopen(argv[2], "w");
    if (out == NULL) {
        fclose(in);
        exit(1);
    }



    char line[MAX_LEN];
    char* lines[1000];
    int numLines = 0;

    while (fgets(line, MAX_LEN, in) != NULL) {
        lines[numLines] = malloc(strlen(line) + 1);
        strcpy(lines[numLines], line);
        numLines++;
    }

    List list = newList();

    for (int i = 0; i < numLines; i++) {
        if (length(list) == 0) {
            append(list, i);
        } else {
            moveFront(list);
            while (index(list) >= 0 && strcmp(lines[get(list)], lines[i]) < 0) {
                moveNext(list);
            }
            if (index(list) >= 0) {
                insertBefore(list, i);
            } else {
                append(list, i);
            }
        }
    }

    moveFront(list);
    while (index(list) >= 0) {


        fprintf(out, "%s", lines[get(list)]);
        moveNext(list);
    }

    for (int i = 0; i < numLines; i++) {
        free(lines[i]);
    }
    freeList(&list);
    fclose(in);
    fclose(out);

    return 0;
}
