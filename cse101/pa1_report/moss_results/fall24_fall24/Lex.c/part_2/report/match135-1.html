* Serves as the main program that reads lines from an input file, sorts them lexicographically using the List ADT, and writes the sorted lines to an output file.
***/

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LINE_LENGTH 1024

int main(int argc, char* argv[]) {
    if (argc != 3) {
        fprintf(stderr, "Usage: Lex <input file> <output file>\n");
        return EXIT_FAILURE;
    }

    char* inputFile = argv[1];
    char* outputFile = argv[2];

    FILE* in = fopen(inputFile, "r");
    if (in == NULL) {
        fprintf(stderr, "Error: Cannot open input file %s\n", inputFile);
        return EXIT_FAILURE;
    }

    int n = 0;
    char buffer[MAX_LINE_LENGTH];
    while (fgets(buffer, MAX_LINE_LENGTH, in) != NULL) {
        n++;
    }



    char** lines = calloc(n, sizeof(char*));
    if (lines == NULL) {
        fprintf(stderr, "Error: Memory allocation failed\n");
        fclose(in);
        return EXIT_FAILURE;
    }

    rewind(in);
    int i = 0;
    while (fgets(buffer, MAX_LINE_LENGTH, in) != NULL) {
        size_t len = strlen(buffer);
        if (len > 0 && buffer[len - 1] == '\n') {
            buffer[len - 1] = '\0';
        }


        lines[i] = malloc((strlen(buffer) + 1) * sizeof(char));
        if (lines[i] == NULL) {
            fprintf(stderr, "Error: Memory allocation failed\n");
            for (int j = 0; j < i; j++) {
                free(lines[j]);
            }
            free(lines);
            fclose(in);
            return EXIT_FAILURE;
        }
        strcpy(lines[i], buffer);
        i++;
    }
    fclose(in);

    List indexList = newList();
    for (int current = 0; current < n; current++) {
        if (length(indexList) == 0) {
            append(indexList, current);
            continue;
        }
        moveFront(indexList);
        bool inserted = false;
        while (index(indexList) >= 0) {
            int listIndexValue = get(indexList);
            if (strcmp(lines[current], lines[listIndexValue]) < 0) {
                insertBefore(indexList, current);
                inserted = true;
                break;
            }
            moveNext(indexList);
        }
        if (!inserted) {
            append(indexList, current);
        }
    }

    FILE* out = fopen(outputFile, "w");
    if (out == NULL) {


        fprintf(stderr, "Error: Cannot open output file %s\n", outputFile);
        for (int j = 0; j < n; j++) {
            free(lines[j]);
        }
        free(lines);
        freeList(&indexList);
        return EXIT_FAILURE;
    }

    moveFront(indexList);
    while (index(indexList) >= 0) {
        int sortedIndex = get(indexList);
        fprintf(out, "%s\n", lines[sortedIndex]);
        moveNext(indexList);
    }
    fclose(out);

    for (int j = 0; j < n; j++) {
        free(lines[j]);
    }
    free(lines);
    freeList(&indexList);

    printf("Sorting completed. Sorted lines written to %s\n", outputFile);

    return EXIT_SUCCESS;
}
