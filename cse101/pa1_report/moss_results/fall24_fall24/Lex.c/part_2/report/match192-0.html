#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LEN 160 // Assume maximum character length per line

int main(int argc, char* argv[]) {
    // Check command line parameters
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        return EXIT_FAILURE;
    }

    FILE *in, *out;
    char line[MAX_LEN];
    int lineCount = 0;

    // Read the file
    in = fopen(argv[1], "r");
    if (in == NULL) {
        fprintf(stderr, "Error: could not open input file %s\n", argv[1]);
        return EXIT_FAILURE;
    }

    // count the lines
    while (fgets(line, MAX_LEN, in) != NULL) {
        lineCount++;
    }
    rewind(in); // reset

    // Allocate an array of strings to store each line
    char **lines = calloc(lineCount, sizeof(char*));


    for (int i = 0; i < lineCount; i++) {
        fgets(line, MAX_LEN, in);
        lines[i] = calloc(strlen(line) + 1, sizeof(char));
        strcpy(lines[i], line); 
    }

    fclose(in); // close file

    // create list
    List list = newList();
    append(list, 0); // add first index

    for (int i = 1; i < lineCount; i++) {
        moveFront(list);
        while (index(list) >= 0 && strcmp(lines[i], lines[get(list)]) > 0) {
            moveNext(list);
        }
        if (index(list) >= 0) {


            insertBefore(list, i);
        } else {
            append(list, i);
        }
    }

    // create the output file
    out = fopen(argv[2], "w");
    if (out == NULL) {
        fprintf(stderr, "Error: could not open output file %s\n", argv[2]);
        return EXIT_FAILURE;
    }

    // write output
    moveFront(list);
    while (index(list) >= 0) {
        fprintf(out, "%s", lines[get(list)]);
        moveNext(list);
    }

    fclose(out); // close the file

    // free the memory
    for (int i = 0; i < lineCount; i++) {
        free(lines[i]);
    }
    free(lines);
    freeList(&list);

    return EXIT_SUCCESS;
}
