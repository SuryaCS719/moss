#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LEN 300
// used example file FILEIO.c as a reference/inspo
int main(int argc, char *argv[]) {
    if (argc != 3) { //check number of command line arguments
        fprintf(stderr, "Usage: %s <input file> <outputfile>\n", argv[0]); 
        exit(EXIT_FAILURE);
    }

    
//opening input and output files
    FILE *in = fopen(argv[1], "r");
    if (in == NULL) {
        fprintf(stderr, "Unable to open file %s for reading\n", argv[1]);
        return (EXIT_FAILURE);
    }

    FILE *out = fopen(argv[2], "w");


        if (out == NULL) {
            fprintf(stderr, "Unable to open file %s for writing\n", argv[2]);
            fclose(in);
            exit(EXIT_FAILURE);
        }
// gather number of lines
        int count = 0;
        char line[MAX_LEN];

        while (fgets(line, MAX_LEN, in) != NULL) {
            count++;
        }

    fclose(in);
    //open input file again with case of file not being able to open
    in = fopen(argv[1], "r");
    if (in == NULL) {
        fprintf(stderr, "Unable to reopen file %s\n", argv[1]);
        exit(EXIT_FAILURE);
    }

//allocate memory for array of strings


    char **lines = calloc(count, sizeof(char *));
    if (lines == NULL) {
        fprintf(stderr, "Allocation failed\n");
        fclose(in);
        fclose(out);
        exit(EXIT_FAILURE);
    }

//read each line and store in array
    for (int i = 0; i < count; i++) {
        if (fgets(line, MAX_LEN, in) != NULL){
            line[strcspn(line, "\n")] = '\0';
            lines[i] = calloc(strlen(line) + 1, sizeof(char));
        
            if (lines[i] != NULL) {
                strcpy(lines[i], line);
            }
        }
    }

//creates new list
    List sort = newList();
    

//sorting using operations from List.c
    for (int i = 0; i < count; i++) {
        moveFront(sort);
        while (index(sort) >= 0 && strcmp(lines[get(sort)], lines[i]) < 0) {
            moveNext(sort);
        }
        if (index(sort) >= 0) {
            insertBefore(sort, i);
        } else {
            append(sort, i);
        }
    }

    //writes lines onto output file
    moveFront(sort);
    while (index(sort) >= 0) {
        fprintf(out, "%s\n", lines[get(sort)]);
        moveNext(sort);
    }
//freeing memory
    for (int i = 0; i < count; i++) {
        free(lines[i]);
    }

    free(lines);
    freeList(&sort);
    fclose(in);
    fclose(out);

    return 0;
}
