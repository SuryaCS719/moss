#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define _GNU_SOURCE
#define MAX_LEN 160

int main(int argc, char* argv[]) {
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        return EXIT_FAILURE;
    }

    FILE* in = fopen(argv[1], "r");
    FILE* out = fopen(argv[2], "w");

    if (in == NULL || out == NULL) {
        fprintf(stderr, "Error: could not open input/output file.\n");
        return EXIT_FAILURE;
    }

    char line[MAX_LEN];
    int lineCount = 0;

    // First pass: count lines
    while (fgets(line, MAX_LEN, in) != NULL) {
        lineCount++;
    }

    // Rewind to read lines into array
    rewind(in);
    char** lines = calloc(lineCount, sizeof(char*));



    for (int i = 0; i < lineCount; i++) {
        fgets(line, MAX_LEN, in);
	lines[i] = malloc(strlen(line) + 1);
	strcpy(lines[i], line);
    }

    // Create a List for the indices
    List indices = newList();
    append(indices, 0);  // Start with first line

    // Insertion sort indices based on lexicographic order
    for (int i = 1; i < lineCount; i++) {
        moveFront(indices);
        while (index(indices) >= 0 && strcmp(lines[i], lines[get(indices)]) > 0) {
            moveNext(indices);
        }
        if (index(indices) == -1) {
            append(indices, i);  // Append if it belongs at the end
        } else {


            insertBefore(indices, i);  // Insert at correct position
        }
    }

    // Write sorted lines to output file
    moveFront(indices);
    while (index(indices) >= 0) {
        fprintf(out, "%s", lines[get(indices)]);
        moveNext(indices);
    }

    // Free allocated memory
    freeList(&indices);
    for (int i = 0; i < lineCount; i++) {
        free(lines[i]);
    }
    free(lines);
    fclose(in);
    fclose(out);

    return EXIT_SUCCESS;
}
