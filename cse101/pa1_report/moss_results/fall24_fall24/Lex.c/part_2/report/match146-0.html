#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LEN 200



int main(int argc, char* argv[]) {
    if (argc != 3) {
        printf("Usage: %s <input file> <output file>\n", argv[0]);
        exit(1);
    }

    FILE* inFile = fopen(argv[1], "r");
    FILE* outFile = fopen(argv[2], "w");
    if (inFile == NULL || outFile == NULL) {
        printf("Error: Unable to open file(s)\n");


        exit(1);
    }

    int lineCount = 0;
    char line[MAX_LEN];
    while (fgets(line, MAX_LEN, inFile) != NULL) {
        lineCount++;
    }
    rewind(inFile);

    char** lines = calloc(lineCount, sizeof(char*));
    for (int i = 0; i < lineCount; i++) {
        fgets(line, MAX_LEN, inFile);
	lines[i] = malloc(strlen(line) + 1);
	strcpy(lines[i], line);
    }

    List indexList = newList();
    append(indexList, 0);

    for (int i = 1; i < lineCount; i++) {
        moveFront(indexList);
        while (index(indexList) >= 0) {
            if (strcmp(lines[i], lines[get(indexList)]) < 0) {
                insertBefore(indexList, i);
                break;
            }
            moveNext(indexList);
        }
        if (index(indexList) < 0) {
            append(indexList, i);
        }
    }

    moveFront(indexList);
    while (index(indexList) >= 0) {
        fprintf(outFile, "%s", lines[get(indexList)]);  
	moveNext(indexList);
    }

    // Free the memory
    for (int i = 0; i < lineCount; i++) {
        free(lines[i]);
    }
    free(lines);  
    freeList(&indexList);

    fclose(inFile);
    fclose(outFile);

    return 0;
}
