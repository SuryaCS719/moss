int main(int argc, char * argv[]) {

    // int token_count, line_count;
    FILE *in, *out;
    // char line[MAX_LEN];
    // char tokenBuffer[MAX_LEN];
    // char* token;

    // Check that there are two command line arguments (other than the program name Lex). 
    // Quit with a usage message to stderr if more than or less than two command line arguments are given
    if( argc != 3 ){
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        exit(1);
    }
    printf("%s %s %s\n", argv[0], argv[1], argv[2]);
    
    // Open files for reading and writing 
    in = fopen(argv[1], "r");
        if( in==NULL ){
        fprintf(stderr, "Unable to open file %s for reading\n", argv[1]);
        exit(1);
    }

    out = fopen(argv[2], "w");
    if( out==NULL ){
        fprintf(stderr, "Unable to open file %s for writing\n", argv[2]);
        exit(1);
    }
    
    // Count the number of lines n in the input file. 


    int n = 0;
    char line[256];
    while(fgets(line, sizeof(line), in)!=NULL){
        n++;
    }

    // printf("number of lines: %d\n", n);             // testing
    rewind(in);
    // Create a string array of length n
    // (Allocate this array from heap memory using functions
    // calloc() or malloc() defined in the header file stdlib.h. 
    char** arr = calloc(n, sizeof(char*));
    
    // Read in the lines of the
    // file as strings, placing them into the array. 
    for(int i=0;i<n;i++){
        fgets(line, sizeof(line), in);
        line[strcspn(line, "\n")] = '\0';
        arr[i] = calloc(strlen(line)+1, sizeof(char));
        // printf("after calloc: %p\n", (void*)arr[i]);
        strcpy(arr[i], line);
    }
    
    // printf("arr at end: \n");
    // for (int i = 0; i < n; i++) {
    //     printf("%s ", arr[i]);
    // }
    // printf("\n");

    // Create a List whose elements are the indices of the above string array. 
    // These indices should be arranged in an order that indirectly sorts the array.

    // int temp[4] = {5, 4, 1, 7};

    List list = newList();

    for (int i = 0; i < n; i++) {
        while ( index(list) != -1 ) {
            if ( strcmp(arr[get(list)], arr[i]) < 0 ) {
                insertAfter(list, i);
                break;
            } else {
                movePrev(list);
            }
        }
        if ( index(list) == -1 ) {
            prepend(list, i);
        } 
        moveBack(list);
    }
    
   
    // printf("after sorting: ");
    // printList(stdout, list);
    // printf("\n");

    // Write the elements to the output file
    moveFront(list);
    while (index(list) != -1) {
        fprintf(out, "%s\n", arr[get(list)]);
        moveNext(list);
    }

    // Free all memory used
    for(int i = 0; i < n; i++) {
        // printf("inside loop: %p %s\n", (void*)arr[i], arr[i]);
        free(arr[i]);  // Free each string
        arr[i] = NULL;
        // printf("after free: %s\n", arr[i]);
    }
    free(arr);  // Free the array itself
    arr = NULL; // Set to NULL to avoid dangling pointers


    // printf("list before free: ");
    // printList(stdout, list);
    // printf("\n");
    // printf("length before free: %d", length(list));
    freeList(&list);  // Free the list and all its nodes
    // printf("freeList() call done\n");
    list = NULL;



    // Close files 
    fclose(in);
    fclose(out);

    return(0);

}
