// order through List ADT, and writes the sorted lines to an output file.

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LINE_LENGTH 256  // Assume maximum length of each line



void usage() {
    fprintf(stderr, "Usage: Lex <input file> <output file>\n");
    exit(EXIT_FAILURE);
}

int main(int argc, char* argv[]) {
    if (argc != 3)
        usage();

    // Open input file
    FILE* inFile = fopen(argv[1], "r");
    if (inFile == NULL) {
        fprintf(stderr, "Error: could not open input file %s\n", argv[1]);
        exit(EXIT_FAILURE);
    }

    // Open output file
    FILE* outFile = fopen(argv[2], "w");
    if (outFile == NULL) {
        fprintf(stderr, "Error: could not open output file %s\n", argv[2]);
        fclose(inFile);
        exit(EXIT_FAILURE);
    }

    // Read all lines and store them in a string array
    char** lines = NULL;
    char buffer[MAX_LINE_LENGTH];
    int lineCount = 0;
    
    // Count the number of lines in the file
    while (fgets(buffer, MAX_LINE_LENGTH, inFile) != NULL) 
        lineCount++;
    
    // Allocate an array of line pointers
    lines = calloc(lineCount, sizeof(char*));
    rewind(inFile);  // Reset the file pointer
    
    // Read all the lines
    int i = 0;
    while (fgets(buffer, MAX_LINE_LENGTH, inFile) != NULL) {
        lines[i] = calloc(strlen(buffer) + 1, sizeof(char));
        strcpy(lines[i], buffer);
        i++;
    }
    
    // Use List to store line indices and sort them
    List list = newList();
    append(list, 0);  // First index

    // Insert sort the indices into the List
    for (i = 1; i < lineCount; i++) {
        moveFront(list);
        while (index(list) >= 0 && strcmp(lines[get(list)], lines[i]) <= 0)
            moveNext(list);
        if (index(list) == -1)
            append(list, i);
        else
            insertBefore(list, i);
    }

    // Output the lines to the file in sorted order according to indices
    moveFront(list);
    while (index(list) >= 0) {
        char* line = lines[get(list)];
        // Check if the line ends with a newline, if not, add one
        if (line[strlen(line) - 1] != '\n')
            fprintf(outFile, "%s\n", line);
        else
            fprintf(outFile, "%s", line);  // line already contains a newline
        moveNext(list);
    }

    // Free memory and close files
    for (i = 0; i < lineCount; i++)
        free(lines[i]);
    
    free(lines);
    freeList(&list);
    fclose(inFile);
    fclose(outFile);

    return 0;
}
