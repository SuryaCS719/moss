#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <assert.h>
#include <stdbool.h>
#include "List.h"

#define MAX_LINE_LENGTH 1024

int main(int argc, char *argv[]) {
    if (argc != 3) {
        fprintf(stderr, "Usage of Lex.c:  input_file output_file\n");
        return EXIT_FAILURE;
    }

    FILE* in = fopen(argv[1], "r");
    FILE* out = fopen(argv[2], "w");
    
    if (in == NULL) {
        fprintf(stderr, "Error opening input file: %s\n", argv[1]);
        return EXIT_FAILURE;
    }


    if (out == NULL) {
        fprintf(stderr, "Error opening output file: %s\n", argv[2]);
        fclose(in);
        return EXIT_FAILURE;
    }

    int count = 0;
    char line[MAX_LINE_LENGTH]; // Buffer to read lines

    // First pass: count lines
    while (fgets(line, sizeof(line), in) != NULL) {
        count++; // Increment the line count
    }

    rewind(in); // Rewind the file to the beginning

    // Allocate memory for lines


    char** lines = calloc(count, sizeof(char*));
    if (lines == NULL) {
        fprintf(stderr, "Memory allocation failed\n");
        fclose(in);
        fclose(out);
        return EXIT_FAILURE;
    }

    // Second pass: store lines
    for (int i = 0; i < count; i++) {
        if (fgets(line, sizeof(line), in) == NULL) {
            fprintf(stderr, "Error reading line %d\n", i + 1);
            fclose(in);
            fclose(out);
            for (int j = 0; j < i; j++) {
                free(lines[j]); // Free any previously allocated lines
            }
            free(lines);
            return EXIT_FAILURE;
        }
        lines[i] = malloc(strlen(line) + 1); // Allocate memory for each line
        if (lines[i] == NULL) {
            fprintf(stderr, "Memory allocation failed\n");
            fclose(in);
            fclose(out);
            for (int j = 0; j < i; j++) {
                free(lines[j]); // Free any previously allocated lines
            }
            free(lines);
            return EXIT_FAILURE;
        }
        strcpy(lines[i], line); // Copy the line into the array
    }

    fclose(in); // Close input file

    // Create the list and insert elements
    List list = newList();
    append(list, 0); // Insert the first index

    for (int i = 1; i < count; i++) {
        moveFront(list);
        while (index(list) != -1 && strcmp(lines[i], lines[get(list)]) > 0) {
            moveNext(list); // Find correct position for the new index
        }
        if (index(list) == -1) {
            append(list, i); // If end of the list, append the index
        } else {
            insertBefore(list, i); // Otherwise, insert before the current element
        }
    }

    // Print lines in lexicographic order
    moveFront(list);
    while (index(list) != -1) {
        fprintf(out, "%s", lines[get(list)]);
        moveNext(list);
    }

    // Free allocated memory
    for (int i = 0; i < count; i++) {
        free(lines[i]); // Free each line
    }
    
    free(lines); // Free the array
    freeList(&list); // Free the List

    fclose(out); // Close output file

    return EXIT_SUCCESS;
}
