#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LEN 160

int main(int argc, char* argv[]) {
    // Check for correct number of command line arguments
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        exit(EXIT_FAILURE);
    }

    // Open the input file for reading
    FILE* inFile = fopen(argv[1], "r");
    if (inFile == NULL) {
        fprintf(stderr, "Unable to open file %s for reading\n", argv[1]);
        exit(EXIT_FAILURE);
    }

    // Open the output file for writing


    FILE* outFile = fopen(argv[2], "w");
    if (outFile == NULL) {
        fprintf(stderr, "Unable to open file %s for writing\n", argv[2]);
        exit(EXIT_FAILURE);
    }

    // Count the number of lines in the input file
    char buffer[MAX_LEN];
    int lineCount = 0;
    while (fgets(buffer, MAX_LEN, inFile) != NULL) {
        lineCount++;
    }

    // Allocate memory for the array of strings
    char** lines = calloc(lineCount, sizeof(char*));
    if (lines == NULL) {
        fprintf(stderr, "Memory allocation failed\n");
        exit(EXIT_FAILURE);
    }

    // Rewind the file to read the lines again
    rewind(inFile);

    // Read the lines into the array, manually copying them
    int i = 0;
    while (fgets(buffer, MAX_LEN, inFile) != NULL) {
        // Allocate memory for each line and copy it
        lines[i] = malloc((strlen(buffer) + 1) * sizeof(char)); 
        if (lines[i] == NULL) {
            fprintf(stderr, "Memory allocation failed\n");
            exit(EXIT_FAILURE);
        }
        strcpy(lines[i], buffer); 
        i++;
    }

    // Create an empty List of indices
    List indexList = newList();
    append(indexList, 0); 

    // Perform an insertion sort on the indices based on the lines
    for (i = 1; i < lineCount; i++) {
        char* currentString = lines[i];
        moveFront(indexList);

        // Traverse the list to find the correct insertion point
        while (index(indexList) >= 0) {
            int currentIndex = get(indexList);
            char* listString = lines[currentIndex];

            if (strcmp(currentString, listString) < 0) {
                insertBefore(indexList, i);
                break;
            }
            moveNext(indexList);
        }

        // If we reached the end of the list, append at the end
        if (index(indexList) == -1) {
            append(indexList, i);
        }
    }

    // Use the ordered list of indices to print lines in sorted order to the output file
    moveFront(indexList);
    while (index(indexList) >= 0) {
        fprintf(outFile, "%s", lines[get(indexList)]);
        moveNext(indexList);
    }

    // Free allocated memory
    for (i = 0; i < lineCount; i++) {
        free(lines[i]);
    }
    free(lines);

    // Close files
    fclose(inFile);
    fclose(outFile);

    // Free the List
    freeList(&indexList);

    return 0;
}
