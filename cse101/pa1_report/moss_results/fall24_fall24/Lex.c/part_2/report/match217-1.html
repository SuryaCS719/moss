* Reads lines from input file, sorts them using the List ADT, and writes to an output file
***/
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"


#define MAX_LEN 200



char *strdup(const char *s);

int main(int argc, char *argv[]) {
    FILE *in, *out;
    char lines[MAX_LEN];

    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        exit(1);
    }

    in = fopen(argv[1], "r");
    if (in == NULL) {
        fprintf(stderr, "Unable to open file %s for reading\n", argv[1]);
        exit(1);
    }
    
    out = fopen(argv[2], "w");
    if (out == NULL) {
        fprintf(stderr, "Unable to open file %s for writing\n", argv[2]);
        exit(1);
    }

    int n = 0;
    while (fgets(lines, MAX_LEN, in) != NULL) {
        n++;
    }

    char **string_arr = malloc(n * sizeof(char *));
    if (string_arr == NULL) {
        fprintf(stderr, "Memory allocation failed\n");
        exit(1);
    }
    rewind(in);

    for (int i = 0; i < n; i++) {
        fgets(lines, MAX_LEN, in);
        string_arr[i] = strdup(lines); 
        if (string_arr[i] == NULL) {
            fprintf(stderr, "Memory allocation failed\n");
            exit(1);
        }
    }



    List l = newList();
    for (int i = 0; i < n; i++) {
        append(l, i);
    }
    clear(l);

    for (int i = 0; i < n; i++) {
        char *temp = string_arr[i];
        moveFront(l);
        while (index(l) >= 0 && strcmp(temp, string_arr[get(l)]) > 0) {
            moveNext(l);
        }
        if (index(l) == -1) {
            append(l, i);
        } else {
            insertBefore(l, i);
        }
    }

    moveFront(l);
    while (index(l) >= 0) {
        fprintf(out, "%s", string_arr[get(l)]);
        moveNext(l);
    }

    for (int i = 0; i < n; i++) {
        free(string_arr[i]);
    }
    free(string_arr);
    freeList(&l);
    fclose(in);
    fclose(out);

    return 0;
}
