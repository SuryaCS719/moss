#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LEN 160

int main(int argc, char* argv[]) {
    // 1. Check that two command line arguments are provided
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        exit(EXIT_FAILURE);
    }

    // 2. Open input and output files
    FILE* in = fopen(argv[1], "r");
    FILE* out = fopen(argv[2], "w");
    if (in == NULL || out == NULL) {
        fprintf(stderr, "Error: Unable to open file %s or %s\n", argv[1], argv[2]);
        exit(EXIT_FAILURE);
    }

    // 3. Count the number of lines in the input file
    int lineCount = 0;
    char line[MAX_LEN];


    while (fgets(line, MAX_LEN, in) != NULL) {
        lineCount++;
    }

    // 4. Allocate memory for array of strings and rewind the file
    char** lines = calloc(lineCount, sizeof(char*));
    rewind(in);

    // 5. Read the lines into the array
    for (int i = 0; i < lineCount; i++) {
        fgets(line, MAX_LEN, in);


        lines[i] = calloc(strlen(line) + 1, sizeof(char));
        strcpy(lines[i], line);
    }

    // 6. Create a new List to hold the indices
    List L = newList();

    // 7. Insertion sort of the indices using List ADT and strcmp for comparison
    append(L, 0);  // Add the first index

    for (int i = 1; i < lineCount; i++) {
        moveFront(L);
        while (index(L) >= 0) {
            if (strcmp(lines[i], lines[get(L)]) < 0) {
                insertBefore(L, i);  // Insert before the cursor
                break;
            }
            moveNext(L);
        }
        if (index(L) < 0) {
            append(L, i);  // If not inserted, append to the end
        }
    }

    // 8. Write the lines in lexicographic order to the output file
    moveFront(L);
    while (index(L) >= 0) {
        fprintf(out, "%s", lines[get(L)]);  // Output sorted lines
        moveNext(L);
    }

    // 9. Free allocated memory and close files
    for (int i = 0; i < lineCount; i++) {
        free(lines[i]);
    }
    free(lines);
    freeList(&L);
    fclose(in);
    fclose(out);

    return 0;
}
