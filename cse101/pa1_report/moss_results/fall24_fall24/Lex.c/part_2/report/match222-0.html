#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

int main(int argc, char* argv[]) {
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        return EXIT_FAILURE;
    }

    FILE* inFile = fopen(argv[1], "r");
    FILE* outFile = fopen(argv[2], "w");

    if (inFile == NULL) {
        fprintf(stderr, "Lex Error: Cannot open file %s for reading.\n", argv[1]);
        return EXIT_FAILURE;
    }
    if (outFile == NULL) {
        fprintf(stderr, "Lex Error: Cannot open file %s for writing.\n", argv[2]);
        fclose(inFile);
        return EXIT_FAILURE;
    }

    int lineCount = 0;
    char buffer[256];



    while (fgets(buffer, sizeof(buffer), inFile) != NULL) {
        lineCount++;
    }

    char** lines = calloc(lineCount, sizeof(char*));
    rewind(inFile);

    for (int i = 0; i < lineCount; i++) {
        fgets(buffer, sizeof(buffer), inFile);


        lines[i] = calloc(strlen(buffer) + 1, sizeof(char));
        strcpy(lines[i], buffer);
    }

    List indexList = newList();
    append(indexList, 0);

    for (int i = 1; i < lineCount; i++) {
        moveFront(indexList);
        while (index(indexList) >= 0) {
            if (strcmp(lines[i], lines[get(indexList)]) < 0) {
                insertBefore(indexList, i);
                break;
            }
            moveNext(indexList);
        }
        if (index(indexList) == -1) {
            append(indexList, i);
        }
    }

    for (moveFront(indexList); index(indexList) >= 0; moveNext(indexList)) {
        fprintf(outFile, "%s", lines[get(indexList)]);
    }

    for (int i = 0; i < lineCount; i++) {
        free(lines[i]);
    }
    free(lines);
    freeList(&indexList);
    fclose(inFile);
    fclose(outFile);

    return 0;
}
