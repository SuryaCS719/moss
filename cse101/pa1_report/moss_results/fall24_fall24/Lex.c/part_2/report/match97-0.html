#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LEN 255

int main(int argc, char* argv[]) {
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        exit(EXIT_FAILURE);
    }

    FILE* in = fopen(argv[1], "r");


    FILE* out = fopen(argv[2], "w");
    if (in == NULL || out == NULL) {
        fprintf(stderr, "Error: cannot open file.\n");
        exit(EXIT_FAILURE);
    }

    fseek(in, 0, SEEK_END);
    long fileSize = ftell(in);
    rewind(in);

    char* buffer = malloc(fileSize + 1);
    fread(buffer, 1, fileSize, in);
    buffer[fileSize] = '\0';

    char* line = strtok(buffer, "\n");
    char** lines = NULL;
    int lineCount = 0;
    while (line != NULL) {
        lines = realloc(lines, (lineCount + 1) * sizeof(char*));
        lines[lineCount] = malloc(strlen(line) +1);
        if (lines[lineCount] != NULL) {
                strcpy(lines[lineCount], line);
        }
        lineCount++;
        line = strtok(NULL, "\n");
    }

    List L = newList();
    append(L, 0);

    for (int i = 1; i < lineCount; i++) {
        moveFront(L);
        while (index(L) >= 0) {
            if (strcmp(lines[i], lines[get(L)]) <= 0) {
                insertBefore(L, i);
                break;
            }
            moveNext(L);
        }

    if (index(L) < 0) {
        append(L, i);
    }
}

    moveFront(L);
    while (index(L) >= 0) {
        fprintf(out, "%s\n", lines[get(L)]);
        moveNext(L);
    }

    for (int i = 0; i < lineCount; i++) {
        free(lines[i]);
    }
    free(lines);
    free(buffer);
    freeList(&L);
    fclose(in);
    fclose(out);

    return 0;
}
