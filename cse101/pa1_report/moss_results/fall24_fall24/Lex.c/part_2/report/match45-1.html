#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"



int main(int argc, char *argv[]) {
    int numLines = 0;
    FILE *in, *out;
    char lines[300];

    if (argc != 3) {
        printf("Usage: %s <input file> <output file>\n", argv[0]);
        exit(1);
    }

    in = fopen(argv[1], "r");
    if (in == NULL) {
        printf("Unable to open file %s for reading\n", argv[1]);
        exit(1);
    }

    while (fgets(lines, sizeof(lines), in) != NULL) {
        numLines+=1;
    }

    char **words=malloc(numLines * sizeof(char *));
    if(words==NULL) {
        fprintf(stderr, "Error: memory allocation failed\n");
    }

    rewind(in);

    int temp = 0;
    while (fgets(lines, 300, in) != NULL && temp < numLines) {

    words[temp] = malloc((strlen(lines) + 1) * sizeof(char));

    if (words[temp] == NULL) {
        fprintf(stderr, "Error: memory allocation failed\n");
        exit(EXIT_FAILURE);
    }

    strcpy(words[temp], lines);

    temp+=1;
}

    fclose(in);

    out = fopen(argv[2], "w");
    if (out == NULL) {
        printf("Unable to open file %s for writing\n", argv[2]);
        exit(1);
    }

    int location;
    List L = newList();
    for(int i=0;i<numLines;++i) {
        append(L,-2);
    }
    for(int i=0;i<numLines;++i) {
        location=0;
        for(int j=0;j<numLines;++j) {
            if(strcmp(words[j],words[i])<0) {
                location+=1;
            }

        }
        moveFront(L);    

        while(index(L)!=location) {
            moveNext(L);
        }
        set(L,i);
    }


    moveFront(L);
    for(int i=0;i<length(L);i+=1){
        fprintf(out, "%s", words[get(L)]);
        moveNext(L);
    }

    fclose(out);

    freeList(&L);
    for (int i = 0; i < numLines; i++) {
        free(words[i]);
    }
    free(words);

    return 0;
}
