#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LEN 255

int main(int argc, char* argv[]) {
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input_file> <output_file>\n", argv[0]);
        exit(1);
    }

    FILE* in = fopen(argv[1], "r");
    if (in == NULL) {
        fprintf(stderr, "Unable to read from file %s.\n", argv[1]);
        exit(1);
    }



    FILE* out = fopen(argv[2], "w");
    if (out == NULL) {
        fprintf(stderr, "Unable to write to file %s.\n", argv[2]);
        fclose(in);
        exit(1);
    }

    int lines = 0;
    char buffer[MAX_LEN];
    while (fgets(buffer, MAX_LEN, in) != NULL) {
        lines++;
    }

    rewind(in);

    char** strArr = malloc(lines * sizeof(char*));
    if (strArr == NULL) {
        fprintf(stderr, "Memory allocation failed.\n");
        fclose(in);
        fclose(out);
        exit(1);
    }

    for (int i = 0; i < lines; i++) {
        fgets(buffer, MAX_LEN, in);
        strArr[i] = malloc((strlen(buffer) + 1) * sizeof(char));
        if (strArr[i] == NULL) {
            fprintf(stderr, "Memory allocation failed.\n");
            fclose(in);
            fclose(out);
            exit(1);
        }
        strcpy(strArr[i], buffer);
    }

    List indexList = newList();
    for (int i = 0; i < lines; i++) {
        moveBack(indexList);
        while (index(indexList) >= 0 && strcmp(strArr[i], strArr[get(indexList)]) < 0) {
            movePrev(indexList);
        }
        if (index(indexList) >= 0) {
            insertAfter(indexList, i);
        } else {
            prepend(indexList, i);
        }
    }

    moveFront(indexList);
    while (index(indexList) >= 0) {
        fprintf(out, "%s", strArr[get(indexList)]);
        moveNext(indexList);
    }

    for (int i = 0; i < lines; i++) {
        free(strArr[i]);
    }
    free(strArr);
    freeList(&indexList);
    fclose(in);
    fclose(out);

    return 0;
}
