//-----------------------------------------------------------------------------
// Lex.c
// Optimized version using List ADT to sort lines from an input file lexicographically.
//-----------------------------------------------------------------------------

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LEN 161

// Inserts an index into the list in sorted order based on the content of lines.


void insertInOrder(List L, int data, char** lines) {
    moveFront(L);
    while (index(L) >= 0) {
        if (strcmp(lines[data], lines[get(L)]) <= 0) {
            insertBefore(L, data);
            return;
        }
        moveNext(L);
    }
    append(L, data); // Append if no earlier position found.
}

int main(int argc, char* argv[]) {
    // Step 1: Verify the number of command line arguments.
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        exit(EXIT_FAILURE);
    }

    // Step 2: Open input and output files.
    FILE* inFile = fopen(argv[1], "r");
    if (inFile == NULL) {
        fprintf(stderr, "Error: Unable to open file %s for reading\n", argv[1]);
        exit(EXIT_FAILURE);
    }



    FILE* outFile = fopen(argv[2], "w");
    if (outFile == NULL) {
        fprintf(stderr, "Error: Unable to open file %s for writing\n", argv[2]);
        fclose(inFile);
        exit(EXIT_FAILURE);
    }

    // Step 3: Read lines from the input file into a dynamically allocated array.
    int lineCount = 0;
    int arraySize = 100; // Start with an initial size.
    char** lines = malloc(arraySize * sizeof(char*));
    if (lines == NULL) {
        fprintf(stderr, "Error: Memory allocation failed\n");
        fclose(inFile);
        fclose(outFile);
        exit(EXIT_FAILURE);
    }

    char line[MAX_LEN];
    while (fgets(line, MAX_LEN, inFile) != NULL) {
        // Expand array if necessary.
        if (lineCount == arraySize) {
            arraySize *= 2;
            lines = realloc(lines, arraySize * sizeof(char*));
            if (lines == NULL) {
                fprintf(stderr, "Error: Memory allocation failed\n");
                fclose(inFile);
                fclose(outFile);
                exit(EXIT_FAILURE);
            }
        }

        // Allocate memory for each line and store it.
        lines[lineCount] = malloc((strlen(line) + 1) * sizeof(char));
        if (lines[lineCount] == NULL) {
            fprintf(stderr, "Error: Memory allocation failed for line storage\n");
            fclose(inFile);
            fclose(outFile);
            exit(EXIT_FAILURE);
        }
        strcpy(lines[lineCount], line);
        lineCount++;
    }

    fclose(inFile);

    // Step 4: Create a List to store indices for sorting.
    List sortedIndices = newList();

    // Step 5: Insert indices into the list to maintain sorted order.
    for (int i = 0; i < lineCount; i++) {
        insertInOrder(sortedIndices, i, lines);
    }

    // Step 6: Write the sorted lines to the output file using the sorted indices.
    moveFront(sortedIndices);
    while (index(sortedIndices) >= 0) {
        int idx = get(sortedIndices);
        fputs(lines[idx], outFile);
        moveNext(sortedIndices);
    }

    // Step 7: Clean up memory.
    for (int i = 0; i < lineCount; i++) {
        free(lines[i]);
    }
    free(lines);
    freeList(&sortedIndices);
    fclose(outFile);

    return EXIT_SUCCESS;
}
