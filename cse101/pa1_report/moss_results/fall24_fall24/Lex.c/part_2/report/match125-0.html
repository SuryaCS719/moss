#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"
#define MAX_LENGTH 100

int main(int argc, char* argv[]) {
    if (argc != 3) {
        fprintf(stderr, "Format: %s <input file> <output file>\n", argv[0]);
        return EXIT_FAILURE;
    }

    FILE* inputFile = fopen(argv[1], "r");
    if (inputFile == NULL) {
        fprintf(stderr, "Can't read the file %s\n", argv[1]);
        return EXIT_FAILURE;
    }



    FILE* outputFile = fopen(argv[2], "w");
    if (outputFile == NULL) {
        fprintf(stderr, "Cant' read the file %s\n", argv[2]);
        fclose(inputFile);
        return EXIT_FAILURE;
    }

    //put lines in an array
    int count = 0;
    char buffer[MAX_LENGTH];
    while (fgets(buffer, sizeof(buffer), inputFile) != NULL) {
        count++;
    }

    rewind(inputFile);
    char** lines = calloc(count, sizeof(char*));
    for (int i = 0; i < count; i++) {
        fgets(buffer, sizeof(buffer), inputFile);
        lines[i] = strdup(buffer);
    }

    //sorting
    List l = newList();
    for (int i = 0; i < count; i++) {
        moveFront(l);
        while (index(l) != -1 && strcmp(lines[get(l)], lines[i]) < 0) {
            moveNext(l);
        }
        if (index(l) == -1) {
            append(l, i);
        } else {
            insertBefore(l, i);
        }
    }

    for (moveFront(l); index(l) != -1; moveNext(l)) {
        fprintf(outputFile, "%s", lines[get(l)]);
    }

    for (int i = 0; i < count; i++) {
        free(lines[i]);
    }
    free(lines);
    freeList(&l);

    fclose(inputFile);
    fclose(outputFile);

    return EXIT_SUCCESS;
}
