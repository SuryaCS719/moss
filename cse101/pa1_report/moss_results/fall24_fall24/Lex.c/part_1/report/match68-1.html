#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h" 

#define MAX_LEN 256

int main(int argc, char* argv[]) {

    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        return 1;
    }



    FILE* infile = fopen(argv[1], "r");
    if (infile == NULL) {
        fprintf(stderr, "Error: Could not open input file %s\n", argv[1]);
        return 1;
    }

    int n = 0;
    char buffer[MAX_LEN];
    while (fgets(buffer, sizeof(buffer), infile) != NULL) {
        n++;
    }

    char** array = (char**)calloc(n, sizeof(char*));
    if (array == NULL) {
        fprintf(stderr, "Memory allocation error\n");
        fclose(infile);
        return 1;
    }

    rewind(infile);
    int i = 0;
    while (fgets(buffer, sizeof(buffer), infile) != NULL) {
        buffer[strcspn(buffer, "\n")] = '\0';
        array[i] = (char*)malloc(strlen(buffer) + 1);
        strcpy(array[i], buffer);
        i++;
    }

    fclose(infile);

    List L = newList();  

    for (i = 0; i < n; i++) {
        if (length(L) == 0) {
            append(L, i); 
        } else {
            moveFront(L);
            while (index(L) >= 0 && strcmp(array[get(L)], array[i]) < 0) {
                moveNext(L);
            }
            if (index(L) >= 0) {
                insertBefore(L, i); 
            } else {
                append(L, i); 
            }
        }
    }

    FILE* outfile = fopen(argv[2], "w");
    if (outfile == NULL) {
        fprintf(stderr, "Error: Could not open output file %s\n", argv[2]);
        freeList(&L);
        for (i = 0; i < n; i++) {
            free(array[i]);
        }
        free(array);
        return 1;
    }

    for (moveFront(L); index(L) >= 0; moveNext(L)) {
        fprintf(outfile, "%s\n", array[get(L)]); 
    }

    fclose(outfile);

    freeList(&L);
    for (i = 0; i < n; i++) {
        free(array[i]);
    }
    free(array);

    return 0;
}
