#include "List.h"
#include <string.h>
#include <stdio.h>
#include <stdlib.h>

#define MAX_LEN 255

int main(int c, char* v[]) {
    if (c != 3) {
        fprintf(stderr, "%s input file, output file\n", v[0]);
        exit(EXIT_FAILURE);
    }

    FILE* input = fopen(v[1], "r");
    FILE* output = fopen(v[2], "w");
    if (input == NULL || output == NULL) {
        fprintf(stderr, "Can't open file\n");
        exit(EXIT_FAILURE);
    }

    char line[MAX_LEN];
    int count = 0;
    
    while (fgets(line, MAX_LEN, input) != NULL) {
        count++;
    }

    char** lines = calloc(count, sizeof(char*));
    fseek(input, 0, SEEK_SET);  

    int i = 0;
    while (fgets(line, MAX_LEN, input) != NULL) {
        lines[i] = calloc(strlen(line) + 1, sizeof(char));
        strcpy(lines[i], line);
        i++;
    }



    List indices = newList();
    append(indices, 0);  

    for (i = 1; i < count; i++) {
        moveFront(indices);
        while (index(indices) >= 0) {
            if (strcmp(lines[i], lines[get(indices)]) < 0) {
                insertBefore(indices, i);
                break;
            }
            moveNext(indices);
        }


        if (index(indices) == -1) {
            append(indices, i);
        }
    }

    moveFront(indices);
    while (index(indices) >= 0) {
        fprintf(output, "%s", lines[get(indices)]);
        moveNext(indices);
    }

    for (i = 0; i < count; i++) {
        free(lines[i]);
    }
    free(lines);
    freeList(&indices);
    fclose(input);
    fclose(output);

    return 0;
}
