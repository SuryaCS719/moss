#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LINE_LENGTH 255



int main(int argc, char* argv[]) {
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        exit(-1);
    }

    FILE* in = fopen(argv[1], "r");
    FILE* out = fopen(argv[2], "w");
    if (in == NULL || out == NULL) {
        fprintf(stderr, "Error: unable to open file %s for R/W.\n", in == NULL ? argv[1] : argv[2]);
        if (out==NULL) { fclose(in); }
        if (in==NULL) { fclose(out); }
	exit(-1);
    }

    char line[MAX_LINE_LENGTH];
    int lineCount = 0;
    while (fgets(line, MAX_LINE_LENGTH, in) != NULL) {
        lineCount++;
    }
    rewind(in);

    
    char** lines = calloc(lineCount, sizeof(char*));
    if (lines == NULL) {
        fprintf(stderr, "Error: unable to calloc.\n");
        fclose(in);
        fclose(out);
        exit(-1);
    }

    for (int i = 0; i < lineCount; i++) {
        fgets(line, MAX_LINE_LENGTH, in);
        lines[i] = calloc(strlen(line) + 1, sizeof(char));
        if (lines[i] == NULL) {
            fprintf(stderr, "Error: unable to calloc for line.\n");
            fclose(in);
            fclose(out);
            exit(-1);
        }
        strcpy(lines[i], line);
    }

    List list = newList();
    append(list, 0);
    
    for (int i = 1; i < lineCount; i++) {
        moveFront(list);
        while (index(list) >= 0 && strcmp(lines[i], lines[get(list)]) > 0) {
            moveNext(list);
        }
        if (index(list) >= 0) {
            insertBefore(list, i);
        } else {
            append(list, i);
        }
    }

    
    for (moveFront(list); index(list) >= 0; moveNext(list)) {
        fprintf(out, "%s", lines[get(list)]);
    }

    for (int i = 0; i < lineCount; i++) {
        free(lines[i]);
    }
    free(lines);
    freeList(&list);
    
    fclose(in);
    fclose(out);

    return 0;
}
