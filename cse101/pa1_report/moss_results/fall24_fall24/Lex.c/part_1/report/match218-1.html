#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LEN 360  // Increase the maximum length of a line to 360 characters

int main(int argc, char *argv[]) {
    FILE *in, *out;
    char line[MAX_LEN];
    int lineCount = 0;
    char **lines;

    // Check command line for correct number of arguments
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        exit(EXIT_FAILURE);
    }

    // Open input file for reading
    if ((in = fopen(argv[1], "r")) == NULL) {
        fprintf(stderr, "Unable to open file %s for reading\n", argv[1]);
        exit(EXIT_FAILURE);
    }

    // Open output file for writing
    if ((out = fopen(argv[2], "w")) == NULL) {
        fprintf(stderr, "Unable to open file %s for writing\n", argv[2]);
        exit(EXIT_FAILURE);
    }

    // Count the number of lines in the input file
    while (fgets(line, MAX_LEN, in) != NULL) {
        lineCount++;
    }

    // Allocate memory for storing the lines
    lines = malloc(lineCount * sizeof(char *));
    if (lines == NULL) {
        fprintf(stderr, "Memory allocation failed for lines array\n");
        exit(EXIT_FAILURE);
    }

    // Rewind input file to read lines
    rewind(in);

    // Read lines from input file into the lines array
    for (int i = 0; i < lineCount; i++) {
        fgets(line, MAX_LEN, in);
        lines[i] = malloc((strlen(line) + 1) * sizeof(char));
        if (lines[i] == NULL) {
            fprintf(stderr, "Memory allocation failed for line %d\n", i);
            exit(EXIT_FAILURE);
        }
        strcpy(lines[i], line);
    }

    // Create a new List to hold the indices of the lines
    List L = newList();


    append(L, 0);

    // Use Insertion Sort to build the List with sorted indices
    for (int i = 1; i < lineCount; i++) {
        moveFront(L);
        while (index(L) >= 0) {
            int j = get(L);
            if (strcmp(lines[i], lines[j]) < 0) {
                insertBefore(L, i);
                break;
            }
            moveNext(L);
        }
        if (index(L) < 0) {
            append(L, i);
        }
    }

    // Write the sorted lines to the output file
    for (moveFront(L); index(L) >= 0; moveNext(L)) {
        fprintf(out, "%s", lines[get(L)]);
    }

    // Free allocated memory
    for (int i = 0; i < lineCount; i++) {
        free(lines[i]);
    }
    free(lines);
    freeList(&L);

    // Close files
    fclose(in);
    fclose(out);

    return 0;
}
