#include<stdio.h>
#include<stdlib.h>
#include<stdbool.h>
#include<string.h>
#include "List.h"

#define MAX_LEN 300



int main(int argc, char* argv[]) {
  
   int line_count = 0;
   FILE *in, *out;
   char buffer[MAX_LEN];
   char **lines = NULL;

   
   if( argc != 3 ){
      fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
      exit(EXIT_FAILURE);
   }

   in = fopen(argv[1], "r");
   if (in==NULL ){
       fprintf(stderr, "Error: Could not open input file %s\n", argv[1]);
       exit(EXIT_FAILURE);
   }

   out = fopen(argv[2], "w");
   if( out==NULL ){
       fprintf(stderr, "Error: Could not open input file %s\n", argv[2]);
       fclose(in);
       exit(EXIT_FAILURE);
   }

   while (fgets(buffer, sizeof(buffer), in) != NULL) { //reads each line from the file into the buffer in order to count the number of lines
       line_count++;  
   }

   lines = (char **)calloc(line_count, sizeof(char *)); //allocates memory for an array of char * pointers and holds the pointers to the actual lines of text 

   if (lines == NULL) {
    fprintf(stderr, "Error: Memory allocation failed\n");
    fclose(in);
    fclose(out);
    return EXIT_FAILURE;
    }

    rewind(in);

    int i = 0;
    while(fgets(buffer, sizeof(buffer), in) != NULL){
        buffer[strcspn(buffer, "\n")] = '\0'; //remove the newline character 
        lines[i] = calloc(strlen(buffer)+1, sizeof(char)); //dynamically allocates memory for the current line of text being stored in buffer

        if (lines[i] == NULL){
            fprintf(stderr, "Error: Memory allocation failed\n");
            fclose(in);
            return EXIT_FAILURE;
        }
        strcpy(lines[i], buffer);
        i++;
    }

    List L = newList();
    append(L,0);

    for(int i = 1; i < line_count; i++){
        moveFront(L);

        while(index(L) >= 0){
            if(strcmp(lines[i], lines[get(L)]) < 0){
                insertBefore(L, i);
                break;
            }
            moveNext(L);
        }


        if(index(L) == -1){
            append(L, i);
        }
    }

    moveFront(L);
    while (index(L)>= 0){
        fprintf(out, "%s\n", lines[get(L)]);  
        moveNext(L);
    }

    for (i = 0; i < line_count; i++) {
        if(lines[i] != NULL){
            free(lines[i]);  
        }
    }

    free(lines);         // Free the array holding pointers to lines
    freeList(&L);        // Free the list
    fclose(in);          // close in file
    fclose(out);         // close out file

    return(0);
}
