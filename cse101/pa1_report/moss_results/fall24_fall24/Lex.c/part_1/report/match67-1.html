#include<stdio.h>
#include<stdlib.h>
#include<stdbool.h>

#include "List.h"



int countLines(char *filename) {
    FILE *fp = fopen(filename, "r");
    if (fp == NULL) {
        fprintf(stderr, "Could not open file '%s'\n", filename);
        exit(1);
    }
    int count = 0;
    char line[1024];
    while (fgets(line, 1024, fp) != NULL) {
        count++;
    }
    fclose(fp);
    return count;
}

void read(char *filename, char **lines, int num_lines) {
    FILE *fp = fopen(filename, "r");
    if (fp == NULL) {
        fprintf(stderr, "Could not open file '%s'\n", filename);
        exit(1);
    }
    char line[1024];
    int i = 0;


    while (fgets(line, 1024, fp) != NULL && i < num_lines) {
        lines[i] = malloc((strlen(line) + 1) * sizeof(char));
        strcpy(lines[i], line);
        i++;
    }
    fclose(fp);
}

int main(int argc, char **argv) {
    if (argc != 3) {
        fprintf(stderr, "Too many arguments\n");


        exit(1);
    }

    // Count number of lines in file
    int num_lines = countLines(argv[1]);

    // Create an array of lines
    char **lines = malloc(num_lines * sizeof(char *));
    if (lines == NULL) {
        fprintf(stderr, "No lines found\n");
        exit(1);
    }

    // Read input lines
    read(argv[1], lines, num_lines);

    // Create a list of indices into the array, sorted by lexicographic order
    List L = newList();
    append(L, 0);
    for (int i = 1; i < num_lines; i++) {
        moveFront(L);
        while (index(L) >= 0 && strcmp(lines[i], lines[get(L)]) > 0) {


            moveNext(L);
        }
        if (index(L) >= 0) {
            insertBefore(L, i);
        } else {
            append(L, i);
        }
    }

    // Write to the output file
    FILE *outputFile = fopen(argv[2], "w");
    if (outputFile == NULL) {
        fprintf(stderr, "Error: could not open file '%s' for writing\n", argv[2]);
        exit(1);
    }
    moveFront(L);
    while (index(L) >= 0) {
        fprintf(outputFile, "%s", lines[get(L)]);
        moveNext(L);
    }
    fclose(outputFile);

    freeList(&L);
    for (int i = 0; i < num_lines; i++) {
        free(lines[i]);
    }
    free(lines);

    return 0;
}
