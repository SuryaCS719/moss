#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<stdbool.h>
#include"List.h"

#define MAX_LEN 300



int main(int argc, char* argv[]){

   int lineCount;
   FILE *in, *out;
   char line[MAX_LEN + 1];

	line[MAX_LEN] = '\0';

   // check command line for correct number of arguments
   if( argc != 3 ){
      printf("Usage: %s <input file> <output file>\n", argv[0]);
      exit(1);
   }

   // open files for reading and writing 
   in = fopen(argv[1], "rb");
      if( in==NULL ){
      printf("Unable to open file %s for reading\n", argv[1]);
      exit(1);
   }

   out = fopen(argv[2], "wb");
   if( out==NULL ){


      printf("Unable to open file %s for writing\n", argv[2]);
      exit(1);
   }

   // read each line of input file, then count and print tokens
   lineCount = 0;
   while( fgets(line, MAX_LEN, in) != NULL)  {
      lineCount++;
   }

	fseek(in, 0, SEEK_SET);

	//allocate string array
	char **strings = (char **)malloc(lineCount * sizeof(char *));

	if (strings != NULL)
	{
		int stringIndex = 0;

	   while( fgets(line, MAX_LEN, in) != NULL)  {
			//line[strcspn(line, "\r\n")] = 0;

			int lineSize = strlen(line);

			strings[stringIndex] = (char *)malloc(lineSize + 1);
			strings[stringIndex][lineSize] = '\0';
			strcpy(strings[stringIndex], line);
			stringIndex++;
		}

		List indexList = newList();

		for (stringIndex = 0; stringIndex < lineCount; stringIndex++) {
			bool found = false;
			char *currentString = strings[stringIndex];
			if (length(indexList) == 0) {
				append(indexList, 0);
				found = true;
			}
			else {
				for(moveFront(indexList); index(indexList)>=0; moveNext(indexList)){
					char *stringToCompare = strings[get(indexList)];
					int compareResult = strcmp(currentString, stringToCompare);

					if (compareResult <= 0) {
						insertBefore(indexList, stringIndex);
						found = true;
						break;
					}
				}
			}

			if (!found)
				append(indexList, stringIndex);
		}

		for(moveFront(indexList); index(indexList)>=0; moveNext(indexList)){
			int stringIndex = get(indexList);

			fprintf(out, "%s", strings[stringIndex]);
		}


		freeList(&indexList);

		for (int i = 0; i < lineCount; i++)
		{
			free(strings[i]);
		}
		
		free(strings);
	}

   // close files 
   fclose(in);
   fclose(out);

   return(0);
}
