#include "List.h"
#include <stdlib.h>
#include <stdbool.h>
#include <string.h>
#include <stdio.h>
#define MAX_LEN 300



int main(int argc, char *argv[]) {
    if (argc != 3) {
        fprintf(stderr, "Lex: Error: call with Lex <input> <output>");
        return 1;
    }
    FILE * file = fopen(argv[1], "r");
    char line[MAX_LEN];

    int n = 0;
    while( fgets(line, MAX_LEN, file) != NULL) {
      n++;
    }
    rewind(file);
   
    char ** s = malloc(sizeof(char*) * n);

    int i = 0;
    while (fgets(line, MAX_LEN, file) != NULL) {
        line[strcspn(line, "\n")] = '\0';  // Remove the newline character
        s[i] = malloc(strlen(line) + 1);   // Allocate memory for the string (plus null terminator)
        if (s[i] == NULL) {
            fprintf(stderr, "Error: Memory allocation failed for line %d.\n", i);
            fclose(file);
            return 1;
        }
        strcpy(s[i], line);  // Copy the line into s[i]
        i++;
    }
    fclose(file);



    // Populate strings
    List mylist = newList();
    char* curr;
    for (int i = 0; i < n; i++) {
        curr = s[i];
        if (length(mylist) == 0) {
            append(mylist, i);
        } else {
            moveFront(mylist);
            while (index(mylist) != -1) {
                if (strcmp(s[get(mylist)], curr) > 0) {
                    insertBefore(mylist, i);
                    break;
                } 
                moveNext(mylist);
            }
            if (index(mylist) == -1) {
                append(mylist, i);
            }
        }
    }

    // print to output
    FILE* out = fopen(argv[2], "w");
    moveFront(mylist);
    while (index(mylist) != -1) {
        fprintf(out, "%s\n", s[get(mylist)]);
        moveNext(mylist);
    }
    fclose(out);


    for (int i = 0; i < n; i++) {
        free(s[i]);
    }
    free(s);
    freeList(&mylist);
    
    return 0;
}
