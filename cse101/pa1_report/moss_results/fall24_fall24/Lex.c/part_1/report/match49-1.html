#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LINE_LENGTH 256  // Assuming a reasonable max line length

int main(int argc, char* argv[]) {
    // Step 1: Check command line arguments
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        exit(EXIT_FAILURE);
    }

    // Open input and output files
    FILE *input = fopen(argv[1], "r");
    if (input == NULL) {
        fprintf(stderr, "Could not open input file %s\n", argv[1]);
        exit(EXIT_FAILURE);
    }



    FILE *output = fopen(argv[2], "w");
    if (output == NULL) {
        fprintf(stderr, "Could not open output file %s\n", argv[2]);
        fclose(input);
        exit(EXIT_FAILURE);
    }

    // Step 2: Count numbers of lines and read lines into the array
    int lineCount = 0;
    char line[MAX_LINE_LENGTH];
    while (fgets(line, MAX_LINE_LENGTH, input) != NULL) {
        lineCount++;
    }

    // Allocate memory for lines
    char** lines = calloc(lineCount, sizeof(char*));
    if (lines == NULL) {
        fprintf(stderr, "Memory allocation failed\n");
        fclose(input);
        fclose(output);
        exit(EXIT_FAILURE);
    }

    // Rewind the file to re-read lines into the array
    rewind(input);
    for (int i = 0; i < lineCount; i++) {
        if (fgets(line, MAX_LINE_LENGTH, input) == NULL) {
            fprintf(stderr, "Error reading line from input file.\n");
            fclose(input);
            fclose(output);
            for (int j = 0; j < i; j++) {  // Free any lines already allocated
                free(lines[j]);
            }
            free(lines);
            exit(EXIT_FAILURE);
        }

        
        lines[i] = malloc(strlen(line) + 1);
        if (lines[i] == NULL) {
            fprintf(stderr, "Memory allocation failed\n");
            fclose(input);
            fclose(output);
            for (int j = 0; j < i; j++) {  // Free any lines already allocated
                free(lines[j]);
            }
            free(lines);
            exit(EXIT_FAILURE);
        }
        strcpy(lines[i], line);
    }

    // Step 3: Create and sort index List
    List indexList = newList();
    for (int i = 0; i < lineCount; i++) {
        if (length(indexList) == 0) {
            append(indexList, i);  // First insertion if list is empty
            // printList(stdout, indexList);
        } else {
            moveFront(indexList);
            while (index(indexList) != -1) {  // Loop until cursor is undefined
                int j = get(indexList);
                if (strcmp(lines[i], lines[j]) < 0) {
                    insertBefore(indexList, i);
                    break;
                }
                moveNext(indexList);
            }
            if (index(indexList) == -1) {
                append(indexList, i);
            }
        }
    }

    // Step 4: Write sorted output to the output file
    moveFront(indexList);
    while (index(indexList) >= 0) {
        int idx = get(indexList);
        // printf("%s", lines[idx]); 
        fprintf(output, "%s", lines[idx]);  // Write in sorted order
        moveNext(indexList);
    }

    // Final cleanup: free memory and close files
    for (int i = 0; i < lineCount; i++) {
        free(lines[i]);
    }
    free(lines);
    freeList(&indexList); //EVERYTHING SEEMS TO WORK BUT THIS
    //ITS DEQUEUING SOMETHING AND CAUSING SEG FAULT

    fclose(input);
    fclose(output);
    
    return 0;
}
