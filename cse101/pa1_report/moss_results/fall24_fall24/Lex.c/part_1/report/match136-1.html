#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LEN 160

int main(int argc, char* argv[]) {
    // Step 1: Check the number of command-line arguments
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        return EXIT_FAILURE;
    }

    // Step 2: Open the input file and output file
    FILE* inFile = fopen(argv[1], "r");
    if (inFile == NULL) {
        fprintf(stderr, "Error: Cannot open input file %s\n", argv[1]);
        return EXIT_FAILURE;
    }

    FILE* outFile = fopen(argv[2], "w");
    if (outFile == NULL) {
        fprintf(stderr, "Error: Cannot open output file %s\n", argv[2]);


        fclose(inFile);
        return EXIT_FAILURE;
    }

    // Step 3: Count the number of lines in the input file
    char line[MAX_LEN];
    int lineCount = 0;
    while (fgets(line, MAX_LEN, inFile) != NULL) {
        lineCount++;
    }

    // Step 4: Allocate memory for an array of strings to hold the lines
    rewind(inFile);  // Reset the file pointer to the beginning of the file
    char** lines = calloc(lineCount, sizeof(char*));


    for (int i = 0; i < lineCount; i++) {
        fgets(line, MAX_LEN, inFile);
        lines[i] = malloc(strlen(line) + 1);  // Allocate exact memory for each line
        strcpy(lines[i], line);
    }

    // Step 5: Create a List to store the indices of the lines array
    List indexList = newList();
    append(indexList, 0);  // Insert the first line index

    // Step 6: Perform indirect insertion sort using strcmp
    for (int i = 1; i < lineCount; i++) {
        moveFront(indexList);
        while (index(indexList) != -1 && strcmp(lines[i], lines[get(indexList)]) > 0) {
            moveNext(indexList);
        }
        if (index(indexList) == -1) {
            append(indexList, i);  // Insert at the end of the list
        } else {
            insertBefore(indexList, i);  // Insert before the cursor
        }
    }

    // Step 7: Write the sorted lines to the output file using the list of indices
    moveFront(indexList);
    while (index(indexList) != -1) {
        fprintf(outFile, "%s", lines[get(indexList)]);
        moveNext(indexList);
    }

    // Step 8: Clean up and free memory
    for (int i = 0; i < lineCount; i++) {
        free(lines[i]);  // Free each line
    }
    free(lines);        // Free the array of strings
    freeList(&indexList); // Free the list
    fclose(inFile);
    fclose(outFile);

    return EXIT_SUCCESS;
}
