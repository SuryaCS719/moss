#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LINE_LENGTH 1000

int main(int argc, char *argv[]) {


    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        return 1;
    }

    FILE *inputFile = fopen(argv[1], "r");
    FILE *outputFile = fopen(argv[2], "w");

    if (!inputFile || !outputFile) {
        perror("Error opening files");
        return 1;
    }

    // Count the number of lines in the input file
    int numLines = 0;
    char line[MAX_LINE_LENGTH];
    while (fgets(line, MAX_LINE_LENGTH, inputFile) != NULL) {
        numLines++;
    }

    // Allocate memory for storing lines
    char **linesArray = (char **)malloc(numLines * sizeof(char *));
    if (!linesArray) {
        perror("Memory allocation failed");
        return 1;
    }

    // Rewind the file to the beginning
    rewind(inputFile);

    // Read each line and store it in the array
    int i = 0;
    while (fgets(line, MAX_LINE_LENGTH, inputFile) != NULL) {
        linesArray[i] = strdup(line); // Allocate memory for each line
        i++;
    }

    // Create a new list using the List ADT
    List tempList = newList();

    for (i = 0; i < numLines; i++) {
        int stepCounter = 0;
        moveFront(tempList);

        // Traverse the list to find the correct position for insertion
        while (index(tempList) >= 0 && strcmp(linesArray[i], linesArray[get(tempList)]) > 0) {
            moveNext(tempList);
            stepCounter++;
        }

        if (index(tempList) == -1) {
            append(tempList, i); // Insert at the end of the list
        } else {
            insertBefore(tempList, i); // Insert before the cursor position
        }
    }

    // Write sorted lines to the output file
    moveFront(tempList);
    while (index(tempList) >= 0) {
        fprintf(outputFile, "%s", linesArray[get(tempList)]);
        moveNext(tempList);
    }

    // Free allocated memory and close files
    for (i = 0; i < numLines; i++) {
        free(linesArray[i]);
    }
    free(linesArray);
    freeList(&tempList);
    fclose(inputFile);
    fclose(outputFile);

    return 0;
}
