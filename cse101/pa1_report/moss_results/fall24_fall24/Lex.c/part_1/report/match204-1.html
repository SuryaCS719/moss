#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

int main(int argc, char* argv[]) {
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        exit(1);
    }

    FILE* inFile = fopen(argv[1], "r");
    if (inFile == NULL) {
        fprintf(stderr, "Error: Cannot open input file %s\n", argv[1]);
        exit(1);
    }



    FILE* outFile = fopen(argv[2], "w");
    if (outFile == NULL) {
        fprintf(stderr, "Error: Cannot open output file %s\n", argv[2]);
        exit(1);
    }

    int lineCount = 0;
    char line[256];
    while (fgets(line, sizeof(line), inFile) != NULL) {
        lineCount++;
    }

    rewind(inFile);

    char** linesArray = calloc(lineCount, sizeof(char*));
    for (int i = 0; i < lineCount; i++) {
        fgets(line, sizeof(line), inFile);
        linesArray[i] = malloc(strlen(line) + 1);
        strcpy(linesArray[i], line);
    }

    List indexList = newList();
    append(indexList, 0);

    for (int i = 1; i < lineCount; i++) {
        moveFront(indexList);
        while (index(indexList) >= 0 && strcmp(linesArray[i], linesArray[get(indexList)]) > 0) {
            moveNext(indexList);
        }
        if (index(indexList) >= 0) {
            insertBefore(indexList, i);
        } else {
            append(indexList, i);
        }
    }

    moveFront(indexList);
    while (index(indexList) >= 0) {
        fprintf(outFile, "%s", linesArray[get(indexList)]);
        moveNext(indexList);
    }

    for (int i = 0; i < lineCount; i++) {
        free(linesArray[i]);
    }
    free(linesArray);
    freeList(&indexList);

    fclose(inFile);
    fclose(outFile);

    return 0;
}
