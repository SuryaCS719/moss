#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LEN 255



int main(int argc, char* argv[]) {
    char str[MAX_LEN];
    List strList = newList();

    // Check for the correct number of arguments
    if (argc != 3) {
        fprintf(stderr, "Incorrect number of arguments.\n");
        exit(EXIT_FAILURE);
    }

    // Open the file to read
    FILE* in = fopen(argv[1], "r");
    if (in == NULL) {
        fprintf(stderr, "Unable to read from file, %s.\n", argv[1]);
        exit(EXIT_FAILURE);
    }

    // Open the file to write
    FILE* out = fopen(argv[2], "w");


    if (out == NULL) {
        fprintf(stderr, "Unable to write to file, %s.\n", argv[2]);
        exit(EXIT_FAILURE);
    }

    // Count lines in in-file
    int lines = 0;
    while (fgets(str, MAX_LEN, in) != NULL) {
        lines++;
    }

    // Reset in-file


    fclose(in);
    in = fopen(argv[1], "r");

    // Initialize string array
    char strArr[lines][MAX_LEN];
    lines = 0;  // Reset lines
    while (fgets(str, MAX_LEN, in) != NULL) {
        strcpy(strArr[lines], str);
        lines++;
    }

    // "Insertion sort"
    for (int i = 0; i < lines; i++) {
        moveFront(strList);
        while (index(strList) > -1 && strcmp(strArr[i], strArr[get(strList)]) > 0) {
            moveNext(strList);
        }


        if (index(strList) == -1) {
            append(strList, i);
        } else {
            insertBefore(strList, i);
        }
    }

    // Print list to out
    moveFront(strList);
    while (index(strList) > -1) {
        fprintf(out, "%s", strArr[get(strList)]);
        moveNext(strList);
    }

    // Clean up
    freeList(&strList);
    strList = NULL;

    fclose(in);
    fclose(out);

    return 0;
}
