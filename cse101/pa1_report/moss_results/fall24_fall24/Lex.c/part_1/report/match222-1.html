#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <stdbool.h>
#include <stdint.h>
#include <stdbool.h>

#include "List.h"
#define MAX_LEN 10000

int main(int argc, char* argv[]){
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        return 1;
    }
    FILE *in, *out;
    in = fopen(argv[1],"r");
    out = fopen(argv[2], "w");
    if (in == NULL) {
        fprintf(stderr, "Error: cannot open input file %s\n", argv[1]);
        return 1;
    }
    if (out == NULL) {
        fprintf(stderr, "Error: cannot open output file %s\n", argv[2]);
        fclose(in);
        return 1;
    }

    size_t n = 0;
    char line[MAX_LEN];
		while(fgets(line, MAX_LEN, in) != NULL){
			n++;
		}	
		rewind(in);
		char** strings = calloc(n,sizeof(line));
		int x = 0;
    while (fgets(line, MAX_LEN, in) != NULL) {
        strings[x] = calloc(strlen(line)+1,sizeof(char));


        strcpy(strings[x], line);
				x++;
    }
		//printf("n:%ld\n",n);

    List indList = newList();
    append(indList, 0); //first element
    for(int i = 1;i < n;i++){
        moveFront(indList); //move pointer to front of list.
        while(index(indList) >= 0){
            int currIn = get(indList);
						//printf("CurrIn:%d\n",currIn);
						//printf("string[i]:%s\n",strings[i]);
            if(strcmp(strings[i],strings[currIn]) < 0){
                insertBefore(indList,i);
                break;
            }
            moveNext(indList);
        }
				if(index(indList) == -1){
					append(indList,i);
				}
    }
    //print
   	moveFront(indList);
    while (index(indList) >= 0) {
				printf("%s\n", strings[get(indList)]);
        fprintf(out, "%s", strings[get(indList)]);  
        moveNext(indList);
    }
		for(int i = 0;i < n;i++){
			free(strings[i]);
		}

    free(strings);
    freeList(&indList);  // Free the list
    fclose(in);
    fclose(out);
    return 0;
}
