#include <stdio.h>
#include <stdlib.h>
#include <string.h>
 



int main(int argc, char *argv[]) 
{
    // Ensure proper usage
    if (argc != 3) {
        printf("Usage: %s <input file> <output file>\n", argv[0]);
        return 1;
    }

    // Open the input file
    FILE *inputFile = fopen(argv[1], "r");
    if (inputFile == NULL) {
        perror("Error opening input file");
        return 1;
    }

    // Open the output file
    FILE *outputFile = fopen(argv[2], "w");
    if (outputFile == NULL) {
        perror("Error opening output file");
        fclose(inputFile);
        return 1;
    }


    char c;
    while ((c = fgetc(inputFile)) != EOF) {
        fputc(c, outputFile);
    }

    

    // Close the files
    fclose(inputFile);
    fclose(outputFile);

    printf("Lexical analysis completed. Output saved to %s.\n", argv[2]);
    return 0;
}
