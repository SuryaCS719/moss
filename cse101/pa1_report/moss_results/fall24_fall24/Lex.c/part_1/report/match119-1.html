#include<stdio.h>

void Lex (char input[256], char output[256]){

    if (input == NULL || output == NULL) {
        fprintf(stderr, "Lex.c: Lex requires exactly 2 command line arguments");
        exit(EXIT_FAILURE);
    }

    scanf("%s", input);
    scanf("%s", output);
    FILE *file1 = fopen(input, "r");
    FILE *file2 = fopen(output, "r");
    char** mArray;

    List outList = newList();

    int lines, i = 0;
    while(!feof(file1)) {
        i = fgetc(file1);
        if (i =='\n') {lines++;}
    }

    i = 0;
    mArray = (char**)malloc(lines * sizeof(char**));
    for (int a = 0; a < lines; a++){

        fgets(mArray[a], 512, file1);
        moveFront(outList);
        Node N;
        for (N = outList->cursor; outList->cursor != NULL; N = N->next){
            
            if (strcmp(mArray[outList->cursor->data], mArray[a])<0){

                insertAfter(outList, a);

            } else if (strcmp(mArray[outList->cursor->data], mArray[a])<0){

                insertBefore(outList, a);

            } else {insertBefore(outList, a);}

        }
    }

    printList(file2, outList);
}
