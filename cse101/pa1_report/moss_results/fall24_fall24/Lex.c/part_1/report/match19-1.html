#include<stdio.h>
#include<stdlib.h>
#include<stdbool.h>
#include<string.h>
#include "List.h"


#define MAX_LEN 300


//so far this main reads the input file and counts the number of words.




int main(int argc, char * argv[]){


   List A = newList();
   int line_count= 0;
   FILE *in, *out;
   char line[MAX_LEN];
   //char* token;


   // check command line for correct number of arguments
   if( argc != 3 ){
      printf("Usage: %s <input file> <output file>\n", argv[0]);
      exit(EXIT_FAILURE);
   }


   // open files for reading and writing
   in = fopen(argv[1], "r");
   if( in==NULL ){
      printf("Unable to open file %s for reading\n", argv[1]);
      exit(EXIT_FAILURE);
   }


   out = fopen(argv[2], "w");
   if( out==NULL ){
      printf("Unable to open file %s for writing\n", argv[2]);
      exit(EXIT_FAILURE);
   }


   //first pass to get the num of tokens
   while (fgets(line, MAX_LEN, in) != NULL){
      line_count++;
     
   }


   //creat space for lines array
   char **linesArray = malloc((line_count +1) * sizeof(char*));
   if(linesArray == NULL){
      exit(EXIT_FAILURE);
   }


   rewind(in);


   
   //read lines into array
   for (int i=0;i<line_count;i++){
      linesArray[i] = (char *)malloc(MAX_LEN * sizeof(char));
      if(linesArray == NULL){
         exit(EXIT_FAILURE);
      }
      if (fgets(linesArray[i], MAX_LEN, in)!= NULL){
         if(i==line_count - 1 &&linesArray[i][strlen(linesArray[i])-1] != '\n'){
            linesArray[i][strlen(linesArray[i])] = '\n';
            linesArray[i][strlen(linesArray[i])+1]= '\0';
         }
      }
   }
   


   append(A,0);
   //places in correct loco
   for(int i = 1; i<line_count; i++){
      moveFront(A);
      while (index(A)>=0 && strcmp(linesArray[get(A)],linesArray[i])<0){
         moveNext(A);
      }
      if(index(A) == -1){
         append(A,i);
      }else{
         insertBefore(A,i);
      }
   }


   //print lines to out
   for(moveFront(A); index(A)>=0; moveNext(A)){
      int ind = get(A);
      fprintf(out,"%s", linesArray[ind]);
   }




   fclose(in);
   fclose(out);


   for (int i=0; i<line_count; i++){
      free(linesArray[i]);
   }
   free(linesArray);
   freeList(&A);
   return 0;


}
