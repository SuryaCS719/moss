* Reads lines from an input file, indirectly sorts them using a List ADT to store indices, and writes the sorted lines to an output file.

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

int main(int argc, char *argv[]) {
    // Step 1: Check for correct number of command-line arguments
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        return EXIT_FAILURE;
    }

    // Step 2: Open the input file


    FILE *inputFile = fopen(argv[1], "r");
    if (inputFile == NULL) {
        fprintf(stderr, "Error: could not open input file %s\n", argv[1]);
        return EXIT_FAILURE;
    }

    // Step 3: Count the number of lines in the input file
    int lineCount = 0;
    char buffer[256];
    while (fgets(buffer, sizeof(buffer), inputFile) != NULL) {
        lineCount++;
    }

    // Handle edge case where file is empty
    if (lineCount == 0) {
        fclose(inputFile);
        fprintf(stderr, "Input file is empty.\n");
        return EXIT_FAILURE;
    }

    // Step 4: Allocate memory for the array of strings


    rewind(inputFile); // Reset file pointer to the beginning of the file
    char **lines = malloc(lineCount * sizeof(char *));
    if (lines == NULL) {
        fprintf(stderr, "Error: memory allocation failed\n");
        fclose(inputFile);
        return EXIT_FAILURE;
    }

    // Step 5: Read lines into the array
    for (int i = 0; i < lineCount; i++) {
        lines[i] = malloc(256 * sizeof(char));
        if (lines[i] == NULL) {
            fprintf(stderr, "Error: memory allocation failed for line %d\n", i);
            for (int j = 0; j < i; j++) free(lines[j]);
            free(lines);
            fclose(inputFile);
            return EXIT_FAILURE;
        }
        fgets(lines[i], 256, inputFile);

        // Remove newline character if it exists
        lines[i][strcspn(lines[i], "\n")] = '\0';
    }

    fclose(inputFile);

    // Step 6: Create the List and sort the indices using Insertion Sort
    List sortedList = newList();
    if (lineCount > 0) {
        append(sortedList, 0); // Add the first index
    }

    for (int i = 1; i < lineCount; i++) {
        moveFront(sortedList);
        while (index(sortedList) >= 0) {
            if (strcmp(lines[i], lines[get(sortedList)]) < 0) {
                insertBefore(sortedList, i);
                break;
            }
            moveNext(sortedList);
        }
        if (index(sortedList) < 0) {
            append(sortedList, i); // If no insertion, append at the end
        }
    }

    // Step 7: Open the output file


    FILE *outputFile = fopen(argv[2], "w");
    if (outputFile == NULL) {
        fprintf(stderr, "Error: could not open output file %s\n", argv[2]);
        freeList(&sortedList);
        for (int i = 0; i < lineCount; i++) free(lines[i]);
        free(lines);
        return EXIT_FAILURE;
    }

    // Step 8: Write the sorted lines to the output file
    moveFront(sortedList);
    while (index(sortedList) >= 0) {
        fprintf(outputFile, "%s\n", lines[get(sortedList)]);
        moveNext(sortedList);
    }

    fclose(outputFile);

    // Step 9: Free all allocated memory
    freeList(&sortedList);
    for (int i = 0; i < lineCount; i++) {
        free(lines[i]);
    }
    free(lines);

    return EXIT_SUCCESS;
}
