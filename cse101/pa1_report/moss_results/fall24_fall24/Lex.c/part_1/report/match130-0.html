#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LEN 160



char* my_strdup(const char* src) {
    char* dst = malloc(strlen(src) + 1);
    if (dst == NULL) return NULL;
    strcpy(dst, src);
    return dst;
}

int main(int argc, char* argv[]) {
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        exit(EXIT_FAILURE);
    }



    FILE* in = fopen(argv[1], "r");
    FILE* out = fopen(argv[2], "w");
    if (in == NULL || out == NULL) {
        fprintf(stderr, "Error: Could not open file.\n");
        exit(EXIT_FAILURE);
    }

    int n = 0;
    char line[MAX_LEN];
    while (fgets(line, MAX_LEN, in) != NULL) n++;

    char** lines = calloc(n, sizeof(char*));
    rewind(in);

    for (int i = 0; i < n; i++) {
        fgets(line, MAX_LEN, in);
        lines[i] = my_strdup(line);
    }

    List L = newList();
    append(L, 0);

    for (int i = 1; i < n; i++) {
        moveFront(L);
        while (index(L) >= 0) {
            if (strcmp(lines[i], lines[get(L)]) < 0) {
                insertBefore(L, i);
                break;
            }
            moveNext(L);
        }
        if (index(L) == -1) append(L, i);
    }

    moveFront(L);
    while (index(L) >= 0) {
        fprintf(out, "%s", lines[get(L)]);
        moveNext(L);
    }

    for (int i = 0; i < n; i++) free(lines[i]);
    free(lines);
    freeList(&L);
    fclose(in);
    fclose(out);

    return 0;
}
