#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LEN 256

int main(int argc, char* argv[]) {
    if (argc != 3) {


        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        exit(1);
    }

    FILE *inFile = fopen(argv[1], "r");
    if (inFile == NULL) {
        exit(1);
    }

    FILE *outFile = fopen(argv[2], "w");
    if (outFile == NULL) {
        fclose(inFile);
        exit(1);
    }

    int n = 0;
    char line[MAX_LEN];
    while (fgets(line, sizeof(line), inFile) != NULL) {
        n++;
    }

    rewind(inFile);

    char **stringArray = (char **)malloc(n * sizeof(char *));
    if (stringArray == NULL) {
        fclose(inFile);
        fclose(outFile);
        exit(1);
    }

    for (int i = 0; i < n; i++) {
        fgets(line, sizeof(line), inFile);
        line[strcspn(line, "\n")] = '\0';

        stringArray[i] = (char *)malloc((strlen(line) + 1) * sizeof(char));
        if (stringArray[i] == NULL) {
            fclose(inFile);
            fclose(outFile);
            exit(1);
        }
        strcpy(stringArray[i], line);
    }

    List indexList = newList();
    for (int i = 0; i < n; i++) {
        moveFront(indexList);
        while (index(indexList) >= 0 && strcmp(stringArray[i], stringArray[get(indexList)]) > 0) {
            moveNext(indexList);
        }
        if (index(indexList) >= 0) {
            insertBefore(indexList, i);
        } else {
            append(indexList, i);
        }
    }

    for (moveFront(indexList); index(indexList) >= 0; moveNext(indexList)) {
        fprintf(outFile, "%s\n", stringArray[get(indexList)]);
    }

    for (int i = 0; i < n; i++) {
        free(stringArray[i]);
    }
    free(stringArray);
    freeList(&indexList);

    fclose(inFile);
    fclose(outFile);

    exit(0);
}
