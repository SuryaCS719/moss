#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LEN 160

void usageMessage() {
    fprintf(stderr, "Usage: Lex <input file> <output file>\n");
}

int main(int argc, char* argv[]) {
    if (argc != 3) {
        usageMessage();
        return 1;
    }

    // Open input and output files
    FILE* in = fopen(argv[1], "r");
    if (in == NULL) {
        fprintf(stderr, "Unable to open file %s for reading\n", argv[1]);
        return 1;
    }

    FILE* out = fopen(argv[2], "w");
    if (out == NULL) {
        fprintf(stderr, "Unable to open file %s for writing\n", argv[2]);
        fclose(in);
        return 1;
    }

    // Step 2: Count lines and store them in an array
    char line[MAX_LEN];
    int lineCount = 0;
    while (fgets(line, MAX_LEN, in) != NULL) {
        lineCount++;
    }
    rewind(in);

    // Allocate memory for the string array
    char** lines = malloc(lineCount * sizeof(char*));


    for (int i = 0; i < lineCount; i++) {
        fgets(line, MAX_LEN, in);
        lines[i] = malloc(strlen(line) + 1);
        strcpy(lines[i], line);
    }

    // Step 3: Create a List to sort the indices of the lines
    List sortedIndices = newList();
    append(sortedIndices, 0);

    for (int i = 1; i < lineCount; i++) {
        moveFront(sortedIndices);
        while (index(sortedIndices) >= 0) {
            int cmp = strcmp(lines[i], lines[get(sortedIndices)]);


            if (cmp < 0) {
                insertBefore(sortedIndices, i);
                break;
            }
            moveNext(sortedIndices);
        }
        if (index(sortedIndices) < 0) {
            append(sortedIndices, i);  // Insert at the end if no insertion happened
        }
    }

    // Step 4: Print sorted lines to the output file
    moveFront(sortedIndices);
    while (index(sortedIndices) >= 0) {
        fprintf(out, "%s", lines[get(sortedIndices)]);
        moveNext(sortedIndices);
    }

    // Free resources
    freeList(&sortedIndices);
    for (int i = 0; i < lineCount; i++) {
        free(lines[i]);
    }
    free(lines);

    fclose(in);
    fclose(out);

    return 0;
}
