#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "list.h"  
#include "list.c"

#define MAX_LEN 300
#define index getCursorIndex  // index -> getCursorIndex



int main(int argc, char *argv[]) {
    int line_count = 0;
    FILE *in, *out;
    char line[MAX_LEN];

    // -------------------------------------
    // Step 1
    if (argc != 3) {
        printf("%s <input file> <output file>\n", argv[0]);
        exit(1);
    }

    in = fopen(argv[1], "r");
    if (in == NULL) {
        printf("Unable to open file %s for reading\n", argv[1]);
        exit(1);
    }

    out = fopen(argv[2], "w");
    if (out == NULL) {
        printf("Unable to open file %s for writing\n", argv[2]);
        fclose(in);  
        exit(1);
    }

    // -------------------------------------
    // Step 2
    while (fgets(line, MAX_LEN, in) != NULL) {
        line_count++;
    }

    // create string array of size line_count
    char **stringArray = malloc(line_count * sizeof(char *));
    if (stringArray == NULL) {
        fprintf(stderr, "error: unable to allocate memory\n");
        fclose(in);
        fclose(out);
        exit(1);
    }

    for (int i = 0; i < line_count; i++) {
        stringArray[i] = malloc(MAX_LEN * sizeof(char));
        if (stringArray[i] == NULL) {
            fprintf(stderr, "error: unable to allocate memory for stringArray[%d]\n", i);
            fclose(in);
            fclose(out);
            exit(1);
        }
    }

    fseek(in, 0, SEEK_SET);

    for (int i = 0; i < line_count; i++) {
        if (fgets(stringArray[i], MAX_LEN, in) != NULL) {
            stringArray[i][strcspn(stringArray[i], "\n")] = '\0';  // Remove newline
        } else {
            fprintf(stderr, "error: unable to read line %d\n", i);
            fclose(in);
            fclose(out);
            exit(1);
        }
    }

    // -------------------------------------
    // Step 3
    List L = newList();
    
    append(L, 0);  // first element in list
    for (int i = 1; i < line_count; i++) {
        moveBack(L);  // move cursor to the back
        while (index(L) >= 0 && strcmp(stringArray[i], stringArray[get(L)]) < 0) {


            movePrev(L);  // move cursor to prev element
        }
        if (index(L) >= 0) {
            insertAfter(L, i);
        } else {
            prepend(L, i);  // insert at the front if no element is smaller
        }
    }

    // -------------------------------------
    // Step 4
    moveFront(L);
    while (index(L) >= 0) {
        if (fprintf(out, "%s\n", stringArray[get(L)]) < 0) {
            fprintf(stderr, "error: unable to write to output file\n");
            fclose(in);
            fclose(out);
            freeList(&L);
            exit(1);
        }
        moveNext(L);
    }

    fflush(out);

    // free mem
    for (int i = 0; i < line_count; i++) {
        free(stringArray[i]);
    }
    free(stringArray);
    freeList(&L);
    fclose(in);
    fclose(out);
    return 0;
}
