* Reads lines from an input file and sorts them lexicographically, writing the sorted lines to an output file
***/

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LEN 255

char *strdup(const char *s);

int main(int argc, char* argv[]) {
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
    }

    FILE* in = fopen(argv[1], "r");
    if (in == NULL) {
        fprintf(stderr, "Error: Cannot open file %s for reading\n", argv[1]);
        exit(EXIT_FAILURE);
    }

    FILE* out = fopen(argv[2], "w");
    if (out == NULL) {
        fprintf(stderr, "Error: Cannot open file %s for writing\n", argv[2]);
        fclose(in);
        exit(EXIT_FAILURE);
    }

    char line[MAX_LEN];
    int lineCount = 0;
    char** lines = NULL;

    while (fgets(line, MAX_LEN, in) != NULL) {
        lineCount++;
    }

    lines = malloc(lineCount * sizeof(char*));
    if (lines == NULL) {
        fprintf(stderr, "Error: Memory allocation failure\n");
        fclose(in);
        fclose(out);
        exit(EXIT_FAILURE);
    }
    
    rewind(in);

    for (int i = 0; i < lineCount; i++) {
        fgets(line, MAX_LEN, in);
        lines[i] = strdup(line); 
        if (lines[i] == NULL) {
            fprintf(stderr, "Error: Memory allocation failure\n");
            fclose(in);
            fclose(out);


            exit(EXIT_FAILURE);
        }
    }

    List list = newList();
    append(list, 0); 

    for (int i = 1; i < lineCount; i++) {
        moveFront(list);
        while (index(list) != -1 && strcmp(lines[get(list)], lines[i]) <= 0) {


            moveNext(list);
        }
        if (index(list) == -1) {
            append(list, i); 
        } 
        else {
            insertBefore(list, i); 
        }
    }

    moveFront(list);
    while (index(list) != -1) {
        fprintf(out, "%s", lines[get(list)]);
        moveNext(list);
    }

    for (int i = 0; i < lineCount; i++) {
        free(lines[i]);
    }

    free(lines);
    freeList(&list);
    fclose(in);
    fclose(out);

    return 0;
}
