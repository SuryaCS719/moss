#include <stdio.h>
#include <stdlib.h>
#include "List.h"
#include <string.h>

int main(int argc, char *argv[])
{
    if (argc != 3)
    {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        return EXIT_FAILURE;
    }

    // printf("no errors with argument\n");

    // open input file for reading
    FILE *in = fopen(argv[1], "r");
    if (in == NULL)
    {
        fprintf(stderr, "Error: unable to open file %s for reading\n", argv[1]);
        return EXIT_FAILURE;
    }

    // printf("no errors with opening input file\n");

    // open output file for writing


    FILE *out = fopen(argv[2], "w");
    if (out == NULL)
    {
        fprintf(stderr, "Error: unable to open file %s for writing\n", argv[2]);
        fclose(in);
        return EXIT_FAILURE;
    }

    // printf("no errors with opening output file\n");

    // count the number of lines in the input file
    int line_count = 0;
    char c;

    for (c = getc(in); c != EOF; c = getc(in))
    {
        if (c == '\n')
        {
            line_count++;
        }
    }

    // reset file pointer to start of the input file
    rewind(in);

    // printf("no errors with counting lines in input file/rewind\n");

    // create array of strings
    char **lines = calloc(line_count, sizeof(char *));


    char buffer[1024];
    for (int i = 0; i < line_count; i++)
    {
        if (fgets(buffer, sizeof(buffer), in) != NULL)
        {
            lines[i] = calloc(strlen(buffer) + 1, sizeof(char));
            strcpy(lines[i], buffer);
        }
    }

    // printf("no errors with array of strings\n");

    // create a List and perform an indirect insertion sort on the indices
    List list = newList();
    // insert first index manually
    append(list, 0);

    // printf("no errors with creating list object\n");

    for (int i = 1; i < line_count; i++)
    {
        moveFront(list);


        while (index(list) >= 0 && strcmp(lines[i], lines[get(list)]) > 0)
        {
            moveNext(list);
        }
        if (index(list) >= 0)
        {
            insertBefore(list, i);
        }
        else
        {
            append(list, i);
        }
    }

    // printf("no errors with sorting list\n");

    // print the lines to the output file
    for (moveFront(list); index(list) >= 0; moveNext(list))
    {
        fprintf(out, "%s", lines[get(list)]);
    }

    // printf("no errors with writing to output file\n");

    // free memory
    for (int i = 0; i < line_count; i++)
    {
        free(lines[i]);
    }
    free(lines);
    freeList(&list);
    fclose(in);
    fflush(out);
    fclose(out);

    // printf("no errors with freeing memory\n");

    return EXIT_SUCCESS;
}
