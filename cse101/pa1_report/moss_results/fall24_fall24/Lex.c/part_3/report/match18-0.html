#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LEN 160

int main(int argc, char* argv[]) {
    // Check command line arguments
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        return EXIT_FAILURE;
    }

    // Open input file


    FILE* in = fopen(argv[1], "r");
    if (in == NULL) {
        fprintf(stderr, "Error: Unable to open file %s for reading\n", argv[1]);
        return EXIT_FAILURE;
    }

    // Count the number of lines in the file
    int lineCount = 0;
    char line[MAX_LEN];
    while (fgets(line, MAX_LEN, in) != NULL) {
        lineCount++;
    }
    rewind(in);

    // Allocate memory to store lines from the file
    char** lines = malloc(lineCount * sizeof(char*));
    for (int i = 0; i < lineCount; i++) {
        lines[i] = malloc(MAX_LEN * sizeof(char));
        fgets(lines[i], MAX_LEN, in);
    }
    fclose(in);

    // Create a List to store the indices of lines
    List indices = newList();
    append(indices, 0);  // Add the first index

    // Insert the rest of the indices into the List in sorted order
    for (int i = 1; i < lineCount; i++) {
        moveFront(indices);
        while (index(indices) >= 0 && strcmp(lines[i], lines[get(indices)]) > 0) {


            moveNext(indices);
        }
        if (index(indices) == -1) {
            append(indices, i);
        } else {
            insertBefore(indices, i);
        }
    }

    // Open the output file and write the lines in the sorted order
    FILE* out = fopen(argv[2], "w");
    if (out == NULL) {
        fprintf(stderr, "Error: Unable to open file %s for writing\n", argv[2]);
        return EXIT_FAILURE;
    }

    moveFront(indices);
    while (index(indices) >= 0) {
        fprintf(out, "%s", lines[get(indices)]);
        moveNext(indices);
    }
    fclose(out);

    // Free dynamically allocated memory
    for (int i = 0; i < lineCount; i++) {
        free(lines[i]);
    }
    free(lines);
    freeList(&indices);

    return EXIT_SUCCESS;
}
