* This file reads lines from an input text file and sorts them lexicographically using the List ADT. It writes these to an output file.

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h" 

#define MAX_LINE_LENGTH 1000

int main(int argc, char *argv[]) {
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        exit(EXIT_FAILURE);
    }

    FILE *in = fopen(argv[1], "r");
    if (in == NULL) {
        fprintf(stderr, "Error: Unable to open file %s for reading\n", argv[1]);
        exit(EXIT_FAILURE);
    }

    char **lines = NULL;
    char buffer[MAX_LINE_LENGTH];
    int lineCount = 0;

    while (fgets(buffer, MAX_LINE_LENGTH, in) != NULL) {
        lineCount++;
    }

    lines = (char **)malloc(lineCount * sizeof(char *));
    rewind(in); 

    int i = 0;
    while (fgets(buffer, MAX_LINE_LENGTH, in) != NULL) {
        buffer[strcspn(buffer, "\n")] = '\0'; 
        lines[i] = malloc(strlen(buffer) + 1); 
        strcpy(lines[i], buffer);             
        i++;
    }
    fclose(in); 

    List list = newList();


    for (i = 0; i < lineCount; i++) {
        moveFront(list);
        while (index(list) >= 0 && strcmp(lines[get(list)], lines[i]) < 0) {
            moveNext(list);
        }

        if (index(list) >= 0) {
            insertBefore(list, i);
        } else {
            append(list, i);
        }
    }

    FILE *out = fopen(argv[2], "w");
    if (out == NULL) {
        fprintf(stderr, "Error: Unable to open file %s for writing\n", argv[2]);


        exit(EXIT_FAILURE);
    }

    moveFront(list);
    while (index(list) >= 0) {
        fprintf(out, "%s\n", lines[get(list)]);
        moveNext(list);
    }

    for (i = 0; i < lineCount; i++) {
        free(lines[i]);
    }
    free(lines);
    freeList(&list);
    fclose(out);

    return 0;
}
