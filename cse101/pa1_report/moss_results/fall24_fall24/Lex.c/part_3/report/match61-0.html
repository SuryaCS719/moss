#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>
#include <string.h>
#include "List.h"

#define MAX_LEN 300

int main(int argc, char** argv) {
    int line_count = 0;


    FILE *in, *out;
    char buff[MAX_LEN];

    // Checking Command Line for correct number of arguments (need two arguments)
    if( argc != 3 ){
        printf("Usage: %s <input file> <output file>\n", argv[0]);
        exit(1);
    }

    // Opening files for reading and writing 
    in = fopen(argv[1], "r");
        if( in==NULL ){
        printf("Unable to open file %s for reading\n", argv[1]);
        exit(1);
    }

    out = fopen(argv[2], "w");
    if( out==NULL ){
        printf("Unable to open file %s for writing\n", argv[2]);
        exit(1);
    }

    // Count number of lines in file
    while (fgets(buff, MAX_LEN, in) != NULL) {
        line_count++;
    }
    rewind(in);

    char** lines = (char**)calloc(line_count, sizeof(char *));
    if(lines==NULL){
        printf("Memory allocation failed");
        exit(1);
    }

    // Read lines and place them into array
    for (int i = 0; i < line_count; i++) {
        if (fgets(buff, MAX_LEN, in) != NULL) {
            lines[i] = (char *)malloc(MAX_LEN + 1); // Allocate memory for each line (+1 for null char)
            if (lines[i] == NULL) {
                printf("Memory allocation failed");
                exit(1);
            }
            strcpy(lines[i], buff);
        }
    }

    List A = newList();
    for (int i = 0; i < line_count; i++) {
        while ((index(A) >=0)) {
            if (strcmp(lines[get(A)], lines[i]) < 0) {
                insertAfter(A, i);
                break;
            } 
            else {
                movePrev(A);
            }
        }
        if (index(A) < 0) {
            prepend(A, i);
        }
        moveBack(A);
    }

    moveFront(A);
    while(index(A)>=0){
        fprintf(out, "%s", lines[get(A)]);
        moveNext(A);
    }

    for (int i = 0; i < line_count; i++) {
        free(lines[i]);
    }
    free(lines);
    freeList(&A);

    // close files 
    fclose(in);
    fclose(out);

    return(0);
}
