#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LINE_LENGTH 300

int main(int argc, char* argv[]) {
    // Handle command line arguments
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        return EXIT_FAILURE;
    }

    // Open the input and output files
    FILE* in = fopen(argv[1], "r");
    FILE* out = fopen(argv[2], "w");

    if (in == NULL || out == NULL) {
        fprintf(stderr, "Error: Unable to open file.\n");
        return EXIT_FAILURE;
    }

    int lineCount = 0;
    char line[MAX_LINE_LENGTH];

    // First pass to count the number of lines
    while (fgets(line, MAX_LINE_LENGTH, in) != NULL) {
        lineCount++;
    }

    // Check if the file is empty
    if (lineCount == 0) {
        fprintf(stderr, "Error: The input file is empty.\n");
        fclose(in);
        fclose(out);
        return EXIT_FAILURE;
    }

    // Allocate memory for lines
    rewind(in);


    char **lines = malloc(lineCount * sizeof(char*));
    if (lines == NULL) {
        fprintf(stderr, "Memory allocation failed for lines array.\n");
        fclose(in);
        fclose(out);
        return EXIT_FAILURE;
    }

    // Read each line, allocate memory, and store the content
    for (int i = 0; i < lineCount; i++) {
        if (fgets(line, MAX_LINE_LENGTH, in) != NULL) {
            // Remove newline character from the line (if any)
            size_t len = strlen(line);
            if (line[len - 1] == '\n') {


                line[len - 1] = '\0';
            }
            lines[i] = malloc((strlen(line) + 1) * sizeof(char));  // +1 for null terminator
            if (lines[i] == NULL) {
                fprintf(stderr, "Memory allocation failed for line %d.\n", i);
                // Free already allocated lines and exit
                for (int j = 0; j < i; j++) {
                    free(lines[j]);
                }
                free(lines);
                fclose(in);
                fclose(out);
                return EXIT_FAILURE;
            }
            strcpy(lines[i], line);  // Copy the content of the line
        } else {
            fprintf(stderr, "Error reading line %d.\n", i);
            // Handle the error and clean up
            for (int j = 0; j < i; j++) {
                free(lines[j]);
            }
            free(lines);
            fclose(in);
            fclose(out);
            return EXIT_FAILURE;
        }
    }

    // Create the List
    List indicesList = newList();
    append(indicesList, 0);

    // Indirect insertion sort
    for (int i = 1; i < lineCount; i++) {
        moveFront(indicesList);
        while (index(indicesList) >= 0) {
            int idx = get(indicesList);
            if (strcmp(lines[i], lines[idx]) < 0) {
                insertBefore(indicesList, i);
                break;
            }
            moveNext(indicesList);
        }
        if (index(indicesList) == -1) {
            append(indicesList, i);
        }
    }

    // Write sorted lines to output file, adding newline characters after each line
    moveFront(indicesList);
    while (index(indicesList) >= 0) {
        fprintf(out, "%s\n", lines[get(indicesList)]);  // Add a newline after each line
        moveNext(indicesList);
    }

    // Clean up
    for (int i = 0; i < lineCount; i++) {
        free(lines[i]);
    }
    free(lines);
    freeList(&indicesList);

    fclose(in);
    fclose(out);

    exit(EXIT_SUCCESS);
}
