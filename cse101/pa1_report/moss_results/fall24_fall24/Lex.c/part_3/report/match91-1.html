#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LINE_LENGTH 256

int main(int argc, char* argv[]) {
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        exit(EXIT_FAILURE);
    }

    char* inputFileName = argv[1];
    char* outputFileName = argv[2];

   
    FILE* inputFile = fopen(inputFileName, "r");
    if (inputFile == NULL) {
        fprintf(stderr, "Error: unable to open input file %s\n", inputFileName);
        exit(EXIT_FAILURE);
    }

    FILE* outputFile = fopen(outputFileName, "w");
    if (outputFile == NULL) {
        fprintf(stderr, "Error: unable to open output file %s\n", outputFileName);
        fclose(inputFile);
        exit(EXIT_FAILURE);
    }

    // reda
    char** lines = NULL;
    int lineCount = 0;
    int capacity = 10;

    lines = malloc(capacity * sizeof(char*));
    if (lines == NULL) {
        fprintf(stderr, "Error: unable to allocate memory for lines\n");
        fclose(inputFile);
        fclose(outputFile);
        exit(EXIT_FAILURE);
    }

    char buffer[MAX_LINE_LENGTH];
    while (fgets(buffer, MAX_LINE_LENGTH, inputFile) != NULL) {
        if (lineCount >= capacity) {
            capacity *= 2;
            lines = realloc(lines, capacity * sizeof(char*));
            if (lines == NULL) {
                fprintf(stderr, "Error: unable to reallocate memory for lines\n");
                fclose(inputFile);
                fclose(outputFile);
                exit(EXIT_FAILURE);
            }
        }


        lines[lineCount] = malloc((strlen(buffer) + 1) * sizeof(char));
        if (lines[lineCount] == NULL) {
            fprintf(stderr, "Error: unable to allocate memory for line\n");
            fclose(inputFile);
            fclose(outputFile);
            exit(EXIT_FAILURE);
        }
        strcpy(lines[lineCount], buffer);
        lineCount++;
    }

    
    List sortedIndices = newList();
    for (int i = 0; i < lineCount; i++) {
        moveFront(sortedIndices);


        while (index(sortedIndices) >= 0 && strcmp(lines[i], lines[get(sortedIndices)]) > 0) {
            moveNext(sortedIndices);
        }
        if (index(sortedIndices) == -1) {
            append(sortedIndices, i);
        } else {
            insertBefore(sortedIndices, i);
        }
    }

    //write to output
    for (moveFront(sortedIndices); index(sortedIndices) >= 0; moveNext(sortedIndices)) {
        fprintf(outputFile, "%s", lines[get(sortedIndices)]);
    }

    
    fclose(inputFile);
    fclose(outputFile);
    for (int i = 0; i < lineCount; i++) {
        free(lines[i]);
    }
    free(lines);
    freeList(&sortedIndices);

    return 0;
}
