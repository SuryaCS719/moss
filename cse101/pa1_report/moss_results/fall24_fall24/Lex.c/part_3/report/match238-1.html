 * Reads lines from an input file, indirectly sorts them using the Insertion Sort algorithm.
 ***/

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LEN 300

int main(int argc, char *argv[]) {
    int lines;
    int i;
    char string[MAX_LEN];
    FILE *in;
    FILE *out;


    List stringList = newList();

    // Check command line for correct number of arguments
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        exit(1);
    }

    // Open input file for reading
    in = fopen(argv[1], "r");
    if (in == NULL) {
        fprintf(stderr, "Unable to read from file %s\n", argv[1]);
        exit(1);
    }

    // Open output file for writing


    out = fopen(argv[2], "w");
    if (out == NULL) {
        fprintf(stderr, "Unable to write to file %s\n", argv[2]);
        exit(1);
    }

    //count the number of lines
    lines = 0;
    while (fgets(string, MAX_LEN, in) != NULL) {
        lines++;
    }

    char **stringArray = calloc(lines, sizeof(char*));
    for (i = 0; i < lines; i++) {
        stringArray[i] = calloc(MAX_LEN, sizeof(char));
    }

    // Reset in
    fclose(in);
    in = fopen(argv[1], "r");

    // Fill stringArray
    lines = 0;
    while (fgets(string, MAX_LEN, in) != NULL) {
        strcpy(stringArray[lines], string);
        lines++;
    }

    //sort
    for (i = 0; i < lines; i++) {

        moveBack(stringList);

        while (index(stringList) > -1
                && strcmp(stringArray[i], stringArray[get(stringList)]) < 0) {

            if (index(stringList) == 0) {
                break;
            }

            movePrev(stringList);
        }

        if (index(stringList) == -1) {
            prepend(stringList, i);
        } else if (strcmp(stringArray[i], stringArray[get(stringList)]) > 0) {
            insertAfter(stringList, i);
        } else {
            prepend(stringList, i);
        }
    }

    // Print stringList to out
    moveFront(stringList);
    while (index(stringList) > -1) {
        // Check if the line already contains a newline at the end
        char *currentLine = stringArray[get(stringList)];
        size_t len = strlen(currentLine);

        if (len > 0 && currentLine[len - 1] != '\n') {

            // if the line does not end with a newline add it

            fprintf(out, "%s\n", currentLine);
        } else {
            fprintf(out, "%s", currentLine);
        }
        moveNext(stringList);
    }

    freeList(&stringList);
    stringList = NULL;

    fclose(in);
    fclose(out);

    for (i = 0; i < lines; i++) {
        free(stringArray[i]);
    }
    free(stringArray);

    return 0;
}
