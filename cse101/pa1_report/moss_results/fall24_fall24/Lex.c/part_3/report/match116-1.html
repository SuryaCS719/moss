#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"


#define MAX_LEN 300

int main(int argc, char *argv[]) {

    FILE *in, *out;
    char line[MAX_LEN];
    int line_count = 0;

    // check command line for correct number of arguments
    if( argc != 3 ){
        printf("Usage: %s <input file> <output file>\n", argv[0]);
        exit(1);
    }

    //Opening input file
    in = fopen(argv[1], "r");
        if( in==NULL ){
        printf("Unable to open file %s for reading\n", argv[1]);
        exit(1);
    }
    

    //Counting lines in the in file


    while (fgets(line, MAX_LEN, in) != NULL) {
        line_count++;
    }

    //Allocating memory for lines
    char **lines = calloc(line_count, sizeof(char*));
    if (lines == NULL) {
        fprintf(stderr, "Error: Memory allocation failed\n");
        exit(EXIT_FAILURE);
    }

    
    rewind(in); //sets the file position indicator to the beginning of file


    List L = newList(); //new list

    


    for (int i = 0; i < line_count; i++) {
        //reading each line
        fgets(line, MAX_LEN, in);
        lines[i] = calloc(strlen(line) + 1, sizeof(char));
        if (lines[i] == NULL) {
            fprintf(stderr, "Error: Memory allocation failed\n");
            exit(EXIT_FAILURE);
        }
        strcpy(lines[i], line); //puts the value just read into the array

        //inserting into correct position in L
        moveFront(L);
        while (index(L) >= 0 && strcmp(lines[i], lines[get(L)]) > 0) { //checking to see if lines[i] should be inserted before or after lines[get(L)]
            moveNext(L);
        }
        if (index(L) >= 0) {
            insertBefore(L, i);
        } else {
            append(L, i);
        }
    }

    fclose(in); //closing in file 

    //opening out file
    out = fopen(argv[2], "w");
    if( out==NULL ){
        printf("Unable to open file %s for writing\n", argv[2]);
        exit(1);
    }

    // Write sorted lines to the output file using the indices in List L
    moveFront(L);
    while (index(L) >= 0) {
        fprintf(out, "%s", lines[get(L)]);
        moveNext(L);
    }
    fclose(out); // Close the output file

    // Free allocated memory
    for (int i = 0; i < line_count; i++) {
        free(lines[i]);
    }
    free(lines);
    freeList(&L);

    return EXIT_SUCCESS;
}
