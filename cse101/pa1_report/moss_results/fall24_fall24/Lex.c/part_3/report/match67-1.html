#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"
#define MAX_LENGTH 256

int main(int argc, char* argv[]) {
	char** lines = NULL;
        char line[MAX_LENGTH];
	
	if (argc != 3) {
        	fprintf(stderr,"Usage: <input file> <output file>\n");
        	exit(EXIT_FAILURE);
    	}

    	//Reading file
    	FILE* inputFile = fopen(argv[1], "r");
    	if (inputFile == NULL) {
        	fprintf(stderr,"Error opening file for reading\n");
        	exit(EXIT_FAILURE);
    	}

      	//Counting lines
	int numof_Lines = 0;
    	while (fgets(line, MAX_LENGTH, inputFile) != NULL) {
        	numof_Lines += 1;
    	}

    	//Allocating memory for lines
    	lines = calloc(numof_Lines, sizeof(char*));

    	//Rewinding and reading lines
    	rewind(inputFile);
    	int indexLines;
	for (int i = 0; i < numof_Lines; i++) {
        	fgets(line, MAX_LENGTH, inputFile);
		indexLines = strlen(line) + 1;
        	
		lines[i] = calloc(indexLines, sizeof(char));
        	strcpy(lines[i], line);
    	}


    	fclose(inputFile); //I FORGOT TO CLOSE :(

    	//Creating the list
	//Sorting indices
    	List L = newList();
    	for (int i = 0; i < numof_Lines; i++) {
	    	if (length(L) == 0){
		    	append(L, i);
	    	}
	    	else{
		    	moveFront(L);
		    	while(index(L) >= 0 && strcmp(lines[get(L)], lines[i]) <= 0) {
			    	moveNext(L);
		    	}


		    	if (index(L) >= 0) {
			    	insertBefore(L, i);
		    	}
		    	else{
			    	append(L, i);
		    	}
	    	}
    	}

    	//Writing to output file
    	FILE* outFile = fopen(argv[2], "w");
    	if (outFile == NULL) {
        	fprintf(stderr,"Unable to open file for writing\n");


        	exit(EXIT_FAILURE);
    	}

    	moveFront(L);
    	while (index(L) >= 0) {
        	fprintf(outFile, "%s", lines[get(L)]);
        	moveNext(L);
    	}

    	fclose(outFile); //CLOSING!!

    	//Freeing memory
    	for (int i = 0; i < numof_Lines; i++) {
        	free(lines[i]);
    	}
    	free(lines);
    	freeList(&L);
	return 0;
}
