#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LEN 300

int main(int argc, char *argv[]) {
    if (argc != 3) {
        fprintf(stderr, "%s\n", argv[0]);
        return EXIT_FAILURE;
    }

    FILE *inputFile = fopen(argv[1], "r");
    if (inputFile == NULL) {
        fprintf(stderr, "%s\n", argv[1]);
        return EXIT_FAILURE;
    }

    FILE *outputFile = fopen(argv[2], "w");


    if (outputFile == NULL) {
        fprintf(stderr, "%s\n", argv[2]);
        fclose(inputFile);
        return EXIT_FAILURE;
    }

    char buffer[MAX_LEN];
    int numlines = 0;

    while (fgets(buffer, MAX_LEN, inputFile) != NULL) {
        numlines++;
    }

    rewind(inputFile);

    char **lineA = (char **)calloc(numlines, sizeof(char *));
    if (lineA == NULL) {
        fprintf(stderr, "fail\n");
        fclose(inputFile);
        fclose(outputFile);
        return EXIT_FAILURE;
    }

    for (int i = 0; i < numlines; i++) {
        fgets(buffer, MAX_LEN, inputFile);
        lineA[i] = (char *)malloc(strlen(buffer) + 1);
        if (lineA[i] == NULL) {
            fprintf(stderr, "Error: Memory allocation failed for line %d\n", i);
            for (int j = 0; j < i; j++) {
                free(lineA[j]);
            }
            free(lineA);
            fclose(inputFile);
            fclose(outputFile);
            return EXIT_FAILURE;
        }
        strcpy(lineA[i], buffer);
    }

    List New = newList();



    for (int i = 0; i < numlines; i++) {
        if (length(New) == 0) {
            append(New, i);
        } else {
            moveFront(New);
            while (index(New) >= 0 && strcmp(lineA[i], lineA[get(New)]) > 0) {
                moveNext(New);
            }
            if (index(New) >= 0) {
                insertBefore(New, i);
            } else {
                append(New, i);
            }
        }
    }

    for (moveFront(New); index(New) >= 0; moveNext(New)) {
        fprintf(outputFile, "%s", lineA[get(New)]);
    }

    for (int i = 0; i < numlines; i++) {
        free(lineA[i]);
    }
    free(lineA);

    fclose(inputFile);
    fclose(outputFile);
    freeList(&New);

    return EXIT_SUCCESS;
}
