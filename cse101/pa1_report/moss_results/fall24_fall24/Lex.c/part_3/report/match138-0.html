#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include"List.h"

#define MAX_LEN 255

int main(int argc, char* argv[]) {
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        return EXIT_FAILURE;
    }

    FILE* in = fopen(argv[1], "r");
    if (in == NULL) {
        fprintf(stderr, "Unable to open input file %s\n", argv[1]);
        return EXIT_FAILURE;
    }

    FILE* out = fopen(argv[2], "w");
    if (out == NULL) {
        fprintf(stderr, "Unable to open output file %s\n", argv[2]);
        fclose(in);
        return EXIT_FAILURE;
    }

    char line[MAX_LEN];
    int count = 0;
    char** lines = malloc(1000 * sizeof(char*));

    while (fgets(line, MAX_LEN, in) != NULL) {
        lines[count] = malloc(strlen(line) + 1); // Allocate memory for the line
	if (lines[count] == NULL) {
		fprintf(stderr, "Memory allocation failed.\n");
		exit(EXIT_FAILURE);
	}
	strcpy(lines[count], line); // Copy the line into the allocated space
	count++;
    }



    List indices = newList();
    append(indices, 0);

    for (int i = 1; i < count; i++) {
        moveFront(indices);
        while (index(indices) >= 0) {
            int j = get(indices);
            if (strcmp(lines[i], lines[j]) < 0) {
                insertBefore(indices, i);
                break;
            }
            moveNext(indices);
        }
        if (index(indices) == -1) {
            append(indices, i);
        }
    }

    moveFront(indices);
    while (index(indices) >= 0) {
        fprintf(out, "%s", lines[get(indices)]);
        moveNext(indices);
    }

    for (int i = 0; i < count; i++) {
        free(lines[i]);
    }
    free(lines);
    freeList(&indices);

    fclose(in);
    fclose(out);

    return EXIT_SUCCESS;
}
