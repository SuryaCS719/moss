#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LEN 160

int main(int argc, char* argv[]) {
    if (argc != 3) {


        printf("Usage: %s <input file> <output file>\n", argv[0]);
        exit(1);
    }

    FILE* in = fopen(argv[1], "r");
    FILE* out = fopen(argv[2], "w");

    if (in == NULL) {
        fprintf(stderr, "Error: could not open input file %s\n", argv[1]);
        exit(1);
    }
    if (out == NULL) {
        fprintf(stderr, "Error: could not open output file %s\n", argv[2]);
        fclose(in);
        exit(1);
    }

    // Count the number of lines in the file
    int n = 0;
    char line[MAX_LEN];
    while (fgets(line, MAX_LEN, in) != NULL) {
        n++;
    }

    // Reset file pointer


    rewind(in);

    // Read lines into array
    char** lines = malloc(n * sizeof(char*));
    for (int i = 0; i < n; i++) {
        fgets(line, MAX_LEN, in);
        lines[i] = malloc(strlen(line) + 1);
        strcpy(lines[i], line);
    }

    // Create a list and insert indices of the lines
    List L = newList();
    append(L, 0);
    for (int i = 1; i < n; i++) {
        moveFront(L);


        while (getindex(L) != -1 && strcmp(lines[get(L)], lines[i]) <= 0) {
            moveNext(L);
        }
        if (getindex(L) == -1) {
            append(L, i);
        } else {
            insertBefore(L, i);
        }
    }

    // Print sorted lines to the output file
    moveFront(L);
    while (getindex(L) != -1) {
        fprintf(out, "%s", lines[get(L)]);
        moveNext(L);
    }

    // Free memory
    for (int i = 0; i < n; i++) {
        free(lines[i]);
    }
    free(lines);
    freeList(&L);

    fclose(in);
    fclose(out);

    return 0;
}
