#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LEN 160

int main(int argc, char* argv[]) {
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        return EXIT_FAILURE;
    }

    FILE* input = fopen(argv[1], "r");


    FILE* output = fopen(argv[2], "w");

    if (input == NULL) {
        fprintf(stderr, "Error: unable to open input file %s\n", argv[1]);
        return EXIT_FAILURE;
    }
    if (output == NULL) {
        fprintf(stderr, "Error: unable to open output file %s\n", argv[2]);
        fclose(input);
        return EXIT_FAILURE;
    }

    char line[MAX_LEN];
    int lineCount = 0;
    while (fgets(line, MAX_LEN, input) != NULL) {
        lineCount++;
    }

    char** lines = malloc(lineCount * sizeof(char*));
    rewind(input); 

    int i = 0;
    while (fgets(line, MAX_LEN, input) != NULL) {
        lines[i] = malloc((strlen(line) + 1) * sizeof(char)); 
        strcpy(lines[i], line);
        i++;
    }

    List indexList = newList();

    for (int j = 0; j < lineCount; j++) {
        moveFront(indexList); 
        while (index(indexList) >= 0 && strcmp(lines[j], lines[get(indexList)]) > 0) {
            moveNext(indexList);
        }
        if (index(indexList) >= 0) {
            insertBefore(indexList, j);
        } else {
            append(indexList, j);
        }
    }

    moveFront(indexList);
    while (index(indexList) >= 0) {
        fprintf(output, "%s", lines[get(indexList)]);
        moveNext(indexList);
    }

    for (int k = 0; k < lineCount; k++) {
        free(lines[k]);
    }
    free(lines);
    freeList(&indexList);

    fclose(input);
    fclose(output);

    return EXIT_SUCCESS;
}
