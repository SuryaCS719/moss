#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LEN 255 

int main(int argc, char* argv[]) {
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        exit(EXIT_FAILURE);
    }

    FILE *input = fopen(argv[1], "r");
    if (input == NULL) {


        fprintf(stderr, "Error: Could not open file(s)\n");
        exit(EXIT_FAILURE);
    }

    FILE *output = fopen(argv[2], "w");
    if (output == NULL) {
        fprintf(stderr, "Error: Could not open file(s)\n");
        fclose(input);
        exit(EXIT_FAILURE);
    }

    char line[MAX_LEN];
    int lineCount = 0;

    // First, count the number of lines in the input file
    while (fgets(line, MAX_LEN, input) != NULL) {
        lineCount++;
    }

    // Allocate memory for storing each line
    char** lines = malloc(lineCount * sizeof(char*));
    rewind(input);

    // Store each line into the array
    for (int i = 0; i < lineCount; i++) {
        fgets(line, MAX_LEN, input);
        lines[i] = malloc(strlen(line) + 1);  // Manually allocate memory
        if (lines[i] == NULL) {
            fprintf(stderr, "Memory allocation error\n");
            exit(EXIT_FAILURE);
        }
        strcpy(lines[i], line);  // Copy the line into the allocated memory
    }

    List L = newList();  // Create a new list to store the indices



    for (int i = 0; i < lineCount; i++) {
        // Insertion sort using List ADT
        if (length(L) == 0) {
            append(L, i);  // Insert the first index if the list is empty
        } else {
            moveFront(L);
            while (index(L) >= 0 && strcmp(lines[get(L)], lines[i]) < 0) {
                moveNext(L);  // Move to the correct insertion point
            }
            if (index(L) == -1) {
                append(L, i);  // If we reach the end, append
            } else {
                insertBefore(L, i);  // Insert before the current element
            }
        }
    }

    // Write the sorted lines to the output file
    for (moveFront(L); index(L) >= 0; moveNext(L)) {
        fprintf(output, "%s", lines[get(L)]);
    }

    // Free allocated memory
    for (int i = 0; i < lineCount; i++) {
        free(lines[i]);
    }
    free(lines);
    freeList(&L);
    fclose(input);
    fclose(output);

    return 0;
}
