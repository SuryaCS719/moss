#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"


int main (int argc, char *argv[]){
    
    //check correct # of arguements
    if(argc != 3){
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        return EXIT_FAILURE;
    }

    //open input file
    FILE *in = fopen(argv[1], "r");
    if(in == NULL){
        fprintf(stderr, "Unable to open file %s for reading\n", argv[1]);
        return EXIT_FAILURE;
    }


    //open output file
    FILE *out = fopen(argv[2], "w");
    if (out == NULL) {
        fprintf(stderr, "Unable to open file %s for writing\n", argv[2]);
        fclose(in);
        return EXIT_FAILURE;
    }

    int line = 0;
    char c;

    //count number of lines in the file
    while((c = getc(in)) != EOF){
        if(c == '\n'){
            line++;
        }
    }

    rewind(in);

    //allocate memory
    char **lines = calloc(line, sizeof(char *));
    
    //null case
    if(lines == NULL){
        fprintf(stderr, " Memory Allocation Failure\n");
        fclose(in);
        fclose(out);
        return EXIT_FAILURE;
    }

    char buffer[2048];
    for(int i = 0; i < line; i++){
        if(fgets(buffer, sizeof(buffer), in) != NULL){
            lines[i] = calloc(strlen(buffer) + 1, sizeof(char));
            if(lines[i] == NULL){
                fprintf(stderr, "Memory Allocation Failure\n");


                for(int j = 0; j < i; j++){
                    free(lines[j]);
                }
                free(lines);
                fclose(in);
                fclose(out);
                return EXIT_FAILURE;
            }
            strcpy(lines[i], buffer);
        }
    }

    //sort the lines (insertion sort)
    List L = newList();
    append(L, 0);

    for(int i = 1; i < line; i++){
        moveFront(L);


        while(index(L) >= 0  && strcmp(lines[i], lines[get(L)]) > 0){
            moveNext(L);
        }   
        if(index(L) >= 0){
            insertBefore(L, i);
        } else {
            append(L, i);
        }
    }

    //write to file
    for(moveFront(L); index(L) >= 0; moveNext(L)){
        fprintf(out, "%s", lines[get(L)]);
    }


    //free memory
    for(int i = 0; i < line; i++){
        free(lines[i]);
    }
    clear(L);
    free(lines);
    freeList(&L);
    fclose(in);
    fclose(out);
    return EXIT_SUCCESS;
}
