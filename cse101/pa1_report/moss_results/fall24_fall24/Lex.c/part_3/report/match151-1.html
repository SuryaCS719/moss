#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LEN 160

int main(int argc, char* argv[]) {

    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        exit(EXIT_FAILURE);
    }

    FILE* inFile = fopen(argv[1], "r");
    if (inFile == NULL) {
        fprintf(stderr, "Unable to open file %s for reading\n", argv[1]);
        exit(EXIT_FAILURE);
    }


    FILE* outFile = fopen(argv[2], "w");
    if (outFile == NULL) {
        fprintf(stderr, "Unable to open file %s for writing\n", argv[2]);
        fclose(inFile);
        exit(EXIT_FAILURE);
    }

    char line[MAX_LEN];
    int lineCount = 0;
    while (fgets(line, MAX_LEN, inFile) != NULL) {
        lineCount++;
    }

    char** lines = malloc(lineCount * sizeof(char*));
    if (lines == NULL) {
        fprintf(stderr, "Memory allocation failed\n");
        fclose(inFile);
        fclose(outFile);
        exit(EXIT_FAILURE);
    }
    rewind(inFile);

    for (int i = 0; i < lineCount; i++) {
        if (fgets(line, MAX_LEN, inFile) != NULL) {
            lines[i] = strdup(line);
            if (lines[i] == NULL) {
                fprintf(stderr, "Memory allocation failed\n");
                for (int j = 0; j < i; j++) {
                    free(lines[j]);
                }
                free(lines);
                fclose(inFile);
                fclose(outFile);


                exit(EXIT_FAILURE);
            }
        }
    }

    List list = newList();
    for (int i = 0; i < lineCount; i++) {
        moveFront(list);
        while (index(list) >= 0) {
            int cmp = strcmp(lines[i], lines[get(list)]);
            if (cmp < 0) {
                insertBefore(list, i);
                break;
            }
            moveNext(list);
        }
        if (index(list) < 0) {
            append(list, i);
        }
    }

    moveFront(list);
    while (index(list) >= 0) {
        fprintf(outFile, "%s", lines[get(list)]);
        moveNext(list);
    }

    for (int i = 0; i < lineCount; i++) {
        free(lines[i]);
    }
    free(lines);
    freeList(&list);
    fclose(inFile);
    fclose(outFile);

    return 0;
}
