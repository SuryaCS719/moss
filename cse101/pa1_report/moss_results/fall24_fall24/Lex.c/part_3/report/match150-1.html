#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

int main(int argc, char* argv[]) {
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        exit(EXIT_FAILURE);
    }

    FILE* in = fopen(argv[1], "r");
    if (in == NULL) {
        fprintf(stderr, "Error: could not open file %s\n", argv[1]);
        exit(EXIT_FAILURE);
    }

    FILE* out = fopen(argv[2], "w");


    if (out == NULL) {
        fprintf(stderr, "Error: could not open file %s\n", argv[2]);
        fclose(in);
        exit(EXIT_FAILURE);
    }

    int lineCount = 0;
    char buffer[256];
    while (fgets(buffer, 256, in) != NULL) {
        lineCount++;
    }
    rewind(in);

    char** lines = calloc(lineCount, sizeof(char*));
    for (int i = 0; i < lineCount; i++) {


        fgets(buffer, 256, in);
        lines[i] = calloc(strlen(buffer) + 1, sizeof(char));
        strcpy(lines[i], buffer);
    }

    List list = newList();
    append(list, 0);

    for (int i = 1; i < lineCount; i++) {
        moveFront(list);
        while (index(list) >= 0 && strcmp(lines[get(list)], lines[i]) < 0) {
            moveNext(list);
        }
        if (index(list) == -1) {
            append(list, i);
        } else {
            insertBefore(list, i);
        }
    }

    moveFront(list);
    while (index(list) >= 0) {
        fprintf(out, "%s", lines[get(list)]);
        moveNext(list);
    }

    for (int i = 0; i < lineCount; i++) {
        free(lines[i]);
    }
    free(lines);
    freeList(&list);
    fclose(in);
    fclose(out);

    return 0;
}
