This file implements a lexical sorting program that reads lines from an input file, stores them in a list, sorts them lexicographically using the List ADT, and writes the sorted output to a file.

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

int main(int argc, char* argv[]) {
    // Check command-line arguments
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        return EXIT_FAILURE;
    }

    // Open input and output files
    FILE* inFile = fopen(argv[1], "r");


    FILE* outFile = fopen(argv[2], "w");
    if (inFile == NULL) {
        fprintf(stderr, "Error: Unable to open input file %s\n", argv[1]);
        return EXIT_FAILURE;
    }
    if (outFile == NULL) {
        fprintf(stderr, "Error: Unable to open output file %s\n", argv[2]);
        fclose(inFile);  // Close input file since we can't proceed
        return EXIT_FAILURE;
    }

    // Read the input file line by line and store it in an array
    char buffer[256];  // Assumes no line exceeds 256 characters
    int lineCount = 0;
    char** lines = NULL;

    // Read each line into the array
    while (fgets(buffer, sizeof(buffer), inFile)) {
        lines = realloc(lines, (lineCount + 1) * sizeof(char*));
        lines[lineCount] = malloc(strlen(buffer) + 1); // Allocate memory for the line


        strcpy(lines[lineCount], buffer);  // Copy buffer into the array
        lineCount++;
    }

    // Create a List for the indices of the array
    List list = newList();

    // Perform indirect sorting based on strcmp
    for (int i = 0; i < lineCount; i++) {
        if (length(list) == 0) {
            append(list, i);  // First element
        } else {
            moveFront(list);
            while (index(list) >= 0 && strcmp(lines[i], lines[get(list)]) > 0) {
                moveNext(list);
            }
            if (index(list) >= 0) {
                insertBefore(list, i);
            } else {
                append(list, i);
            }
        }
    }

    // Write the sorted lines to the output file using the sorted indices
    for (moveFront(list); index(list) >= 0; moveNext(list)) {
        fprintf(outFile, "%s", lines[get(list)]);
    }

    // Cleanup
    freeList(&list);
    for (int i = 0; i < lineCount; i++) {
        free(lines[i]);
    }
    free(lines);

    // Close files
    fclose(inFile);
    fclose(outFile);

    return EXIT_SUCCESS;
}
