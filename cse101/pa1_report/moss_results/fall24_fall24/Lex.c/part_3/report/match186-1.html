#include "List.h"
#include <string.h>
#include <stdio.h>
#include <stdlib.h>
#include <assert.h>
#include <stdbool.h>
char *strdup(const char *s);

int main(int argc, char *argv[]) {
    FILE *in, *out;



    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        exit(1);
    }

    in = fopen(argv[1], "r");
    out = fopen(argv[2], "w");
    if (in == NULL) {
        fprintf(stderr, "Error: Can't open file %s for reading \n", argv[1]);
        exit(1);
    }  
    if (out == NULL) {
        fprintf(stderr, "Error: Can't open file %s for writing \n", argv[2]);
        fclose(in);
        exit(1);
    }

    char line[256];
    int n = 0;

    while (fgets(line, 256, in) != NULL) {
        n++;
    } 

    char **array = malloc(n * sizeof(char *));

    rewind(in);
    for (int i = 0; i < n; i ++) {
        fgets(line, 256, in);
        array[i] = strdup(line);
    }

    List indices = newList();

    for (int i = 0; i < n; i ++) {
        if (length(indices) == 0) {
            append(indices, i);
        } else {
            moveFront(indices);
            while (index(indices)>= 0 && strcmp(array[i], array[get(indices)]) > 0) {
                moveNext(indices);
            }
            if (index(indices) == -1) {
                append(indices, i);
            } else {
                insertBefore(indices, i);
            }
        }
    }

    moveFront(indices);
    while (index(indices) >= 0) {
        fprintf(out, "%s", array[get(indices)]);
        moveNext(indices);
    }

    for (int i = 0; i < n; i ++) {
        free(array[i]);
    }
    free(array);
    freeList(&indices);

    fclose(in);
    fclose(out);
    return 0;
}
