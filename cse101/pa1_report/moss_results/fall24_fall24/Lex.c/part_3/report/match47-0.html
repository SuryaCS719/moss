#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LEN 160

int main(int argc, char* argv[]) {
   
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        return EXIT_FAILURE;
    }

   
    FILE* inFile = fopen(argv[1], "r");
    if (inFile == NULL) {
        fprintf(stderr, "Error: Cannot open input file %s\n", argv[1]);
        return EXIT_FAILURE;
    }



    FILE* outFile = fopen(argv[2], "w");
    if (outFile == NULL) {
        fprintf(stderr, "Error: Cannot open output file %s\n", argv[2]);
        fclose(inFile);
        return EXIT_FAILURE;
    }

    
    char line[MAX_LEN];
    int lineCount = 0;
    while (fgets(line, MAX_LEN, inFile) != NULL) {
        lineCount++;
    }

    rewind(inFile); 
    char** lines = calloc(lineCount, sizeof(char*));
    for (int i = 0; i < lineCount; i++) {
        fgets(line, MAX_LEN, inFile);
        lines[i] = malloc(strlen(line) + 1);  


        strcpy(lines[i], line);
    }

    List indexList = newList();
    append(indexList, 0);  

    
    for (int i = 1; i < lineCount; i++) {
        moveFront(indexList);
        while (index(indexList) != -1 && strcmp(lines[i], lines[get(indexList)]) > 0) {


            moveNext(indexList);
        }
        if (index(indexList) == -1) {
            append(indexList, i);  
        } else {
            insertBefore(indexList, i);  
        }
    }

   
    moveFront(indexList);
    while (index(indexList) != -1) {
        fprintf(outFile, "%s", lines[get(indexList)]);
        moveNext(indexList);
    }

   
    for (int i = 0; i < lineCount; i++) {
        free(lines[i]);  
    }
    free(lines);       
    freeList(&indexList); 
    fclose(inFile);
    fclose(outFile);

    return EXIT_SUCCESS;
}
