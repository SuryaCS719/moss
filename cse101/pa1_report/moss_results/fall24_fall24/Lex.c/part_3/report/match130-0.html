#include <string.h>
#include <stdlib.h>
#include <stdio.h>
#include "List.h"
#define MAX_LEN 100000


char* duplicate_string(const char* input) {
    size_t len = strlen(input) + 1;  // Get the length of the input string
    char* result = malloc(len);  // Allocate memory for the copy
    if (result != NULL) {
        memcpy(result, input, len);  // Copy the string using memcpy
    }
    return result;
}

int main(int argc, char *argv[]) {

    FILE *input_file, *output_file;
    char line[MAX_LEN];
    char **lines;
    int total_lines = 0;
    List list = newList();


    if (argc != 3) {
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        return 1;
    }


    input_file = fopen(argv[1], "r");
    if (input_file == NULL) {
        fprintf(stderr, "Error: Could not open file %s for reading.\n", argv[1]);
        return 1;
    }

    output_file = fopen(argv[2], "w");
    if (output_file == NULL) {
        fprintf(stderr, "Error: Could not open file %s for writing.\n", argv[2]);
        fclose(input_file);
        return 1;
    }


    while (fgets(line, MAX_LEN, input_file) != NULL) {
        total_lines++;
    }
    lines = calloc(total_lines, sizeof(char*));

    if (lines == NULL) {
        fprintf(stderr, "Error: Memory allocation failed for lines.\n");
        fclose(input_file);
        fclose(output_file);
        return 1;
    }


    rewind(input_file);
    for (int i = 0; i < total_lines; i++) {
        fgets(line, MAX_LEN, input_file);
        lines[i] = duplicate_string(line);  // Store each line using the custom string duplication function
        if (lines[i] == NULL) {
            fprintf(stderr, "Error: Memory allocation failed for line %d.\n", i);
            for (int j = 0; j < i; j++) {
                free(lines[j]);  // Free previously allocated lines
            }
            free(lines);
            fclose(input_file);
            fclose(output_file);
            return 1;
        }
    }

    append(list, 0);  // Append the first index (0) to the list



    for (int i = 1; i < total_lines; i++) {
        moveFront(list);  // Start comparison from the front of the list
        while (index(list) != -1 && strcmp(lines[i], lines[get(list)]) > 0) {
            moveNext(list);
        }

        // Insert the index before or append if we reach the end of the list
        if (index(list) != -1) {
            insertBefore(list, i);
        } else {
            append(list, i);
        }
    }


    for (moveFront(list); index(list) != -1; moveNext(list)) {
        fprintf(output_file, "%s", lines[get(list)]);
    }


    for (int i = 0; i < total_lines; i++) {
        free(lines[i]);
    }
    free(lines);
    freeList(&list);


    fclose(input_file);
    fclose(output_file);

    return 0;
}
