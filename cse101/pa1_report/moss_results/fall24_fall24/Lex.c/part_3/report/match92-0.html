#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>
#include <string.h>

#include "List.h"

#define MAX_LEN 256

int main(int argc, char* argv[]) {
	// count number of inputs
	if (argc != 3) {
		fprintf(stderr, "Syntax: %s {input file} {output file}\n", argv[0]);
		exit(EXIT_FAILURE);	
	}

	// open input file


	FILE* in = fopen(argv[1], "r");
	if (in == NULL) {
		fprintf(stderr, "Unable to OPEN Input File %s\n", argv[1]);
		exit(EXIT_FAILURE);
	}

	// read and store lines from input file into an array
	char** lines = NULL;
	char line[MAX_LEN];
	int numL = 0;

	// counts number of lines from the input file
	while (fgets(line, MAX_LEN, in) != NULL) {
		numL++;
	}
	
	if (numL == 0) {
		fprintf(stderr, "Input File is EMPTY\n");
		exit(EXIT_FAILURE);
	}

	// dynamically allocate memory for lines
	lines = calloc(numL, sizeof(char*));
	if (lines == NULL) {
		fprintf(stderr, "Memory Allocation Failed\n");
		fclose(in);
		exit(EXIT_FAILURE);
	}

	// read lines into the array
	rewind(in);
	int i = 0;
	while (fgets(line, MAX_LEN, in) != NULL) {
		lines[i] = calloc(strlen(line) + 1, sizeof(char));
		if (lines[i] == NULL) {
			fprintf(stderr, "Line Memory Allocation Failed\n");
			for (int j = 0; j < i; j++) {
				free(lines[j]);
			}
			free(lines);
                	fclose(in);
			exit(EXIT_FAILURE);
		}
		strcpy(lines[i], line);
		i++;
	}
	fclose(in);

	// creates a new list to insert sorted indices
	List L = newList();
	for (int i = 0; i < numL; i++) {
		if (length(L) == 0) {
			append(L, i);
		} else {
			moveFront(L);
			while (index(L) >= 0) {
				int cmp = strcmp(lines[get(L)], lines[i]);
				if (cmp > 0) break;
				moveNext(L);
			}


			if (index(L) >= 0) {
				insertBefore(L, i);
			} else {
				append(L, i);
			}	
		}	

	}

	// opens the output file
	FILE* out = fopen(argv[2], "w");
	if (out == NULL) {
		fprintf(stderr, "Unable to OPEN Output File %s\n", argv[2]);
                exit(EXIT_FAILURE);
	
		for (int i = 0; i < numL; i++) {
		free(lines[i]);
		}
		free(lines);
		freeList(&L);
		exit(EXIT_FAILURE);
	}

	// print the list onto the output file
	moveFront(L);
	while (index(L) >= 0) {
		fprintf(out, "%s", lines[get(L)]);
		moveNext(L);
	}
	fclose(out);

	// free up the used memory
	for (int i = 0; i < numL; i++) {
		free(lines[i]);
	}

	free(lines);
	freeList(&L);
	
	return 0;
}
